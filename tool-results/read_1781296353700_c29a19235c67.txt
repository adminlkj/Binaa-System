   190→type ViewMode = 'list' | 'create' | 'edit' | 'detail'
   191→
   192→export function RentalContractsModule() {
   193→  const { lang } = useAppStore()
   194→  const queryClient = useQueryClient()
   195→  const [viewMode, setViewMode] = useState<ViewMode>('list')
   196→  const [selectedId, setSelectedId] = useState<string | null>(null)
   197→  const [searchQuery, setSearchQuery] = useState('')
   198→  const [statusFilter, setStatusFilter] = useState('all')
   199→
   200→  // Fetch contracts
   201→  const { data: contracts = [], isLoading, refetch } = useQuery<RentalContract[]>({
   202→    queryKey: ['rental-contracts', statusFilter],
   203→    queryFn: async () => {
   204→      const params = new URLSearchParams()
   205→      if (statusFilter !== 'all') params.set('status', statusFilter)
   206→      const r = await fetch(`/api/equipment/rental-contracts?${params.toString()}`)
   207→      if (!r.ok) throw new Error()
   208→      return r.json()
   209→    },
   210→  })
   211→
   212→  // Fetch equipment options
   213→  const { data: equipmentOptions = [] } = useQuery<EquipmentOption[]>({
   214→    queryKey: ['equipment-options'],
   215→    queryFn: async () => {
   216→      const r = await fetch('/api/equipment')
   217→      if (!r.ok) return []
   218→      return r.json()
   219→    },
   220→  })
   221→
   222→  // Fetch client options
   223→  const { data: clientOptions = [] } = useQuery<ClientOption[]>({
   224→    queryKey: ['client-options'],
   225→    queryFn: async () => {
   226→      const r = await fetch('/api/clients?simple=true')
   227→      if (!r.ok) return []
   228→      return r.json()
   229→    },
   230→  })
   231→
   232→  // Fetch project options
   233→  const { data: projectOptions = [] } = useQuery<ProjectOption[]>({
   234→    queryKey: ['project-options'],
   235→    queryFn: async () => {
   236→      const r = await fetch('/api/projects/list')
   237→      if (!r.ok) return []
   238→      return r.json()
   239→    },
   240→  })
   241→
   242→  // Filter contracts by search
   243→  const filteredContracts = useMemo(() => {
   244→    if (!searchQuery) return contracts
   245→    const q = searchQuery.toLowerCase()
   246→    return contracts.filter(c =>
   247→      c.contractNo.toLowerCase().includes(q) ||
   248→      c.equipment?.name?.toLowerCase().includes(q) ||
   249→      c.equipment?.nameAr?.includes(q) ||
   250→      c.client?.name?.toLowerCase().includes(q) ||
   251→      c.purchaseOrderNo?.toLowerCase().includes(q)
   252→    )
   253→  }, [contracts, searchQuery])
   254→
   255→  // KPIs
   256→  const totalContracts = contracts.length
   257→  const activeContracts = contracts.filter(c => c.status === 'ACTIVE').length
   258→  const draftContracts = contracts.filter(c => c.status === 'DRAFT').length
   259→  const expiredContracts = contracts.filter(c => c.status === 'EXPIRED').length
   260→
   261→  // Handlers
   262→  const handleCreate = () => {
   263→    setSelectedId(null)
   264→    setViewMode('create')
   265→  }
   266→
   267→  const handleEdit = (id: string) => {
   268→    setSelectedId(id)
   269→    setViewMode('edit')
   270→  }
   271→
   272→  const handleViewDetail = (id: string) => {
   273→    setSelectedId(id)
   274→    setViewMode('detail')
   275→  }
   276→
   277→  const handleBack = () => {
   278→    setSelectedId(null)
   279→    setViewMode('list')
   280→  }
   281→
   282→  // ============ LIST VIEW ============
   283→  if (viewMode === 'list') {
   284→    return (
   285→      <ModuleLayout
   286→        title={{ ar: 'عقود التأجير', en: 'Rental Contracts' }}
   287→        subtitle={{ ar: 'إدارة عقود تأجير المعدات والآليات', en: 'Manage equipment and machinery rental contracts' }}
   288→        actions={
   289→          <Button onClick={handleCreate} className="bg-emerald-600 hover:bg-emerald-700 gap-2">
   290→            <Plus className="size-4" />
   291→            {t('عقد جديد', 'New Contract', lang)}
   292→          </Button>
   293→        }
   294→      >
   295→        {/* KPI Cards */}
   296→        <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   297→          <Card className="bg-emerald-50 border-emerald-200">
   298→            <CardContent className="p-4 text-center">
   299→              <div className="flex items-center justify-center gap-2 mb-1">
   300→                <FileText className="size-4 text-emerald-600" />
   301→                <p className="text-xs text-emerald-600">{t('إجمالي العقود', 'Total Contracts', lang)}</p>
   302→              </div>
   303→              <p className="text-2xl font-bold text-emerald-700">{formatNumber(totalContracts)}</p>
   304→            </CardContent>
   305→          </Card>
   306→          <Card className="bg-teal-50 border-teal-200">
   307→            <CardContent className="p-4 text-center">
   308→              <div className="flex items-center justify-center gap-2 mb-1">
   309→                <CheckCircle className="size-4 text-teal-600" />
   310→                <p className="text-xs text-teal-600">{t('عقود فعالة', 'Active', lang)}</p>
   311→              </div>
   312→              <p className="text-2xl font-bold text-teal-700">{formatNumber(activeContracts)}</p>
   313→            </CardContent>
   314→          </Card>
   315→          <Card className="bg-yellow-50 border-yellow-200">
   316→            <CardContent className="p-4 text-center">
   317→              <div className="flex items-center justify-center gap-2 mb-1">
   318→                <Edit3 className="size-4 text-yellow-600" />
   319→                <p className="text-xs text-yellow-600">{t('مسودات', 'Drafts', lang)}</p>
   320→              </div>
   321→              <p className="text-2xl font-bold text-yellow-700">{formatNumber(draftContracts)}</p>
   322→            </CardContent>
   323→          </Card>
   324→          <Card className="bg-gray-50 border-gray-200">
   325→            <CardContent className="p-4 text-center">
   326→              <div className="flex items-center justify-center gap-2 mb-1">
   327→                <Clock className="size-4 text-gray-500" />
   328→                <p className="text-xs text-gray-500">{t('منتهية', 'Expired', lang)}</p>
   329→              </div>
   330→              <p className="text-2xl font-bold text-gray-600">{formatNumber(expiredContracts)}</p>
   331→            </CardContent>
   332→          </Card>
   333→        </div>
   334→
   335→        {/* Filters */}
   336→        <div className="flex flex-wrap items-center gap-3">
   337→          <div className="relative flex-1 min-w-[200px]">
   338→            <Search className="absolute right-3 top-1/2 -translate-y-1/2 size-4 text-muted-foreground" />
   339→            <Input
   340→              value={searchQuery}
   341→              onChange={e => setSearchQuery(e.target.value)}
   342→              placeholder={t('بحث برقم العقد، العميل، المعدة...', 'Search by contract no, client, equipment...', lang)}
   343→              className="pr-9"
   344→            />
   345→          </div>
   346→          <Select value={statusFilter} onValueChange={setStatusFilter}>
   347→            <SelectTrigger className="w-[160px]">
   348→              <SelectValue placeholder={t('الحالة', 'Status', lang)} />
   349→            </SelectTrigger>
   350→            <SelectContent>
   351→              <SelectItem value="all">{t('الكل', 'All', lang)}</SelectItem>
   352→              {Object.entries(contractStatusConfig).map(([k, v]) => (
   353→                <SelectItem key={k} value={k}>{v.label[lang]}</SelectItem>
   354→              ))}
   355→            </SelectContent>
   356→          </Select>
   357→          <Button variant="outline" size="icon" onClick={() => refetch()}>
   358→            <RefreshCw className="size-4" />
   359→          </Button>
   360→        </div>
   361→
   362→        {/* Contracts Table */}
   363→        <Card>
   364→          <CardContent className="p-0">
   365→            {isLoading ? (
   366→              <div className="p-4"><TableSkeleton /></div>
   367→            ) : filteredContracts.length === 0 ? (
   368→              <div className="p-8 text-center text-muted-foreground">
   369→                <FileText className="size-12 mx-auto mb-2 opacity-30" />
   370→                <p>{t('لا توجد عقود تأجير', 'No rental contracts found', lang)}</p>
   371→              </div>
   372→            ) : (
   373→              <div className="overflow-x-auto">
   374→                <Table>
   375→                  <TableHeader>
   376→                    <TableRow>
   377→                      <TableHead className="text-right">{t('رقم العقد', 'Contract No.', lang)}</TableHead>
   378→                      <TableHead className="text-right">{t('العميل', 'Client', lang)}</TableHead>
   379→                      <TableHead className="text-right">{t('المعدة', 'Equipment', lang)}</TableHead>
   380→                      <TableHead className="text-right">{t('نوع التسعير', 'Pricing', lang)}</TableHead>
   381→                      <TableHead className="text-right">{t('القيمة', 'Value', lang)}</TableHead>
   382→                      <TableHead className="text-right">{t('الحالة', 'Status', lang)}</TableHead>
   383→                      <TableHead className="text-right">{t('تاريخ البداية', 'Start Date', lang)}</TableHead>
   384→                      <TableHead className="text-right">{t('الإجراءات', 'Actions', lang)}</TableHead>
   385→                    </TableRow>
   386→                  </TableHeader>
   387→                  <TableBody>
   388→                    {filteredContracts.map(contract => {
   389→                      const pricingLabel = pricingTypeOptions.find(p => p.value === contract.pricingType)
   390→                      const contractValue = contract.pricingType === 'HOURLY' ? (contract.referenceRate ?? 0)
   391→                        : contract.pricingType === 'DAILY' ? (contract.dailyRate ?? 0)
   392→                        : contract.pricingType === 'MONTHLY' ? (contract.monthlyRate ?? 0)
   393→                        : (contract.lumpSumAmount ?? 0)
   394→                      return (
   395→                        <TableRow key={contract.id} className="cursor-pointer hover:bg-muted/50" onClick={() => handleViewDetail(contract.id)}>
   396→                          <TableCell className="font-mono font-semibold text-emerald-700">{contract.contractNo}</TableCell>
   397→                          <TableCell>
   398→                            <p className="font-medium">{contract.client?.name || '—'}</p>
   399→                          </TableCell>
   400→                          <TableCell>
   401→                            <div>
   402→                              <p className="font-medium">{contract.equipment?.nameAr || contract.equipment?.name}</p>
   403→                              <p className="text-xs text-muted-foreground">{contract.equipment?.code}</p>
   404→                            </div>
   405→                          </TableCell>
   406→                          <TableCell>
   407→                            <Badge variant="outline" className="text-xs">
   408→                              {pricingLabel ? pricingLabel[lang] : contract.pricingType}
   409→                            </Badge>
   410→                          </TableCell>
   411→                          <TableCell className="font-medium">
   412→                            <MoneyDisplay value={contractValue} lang={lang} size="sm" bold />
   413→                          </TableCell>
   414→                          <TableCell><ContractStatusBadge status={contract.status} lang={lang} /></TableCell>
   415→                          <TableCell>{formatDate(contract.startDate, lang)}</TableCell>
   416→                          <TableCell>
   417→                            <div className="flex items-center gap-1" onClick={e => e.stopPropagation()}>
   418→                              <Button variant="ghost" size="icon" className="size-8" onClick={() => handleViewDetail(contract.id)}>
   419→                                <Eye className="size-4" />
   420→                              </Button>
   421→                              {(contract.status === 'DRAFT' || contract.status === 'UNDER_REVIEW') && (
   422→                                <Button variant="ghost" size="icon" className="size-8" onClick={() => handleEdit(contract.id)}>
   423→                                  <Edit3 className="size-4" />
   424→                                </Button>
   425→                              )}
   426→                            </div>
   427→                          </TableCell>
   428→                        </TableRow>
   429→                      )
   430→                    })}
   431→                  </TableBody>
   432→                </Table>
   433→              </div>
   434→            )}
   435→          </CardContent>
   436→        </Card>
   437→      </ModuleLayout>
   438→    )
   439→  }
   440→
   441→  // ============ CREATE / EDIT VIEW ============
   442→  if (viewMode === 'create' || viewMode === 'edit') {
   443→    return (
   444→      <ContractFormView
   445→        mode={viewMode}
   446→        contractId={selectedId}
   447→        onBack={handleBack}
   448→        equipmentOptions={equipmentOptions}
   449→        clientOptions={clientOptions}
   450→        projectOptions={projectOptions}
   451→      />
   452→    )
   453→  }
   454→
   455→  // ============ DETAIL VIEW ============
   456→  if (viewMode === 'detail' && selectedId) {
   457→    return (
   458→      <ContractDetailView
   459→        contractId={selectedId}
   460→        onBack={handleBack}
   461→        onEdit={() => handleEdit(selectedId)}
   462→      />
   463→    )
   464→  }
   465→
   466→  return null
   467→}

   469→// ============ Contract Form View (6 Sections) ============
   470→function ContractFormView({
   471→  mode, contractId, onBack, equipmentOptions, clientOptions, projectOptions,
   472→}: {
   473→  mode: 'create' | 'edit'
   474→  contractId: string | null
   475→  onBack: () => void
   476→  equipmentOptions: EquipmentOption[]
   477→  clientOptions: ClientOption[]
   478→  projectOptions: ProjectOption[]
   479→}) {
   480→  const { lang } = useAppStore()
   481→  const queryClient = useQueryClient()
   482→
   483→  // For edit mode, fetch the contract
   484→  const { data: editContract, isLoading: editLoading } = useQuery<RentalContract>({
   485→    queryKey: ['rental-contract', contractId],
   486→    queryFn: async () => {
   487→      const r = await fetch(`/api/equipment/rental-contracts/${contractId}`)
   488→      if (!r.ok) throw new Error()
   489→      return r.json()
   490→    },
   491→    enabled: mode === 'edit' && !!contractId,
   492→  })
   493→
   494→  // ═══════ Form State ═══════
   495→  // Section 1: Contract Data
   496→  const [equipmentId, setEquipmentId] = useState('')
   497→  const [clientId, setClientId] = useState('')
   498→  const [projectId, setProjectId] = useState('')
   499→  const [startDate, setStartDate] = useState('')
   500→  const [endDate, setEndDate] = useState('')
   501→
   502→  // Section 2: Work Location
   503→  const [workCity, setWorkCity] = useState('')
   504→  const [workLocation, setWorkLocation] = useState('')
   505→  const [siteSupervisor, setSiteSupervisor] = useState('')
   506→  const [siteSupervisorPhone, setSiteSupervisorPhone] = useState('')
   507→
   508→  // Section 3: Pricing
   509→  const [pricingType, setPricingType] = useState('HOURLY')
   510→  const [referenceRate, setReferenceRate] = useState('')
   511→  const [referenceHours, setReferenceHours] = useState('')
   512→  const [dailyRate, setDailyRate] = useState('')
   513→  const [monthlyRate, setMonthlyRate] = useState('')
   514→  const [lumpSumAmount, setLumpSumAmount] = useState('')
   515→
   516→  // Section 4: Additional Costs
   517→  const [deliveryFeesType, setDeliveryFeesType] = useState('NONE')
   518→  const [deliveryFees, setDeliveryFees] = useState('')
   519→  const [deliveryFeesTaxable, setDeliveryFeesTaxable] = useState(true)
   520→  const [fuelResponsibility, setFuelResponsibility] = useState('ON_CLIENT')
   521→  const [operationMode, setOperationMode] = useState('WITHOUT_DRIVER')
   522→  const [insuranceResponsibility, setInsuranceResponsibility] = useState('ON_CLIENT')
   523→
   524→  // Section 5: Reference Documents
   525→  const [purchaseOrderNo, setPurchaseOrderNo] = useState('')
   526→  const [quotationNo, setQuotationNo] = useState('')
   527→
   528→  // Section 6: Terms
   529→  const [paymentDuration, setPaymentDuration] = useState('')
   530→  const [additionalTerms, setAdditionalTerms] = useState('')
   531→  const [notes, setNotes] = useState('')
   532→
   533→  // Populate form for edit mode
   534→  useEffect(() => {
   535→    /* eslint-disable react-hooks/set-state-in-effect -- Legitimate: populate form fields from API data on load */
   536→    if (mode === 'edit' && editContract) {
   537→      const c = editContract
   538→      setEquipmentId(c.equipmentId)
   539→      setClientId(c.clientId)
   540→      setProjectId(c.projectId || '')
   541→      setStartDate(c.startDate ? new Date(c.startDate).toISOString().split('T')[0] : '')
   542→      setEndDate(c.endDate ? new Date(c.endDate).toISOString().split('T')[0] : '')
   543→      setWorkCity(c.workCity || '')
   544→      setWorkLocation(c.workLocation || '')
   545→      setSiteSupervisor(c.siteSupervisor || '')
   546→      setSiteSupervisorPhone(c.siteSupervisorPhone || '')
   547→      setPricingType(c.pricingType || 'HOURLY')
   548→      setReferenceRate(String(c.referenceRate || ''))
   549→      setReferenceHours(String(c.referenceHours || ''))
   550→      setDailyRate(String(c.dailyRate || ''))
   551→      setMonthlyRate(String(c.monthlyRate || ''))
   552→      setLumpSumAmount(String(c.lumpSumAmount || ''))
   553→      setDeliveryFeesType(c.deliveryFeesType || 'NONE')
   554→      setDeliveryFees(String(c.deliveryFees || ''))
   555→      setDeliveryFeesTaxable(c.deliveryFeesTaxable !== false)
   556→      setFuelResponsibility(c.fuelResponsibility || 'ON_CLIENT')
   557→      setOperationMode(c.operationMode || 'WITHOUT_DRIVER')
   558→      setInsuranceResponsibility(c.insuranceResponsibility || 'ON_CLIENT')
   559→      setPurchaseOrderNo(c.purchaseOrderNo || '')
   560→      setQuotationNo(c.quotationNo || '')
   561→      setPaymentDuration(c.paymentDuration || '')
   562→      setAdditionalTerms(c.additionalTerms || '')
   563→      setNotes(c.notes || '')
   564→    }
   565→  }, [mode, editContract])
   566→

   568→  const refRate = parseFloat(referenceRate) || 0
   569→  const refHours = parseFloat(referenceHours) || 0
   570→  const calculatedHourlyRate = (pricingType === 'HOURLY' && refHours > 0) ? refRate / refHours : 0
   571→

   574→    mutationFn: (data: Record<string, unknown>) =>

   576→        method: 'POST',
   577→        headers: { 'Content-Type': 'application/json' },
   578→        body: JSON.stringify(data),
   579→      }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   580→    onSuccess: () => {
   581→      queryClient.invalidateQueries({ queryKey: ['rental-contracts'] })
   582→      queryClient.invalidateQueries({ queryKey: ['equipment'] })
   583→      onBack()
   584→    },
   585→  })
   586→
   587→  // Update mutation
   588→  const updateMutation = useMutation({
   589→    mutationFn: (data: Record<string, unknown> & { id: string }) =>
   590→      fetch(`/api/equipment/rental-contracts/${data.id}`, {
   591→        method: 'PATCH',
   592→        headers: { 'Content-Type': 'application/json' },
   593→        body: JSON.stringify(data),
   594→      }).then(r => { if (!r.ok) throw new Error(); return r.json() }),

   596→      queryClient.invalidateQueries({ queryKey: ['rental-contracts'] })
   597→      queryClient.invalidateQueries({ queryKey: ['rental-contract', contractId] })
   598→      queryClient.invalidateQueries({ queryKey: ['equipment'] })
   599→      onBack()
   600→    },
   601→  })
   602→

   604→    e.preventDefault()
   605→    const data: Record<string, unknown> = {
   606→      equipmentId,
   607→      clientId,
   608→      projectId: projectId || null,
   609→      startDate,
   610→      endDate: endDate || null,
   611→      // Pricing
   612→      pricingType,
   613→      referenceRate: parseFloat(referenceRate) || 0,
   614→      referenceHours: parseFloat(referenceHours) || 0,
   615→      hourlyRate: calculatedHourlyRate,
   616→      dailyRate: parseFloat(dailyRate) || 0,
   617→      monthlyRate: parseFloat(monthlyRate) || 0,
   618→      lumpSumAmount: parseFloat(lumpSumAmount) || 0,
   619→      // Work Location
   620→      workCity: workCity || null,
   621→      workLocation: workLocation || null,
   622→      siteSupervisor: siteSupervisor || null,
   623→      siteSupervisorPhone: siteSupervisorPhone || null,
   624→      // Delivery
   625→      deliveryFeesType,
   626→      deliveryFees: parseFloat(deliveryFees) || 0,
   627→      deliveryFeesTaxable,
   628→      // Operation
   629→      operationMode,
   630→      fuelResponsibility: fuelResponsibility || null,
   631→      insuranceResponsibility: insuranceResponsibility || null,
   632→      // Reference Documents
   633→      purchaseOrderNo: purchaseOrderNo || null,
   634→      quotationNo: quotationNo || null,
   635→      // Terms
   636→      paymentDuration: paymentDuration || null,
   637→      additionalTerms: additionalTerms || null,
   638→      notes: notes || null,
   639→      status: 'DRAFT',
   640→    }
   641→
   642→    if (mode === 'create') {
   643→      createMutation.mutate(data)
   644→    } else if (contractId) {
   645→      updateMutation.mutate({ ...data, id: contractId })
   646→    }
   647→  }
   648→
   649→  const isSubmitting = createMutation.isPending || updateMutation.isPending
   650→  const canSubmit = equipmentId && clientId && startDate && (
   651→    (pricingType === 'HOURLY' && referenceRate && referenceHours) ||
   652→    (pricingType === 'DAILY' && dailyRate) ||
   653→    (pricingType === 'MONTHLY' && monthlyRate) ||
   654→    (pricingType === 'LUMP_SUM' && lumpSumAmount)
   655→  )
   656→
   657→  if (mode === 'edit' && editLoading) {
   658→    return <div className="p-4"><TableSkeleton rows={6} /></div>
   659→  }
   660→
   661→  return (
   662→    <ModuleLayout
   663→      title={{
   664→        ar: mode === 'create' ? 'عقد تأجير جديد' : 'تعديل عقد التأجير',
   665→        en: mode === 'create' ? 'New Rental Contract' : 'Edit Rental Contract',
   666→      }}
   667→      subtitle={mode === 'edit' && editContract ? { ar: editContract.contractNo, en: editContract.contractNo } : undefined}
   668→      actions={
   669→        <Button variant="outline" size="icon" onClick={onBack}>
   670→          <ArrowRight className="size-4" />
   671→        </Button>
   672→      }
   673→    >
   674→      <form onSubmit={handleSubmit} className="space-y-6">
   675→
   676→        {/* ═══════════════════════════════════════════════════════════
   677→            القسم الأول: بيانات العقد (Contract Data)
   678→        ═══════════════════════════════════════════════════════════ */}
   679→        <Card>
   680→          <CardHeader className="pb-3">
   681→            <CardTitle className="flex items-center gap-2 text-emerald-700">
   682→              <FileText className="size-5" />
   683→              {t('القسم الأول: بيانات العقد', 'Section 1: Contract Data', lang)}
   684→            </CardTitle>
   685→          </CardHeader>
   686→          <CardContent className="space-y-4">
   687→            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
   688→              {/* رقم العقد (read-only, auto-generated) */}
   689→              <div className="space-y-2">
   690→                <Label className="text-right block">
   691→                  {t('رقم العقد', 'Contract No.', lang)}
   692→                </Label>
   693→                <Input
   694→                  value={mode === 'edit' && editContract ? editContract.contractNo : t('تلقائي', 'Auto-generated', lang)}
   695→                  readOnly
   696→                  className="bg-muted"
   697→                />
   698→              </div>
   699→
   700→              {/* العميل */}
   701→              <div className="space-y-2">
   702→                <Label className="text-right block">
   703→                  {t('العميل *', 'Client *', lang)}
   704→                </Label>
   705→                <Select value={clientId} onValueChange={setClientId} required>
   706→                  <SelectTrigger>
   707→                    <SelectValue placeholder={t('اختر العميل', 'Select client', lang)} />
   708→                  </SelectTrigger>
   709→                  <SelectContent>
   710→                    {clientOptions.map(c => (
   711→                      <SelectItem key={c.id} value={c.id}>
   712→                        {c.nameAr || c.name}
   713→                        <span className="text-muted-foreground text-xs mr-1">({c.code})</span>
   714→                      </SelectItem>
   715→                    ))}
   716→                  </SelectContent>
   717→                </Select>
   718→              </div>
   719→
   720→              {/* المشروع */}
   721→              <div className="space-y-2">
   722→                <Label className="text-right block">
   723→                  {t('المشروع', 'Project', lang)}
   724→                  <span className="text-muted-foreground text-xs"> ({t('اختياري', 'optional', lang)})</span>
   725→                </Label>
   726→                <Select value={projectId} onValueChange={setProjectId}>
   727→                  <SelectTrigger>
   728→                    <SelectValue placeholder={t('اختر المشروع', 'Select project', lang)} />
   729→                  </SelectTrigger>
   730→                  <SelectContent>
   731→                    {projectOptions.map(p => (
   732→                      <SelectItem key={p.id} value={p.id}>
   733→                        {p.nameAr || p.name}
   734→                        <span className="text-muted-foreground text-xs mr-1">({p.code})</span>
   735→                      </SelectItem>
   736→                    ))}
   737→                  </SelectContent>
   738→                </Select>
   739→              </div>
   740→
   741→              {/* المعدة */}
   742→              <div className="space-y-2">
   743→                <Label className="text-right block">
   744→                  {t('المعدة *', 'Equipment *', lang)}
   745→                </Label>
   746→                <Select value={equipmentId} onValueChange={setEquipmentId} required>
   747→                  <SelectTrigger>
   748→                    <SelectValue placeholder={t('اختر المعدة', 'Select equipment', lang)} />
   749→                  </SelectTrigger>
   750→                  <SelectContent>
   751→                    {equipmentOptions.map(eq => (
   752→                      <SelectItem key={eq.id} value={eq.id}>
   753→                        {eq.nameAr || eq.name}
   754→                        <span className="text-muted-foreground text-xs mr-1">({eq.code})</span>
   755→                      </SelectItem>
   756→                    ))}
   757→                  </SelectContent>
   758→                </Select>
   759→              </div>
   760→
   761→              {/* نوع النشاط */}
   762→              <div className="space-y-2">
   763→                <Label className="text-right block">
   764→                  {t('نوع النشاط', 'Activity Type', lang)}
   765→                </Label>
   766→                <div className="flex items-center gap-2 h-10 px-3 border rounded-md bg-muted">
   767→                  <div className="size-4 rounded border border-primary bg-primary text-primary-foreground flex items-center justify-center">
   768→                    <svg className="size-3" fill="none" viewBox="0 0 24 24" stroke="currentColor" strokeWidth={3}><path strokeLinecap="round" strokeLinejoin="round" d="M5 13l4 4L19 7" /></svg>
   769→                  </div>
   770→                  <span className="text-sm font-medium">{t('تأجير معدات', 'Equipment Rental', lang)}</span>
   771→                </div>
   772→              </div>
   773→
   774→              {/* تاريخ البداية */}
   775→              <div className="space-y-2">
   776→                <Label className="text-right block">
   777→                  {t('تاريخ البداية *', 'Start Date *', lang)}
   778→                </Label>
   779→                <Input type="date" value={startDate} onChange={e => setStartDate(e.target.value)} required />
   780→              </div>
   781→
   782→              {/* تاريخ النهاية */}
   783→              <div className="space-y-2">
   784→                <Label className="text-right block">
   785→                  {t('تاريخ النهاية', 'End Date', lang)}
   786→                  <span className="text-muted-foreground text-xs"> ({t('اختياري', 'optional', lang)})</span>
   787→                </Label>
   788→                <Input type="date" value={endDate} onChange={e => setEndDate(e.target.value)} />
   789→              </div>
   790→            </div>
   791→          </CardContent>
   792→        </Card>
   793→
   794→        {/* ═══════════════════════════════════════════════════════════
   795→            القسم الثاني: موقع العمل (Work Location)
   796→        ═══════════════════════════════════════════════════════════ */}
   797→        <Card>
   798→          <CardHeader className="pb-3">
   799→            <CardTitle className="flex items-center gap-2 text-cyan-700">
   800→              <MapPin className="size-5" />
   801→              {t('القسم الثاني: موقع العمل', 'Section 2: Work Location', lang)}
   802→            </CardTitle>
   803→          </CardHeader>
   804→          <CardContent>
   805→            <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   806→              <div className="space-y-2">
   807→                <Label className="text-right block">
   808→                  {t('مدينة المشروع', 'Project City', lang)}
   809→                </Label>
   810→                <Input
   811→                  value={workCity}
   812→                  onChange={e => setWorkCity(e.target.value)}
   813→                  placeholder={t('مثال: الرياض', 'e.g. Riyadh', lang)}
   814→                />
   815→              </div>
   816→              <div className="space-y-2">
   817→                <Label className="text-right block">
   818→                  {t('موقع العمل', 'Work Location', lang)}
   819→                </Label>
   820→                <Input
   821→                  value={workLocation}
   822→                  onChange={e => setWorkLocation(e.target.value)}
   823→                  placeholder={t('وصف الموقع', 'Location description', lang)}
   824→                />
   825→              </div>
   826→              <div className="space-y-2">
   827→                <Label className="text-right block flex items-center gap-1">
   828→                  <User className="size-3.5" />
   829→                  {t('اسم المسؤول', 'Site Supervisor', lang)}
   830→                </Label>
   831→                <Input
   832→                  value={siteSupervisor}
   833→                  onChange={e => setSiteSupervisor(e.target.value)}
   834→                  placeholder={t('اسم المسؤول بالموقع', 'Supervisor name', lang)}
   835→                />
   836→              </div>
   837→              <div className="space-y-2">
   838→                <Label className="text-right block flex items-center gap-1">
   839→                  <Phone className="size-3.5" />
   840→                  {t('رقم الجوال', 'Phone Number', lang)}
   841→                </Label>
   842→                <Input
   843→                  value={siteSupervisorPhone}
   844→                  onChange={e => setSiteSupervisorPhone(e.target.value)}
   845→                  placeholder="05XXXXXXXX"
   846→                  dir="ltr"
   847→                />
   848→              </div>
   849→            </div>
   850→          </CardContent>
   851→        </Card>
   852→
   853→        {/* ═══════════════════════════════════════════════════════════
   854→            القسم الثالث: التسعير (Pricing)
   855→        ═══════════════════════════════════════════════════════════ */}
   856→        <Card>
   857→          <CardHeader className="pb-3">
   858→            <CardTitle className="flex items-center gap-2 text-teal-700">
   859→              <DollarSign className="size-5" />
   860→              {t('القسم الثالث: التسعير', 'Section 3: Pricing', lang)}
   861→            </CardTitle>
   862→          </CardHeader>
   863→          <CardContent className="space-y-4">
   864→            {/* نوع التسعير */}
   865→            <div className="space-y-2">
   866→              <Label className="text-right block">
   867→                {t('نوع التسعير *', 'Pricing Type *', lang)}
   868→              </Label>
   869→              <Select value={pricingType} onValueChange={setPricingType}>
   870→                <SelectTrigger className="w-full sm:w-[280px]">
   871→                  <SelectValue />
   872→                </SelectTrigger>
   873→                <SelectContent>
   874→                  {pricingTypeOptions.map(opt => (
   875→                    <SelectItem key={opt.value} value={opt.value}>{opt[lang]}</SelectItem>
   876→                  ))}
   877→                </SelectContent>
   878→              </Select>
   879→            </div>
   880→
   881→            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
   882→              {/* القيمة الشهرية المرجعية */}
   883→              <div className="space-y-2">
   884→                <Label className="text-right block">
   885→                  {t('القيمة الشهرية المرجعية *', 'Monthly Reference Rate *', lang)}
   886→                </Label>
   887→                <Input
   888→                  type="number"
   889→                  min="0"
   890→                  step="0.0001"
   891→                  value={referenceRate}
   892→                  onChange={e => setReferenceRate(e.target.value)}
   893→                  dir="ltr"
   894→                  placeholder={t('مثال: 240,000', 'e.g. 240,000', lang)}
   895→                  required
   896→                />
   897→              </div>
   898→
   899→              {/* ساعات التسعير المرجعية - only for HOURLY */}
   900→              {pricingType === 'HOURLY' && (
   901→                <div className="space-y-2">
   902→                  <Label className="text-right block">
   903→                    {t('ساعات التسعير المرجعية *', 'Reference Hours *', lang)}
   904→                  </Label>
   905→                  <Input
   906→                    type="number"
   907→                    min="0"
   908→                    step="0.5"
   909→                    value={referenceHours}
   910→                    onChange={e => setReferenceHours(e.target.value)}
   911→                    dir="ltr"
   912→                    placeholder={t('مثال: 260', 'e.g. 260', lang)}
   913→                    required
   914→                  />
   915→                </div>
   916→              )}
   917→
   918→              {/* سعر الساعة - auto-calculated, read-only for HOURLY */}
   919→              {pricingType === 'HOURLY' && (
   920→                <div className="space-y-2">
   921→                  <Label className="text-right block">
   922→                    {t('سعر الساعة (محسوب)', 'Hourly Rate (calculated)', lang)}
   923→                  </Label>
   924→                  <Input
   925→                    value={calculatedHourlyRate > 0 ? (calculatedHourlyRate ?? 0).toFixed(4) : ''}
   926→                    readOnly
   927→                    className="bg-emerald-50 border-emerald-200 font-bold text-emerald-700"
   928→                    dir="ltr"
   929→                  />
   930→                </div>
   931→              )}
   932→
   933→              {/* السعر اليومي - for DAILY */}
   934→              {pricingType === 'DAILY' && (
   935→                <div className="space-y-2">
   936→                  <Label className="text-right block">
   937→                    {t('السعر اليومي *', 'Daily Rate *', lang)}
   938→                  </Label>
   939→                  <Input
   940→                    type="number"
   941→                    min="0"
   942→                    step="0.0001"
   943→                    value={dailyRate}
   944→                    onChange={e => setDailyRate(e.target.value)}
   945→                    dir="ltr"
   946→                    placeholder={t('مثال: 2,500', 'e.g. 2,500', lang)}
   947→                    required
   948→                  />
   949→                </div>
   950→              )}
   951→
   952→              {/* السعر الشهري - for MONTHLY */}
   953→              {pricingType === 'MONTHLY' && (
   954→                <div className="space-y-2">
   955→                  <Label className="text-right block">
   956→                    {t('السعر الشهري *', 'Monthly Rate *', lang)}
   957→                  </Label>
   958→                  <Input
   959→                    type="number"
   960→                    min="0"
   961→                    step="0.0001"
   962→                    value={monthlyRate}
   963→                    onChange={e => setMonthlyRate(e.target.value)}
   964→                    dir="ltr"
   965→                    placeholder={t('مثال: 75,000', 'e.g. 75,000', lang)}
   966→                    required
   967→                  />
   968→                </div>
   969→              )}
   970→
   971→              {/* المبلغ المقطوع - for LUMP_SUM */}
   972→              {pricingType === 'LUMP_SUM' && (

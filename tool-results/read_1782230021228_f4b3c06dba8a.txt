    11→/**
    12→ * Build a map of project code → cost center ID for GL-based project queries.
    13→ */
    14→async function buildProjectCostCenterMap(projectIds: string[]) {
    15→  if (projectIds.length === 0) return new Map<string, string>()
    16→  const projects = await db.project.findMany({
    17→    where: { id: { in: projectIds } },
    18→    select: { id: true, code: true },
    19→  })
    20→  const codes = projects.map(p => p.code)
    21→  const costCenters = await db.costCenter.findMany({
    22→    where: { code: { in: codes } },
    23→    select: { id: true, code: true },
    24→  })
    25→  // Map: project.id → costCenter.id
    26→  const projectCodeToId = new Map(projects.map(p => [p.code, p.id]))
    27→  const result = new Map<string, string>()
    28→  for (const cc of costCenters) {
    29→    const projectId = projectCodeToId.get(cc.code)
    30→    if (projectId) result.set(projectId, cc.id)
    31→  }
    32→  return result
    33→}

    35→/**
    36→ * Get GL-based revenue and costs for projects, returns Map<projectId, {revenue, costs}>
    37→ */
    38→async function getProjectGLBalances(projectIds: string[]) {
    39→  const result = new Map<string, { revenue: number; costs: number }>()
    40→  if (projectIds.length === 0) return result
    41→
    42→  const ccMap = await buildProjectCostCenterMap(projectIds)
    43→  const costCenterIds = [...ccMap.values()]
    44→
    45→  if (costCenterIds.length === 0) {
    46→    // Initialize with zeros for all projects
    47→    for (const pid of projectIds) result.set(pid, { revenue: 0, costs: 0 })
    48→    return result
    49→  }
    50→
    51→  // Single GL query for all project cost centers
    52→  const lines = await db.journalLine.findMany({
    53→    where: {
    54→      costCenterId: { in: costCenterIds },
    55→      journalEntry: { status: 'POSTED' },
    56→      account: { type: { in: ['REVENUE', 'EXPENSE'] } },
    57→    },
    58→    include: { account: { select: { type: true } }, costCenter: { select: { code: true } } },
    59→  })
    60→
    61→  // Build project code → GL data map
    62→  const glMap = new Map<string, { revenue: number; costs: number }>()
    63→  for (const line of lines) {
    64→    const ccCode = line.costCenter?.code
    65→    if (!ccCode) continue
    66→    if (!glMap.has(ccCode)) glMap.set(ccCode, { revenue: 0, costs: 0 })
    67→    const entry = glMap.get(ccCode)!
    68→    if (line.account.type === 'REVENUE') {
    69→      entry.revenue += toNumber(line.credit) - toNumber(line.debit)
    70→    } else if (line.account.type === 'EXPENSE') {
    71→      entry.costs += toNumber(line.debit) - toNumber(line.credit)
    72→    }
    73→  }
    74→
    75→  // Map back to project IDs
    76→  const projects = await db.project.findMany({
    77→    where: { id: { in: projectIds } },
    78→    select: { id: true, code: true },
    79→  })
    80→  for (const p of projects) {
    81→    const glData = glMap.get(p.code) || { revenue: 0, costs: 0 }
    82→    result.set(p.id, glData)
    83→  }
    84→  // Fill zeros for projects without cost centers
    85→  for (const pid of projectIds) {
    86→    if (!result.has(pid)) result.set(pid, { revenue: 0, costs: 0 })
    87→  }
    88→  return result
    89→}

    91→/**
    92→ * Get GL aggregation by account type with optional activity type filter.
    93→ */
    94→async function getGLBalanceByType(
    95→  accountType: string,
    96→  options?: {
    97→    activityType?: 'CONSTRUCTION' | 'EQUIPMENT_RENTAL'
    98→    dateFrom?: Date
    99→    dateTo?: Date
   100→  }
   101→): Promise<number> {
   102→  const accountWhere: Record<string, unknown> = { type: accountType, allowPosting: true }
   103→  if (options?.activityType) {
   104→    accountWhere.activityType = { in: [options.activityType, 'BOTH'] }
   105→  }
   106→
   107→  const jeWhere: Record<string, unknown> = { status: 'POSTED' }
   108→  if (options?.dateFrom || options?.dateTo) {
   109→    jeWhere.date = {}
   110→    if (options.dateFrom) jeWhere.date.gte = options.dateFrom
   111→    if (options.dateTo) jeWhere.date.lte = options.dateTo
   112→  }
   113→
   114→  const result = await db.journalLine.aggregate({
   115→    _sum: { debit: true, credit: true },
   116→    where: { account: accountWhere, journalEntry: jeWhere },
   117→  })
   118→
   119→  const totalDebit = toNumber(result._sum.debit)
   120→  const totalCredit = toNumber(result._sum.credit)
   121→  const isCreditNormal = accountType === 'REVENUE' || accountType === 'LIABILITY' || accountType === 'EQUITY'
   122→  return isCreditNormal ? totalCredit - totalDebit : totalDebit - totalCredit
   123→}

   125→/**
   126→ * Get GL balance for specific account codes.
   127→ */
   128→async function getGLBalanceForCodes(
   129→  codes: string[],
   130→  dateFrom?: Date,
   131→  dateTo?: Date
   132→): Promise<number> {
   133→  if (codes.length === 0) return 0
   134→
   135→  const accounts = await db.account.findMany({
   136→    where: { code: { in: codes }, isActive: true },
   137→    select: { id: true, type: true },
   138→  })
   139→  if (accounts.length === 0) return 0
   140→
   141→  const jeWhere: Record<string, unknown> = { status: 'POSTED' }
   142→  if (dateFrom || dateTo) {
   143→    jeWhere.date = {}
   144→    if (dateFrom) jeWhere.date.gte = dateFrom
   145→    if (dateTo) jeWhere.date.lte = dateTo
   146→  }
   147→
   148→  const agg = await db.journalLine.aggregate({
   149→    _sum: { debit: true, credit: true },
   150→    where: {
   151→      accountId: { in: accounts.map(a => a.id) },
   152→      journalEntry: jeWhere,
   153→    },
   154→  })
   155→
   156→  // Use the first account's type to determine normal balance (all should be same type)
   157→  const totalDebit = toNumber(agg._sum.debit)
   158→  const totalCredit = toNumber(agg._sum.credit)
   159→  const accountType = accounts[0]?.type || 'ASSET'
   160→  const isCreditNormal = accountType === 'REVENUE' || accountType === 'LIABILITY' || accountType === 'EQUITY'
   161→  return isCreditNormal ? totalCredit - totalDebit : totalDebit - totalCredit
   162→}

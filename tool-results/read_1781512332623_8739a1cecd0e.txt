   269→// ============ Status Badge for Contracts ============
   270→function ContractStatusBadge({ status, lang }: { status: string; lang: 'ar' | 'en' }) {
   271→  const config = statusOptions.find(s => s.value === status)
   272→  if (!config) return <StatusBadge status={status} lang={lang} />
   273→  return (
   274→    <span className={`inline-flex items-center rounded-full px-2.5 py-0.5 text-xs font-medium ${config.color}`}>
   275→      {config.label[lang]}
   276→    </span>
   277→  )
   278→}

   308→// ============ Contract Form Page ============
   309→function ContractFormPage({
   310→  form, setForm, onSubmit, onBack, isEdit, isLoading, clients, projects,
   311→}: {
   312→  form: ContractFormData; setForm: React.Dispatch<React.SetStateAction<ContractFormData>>
   313→  onSubmit: (e: React.FormEvent) => void; onBack: () => void
   314→  isEdit: boolean; isLoading: boolean
   315→  clients: ClientOption[]; projects: ProjectOption[]
   316→}) {
   317→  const { lang } = useAppStore()
   318→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   319→

   321→  const rate = parseFloat(form.vatRate) || 0
   322→  const vatAmount = Math.round(val * rate * 100) / 100
   323→  const totalValue = Math.round((val + vatAmount) * 100) / 100
   324→
   325→  // Filter projects by selected client
   326→  const filteredProjects = form.clientId
   327→    ? projects.filter(p => !p.clientId || p.clientId === form.clientId)
   328→    : projects
   329→
   330→  return (
   331→    <div className="space-y-6">
   332→      <div className="flex items-center gap-3">
   333→        <Button variant="outline" size="icon" onClick={onBack}>
   334→          <ArrowRight className="size-4" />

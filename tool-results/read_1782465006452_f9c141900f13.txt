   171→function ExpenseFormDialog({
   172→  open, onOpenChange, projects, activeTab: initialTab,
   173→}: ExpenseFormDialogProps) {
   174→  const { lang } = useAppStore()
   175→  const queryClient = useQueryClient()
   176→
   177→  const [tab, setTab] = useState<'project' | 'admin'>(initialTab)
   178→  const [projectId, setProjectId] = useState('')
   179→  const [category, setCategory] = useState('')
   180→  const [description, setDescription] = useState('')
   181→  const [amount, setAmount] = useState('')
   182→  const [vatAmount, setVatAmount] = useState('')
   183→  const [date, setDate] = useState(() => {
   184→    // Default to today's ISO date (yyyy-mm-dd) so the date picker never renders 0/0/0.
   185→    const now = new Date()
   186→    const y = now.getFullYear()
   187→    const m = String(now.getMonth() + 1).padStart(2, '0')
   188→    const d = String(now.getDate()).padStart(2, '0')
   189→    return `${y}-${m}-${d}`
   190→  })
   191→  const [reference, setReference] = useState('')
   192→  const [payFrom, setPayFrom] = useState('TREASURY')
   193→  const [attachmentPath, setAttachmentPath] = useState('')
   194→  const [vatRate, setVatRate] = useState('0.15')
   195→
   196→  // Account-based fields (replacing hardcoded payFrom & category)
   197→  const [payingAccountId, setPayingAccountId] = useState<string | null>(null)
   198→  const [payingAccountCode, setPayingAccountCode] = useState('')
   199→  const [payingAccountName, setPayingAccountName] = useState('')
   200→  const [expenseAccountId, setExpenseAccountId] = useState<string | null>(null)
   201→  const [expenseAccountCode, setExpenseAccountCode] = useState('')
   202→  const [expenseAccountNameAr, setExpenseAccountNameAr] = useState('')
   203→  const [expenseAccountRole, setExpenseAccountRole] = useState<string | null>(null)
   204→  const [expenseAccountActivityType, setExpenseAccountActivityType] = useState<string | null>(null)
   205→  // Cost center (مركز التكلفة) — optional, linked to journal lines
   206→  const [costCenterId, setCostCenterId] = useState<string>('')
   207→
   208→  // Fetch cost centers for selection
   209→  const { data: costCenters = [] } = useQuery<{ id: string; code: string; name: string; parentId: string | null }[]>({
   210→    queryKey: ['cost-centers-list'],
   211→    queryFn: async () => {
   212→      const res = await fetch('/api/cost-centers')
   213→      if (!res.ok) return []
   214→      return res.json()
   215→    },
   216→  })
   217→
   218→  React.useEffect(() => {
   219→    if (open) {
   220→      setTab(initialTab)
   221→      setProjectId('')
   222→      setCategory(''); setDescription('')
   223→      setAmount(''); setVatAmount('')
   224→      // Reset date to today's ISO date so the picker always shows a valid value.
   225→      const now = new Date()
   226→      const y = now.getFullYear()
   227→      const m = String(now.getMonth() + 1).padStart(2, '0')
   228→      const d = String(now.getDate()).padStart(2, '0')
   229→      setDate(`${y}-${m}-${d}`)
   230→      setReference(''); setPayFrom('TREASURY')
   231→      setAttachmentPath(''); setVatRate('0.15')
   232→      setPayingAccountId(null); setPayingAccountCode(''); setPayingAccountName('')
   233→      setExpenseAccountId(null); setExpenseAccountCode(''); setExpenseAccountNameAr('')
   234→      setExpenseAccountRole(null); setExpenseAccountActivityType(null)
   235→      setCostCenterId('')
   236→    }
   237→  }, [open, initialTab])
   238→
   239→  // Auto-calc VAT if applicable
   240→  const parsedAmount = parseFloat(amount) || 0
   241→  const parsedVatRate = parseFloat(vatRate) || 0.15
   242→  const autoVat = useMemo(() => {
   243→    const manualVat = parseFloat(vatAmount)
   244→    if (!isNaN(manualVat)) return manualVat
   245→    return parsedAmount * parsedVatRate
   246→  }, [amount, vatAmount, parsedAmount, parsedVatRate])
   247→
   248→  const totalAmount = parsedAmount + autoVat
   249→
   250→  // Compute JE preview lines
   251→  const jeLines = useMemo<JePreviewLine[]>(() => {
   252→    if (parsedAmount <= 0 || !expenseAccountId || !payingAccountId) return []
   253→    const lines: JePreviewLine[] = []
   254→    // Debit: Expense account
   255→    lines.push({
   256→      accountCode: expenseAccountCode,
   257→      accountNameAr: expenseAccountNameAr,
   258→      debit: parsedAmount,
   259→      credit: 0,
   260→    })
   261→    // Debit: VAT Input (1410) if VAT > 0
   262→    if (autoVat > 0) {
   263→      lines.push({
   264→        accountCode: '1410',
   265→        accountNameAr: 'ضريبة مدخلات',
   266→        debit: autoVat,
   267→        credit: 0,
   268→      })
   269→    }
   270→    // Credit: Paying account
   271→    lines.push({
   272→      accountCode: payingAccountCode,
   273→      accountNameAr: payingAccountName,
   274→      debit: 0,
   275→      credit: totalAmount,
   276→    })
   277→    return lines
   278→  }, [parsedAmount, autoVat, totalAmount, expenseAccountId, expenseAccountCode, expenseAccountNameAr, payingAccountId, payingAccountCode, payingAccountName])
   279→
   280→  const createMutation = useMutation({
   281→    mutationFn: (data: Record<string, unknown>) =>
   282→      fetch('/api/expenses', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) })
   283→        .then(r => { if (!r.ok) throw new Error(); return r.json() }),
   284→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['expenses'] }); onOpenChange(false) },
   285→  })
   286→
   287→  const handleSubmit = (e: React.FormEvent) => {
   288→    e.preventDefault()
   289→    // Auto-determine expenseType from account activityType
   290→    let expenseType: string
   291→    if (expenseAccountActivityType === 'CONSTRUCTION' || expenseAccountActivityType === 'EQUIPMENT_RENTAL') {
   292→      expenseType = 'PROJECT'
   293→    } else if (expenseAccountActivityType === 'BOTH') {
   294→      expenseType = 'INTERNAL'
   295→    } else {
   296→      // Fallback: determine from account code prefix (7xxx = PROJECT, 8xxx = INTERNAL)
   297→      expenseType = expenseAccountCode.startsWith('7') ? 'PROJECT' : 'INTERNAL'
   298→    }
   299→    const isProject = expenseType === 'PROJECT'
   300→    createMutation.mutate({
   301→      projectId: isProject ? projectId : null,
   302→      expenseType,
   303→      costCenterId: costCenterId || undefined,
   304→      category, description, amount,
   305→      vatRate: parsedVatRate,
   306→      vatAmount: autoVat || null,
   307→      totalAmount,
   308→      date, reference: reference || null,
   309→      payFrom,
   310→      attachmentPath: attachmentPath || null,
   311→      // Account-based fields
   312→      accountId: expenseAccountId,
   313→      payingAccountId,
   314→      payingAccountCode,
   315→      payingAccountName,
   316→      expenseAccountCode,
   317→      expenseAccountNameAr,
   318→    })
   319→  }
   320→
   321→  return (
   322→    <Dialog open={open} onOpenChange={onOpenChange}>
   323→      <DialogContent className="max-w-lg max-h-[90vh] overflow-y-auto">
   324→        <DialogHeader>
   325→          <DialogTitle>{t(lang, 'مصروف جديد', 'New Expense')}</DialogTitle>
   326→          <DialogDescription>{t(lang, 'إضافة مصروف جديد', 'Add a new expense')}</DialogDescription>
   327→        </DialogHeader>
   328→        <form onSubmit={handleSubmit} className="space-y-4">
   329→          {/* Type toggle */}
   330→          <div className="flex gap-2 p-1 bg-gray-100 rounded-lg">
   331→            <button
   332→              type="button"
   333→              onClick={() => { setTab('project'); setCategory(''); setProjectId(''); setExpenseAccountId(null); setExpenseAccountCode(''); setExpenseAccountNameAr(''); setExpenseAccountRole(null); setExpenseAccountActivityType(null) }}
   334→              className={`flex-1 flex items-center justify-center gap-2 px-3 py-2 rounded-md text-sm font-medium transition-all ${
   335→                tab === 'project' ? 'bg-emerald-600 text-white shadow-sm' : 'text-gray-600 hover:text-gray-800'
   336→              }`}
   337→            >
   338→              <Building2 className="size-4" />
   339→              {t(lang, 'مصروف مشروع', 'Project Expense')}
   340→            </button>
   341→            <button
   342→              type="button"
   343→              onClick={() => { setTab('admin'); setCategory(''); setProjectId(''); setExpenseAccountId(null); setExpenseAccountCode(''); setExpenseAccountNameAr(''); setExpenseAccountRole(null); setExpenseAccountActivityType(null) }}
   344→              className={`flex-1 flex items-center justify-center gap-2 px-3 py-2 rounded-md text-sm font-medium transition-all ${
   345→                tab === 'admin' ? 'bg-emerald-600 text-white shadow-sm' : 'text-gray-600 hover:text-gray-800'
   346→              }`}
   347→            >
   348→              <Briefcase className="size-4" />
   349→              {t(lang, 'مصروف إداري', 'Admin Expense')}
   350→            </button>
   351→          </div>
   352→
   353→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   354→            {/* Project selector - only for project tab */}
   355→            {tab === 'project' && (
   356→              <div className="space-y-2">
   357→                <Label>{t(lang, 'المشروع *', 'Project *')}</Label>
   358→                <Select value={projectId} onValueChange={setProjectId}>
   359→                  <SelectTrigger className="w-full"><SelectValue placeholder={t(lang, 'اختر المشروع', 'Select project')} /></SelectTrigger>
   360→                  <SelectContent>
   361→                    {projects.map(p => <SelectItem key={p.id} value={p.id}>{p.name}</SelectItem>)}
   362→                  </SelectContent>
   363→                </Select>
   364→              </div>
   365→            )}
   366→
   367→            {/* Admin info */}
   368→            {tab === 'admin' && (
   369→              <div className="space-y-2 sm:col-span-2">
   370→                <div className="flex items-center gap-2 rounded-lg bg-amber-50 border border-amber-200 px-3 py-2">
   371→                  <Briefcase className="size-4 text-amber-600 shrink-0" />
   372→                  <span className="text-sm text-amber-700">
   373→                    {t(lang, 'المصروفات الإدارية لا ترتبط بمشروع معين', 'Administrative expenses are not linked to a specific project')}

   375→                </div>
   376→              </div>

   378→
   379→            {/* Expense Account selector (replaces hardcoded category + activity type) */}

   381→              roles={['PROJECT_COST', 'SUBCONTRACTOR_COST', 'FUEL_EXPENSE', 'MAINTENANCE_EXPENSE', 'DRIVER_EXPENSE', 'TRANSPORT_EXPENSE', 'RENTAL_DEPRECIATION', 'PAYROLL_EXPENSE', 'GOSI_EXPENSE', 'ADMIN_EXPENSE', 'DEPRECIATION_EXPENSE', 'ZAKAT_EXPENSE']}

   383→              onValueChange={(id, account) => {
   384→                setExpenseAccountId(id)
   385→                setExpenseAccountCode(account.code)
   386→                setExpenseAccountNameAr(account.nameAr || account.name)
   387→                setExpenseAccountRole(account.accountRole)
   388→                setExpenseAccountActivityType(account.activityType ?? null)

   392→                // Auto-switch tab based on account code prefix (7xxx = project, 8xxx = admin)
   393→                const isProjectAccount = account.code.startsWith('7')
   394→                if (isProjectAccount) {
   395→                  setTab('project')
   396→                } else {
   397→                  setTab('admin')
   398→                  setProjectId('')
   399→                }
   400→              }}
   401→              label={t(lang, 'الحساب المحاسبي *', 'Accounting Account *')}
   402→              placeholder={t(lang, 'اختر الحساب المحاسبي...', 'Select accounting account...')}
   403→              className="sm:col-span-2"
   404→            />
   405→            {/* Selected account info badge */}
   406→            {expenseAccountId && (

   409→                <span className="text-sm text-emerald-700">{expenseAccountNameAr}</span>
   410→                {category && (
   411→                  <Badge className="bg-gray-100 text-gray-700 border-0 text-xs mr-auto">
   412→                    {allCategoryLabels[category]?.[lang] || category}
   413→                  </Badge>
   414→                )}
   415→              </div>
   416→            )}
   417→            <div className="space-y-2 sm:col-span-2">
   418→              <Label>{t(lang, 'الوصف *', 'Description *')}</Label>

   421→            <div className="space-y-2">

   429→            <div className="space-y-2">
   430→              <Label>{t(lang, 'التاريخ *', 'Date *')}</Label>

   433→            {/* Paying Account selector (replaces hardcoded payFrom) */}

   435→              roles={['CASH', 'BANK']}
   436→              value={payingAccountId}
   437→              onValueChange={(id, account) => {

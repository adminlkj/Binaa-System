    61→    61→// Normal balance by account type
    62→    62→const NORMAL_BALANCE: Record<string, 'DEBIT' | 'CREDIT'> = {
    63→    63→  ASSET: 'DEBIT',
    64→    64→  LIABILITY: 'CREDIT',
    65→    65→  EQUITY: 'CREDIT',
    66→    66→  REVENUE: 'CREDIT',

    78→    78→function postedLinesWhere(extra: Prisma.JournalLineWhereInput = {}): Prisma.JournalLineWhereInput {
    79→    79→  return {
    80→    80→    deletedAt: null,
    81→    81→    journalEntry: {
    82→    82→      status: 'POSTED',
    83→    83→      deletedAt: null,
    84→    84→    },
    85→    85→    ...extra,
    86→    86→  }
    87→    87→}
    88→    88→
    89→    89→function dateRangeFilter(range?: DateRange): Prisma.JournalLineWhereInput {
    90→    90→  if (!range || (!range.from && !range.to)) return {}
    91→    91→  return {
    92→    92→    journalEntry: {
    93→    93→      status: 'POSTED',
    94→    94→      deletedAt: null,
    95→    95→      date: {
    96→    96→        ...(range.from && { gte: range.from }),
    97→    97→        ...(range.to && { lte: range.to }),
    98→    98→      },
    99→    99→    },
   100→   100→  }
   101→   101→}

   107→   107→/**
   108→   108→ * Get balances for all accounts of given types, sourced from posted JEs.
   109→   109→ * Returns signed balances (respecting normal balance).
   110→   110→ */
   111→   111→export async function getAccountBalancesByType(
   112→   112→  types: string[],
   113→   113→  range?: DateRange
   114→   114→): Promise<AccountBalance[]> {
   115→   115→  const accounts = await db.account.findMany({
   116→   116→    where: { type: { in: types }, isActive: true },
   117→   117→    select: { id: true, code: true, name: true, nameAr: true, type: true, accountRole: true, activityType: true },
   118→   118→    orderBy: { code: 'asc' },
   119→   119→  })
   120→   120→  if (accounts.length === 0) return []
   121→   121→
   122→   122→  const accountIds = accounts.map(a => a.id)
   123→   123→
   124→   124→  // Aggregate journal lines in one query
   125→   125→  const aggregated = await db.journalLine.groupBy({
   126→   126→    by: ['accountId'],
   127→   127→    _sum: { debit: true, credit: true },
   128→   128→    where: postedLinesWhere({
   129→   129→      accountId: { in: accountIds },
   130→   130→      ...dateRangeFilter(range),
   131→   131→    }),
   132→   132→  })
   133→   133→
   134→   134→  const sumMap = new Map<string, { debit: number; credit: number }>()
   135→   135→  for (const a of aggregated) {
   136→   136→    sumMap.set(a.accountId, {
   137→   137→      debit: toNumber(a._sum.debit),
   138→   138→      credit: toNumber(a._sum.credit),
   139→   139→    })
   140→   140→  }
   141→   141→
   142→   142→  return accounts.map(a => {
   143→   143→    const sums = sumMap.get(a.id) || { debit: 0, credit: 0 }
   144→   144→    const sign = signForType(a.type)
   145→   145→    return {
   146→   146→      accountId: a.id,
   147→   147→      code: a.code,
   148→   148→      name: a.name,
   149→   149→      nameAr: a.nameAr,
   150→   150→      type: a.type,
   151→   151→      accountRole: a.accountRole,
   152→   152→      activityType: a.activityType,
   153→   153→      totalDebit: sums.debit,
   154→   154→      totalCredit: sums.credit,
   155→   155→      balance: sign * (sums.debit - sums.credit),
   156→   156→    }
   157→   157→  })
   158→   158→}

   161→   161→ * Get a single total balance for all accounts matching a role list.

   163→   163→export async function getBalanceByRole(
   164→   164→  roles: string[],
   165→   165→  range?: DateRange
   166→   166→): Promise<number> {
   167→   167→  if (roles.length === 0) return 0
   168→   168→  const accounts = await db.account.findMany({
   169→   169→    where: { accountRole: { in: roles }, isActive: true },
   170→   170→    select: { id: true, type: true },
   171→   171→  })
   172→   172→  if (accounts.length === 0) return 0
   173→   173→  const accountIds = accounts.map(a => a.id)
   174→   174→
   175→   175→  const agg = await db.journalLine.aggregate({
   176→   176→    _sum: { debit: true, credit: true },
   177→   177→    where: postedLinesWhere({
   178→   178→      accountId: { in: accountIds },
   179→   179→      ...dateRangeFilter(range),
   180→   180→    }),
   181→   181→  })
   182→   182→
   183→   183→  const totalDebit = toNumber(agg._sum.debit)
   184→   184→  const totalCredit = toNumber(agg._sum.credit)
   185→   185→  // Use the type of the first account as representative (roles map to a single type normally)
   186→   186→  const type = accounts[0]?.type || 'EXPENSE'
   187→   187→  return signForType(type) * (totalDebit - totalCredit)
   188→   188→}

   226→   226→export async function getTrialBalance(range?: DateRange): Promise<{
   227→   227→  rows: TrialBalanceRow[]
   228→   228→  totals: { totalDebit: number; totalCredit: number; totalNetDebit: number; totalNetCredit: number; isBalanced: boolean }
   229→   229→}> {
   230→   230→  const accounts = await db.account.findMany({
   231→   231→    where: { isActive: true },
   232→   232→    select: { id: true, code: true, name: true, nameAr: true, type: true, accountRole: true, activityType: true },
   233→   233→    orderBy: { code: 'asc' },
   234→   234→  })
   235→   235→
   236→   236→  if (accounts.length === 0) {
   237→   237→    return { rows: [], totals: { totalDebit: 0, totalCredit: 0, totalNetDebit: 0, totalNetCredit: 0, isBalanced: true } }
   238→   238→  }
   239→   239→
   240→   240→  const accountIds = accounts.map(a => a.id)
   241→   241→  const aggregated = await db.journalLine.groupBy({
   242→   242→    by: ['accountId'],
   243→   243→    _sum: { debit: true, credit: true },
   244→   244→    where: postedLinesWhere({
   245→   245→      accountId: { in: accountIds },
   246→   246→      ...dateRangeFilter(range),
   247→   247→    }),
   248→   248→  })
   249→   249→
   250→   250→  const sumMap = new Map<string, { debit: number; credit: number }>()
   251→   251→  for (const a of aggregated) {
   252→   252→    sumMap.set(a.accountId, {
   253→   253→      debit: toNumber(a._sum.debit),
   254→   254→      credit: toNumber(a._sum.credit),
   255→   255→    })
   256→   256→  }
   257→   257→
   258→   258→  const rows: TrialBalanceRow[] = []
   259→   259→  let totalDebit = 0
   260→   260→  let totalCredit = 0
   261→   261→  let totalNetDebit = 0
   262→   262→  let totalNetCredit = 0
   263→   263→
   264→   264→  for (const a of accounts) {
   265→   265→    const sums = sumMap.get(a.id) || { debit: 0, credit: 0 }
   266→   266→    const net = sums.debit - sums.credit
   267→   267→    if (Math.abs(net) < 0.005 && sums.debit === 0 && sums.credit === 0) continue // skip zero accounts
   268→   268→    const sign = signForType(a.type)
   269→   269→    const balance = sign * net
   270→   270→    const netDebit = balance > 0 ? balance : 0
   271→   271→    const netCredit = balance < 0 ? Math.abs(balance) : 0
   272→   272→    rows.push({
   273→   273→      accountId: a.id,
   274→   274→      code: a.code,
   275→   275→      name: a.name,
   276→   276→      nameAr: a.nameAr,
   277→   277→      type: a.type,
   278→   278→      accountRole: a.accountRole,
   279→   279→      activityType: a.activityType,
   280→   280→      totalDebit: sums.debit,
   281→   281→      totalCredit: sums.credit,
   282→   282→      balance,
   283→   283→      netDebit,
   284→   284→      netCredit,
   285→   285→    })
   286→   286→    totalDebit += sums.debit
   287→   287→    totalCredit += sums.credit
   288→   288→    totalNetDebit += netDebit
   289→   289→    totalNetCredit += netCredit
   290→   290→  }
   291→   291→
   292→   292→  return {
   293→   293→    rows,
   294→   294→    totals: {
   295→   295→      totalDebit,
   296→   296→      totalCredit,
   297→   297→      totalNetDebit,
   298→   298→      totalNetCredit,
   299→   299→      isBalanced: Math.abs(totalDebit - totalCredit) < 0.01,
   300→   300→    },
   301→   301→  }
   302→   302→}

   308→   308→export interface IncomeStatementData {
   309→   309→  revenue: { accounts: AccountBalance[]; total: number }
   310→   310→  expenses: { accounts: AccountBalance[]; total: number }
   311→   311→  grossProfit: number // revenue - direct project costs
   312→   312→  netIncome: number // revenue - all expenses
   313→   313→  netProfitMargin: number
   314→   314→}
   315→   315→
   316→   316→export async function getIncomeStatement(range?: DateRange): Promise<IncomeStatementData> {
   317→   317→  const [revenueAccounts, expenseAccounts] = await Promise.all([
   318→   318→    getAccountBalancesByType(['REVENUE'], range),
   319→   319→    getAccountBalancesByType(['EXPENSE'], range),
   320→   320→  ])
   321→   321→
   322→   322→  const totalRevenue = revenueAccounts.reduce((s, a) => s + a.balance, 0)
   323→   323→  const totalExpenses = expenseAccounts.reduce((s, a) => s + a.balance, 0)
   324→   324→
   325→   325→  // Gross profit = revenue - direct project costs (PROJECT_COST + SUBCONTRACTOR_COST)
   326→   326→  const directCostRoles = ['PROJECT_COST', 'SUBCONTRACTOR_COST', 'FUEL_EXPENSE', 'MAINTENANCE_EXPENSE', 'DRIVER_EXPENSE', 'TRANSPORT_EXPENSE', 'RENTAL_DEPRECIATION']
   327→   327→  const grossCosts = expenseAccounts
   328→   328→    .filter(a => a.accountRole && directCostRoles.includes(a.accountRole))
   329→   329→    .reduce((s, a) => s + a.balance, 0)
   330→   330→  const grossProfit = totalRevenue - grossCosts
   331→   331→  const netIncome = totalRevenue - totalExpenses
   332→   332→  const netProfitMargin = totalRevenue > 0 ? (netIncome / totalRevenue) * 100 : 0
   333→   333→
   334→   334→  return {
   335→   335→    revenue: { accounts: revenueAccounts, total: totalRevenue },
   336→   336→    expenses: { accounts: expenseAccounts, total: totalExpenses },
   337→   337→    grossProfit,
   338→   338→    netIncome,
   339→   339→    netProfitMargin,
   340→   340→  }
   341→   341→}

   347→   347→export interface BalanceSheetData {
   348→   348→  assets: { accounts: AccountBalance[]; total: number }
   349→   349→  liabilities: { accounts: AccountBalance[]; total: number }
   350→   350→  equity: { accounts: AccountBalance[]; total: number }
   351→   351→  totalLiabilitiesAndEquity: number
   352→   352→  isBalanced: boolean
   353→   353→  /** Net income for the current period (revenue - expenses), shown as "Current Year Earnings" in equity */
   354→   354→  currentYearEarnings: number
   355→   355→}
   356→   356→
   357→   357→export async function getBalanceSheet(asOfDate?: Date): Promise<BalanceSheetData> {
   358→   358→  const range: DateRange = asOfDate ? { to: asOfDate } : {}
   359→   359→  const [assets, liabilities, equity, income] = await Promise.all([
   360→   360→    getAccountBalancesByType(['ASSET'], range),
   361→   361→    getAccountBalancesByType(['LIABILITY'], range),
   362→   362→    getAccountBalancesByType(['EQUITY'], range),
   363→   363→    getIncomeStatement(range),
   364→   364→  ])
   365→   365→
   366→   366→  const totalAssets = assets.reduce((s, a) => s + a.balance, 0)
   367→   367→  const totalLiabilities = liabilities.reduce((s, a) => s + a.balance, 0)
   368→   368→
   369→   369→  // Net income (revenue - expenses) for the period must be included in equity
   370→   370→  // as "Current Year Earnings" until formally closed at period end.
   371→   371→  // This is required by the accounting equation: Assets = Liabilities + Equity
   372→   372→  // where Equity = Contributed Capital + Retained Earnings + Current Period Net Income.
   373→   373→  const currentYearEarnings = income.netIncome
   374→   374→
   375→   375→  // Append a synthetic "Current Year Earnings" equity row so the user sees it.
   376→   376→  // (If an account 5300 "أرباح/خسائر السنة الحالية" already has posted lines,
   377→   377→  // those are shown separately — this synthetic row is the unclosed P&L balance.)
   378→   378→  const equityWithEarnings: AccountBalance[] = [...equity]
   379→   379→  if (Math.abs(currentYearEarnings) > 0.005) {
   380→   380→    equityWithEarnings.push({
   381→   381→      accountId: '__current_year_earnings__',
   382→   382→      code: '5300',
   383→   383→      name: 'Current Year Earnings (Unclosed P&L)',
   384→   384→      nameAr: 'أرباح (خسائر) السنة الحالية - غير مُقفلة',
   385→   385→      type: 'EQUITY',
   386→   386→      accountRole: 'CURRENT_YEAR_EARNINGS',
   387→   387→      activityType: 'BOTH',
   388→   388→      totalDebit: currentYearEarnings < 0 ? Math.abs(currentYearEarnings) : 0,
   389→   389→      totalCredit: currentYearEarnings > 0 ? currentYearEarnings : 0,
   390→   390→      balance: currentYearEarnings, // equity is credit-normal, so positive net income = positive balance
   391→   391→    })
   392→   392→  }
   393→   393→
   394→   394→  const totalEquity = equityWithEarnings.reduce((s, a) => s + a.balance, 0)
   395→   395→  const totalLiabilitiesAndEquity = totalLiabilities + totalEquity
   396→   396→
   397→   397→  return {
   398→   398→    assets: { accounts: assets, total: totalAssets },
   399→   399→    liabilities: { accounts: liabilities, total: totalLiabilities },
   400→   400→    equity: { accounts: equityWithEarnings, total: totalEquity },
   401→   401→    totalLiabilitiesAndEquity,
   402→   402→    isBalanced: Math.abs(totalAssets - totalLiabilitiesAndEquity) < 0.01,
   403→   403→    currentYearEarnings,
   404→   404→  }
   405→   405→}

   420→   420→
   421→   421→export async function getCashFlow(range?: DateRange): Promise<CashFlowData> {
   422→   422→  // Cash & bank accounts
   423→   423→  const accounts = await db.account.findMany({
   424→   424→    where: { accountRole: { in: ['CASH', 'BANK'] }, isActive: true },
   425→   425→    select: { id: true, code: true, name: true, nameAr: true },
   426→   426→    orderBy: { code: 'asc' },
   427→   427→  })
   428→   428→
   429→   429→  if (accounts.length === 0) {
   430→   430→    return { inflows: 0, outflows: 0, netCashFlow: 0, openingBalance: 0, closingBalance: 0, byAccount: [], monthly: [] }
   431→   431→  }
   432→   432→
   433→   433→  const accountIds = accounts.map(a => a.id)
   434→   434→
   435→   435→  // Lines within the period
   436→   436→  const periodLines = await db.journalLine.findMany({
   437→   437→    where: postedLinesWhere({
   438→   438→      accountId: { in: accountIds },
   439→   439→      ...dateRangeFilter(range),
   440→   440→    }),
   441→   441→    include: { journalEntry: { select: { date: true } }, account: { select: { code: true, name: true, nameAr: true } } },
   442→   442→    orderBy: { journalEntry: { date: 'asc' } },
   443→   443→  })
   444→   444→
   445→   445→  // Opening balance = sum of all lines BEFORE range.from
   446→   446→  let openingWhere: Prisma.JournalLineWhereInput = postedLinesWhere({ accountId: { in: accountIds } })
   447→   447→  if (range?.from) {
   448→   448→    openingWhere = {
   449→   449→      deletedAt: null,
   450→   450→      journalEntry: {
   451→   451→        status: 'POSTED',
   452→   452→        deletedAt: null,
   453→   453→        date: { lt: range.from },
   454→   454→      },
   455→   455→    }
   456→   456→  }
   457→   457→  const openingAgg = await db.journalLine.aggregate({
   458→   458→    _sum: { debit: true, credit: true },
   459→   459→    where: openingWhere,
   460→   460→  })
   461→   461→  const openingBalance = toNumber(openingAgg._sum.debit) - toNumber(openingAgg._sum.credit)
   462→   462→
   463→   463→  // Aggregate by account
   464→   464→  const byAccountMap = new Map<string, { code: string; name: string; nameAr: string | null; inflows: number; outflows: number; net: number }>()
   465→   465→  for (const a of accounts) {
   466→   466→    byAccountMap.set(a.id, { code: a.code, name: a.name, nameAr: a.nameAr, inflows: 0, outflows: 0, net: 0 })
   467→   467→  }
   468→   468→
   469→   469→  // Monthly aggregation
   470→   470→  const monthlyMap = new Map<string, { inflows: number; outflows: number; net: number }>()
   471→   471→
   472→   472→  let totalInflows = 0
   473→   473→  let totalOutflows = 0
   474→   474→
   475→   475→  for (const line of periodLines) {
   476→   476→    const debit = toNumber(line.debit)
   477→   477→    const credit = toNumber(line.credit)
   478→   478→    const acc = byAccountMap.get(line.accountId)
   479→   479→    if (acc) {
   480→   480→      acc.inflows += debit
   481→   481→      acc.outflows += credit
   482→   482→      acc.net += debit - credit
   483→   483→    }
   484→   484→    totalInflows += debit
   485→   485→    totalOutflows += credit
   486→   486→
   487→   487→    const monthKey = line.journalEntry.date.toISOString().slice(0, 7)
   488→   488→    if (!monthlyMap.has(monthKey)) monthlyMap.set(monthKey, { inflows: 0, outflows: 0, net: 0 })
   489→   489→    const m = monthlyMap.get(monthKey)!
   490→   490→    m.inflows += debit
   491→   491→    m.outflows += credit
   492→   492→    m.net += debit - credit
   493→   493→  }
   494→   494→
   495→   495→  const closingBalance = openingBalance + (totalInflows - totalOutflows)
   496→   496→
   497→   497→  return {
   498→   498→    inflows: totalInflows,
   499→   499→    outflows: totalOutflows,
   500→   500→    netCashFlow: totalInflows - totalOutflows,
   501→   501→    openingBalance,
   502→   502→    closingBalance,
   503→   503→    byAccount: Array.from(byAccountMap.values()),
   504→   504→    monthly: Array.from(monthlyMap.entries())
   505→   505→      .sort(([a], [b]) => a.localeCompare(b))
   506→   506→      .map(([month, v]) => ({ month, ...v })),
   507→   507→  }
   508→   508→}

   528→   528→  openingBalance: number

   530→   530→  totalDebit: number
   531→   531→  totalCredit: number
   532→   532→  closingBalance: number

   534→   534→
   535→   535→export async function getGeneralLedger(
   536→   536→  accountId: string,
   537→   537→  range?: DateRange
   538→   538→): Promise<GeneralLedgerData | null> {
   539→   539→  const account = await db.account.findUnique({
   540→   540→    where: { id: accountId },
   541→   541→    select: { id: true, code: true, name: true, nameAr: true, type: true },
   542→   542→  })
   543→   543→  if (!account) return null
   544→   544→
   545→   545→  // Opening balance (before range.from)
   546→   546→  let openingBalance = 0
   547→   547→  if (range?.from) {
   548→   548→    const openingAgg = await db.journalLine.aggregate({
   549→   549→      _sum: { debit: true, credit: true },
   550→   550→      where: {
   551→   551→        deletedAt: null,
   552→   552→        accountId,
   553→   553→        journalEntry: { status: 'POSTED', deletedAt: null, date: { lt: range.from } },
   554→   554→      },
   555→   555→    })
   556→   556→    const d = toNumber(openingAgg._sum.debit)
   557→   557→    const c = toNumber(openingAgg._sum.credit)
   558→   558→    openingBalance = signForType(account.type) * (d - c)
   559→   559→  }
   560→   560→
   561→   561→  // Period lines
   562→   562→  const lines = await db.journalLine.findMany({
   563→   563→    where: {
   564→   564→      deletedAt: null,
   565→   565→      accountId,
   566→   566→      journalEntry: {
   567→   567→        status: 'POSTED',
   568→   568→        deletedAt: null,
   569→   569→        ...(range && (range.from || range.to) && {
   570→   570→          date: {
   571→   571→            ...(range.from && { gte: range.from }),
   572→   572→            ...(range.to && { lte: range.to }),
   573→   573→          },
   574→   574→        }),
   575→   575→      },
   576→   576→    },
   577→   577→    include: {
   578→   578→      journalEntry: { select: { entryNo: true, date: true, description: true, sourceType: true } },
   579→   579→      costCenter: { select: { code: true } },
   580→   580→    },
   581→   581→    orderBy: { journalEntry: { date: 'asc' } },
   582→   582→  })
   583→   583→
   584→   584→  let running = openingBalance
   585→   585→  const sign = signForType(account.type)
   586→   586→  const glLines: GeneralLedgerLine[] = []
   587→   587→  let totalDebit = 0
   588→   588→  let totalCredit = 0
   589→   589→
   590→   590→  for (const l of lines) {
   591→   591→    const debit = toNumber(l.debit)
   592→   592→    const credit = toNumber(l.credit)
   593→   593→    running += sign * (debit - credit)
   594→   594→    totalDebit += debit
   595→   595→    totalCredit += credit
   596→   596→    glLines.push({
   597→   597→      date: l.journalEntry.date,
   598→   598→      entryNo: l.journalEntry.entryNo,
   599→   599→      description: l.journalEntry.description,
   600→   600→      lineDescription: l.description,
   601→   601→      debit,
   602→   602→      credit,
   603→   603→      balance: running,
   604→   604→      sourceType: l.journalEntry.sourceType,
   605→   605→      costCenterCode: l.costCenter?.code || null,
   606→   606→    })
   607→   607→  }
   608→   608→
   609→   609→  return {
   610→   610→    account: { id: account.id, code: account.code, name: account.name, nameAr: account.nameAr, type: account.type },
   611→   611→    openingBalance,
   612→   612→    lines: glLines,
   613→   613→    totalDebit,
   614→   614→    totalCredit,
   615→   615→    closingBalance: running,
   616→   616→  }
   617→   617→}
   618→   618→

   620→   620→// Cost Center / Project reports
   621→   621→// ---------------------------------------------------------------------------
   622→   622→
   623→   623→/**
   624→   624→ * Build a map of project.id → costCenter.id.
   625→   625→ * Uses the direct Project.costCenterId link first, then falls back to code/name matching.
   626→   626→ */
   627→   627→export async function buildProjectCostCenterMap(projectIds: string[]): Promise<Map<string, string>> {
   628→   628→  if (projectIds.length === 0) return new Map()
   629→   629→  const projects = await db.project.findMany({
   630→   630→    where: { id: { in: projectIds } },
   631→   631→    select: { id: true, code: true, name: true, nameAr: true, costCenterId: true },
   632→   632→  })
   633→   633→  const result = new Map<string, string>()
   634→   634→
   635→   635→  // 1. Direct link via costCenterId
   636→   636→  const linkedProjects = projects.filter(p => p.costCenterId)
   637→   637→  for (const p of linkedProjects) result.set(p.id, p.costCenterId!)
   638→   638→
   639→   639→  // 2. Fallback for unlinked: match by code, then by name
   640→   640→  const unlinked = projects.filter(p => !p.costCenterId)
   641→   641→  if (unlinked.length > 0) {
   642→   642→    const costCenters = await db.costCenter.findMany({ select: { id: true, code: true, name: true } })
   643→   643→    for (const p of unlinked) {
   644→   644→      let cc = costCenters.find(c => c.code === p.code)
   645→   645→      if (!cc) {
   646→   646→        const pName = (p.nameAr || p.name || '').trim()
   647→   647→        cc = costCenters.find(c => pName.includes(c.name) || c.name.includes(pName))
   648→   648→      }
   649→   649→      if (!cc) {
   650→   650→        const m = p.code.match(/(\d+)/)
   651→   651→        if (m) cc = costCenters.find(c => c.code.includes(m[1]))
   652→   652→      }
   653→   653→      if (cc) result.set(p.id, cc.id)
   654→   654→    }
   655→   655→  }
   656→   656→  return result
   657→   657→}

   659→   659→/**
   660→   660→ * Get per-project revenue & cost from posted JEs via cost center.
   661→   661→ */
   662→   662→export async function getProjectBalances(
   663→   663→  projectIds: string[],
   664→   664→  range?: DateRange
   665→   665→): Promise<Map<string, { revenue: number; costs: number; costCenterId: string | null }>> {
   666→   666→  const result = new Map<string, { revenue: number; costs: number; costCenterId: string | null }>()
   667→   667→  for (const pid of projectIds) result.set(pid, { revenue: 0, costs: 0, costCenterId: null })
   668→   668→
   669→   669→  const ccMap = await buildProjectCostCenterMap(projectIds)
   670→   670→  const costCenterIds = [...ccMap.values()]
   671→   671→  if (costCenterIds.length === 0) return result
   672→   672→
   673→   673→  // Reverse map: costCenterId → projectId
   674→   674→  const ccToProject = new Map<string, string>()
   675→   675→  for (const [pid, ccId] of ccMap) ccToProject.set(ccId, pid)
   676→   676→  // Update costCenterId in result
   677→   677→  for (const [pid, ccId] of ccMap) {
   678→   678→    const r = result.get(pid)
   679→   679→    if (r) r.costCenterId = ccId
   680→   680→  }
   681→   681→
   682→   682→  const lines = await db.journalLine.findMany({
   683→   683→    where: postedLinesWhere({
   684→   684→      costCenterId: { in: costCenterIds },
   685→   685→      ...dateRangeFilter(range),
   686→   686→    }),
   687→   687→    include: { account: { select: { type: true } }, costCenter: { select: { id: true } } },
   688→   688→  })
   689→   689→
   690→   690→  for (const l of lines) {
   691→   691→    const pid = ccToProject.get(l.costCenterId || '')
   692→   692→    if (!pid) continue
   693→   693→    const r = result.get(pid)
   694→   694→    if (!r) continue
   695→   695→    const debit = toNumber(l.debit)
   696→   696→    const credit = toNumber(l.credit)
   697→   697→    if (l.account.type === 'REVENUE') {
   698→   698→      r.revenue += credit - debit
   699→   699→    } else if (l.account.type === 'EXPENSE') {
   700→   700→      r.costs += debit - credit
   701→   701→    }
   702→   702→  }
   703→   703→
   704→   704→  return result
   705→   705→}

   707→   707→/**
   708→   708→ * Detailed project cost breakdown by account role, from posted JEs.
   709→   709→ */
   710→   710→export async function getProjectCostBreakdown(
   711→   711→  projectId: string,
   712→   712→  range?: DateRange
   713→   713→): Promise<{
   714→   714→  costCenterId: string | null
   715→   715→  byRole: Map<string, number>
   716→   716→  total: number
   717→   717→  revenue: number
   718→   718→}> {
   719→   719→  const ccMap = await buildProjectCostCenterMap([projectId])
   720→   720→  const costCenterId = ccMap.get(projectId) || null
   721→   721→
   722→   722→  const byRole = new Map<string, number>()
   723→   723→  let total = 0
   724→   724→  let revenue = 0
   725→   725→
   726→   726→  if (!costCenterId) {
   727→   727→    return { costCenterId, byRole, total, revenue }
   728→   728→  }
   729→   729→
   730→   730→  const lines = await db.journalLine.findMany({
   731→   731→    where: postedLinesWhere({
   732→   732→      costCenterId,
   733→   733→      ...dateRangeFilter(range),
   734→   734→    }),
   735→   735→    include: { account: { select: { type: true, accountRole: true } } },
   736→   736→  })
   737→   737→
   738→   738→  for (const l of lines) {
   739→   739→    const debit = toNumber(l.debit)
   740→   740→    const credit = toNumber(l.credit)
   741→   741→    if (l.account.type === 'EXPENSE') {
   742→   742→      const amt = debit - credit
   743→   743→      const role = l.account.accountRole || 'OTHER'
   744→   744→      byRole.set(role, (byRole.get(role) || 0) + amt)
   745→   745→      total += amt
   746→   746→    } else if (l.account.type === 'REVENUE') {
   747→   747→      revenue += credit - debit
   748→   748→    }
   749→   749→  }
   750→   750→
   751→   751→  return { costCenterId, byRole, total, revenue }
   752→   752→}

   758→   758→export async function getCostCenterReport(range?: DateRange): Promise<CostCenterBalance[]> {
   759→   759→  const costCenters = await db.costCenter.findMany({
   760→   760→    where: { isActive: true },
   761→   761→    select: { id: true, code: true, name: true },
   762→   762→    orderBy: { code: 'asc' },
   763→   763→  })
   764→   764→  if (costCenters.length === 0) return []
   765→   765→
   766→   766→  const ccIds = costCenters.map(c => c.id)
   767→   767→
   768→   768→  const lines = await db.journalLine.findMany({
   769→   769→    where: postedLinesWhere({
   770→   770→      costCenterId: { in: ccIds },
   771→   771→      ...dateRangeFilter(range),
   772→   772→    }),
   773→   773→    include: { account: { select: { type: true } }, costCenter: { select: { id: true } } },
   774→   774→  })
   775→   775→
   776→   776→  const map = new Map<string, { revenue: number; costs: number }>()
   777→   777→  for (const cc of costCenters) map.set(cc.id, { revenue: 0, costs: 0 })
   778→   778→
   779→   779→  for (const l of lines) {
   780→   780→    const entry = map.get(l.costCenterId || '')
   781→   781→    if (!entry) continue
   782→   782→    const debit = toNumber(l.debit)
   783→   783→    const credit = toNumber(l.credit)
   784→   784→    if (l.account.type === 'REVENUE') entry.revenue += credit - debit
   785→   785→    else if (l.account.type === 'EXPENSE') entry.costs += debit - credit
   786→   786→  }
   787→   787→
   788→   788→  return costCenters.map(cc => {
   789→   789→    const e = map.get(cc.id)!
   790→   790→    return {
   791→   791→      costCenterId: cc.id,
   792→   792→      code: cc.code,
   793→   793→      name: cc.name,
   794→   794→      revenue: e.revenue,
   795→   795→      costs: e.costs,
   796→   796→      net: e.revenue - e.costs,
   797→   797→    }
   798→   798→  })
   799→   799→}

   805→   805→export interface VATReconciliationData {
   806→   806→  outputVat: number // VAT_OUTPUT credit balance
   807→   807→  inputVat: number // VAT_INPUT debit balance
   808→   808→  netVatDue: number // output - input (if positive, payable)
   809→   809→  vatSettlement: number
   810→   810→  outputAccounts: AccountBalance[]
   811→   811→  inputAccounts: AccountBalance[]
   812→   812→}
   813→   813→
   814→   814→export async function getVATReconciliation(range?: DateRange): Promise<VATReconciliationData> {
   815→   815→  const [outputAccounts, inputAccounts] = await Promise.all([
   816→   816→    getAccountBalancesByType(['LIABILITY'], range).then(list =>
   817→   817→      list.filter(a => ['VAT_OUTPUT', 'VAT_DUE', 'VAT_SETTLEMENT'].includes(a.accountRole || ''))
   818→   818→    ),
   819→   819→    getAccountBalancesByType(['ASSET'], range).then(list =>
   820→   820→      list.filter(a => a.accountRole === 'VAT_INPUT')
   821→   821→    ),
   822→   822→  ])
   823→   823→
   824→   824→  const outputVat = outputAccounts.reduce((s, a) => s + a.balance, 0)
   825→   825→  const inputVat = inputAccounts.reduce((s, a) => s + a.balance, 0)
   826→   826→  const netVatDue = outputVat - inputVat
   827→   827→
   828→   828→  return {
   829→   829→    outputVat,
   830→   830→    inputVat,
   831→   831→    netVatDue,
   832→   832→    vatSettlement: netVatDue,
   833→   833→    outputAccounts,
   834→   834→    inputAccounts,
   835→   835→  }
   836→   836→}

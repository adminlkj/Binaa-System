    32→// ============ Arabic/English Month Names ============

    34→const englishMonths = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December']
    35→
    36→function formatMonthYear(month: number, year: number, lang: 'ar' | 'en'): string {
    37→  if (lang === 'ar') return `${arabicMonths[month - 1]} ${year}`
    38→  return `${englishMonths[month - 1]} ${year}`
    39→}

   144→// ============ Create Rental Invoice Page ============
   145→function CreateRentalInvoicePage({
   146→  approvedTimesheets, onBack,
   147→}: {
   148→  approvedTimesheets: TimesheetOption[]; onBack: () => void
   149→}) {
   150→  const queryClient = useQueryClient()
   151→  const { lang } = useAppStore()
   152→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   153→
   154→  const [timesheetId, setTimesheetId] = useState('')
   155→  const [date, setDate] = useState('')
   156→  const [dueDate, setDueDate] = useState('')
   157→  const [notes, setNotes] = useState('')
   158→
   159→  // Selected timesheet - RULE: User selects ONLY a Timesheet
   160→  const selectedTimesheet = approvedTimesheets.find(ts => ts.id === timesheetId)
   161→
   162→  // RULE: System auto-fills from contract and timesheet
   163→  // Auto-filled from contract: client, project, equipment, contract no, sales order no, hourly rate, delivery fees, payment terms
   164→  // Auto-filled from timesheet: month, year, operating hours
   165→  // RULE: Rate is read-only (from contract)
   166→  const contract = selectedTimesheet?.contract
   167→  const rental = selectedTimesheet?.rental
   168→  const hourlyRate = rental?.hourlyRate || contract?.hourlyRate || 0
   169→  const operatingHours = selectedTimesheet?.operatingHours || 0
   170→  const deliveryFees = rental?.deliveryFees || contract?.deliveryFees || 0
   171→  const deliveryFeesTaxable = rental?.deliveryFeesTaxable ?? contract?.deliveryFeesTaxable ?? true
   172→  const clientId = rental?.clientId || contract?.clientId || ''
   173→  const clientName = selectedTimesheet?.clientName || selectedTimesheet?.clientNameAr || ''
   174→  const projectId = selectedTimesheet?.projectId || ''
   175→  const projectName = selectedTimesheet?.project?.name || ''
   176→  const equipmentName = selectedTimesheet?.equipment?.nameAr || selectedTimesheet?.equipment?.name || ''
   177→  const contractNo = contract?.contractNo || ''
   178→  const salesOrderNo = rental?.salesOrderNo || contract?.salesOrderNo || ''
   179→  const paymentTerms = rental?.paymentDuration || contract?.paymentTerms || '30 days'
   180→
   181→  // RULE: System auto-calculates: hours × rate = subtotal
   182→  const subtotal = operatingHours * hourlyRate
   183→  const vatRate = 0.15
   184→  const rentalVat = Math.round(subtotal * vatRate * 100) / 100
   185→  // Delivery fees (with/without VAT based on contract setting)
   186→  const deliveryVat = deliveryFeesTaxable && deliveryFees > 0 ? Math.round(deliveryFees * vatRate * 100) / 100 : 0
   187→  const totalVat = rentalVat + deliveryVat
   188→  const totalAmount = subtotal + deliveryFees + totalVat
   189→
   190→  const createMutation = useMutation({
   191→    mutationFn: (data: Record<string, unknown>) =>
   192→      fetch('/api/sales-invoices', {
   193→        method: 'POST',
   194→        headers: { 'Content-Type': 'application/json' },
   195→        body: JSON.stringify(data),
   196→      }).then(async r => {
   197→        if (!r.ok) {
   198→          const err = await r.json().catch(() => ({}))
   199→          throw new Error((err as { error?: string }).error || 'فشل في إنشاء الفاتورة')
   200→        }
   201→        return r.json()
   202→      }),
   203→    onSuccess: () => {
   204→      queryClient.invalidateQueries({ queryKey: ['rental-invoices'] })
   205→      queryClient.invalidateQueries({ queryKey: ['equipment-timesheets'] })
   206→      onBack()
   207→    },
   208→  })
   209→
   210→  const handleSubmit = (e: React.FormEvent) => {
   211→    e.preventDefault()
   212→    if (!timesheetId || !date || !dueDate || !clientId) return
   213→
   214→    // Accounting Integration: When rental invoice is created,
   215→    // the accounting engine should create journal entries:
   216→    // Debit: Accounts Receivable (totalAmount)
   217→    // Credit: Rental Revenue (subtotal)
   218→    // Credit: Delivery Revenue (deliveryFees) if applicable
   219→    // Credit: Output VAT (totalVat)
   220→    // Also, timesheet status should be updated to INVOICED
   221→    // This is handled via POST /api/journal-entries with sourceType: SALES_INVOICE
   222→    createMutation.mutate({
   223→      sourceType: 'TIMESHEET',
   224→      timesheetId,
   225→      date,
   226→      dueDate,
   227→      notes,
   228→    })
   229→  }
   230→
   231→  // RULE: Cannot create invoice without an approved timesheet
   232→  if (approvedTimesheets.length === 0) {
   233→    return (
   234→      <div className="space-y-6">
   235→        <div className="flex items-center gap-3">
   236→          <Button variant="outline" size="icon" onClick={onBack}>
   237→            <ArrowRight className="size-4" />
   238→          </Button>
   239→          <div className="flex-1">
   240→            <h1 className="text-2xl font-bold text-gray-900">{t('فاتورة إيجار جديدة', 'New Rental Invoice')}</h1>
   241→          </div>
   242→        </div>
   243→        <Card className="bg-amber-50 border-amber-200">
   244→          <CardContent className="p-6 text-center">
   245→            <p className="text-amber-700 font-medium">{t(labels.noApprovedTimesheets.ar, labels.noApprovedTimesheets.en)}</p>
   246→            <p className="text-sm text-amber-600 mt-1">{t('يجب اعتماد سجل ساعات عمل أولاً من وحدة ساعات العمل', 'An approved timesheet must be created first from the Timesheets module')}</p>
   247→          </CardContent>
   248→        </Card>
   249→      </div>
   250→    )
   251→  }
   252→
   253→  return (
   254→    <div className="space-y-6">
   255→      <div className="flex items-center gap-3">
   256→        <Button variant="outline" size="icon" onClick={onBack}>
   257→          <ArrowRight className="size-4" />
   258→        </Button>
   259→        <div className="flex-1">
   260→          <h1 className="text-2xl font-bold text-gray-900">{t('فاتورة إيجار جديدة', 'New Rental Invoice')}</h1>
   261→          <p className="text-sm text-muted-foreground">{t('إنشاء فاتورة إيجار من سجل ساعات عمل معتمد', 'Create rental invoice from an approved timesheet')}</p>
   262→        </div>
   263→      </div>
   264→
   265→      <form onSubmit={handleSubmit} className="space-y-6">
   266→        {/* RULE: User selects ONLY a Timesheet */}
   267→        <Card>
   268→          <CardHeader className="pb-4">
   269→            <CardTitle className="text-lg">{t('اختيار سجل الساعات', 'Select Timesheet')}</CardTitle>
   270→          </CardHeader>
   271→          <CardContent className="space-y-4">
   272→            <div className="space-y-2">
   273→              <Label>{t(labels.selectTimesheet.ar, labels.selectTimesheet.en)} *</Label>
   274→              <Select value={timesheetId} onValueChange={setTimesheetId}>
   275→                <SelectTrigger className="w-full"><SelectValue placeholder={t(labels.selectTimesheet.ar, labels.selectTimesheet.en)} /></SelectTrigger>
   276→                <SelectContent>
   277→                  {approvedTimesheets.map(ts => (
   278→                    <SelectItem key={ts.id} value={ts.id}>
   279→                      {ts.equipment?.name || '—'} - {formatMonthYear(ts.month, ts.year, lang)} ({formatNumber(ts.operatingHours)} {t('ساعة', 'hrs')}) [{ts.contract?.contractNo}]
   280→                    </SelectItem>
   281→                  ))}
   282→                </SelectContent>
   283→              </Select>
   284→            </div>
   285→
   286→            {/* Auto-filled info from timesheet & contract */}
   287→            {selectedTimesheet && (
   288→              <div className="p-4 rounded-lg border bg-blue-50 space-y-3">
   289→                <div className="flex items-center gap-2 text-sm font-medium text-blue-700 mb-2">
   290→                  <FileText className="size-4" />
   291→                  {t(labels.autoFilledFromContract.ar, labels.autoFilledFromContract.en)} / {t(labels.autoFilledFromTimesheet.ar, labels.autoFilledFromTimesheet.en)}
   292→                </div>
   293→                <div className="grid grid-cols-2 sm:grid-cols-4 gap-3 text-sm">
   294→                  <div><span className="text-muted-foreground">{t(labels.client.ar, labels.client.en)}:</span><p className="font-medium">{clientName}</p></div>
   295→                  <div><span className="text-muted-foreground">{t(labels.project.ar, labels.project.en)}:</span><p className="font-medium">{projectName}</p></div>
   296→                  <div><span className="text-muted-foreground">{t(labels.equipment.ar, labels.equipment.en)}:</span><p className="font-medium">{equipmentName}</p></div>
   297→                  <div><span className="text-muted-foreground">{t(labels.deliveryMonth.ar, labels.deliveryMonth.en)}:</span><p className="font-medium">{formatMonthYear(selectedTimesheet.month, selectedTimesheet.year, lang)}</p></div>
   298→                  <div><span className="text-muted-foreground">{t(labels.contractNo.ar, labels.contractNo.en)}:</span><p className="font-medium font-mono">{contractNo}</p></div>
   299→                  <div><span className="text-muted-foreground">{t(labels.salesOrderNo.ar, labels.salesOrderNo.en)}:</span><p className="font-medium font-mono">{salesOrderNo || '—'}</p></div>
   300→                  <div><span className="text-muted-foreground">{t(labels.paymentTerms.ar, labels.paymentTerms.en)}:</span><p className="font-medium">{paymentTerms}</p></div>
   301→                </div>
   302→              </div>
   303→            )}
   304→          </CardContent>
   305→        </Card>
   306→
   307→        {/* Operating Hours & Rate - RULE: No manual editing of rate or hours */}
   308→        <Card>
   309→          <CardHeader className="pb-4">
   310→            <CardTitle className="text-lg">{t('ساعات التشغيل والسعر', 'Operating Hours & Rate')}</CardTitle>

   313→            <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   314→              <div className="space-y-2">
   315→                <Label>{t(labels.operatingHours.ar, labels.operatingHours.en)}</Label>

   318→                <p className="text-xs text-muted-foreground">{t('من سجل الساعات - للقراءة فقط', 'From timesheet - read only')}</p>

   338→          <Card>
   339→            <CardHeader className="pb-4">
   340→              <CardTitle className="text-lg">{t(labels.deliveryFees.ar, labels.deliveryFees.en)}</CardTitle>
   341→            </CardHeader>

   350→                    {deliveryFeesTaxable ? t('خاضعة للضريبة', 'Taxable') : t('غير خاضعة للضريبة', 'Not Taxable')}
   351→                  </Badge>
   352→                </div>
   353→              </div>
   354→            </CardContent>

   358→        {/* Dates */}
   359→        <Card>

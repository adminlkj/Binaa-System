   258→/**
   259→ * Resolve cost center ID from a project ID.
   260→ * Project IDs and Cost Center IDs are different entities in the database.
   261→ * This function looks up the project's code and finds the matching cost center.
   262→ * Returns undefined if no matching cost center is found.
   263→ */
   264→export async function resolveCostCenterId(projectId: string | undefined | null): Promise<string | undefined> {
   265→  if (!projectId) return undefined
   266→  const project = await db.project.findUnique({ where: { id: projectId }, select: { code: true } })
   267→  if (!project) return undefined
   268→  const costCenter = await db.costCenter.findFirst({ where: { code: project.code } })
   269→  return costCenter?.id
   270→}

   254→function AccountStatementDialog({ account, open, onClose }: {
   255→  account: Account | null; open: boolean; onClose: () => void
   256→}) {
   257→  const { lang } = useAppStore()
   258→  const [dateFrom, setDateFrom] = useState('')
   259→  const [dateTo, setDateTo] = useState('')
   260→
   261→  const { data: statementData, isLoading, isError, refetch } = useQuery<AccountStatementData | null>({
   262→    queryKey: ['account-statement-detail', account?.id, dateFrom, dateTo],
   263→    queryFn: async () => {
   264→      if (!account) return null
   265→      const params = new URLSearchParams({ accountId: account.id })
   266→      if (dateFrom) params.set('dateFrom', dateFrom)
   267→      if (dateTo) params.set('dateTo', dateTo)
   268→      const res = await fetch(`/api/accounts/statement?${params}`)
   269→      if (!res.ok) throw new Error()
   270→      return res.json()
   271→    },
   272→    enabled: !!account && open,
   273→  })
   274→
   275→  // Reset dates when account changes
   276→  React.useEffect(() => {
   277→    if (account) {
   278→      setDateFrom('')
   279→      setDateTo('')
   280→    }
   281→  }, [account?.id])
   282→
   283→  if (!account) return null
   284→
   285→  const statement = statementData as AccountStatementData | null
   286→
   287→  return (
   288→    <Dialog open={open} onOpenChange={onClose}>
   289→      <DialogContent className="max-w-4xl max-h-[90vh] overflow-hidden flex flex-col">
   290→        <DialogHeader>
   291→          <DialogTitle className="flex items-center gap-2">
   292→            <FileSearch className="size-5 text-emerald-600" />

   555→    const uniqueAccounts = new Map<string, { code: string; name: string; nameAr: string | null; totalDebit: number; totalCredit: number }>()
   556→    for (const line of entry.lines) {
   557→      const existing = uniqueAccounts.get(line.accountId)
   558→      if (existing) {
   559→        existing.totalDebit += line.debit
   560→        existing.totalCredit += line.credit
   561→      } else {
   562→        uniqueAccounts.set(line.accountId, {
   563→          code: line.account.code,
   564→          name: line.account.name,
   565→          nameAr: line.account.nameAr,
   566→          totalDebit: line.debit,
   567→          totalCredit: line.credit,
   568→        })
   569→      }
   570→    }
   571→
   572→    // For each unique account, compute before/after balance
   573→    const impactItems: {
   574→      accountId: string; code: string; name: string; nameAr: string | null
   575→      beforeBalance: number; movementDebit: number; movementCredit: number; afterBalance: number

   579→      const accountData = accounts.find(a => a.id === accountId)

   584→      // Net movement effect on this account

   588→        ? info.totalDebit - info.totalCredit

   591→      let beforeBalance: number

   594→      if (isPosted) {

   597→        afterBalance = currentBalance
   598→      } else {
   599→        // Current balance does NOT include this JE's effect (DRAFT)

   606→        code: info.code,
   607→        name: info.name,

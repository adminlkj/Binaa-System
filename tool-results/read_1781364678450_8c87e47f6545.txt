   489→   489→export async function initializeChartOfAccounts() {
   490→   490→  let created = 0
   491→   491→  let updated = 0
   492→   492→
   493→   493→  // Create parent accounts first, then children (sorted by code length then code)
   494→   494→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   495→   495→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   496→   496→    return a.code.localeCompare(b.code)
   497→   497→  })
   498→   498→
   499→   499→  for (const tmpl of sorted) {
   500→   500→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   501→   501→    if (existing) {
   502→   502→      // Update existing account with new fields
   503→   503→      await db.account.update({
   504→   504→        where: { code: tmpl.code },
   505→   505→        data: {
   506→   506→          name: tmpl.name,
   507→   507→          nameAr: tmpl.nameAr,
   508→   508→          type: tmpl.type,
   509→   509→          activityType: tmpl.activityType || null,
   510→   510→          isSystem: tmpl.isSystem || false,
   511→   511→          allowPosting: tmpl.allowPosting || false,
   512→   512→          level: tmpl.level || 0,
   513→   513→          isActive: true,
   514→   514→        },
   515→   515→      })
   516→   516→      // Update parent reference if needed
   517→   517→      if (tmpl.parentId) {
   518→   518→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   519→   519→        if (parent && existing.parentId !== parent.id) {
   520→   520→          await db.account.update({
   521→   521→            where: { code: tmpl.code },
   522→   522→            data: { parentId: parent.id },
   523→   523→          })
   524→   524→        }
   525→   525→      }
   526→   526→      updated++
   527→   527→    } else {
   528→   528→      await ensureAccountExists(tmpl)
   529→   529→      created++
   530→   530→    }
   531→   531→  }
   532→   532→
   533→   533→  const total = await db.account.count()
   534→   534→  return { created, updated, total }
   535→   535→}

   537→   537→// ============ JOURNAL ENTRY CREATION ============
   538→   538→
   539→   539→export async function createJournalEntry(template: JournalEntryTemplate) {
   540→   540→  // Validate: total debits must equal total credits
   541→   541→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   542→   542→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   543→   543→
   544→   544→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   545→   545→    throw new Error(
   546→   546→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   547→   547→    )
   548→   548→  }
   549→   549→
   550→   550→  // Ensure all referenced accounts exist
   551→   551→  for (const line of template.lines) {
   552→   552→    const account = await getAccountByCode(line.accountCode)
   553→   553→    if (!account) {
   554→   554→      throw new Error(`Account not found: ${line.accountCode}`)
   555→   555→    }
   556→   556→  }
   557→   557→
   558→   558→  // Create the journal entry with lines
   559→   559→  const entry = await db.journalEntry.create({
   560→   560→    data: {
   561→   561→      entryNo: template.entryNo,
   562→   562→      date: template.date,
   563→   563→      description: template.description,
   564→   564→      status: 'POSTED', // Auto-entries are posted immediately
   565→   565→      lines: {
   566→   566→        create: await Promise.all(
   567→   567→          template.lines.map(async (line) => {
   568→   568→            const account = await getAccountByCode(line.accountCode)
   569→   569→            return {
   570→   570→              accountId: account!.id,
   571→   571→              costCenterId: line.costCenterId,
   572→   572→              debit: line.debit,
   573→   573→              credit: line.credit,
   574→   574→              description: line.description,
   575→   575→            }
   576→   576→          })
   577→   577→        ),
   578→   578→      },
   579→   579→    },
   580→   580→    include: { lines: true },
   581→   581→  })
   582→   582→
   583→   583→  return entry
   584→   584→}

  1551→  1551→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1552→  1552→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1553→  1553→  if (dateFrom || dateTo) {
  1554→  1554→    dateFilter.date = {
  1555→  1555→      ...(dateFrom && { gte: dateFrom }),
  1556→  1556→      ...(dateTo && { lte: dateTo }),
  1557→  1557→    }
  1558→  1558→  }
  1559→  1559→  const entries = await db.journalEntry.findMany({
  1560→  1560→    where: {
  1561→  1561→      status: 'POSTED',
  1562→  1562→      ...dateFilter,
  1563→  1563→    },
  1564→  1564→    include: {
  1565→  1565→      lines: {
  1566→  1566→        include: {
  1567→  1567→          account: true,
  1568→  1568→        },
  1569→  1569→      },
  1570→  1570→    },
  1571→  1571→  })
  1572→  1572→
  1573→  1573→  // Aggregate by account
  1574→  1574→  const accountBalances = new Map<string, {
  1575→  1575→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1576→  1576→    totalDebit: number
  1577→  1577→    totalCredit: number
  1578→  1578→  }>()
  1579→  1579→
  1580→  1580→  for (const entry of entries) {
  1581→  1581→    for (const line of entry.lines) {
  1582→  1582→      const key = line.accountId
  1583→  1583→      if (!accountBalances.has(key)) {
  1584→  1584→        accountBalances.set(key, {
  1585→  1585→          account: line.account,
  1586→  1586→          totalDebit: 0,
  1587→  1587→          totalCredit: 0,
  1588→  1588→        })
  1589→  1589→      }
  1590→  1590→      const bal = accountBalances.get(key)!
  1591→  1591→      bal.totalDebit += line.debit
  1592→  1592→      bal.totalCredit += line.credit
  1593→  1593→    }
  1594→  1594→  }
  1595→  1595→
  1596→  1596→  // Calculate net balance for each account
  1597→  1597→  const results = Array.from(accountBalances.values()).map(bal => {
  1598→  1598→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1599→  1599→    let netDebit = 0
  1600→  1600→    let netCredit = 0
  1601→  1601→
  1602→  1602→    if (normalBalance === 'DEBIT') {
  1603→  1603→      const net = bal.totalDebit - bal.totalCredit
  1604→  1604→      if (net >= 0) netDebit = net
  1605→  1605→      else netCredit = Math.abs(net)
  1606→  1606→    } else {
  1607→  1607→      const net = bal.totalCredit - bal.totalDebit
  1608→  1608→      if (net >= 0) netCredit = net
  1609→  1609→      else netDebit = Math.abs(net)
  1610→  1610→    }
  1611→  1611→
  1612→  1612→    return {
  1613→  1613→      ...bal,
  1614→  1614→      netDebit,
  1615→  1615→      netCredit,
  1616→  1616→    }
  1617→  1617→  })
  1618→  1618→
  1619→  1619→  // Sort by account code
  1620→  1620→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1621→  1621→
  1622→  1622→  return results
  1623→  1623→}

  1637→  1637→
  1638→  1638→  const totalDebit = lines.reduce((sum, l) => sum + l.debit, 0)

  1687→  1687→export async function getGeneralLedger(accountCode: string, dateFrom?: Date, dateTo?: Date) {
  1688→  1688→  const account = await getAccountByCode(accountCode)
  1689→  1689→  if (!account) return []
  1690→  1690→
  1691→  1691→  const glDateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1692→  1692→  if (dateFrom || dateTo) {
  1693→  1693→    glDateFilter.date = {
  1694→  1694→      ...(dateFrom && { gte: dateFrom }),
  1695→  1695→      ...(dateTo && { lte: dateTo }),
  1696→  1696→    }
  1697→  1697→  }
  1698→  1698→  const lines = await db.journalLine.findMany({
  1699→  1699→    where: {
  1700→  1700→      accountId: account.id,
  1701→  1701→      journalEntry: {
  1702→  1702→        status: 'POSTED',
  1703→  1703→        ...glDateFilter,
  1704→  1704→      },
  1705→  1705→    },
  1706→  1706→    include: { journalEntry: true },
  1707→  1707→    orderBy: { journalEntry: { date: 'asc' } },
  1708→  1708→  })
  1709→  1709→
  1710→  1710→  let runningBalance = 0
  1711→  1711→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1712→  1712→
  1713→  1713→  return lines.map(line => {
  1714→  1714→    if (normalBalance === 'DEBIT') {
  1715→  1715→      runningBalance += line.debit - line.credit
  1716→  1716→    } else {
  1717→  1717→      runningBalance += line.credit - line.debit
  1718→  1718→    }
  1719→  1719→
  1720→  1720→    return {
  1721→  1721→      date: line.journalEntry.date,
  1722→  1722→      entryNo: line.journalEntry.entryNo,
  1723→  1723→      description: line.journalEntry.description,
  1724→  1724→      debit: line.debit,
  1725→  1725→      credit: line.credit,
  1726→  1726→      balance: runningBalance,
  1727→  1727→    }
  1728→  1728→  })
  1729→  1729→}

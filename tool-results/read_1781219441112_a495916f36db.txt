   272→// ============ Distribution Form Dialog ============
   273→function DistributionFormDialog({ open, onOpenChange, projects, employees, equipment, teams }: {
   274→  open: boolean; onOpenChange: (open: boolean) => void
   275→  projects: Project[]; employees: Employee[]; equipment: Equipment[]; teams: WorkTeam[]
   276→}) {
   277→  const queryClient = useQueryClient()
   278→  const [form, setForm] = useState<DistributionFormData>(defaultForm)
   279→  const { lang } = useAppStore()
   280→
   281→  const resourceOptions = React.useMemo(() => {
   282→    switch (form.resourceType) {
   283→      case 'EMPLOYEE': return employees.map(e => ({ id: e.id, name: e.name, code: e.code }))
   284→      case 'TEAM': return teams.map(t => ({ id: t.id, name: t.name, code: t.code }))
   285→      case 'EQUIPMENT': return equipment.map(e => ({ id: e.id, name: e.name, code: e.code }))
   286→      default: return []
   287→    }
   288→  }, [form.resourceType, employees, teams, equipment])
   289→
   290→  React.useEffect(() => {
   291→    if (open) setForm(defaultForm)
   292→  }, [open])
   293→
   294→  React.useEffect(() => {
   295→    setForm(f => ({ ...f, resourceId: '' }))
   296→  }, [form.resourceType])
   297→
   298→  const createMutation = useMutation({
   299→    mutationFn: (data: Record<string, unknown>) => fetch('/api/resource-distribution', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   300→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['resource-distribution'] }); onOpenChange(false) },
   301→  })
   302→
   303→  const handleSubmit = (e: React.FormEvent) => {
   304→    e.preventDefault()
   305→    createMutation.mutate({
   306→      ...form,
   307→      endDate: form.endDate || null,
   308→    })
   309→  }
   310→
   311→  return (
   312→    <Dialog open={open} onOpenChange={onOpenChange}>
   313→      <DialogContent className="max-w-lg max-h-[90vh] overflow-y-auto">
   314→        <DialogHeader>
   315→          <DialogTitle>{t('توزيع مورد', 'Allocate Resource', lang)}</DialogTitle>
   316→          <DialogDescription>{t('توزيع مورد على مشروع', 'Allocate a resource to a project', lang)}</DialogDescription>
   317→        </DialogHeader>
   318→        <form onSubmit={handleSubmit} className="space-y-4">
   319→          <div className="space-y-2">
   320→            <Label>{t('المشروع *', 'Project *', lang)}</Label>
   321→            <Select value={form.projectId} onValueChange={v => setForm(f => ({ ...f, projectId: v }))}>
   322→              <SelectTrigger><SelectValue placeholder={t('اختر المشروع', 'Select project', lang)} /></SelectTrigger>
   323→              <SelectContent>
   324→                {projects.map(p => <SelectItem key={p.id} value={p.id}>{p.name}</SelectItem>)}
   325→              </SelectContent>
   326→            </Select>
   327→          </div>
   328→
   329→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   330→            <div className="space-y-2">
   331→              <Label>{t('نوع المورد *', 'Resource Type *', lang)}</Label>
   332→              <Select value={form.resourceType} onValueChange={v => setForm(f => ({ ...f, resourceType: v }))}>
   333→                <SelectTrigger><SelectValue placeholder={t('اختر النوع', 'Select type', lang)} /></SelectTrigger>
   334→                <SelectContent>
   335→                  {Object.entries(resourceTypeConfig).map(([k, v]) => (
   336→                    <SelectItem key={k} value={k}>{v.label[lang]}</SelectItem>
   337→                  ))}
   338→                </SelectContent>
   339→              </Select>
   340→            </div>
   341→            <div className="space-y-2">
   342→              <Label>{t('المورد *', 'Resource *', lang)}</Label>
   343→              <Select value={form.resourceId} onValueChange={v => setForm(f => ({ ...f, resourceId: v }))} disabled={!form.resourceType}>
   344→                <SelectTrigger><SelectValue placeholder={form.resourceType ? t('اختر المورد', 'Select resource', lang) : t('اختر النوع أولاً', 'Select type first', lang)} /></SelectTrigger>
   345→                <SelectContent>
   346→                  {resourceOptions.map(r => <SelectItem key={r.id} value={r.id}>{r.name} ({r.code})</SelectItem>)}
   347→                </SelectContent>
   348→              </Select>
   349→            </div>
   350→          </div>
   351→
   352→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   353→            <div className="space-y-2"><Label>{t('تاريخ البداية *', 'Start Date *', lang)}</Label><Input type="date" value={form.startDate} onChange={e => setForm(f => ({ ...f, startDate: e.target.value }))} required /></div>
   354→            <div className="space-y-2"><Label>{t('تاريخ النهاية', 'End Date', lang)}</Label><Input type="date" value={form.endDate} onChange={e => setForm(f => ({ ...f, endDate: e.target.value }))} /></div>
   355→          </div>
   356→
   357→          <DialogFooter>
   358→            <Button type="button" variant="outline" onClick={() => onOpenChange(false)}>{t('إلغاء', 'Cancel', lang)}</Button>
   359→            <Button type="submit" disabled={createMutation.isPending || !form.projectId || !form.resourceType || !form.resourceId || !form.startDate} className="bg-emerald-600 hover:bg-emerald-700">{createMutation.isPending ? t('جاري الحفظ...', 'Saving...', lang) : t('توزيع', 'Allocate', lang)}</Button>
   360→          </DialogFooter>
   361→        </form>
   362→      </DialogContent>
   363→    </Dialog>
   364→  )
   365→}

   405→
   406→  // Filter
   407→  const filtered = distributions.filter(d => {
   408→    if (filterProject && d.projectId !== filterProject) return false
   409→    if (!search) return true
   410→    const s = search.toLowerCase()
   411→    return d.project.name.toLowerCase().includes(s) || ((d.resource as Record<string, unknown>)?.name as string)?.toLowerCase().includes(s)
   412→  })
   413→
   414→  // Summary counts
   415→  const totalEmployees = distributions.filter(d => d.resourceType === 'EMPLOYEE').length
   416→  const totalTeams = distributions.filter(d => d.resourceType === 'TEAM').length
   417→  const totalEquipment = distributions.filter(d => d.resourceType === 'EQUIPMENT').length
   418→
   419→  // Group by project for visual grid
   420→  const projectGroups = React.useMemo(() => {
   421→    const groups: Record<string, { project: Project; items: ResourceDistribution[] }> = {}
   422→    filtered.forEach(d => {
   423→      if (!groups[d.projectId]) {
   424→        groups[d.projectId] = { project: d.project, items: [] }
   425→      }
   426→      groups[d.projectId].items.push(d)
   427→    })
   428→    return Object.values(groups)
   429→  }, [filtered])
   430→
   431→  const printData = useMemo(() => ({
   432→    columns: [
   433→      { key: 'projectName', label: lang === 'ar' ? 'المشروع' : 'Project' },
   434→      { key: 'resourceType', label: lang === 'ar' ? 'نوع المورد' : 'Resource Type' },
   435→      { key: 'resourceName', label: lang === 'ar' ? 'المورد' : 'Resource' },
   436→      { key: 'startDate', label: lang === 'ar' ? 'تاريخ البداية' : 'Start Date' },
   437→      { key: 'endDate', label: lang === 'ar' ? 'تاريخ النهاية' : 'End Date' },
   438→    ],
   439→    rows: filtered.map(d => ({
   440→      projectName: d.project.name,
   441→      resourceType: resourceTypeConfig[d.resourceType]?.label[lang] || d.resourceType,
   442→      resourceName: (d.resource as Record<string, unknown>)?.name as string || '—',
   443→      startDate: d.startDate,
   444→      endDate: d.endDate || '—',
   445→    })),
   446→    infoItems: [
   447→      { label: lang === 'ar' ? 'تاريخ الطباعة' : 'Print Date', value: new Date().toLocaleDateString() },
   448→      { label: lang === 'ar' ? 'عدد التوزيعات' : 'Distributions', value: String(filtered.length) },
   449→    ],
   450→  }), [filtered, lang])
   451→
   452→  const handleExport = () => {
   453→    const columns: CSVColumn[] = [
   454→      { key: 'projectName', label: t('المشروع', 'Project', lang) },
   455→      { key: 'resourceType', label: t('نوع المورد', 'Resource Type', lang) },
   456→      { key: 'resourceName', label: t('المورد', 'Resource', lang) },
   457→      { key: 'startDate', label: t('تاريخ البداية', 'Start Date', lang) },
   458→      { key: 'endDate', label: t('تاريخ النهاية', 'End Date', lang) },
   459→    ]
   460→    exportToCSV(filtered.map(d => ({
   461→      projectName: d.project.name,
   462→      resourceType: resourceTypeConfig[d.resourceType]?.label[lang] || d.resourceType,
   463→      resourceName: (d.resource as Record<string, unknown>)?.name as string || '',
   464→      startDate: d.startDate, endDate: d.endDate || '',
   465→    })), `resource-distribution-${new Date().toISOString().slice(0, 10)}`, columns)
   466→  }
   467→
   468→  // If a project is selected, show the cost sheet
   469→  if (selectedProject) {
   470→    return (
   471→      <ModuleLayout
   472→        title={{ ar: 'كشف تكاليف المشروع', en: 'Project Cost Sheet' }}
   473→        subtitle={{ ar: selectedProject.name, en: selectedProject.name }}
   474→      >
   475→        <ProjectCostSheet project={selectedProject} onBack={() => setSelectedProject(null)} />
   476→      </ModuleLayout>
   477→    )
   478→  }
   479→
   480→  return (
   481→    <ModuleLayout
   482→      title={{ ar: 'توزيع الموارد', en: 'Resource Distribution' }}
   483→      subtitle={{ ar: 'توزيع الموارد على المشاريع وتحسين استخدامها', en: 'Allocate resources to projects and optimize utilization' }}
   484→      actions={
   485→        <div className="flex items-center gap-2">
   486→          <PrintButton type="resource-distribution" data={printData} size="icon" />
   487→          <Button variant="outline" size="icon" onClick={handleExport}><Download className="size-4" /></Button>
   488→          <Button variant="outline" size="icon" onClick={() => refetch()}><RefreshCw className="size-4" /></Button>
   489→          <Button className="gap-2 bg-emerald-600 hover:bg-emerald-700" onClick={() => setDialogOpen(true)}><Plus className="size-4" />{t('توزيع مورد', 'Allocate Resource', lang)}</Button>
   490→        </div>
   491→      }
   492→    >
   493→      {/* Summary Cards */}
   494→      <div className="grid grid-cols-1 sm:grid-cols-3 gap-3">
   495→        <Card className="bg-teal-50 border-teal-200">
   496→          <CardContent className="p-3 flex items-center gap-3">
   497→            <div className="flex size-10 items-center justify-center rounded-lg bg-teal-100"><Users className="size-5 text-teal-700" /></div>
   498→            <div>
   499→              <p className="text-xs text-teal-600">{t('الموظفون الموزعون', 'Allocated Employees', lang)}</p>
   500→              <p className="text-xl font-bold text-teal-700">{totalEmployees}</p>
   501→            </div>
   502→          </CardContent>
   503→        </Card>
   504→        <Card className="bg-violet-50 border-violet-200">
   505→          <CardContent className="p-3 flex items-center gap-3">
   506→            <div className="flex size-10 items-center justify-center rounded-lg bg-violet-100"><Users2 className="size-5 text-violet-700" /></div>
   507→            <div>
   508→              <p className="text-xs text-violet-600">{t('فرق العمل الموزعة', 'Allocated Teams', lang)}</p>
   509→              <p className="text-xl font-bold text-violet-700">{totalTeams}</p>
   510→            </div>
   511→          </CardContent>
   512→        </Card>
   513→        <Card className="bg-orange-50 border-orange-200">
   514→          <CardContent className="p-3 flex items-center gap-3">
   515→            <div className="flex size-10 items-center justify-center rounded-lg bg-orange-100"><Truck className="size-5 text-orange-700" /></div>
   516→            <div>
   517→              <p className="text-xs text-orange-600">{t('المعدات الموزعة', 'Allocated Equipment', lang)}</p>
   518→              <p className="text-xl font-bold text-orange-700">{totalEquipment}</p>
   519→            </div>
   520→          </CardContent>
   521→        </Card>
   522→      </div>
   523→
   524→      {/* Search & Filter */}
   525→      <Card><CardContent className="p-4">
   526→        <div className="flex flex-col sm:flex-row gap-3">
   527→          <div className="relative flex-1"><Search className="absolute right-3 top-1/2 -translate-y-1/2 size-4 text-muted-foreground" /><Input placeholder={t('بحث بالمشروع أو المورد...', 'Search by project or resource...', lang)} value={search} onChange={e => setSearch(e.target.value)} className="pr-9" /></div>
   528→          <Select value={filterProject || 'ALL'} onValueChange={v => setFilterProject(v === 'ALL' ? '' : v)}>
   529→            <SelectTrigger className="w-full sm:w-[200px]"><SelectValue placeholder={t('كل المشاريع', 'All Projects', lang)} /></SelectTrigger>
   530→            <SelectContent>
   531→              <SelectItem value="ALL">{t('كل المشاريع', 'All Projects', lang)}</SelectItem>
   532→              {projects.map(p => <SelectItem key={p.id} value={p.id}>{p.name}</SelectItem>)}
   533→            </SelectContent>
   534→          </Select>
   535→        </div>
   536→      </CardContent></Card>
   537→
   538→      {/* Visual Grid by Project with Cost Sheet Button */}
   539→      {projectGroups.length > 0 && (
   540→        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
   541→          {projectGroups.map(group => (
   542→            <Card key={group.project.id} className="border-l-4 border-l-emerald-500">
   543→              <CardContent className="p-4">
   544→                <div className="flex items-center justify-between mb-3">
   545→                  <h3 className="font-semibold text-sm">
   546→                    <div className="flex items-center gap-1">
   547→                      {group.project.name}
   548→                      {group.project.projectType && <ProjectTypeBadge projectType={group.project.projectType} lang={lang} />}
   549→                    </div>
   550→                  </h3>
   551→                  <Button
   552→                    variant="outline"
   553→                    size="sm"
   554→                    className="gap-1 text-xs border-emerald-300 text-emerald-700 hover:bg-emerald-50"
   555→                    onClick={() => setSelectedProject(group.project)}
   556→                  >
   557→                    <BarChart3 className="size-3" />
   558→                    {t('كشف التكاليف', 'Cost Sheet', lang)}
   559→                  </Button>
   560→                </div>
   561→                <div className="space-y-2">
   562→                  {group.items.map(item => (
   563→                    <div key={item.id} className="flex items-center justify-between gap-2 p-2 rounded-lg bg-gray-50">
   564→                      <div className="flex items-center gap-2">
   565→                        <ResourceTypeBadge type={item.resourceType} lang={lang} />
   566→                        <span className="text-sm font-medium">{(item.resource as Record<string, unknown>)?.name as string || '—'}</span>
   567→                      </div>
   568→                      <div className="flex items-center gap-2">
   569→                        <span className="text-xs text-muted-foreground">{formatDate(item.startDate, lang)}</span>
   570→                        <Button variant="ghost" size="icon" className="size-6 text-rose-500 hover:text-rose-700" onClick={() => { if (confirm(t('هل أنت متأكد من إلغاء التوزيع؟', 'Are you sure you want to remove this allocation?', lang))) deleteMutation.mutate(item.id) }}><Trash2 className="size-3" /></Button>
   571→                      </div>
   572→                    </div>
   573→                  ))}
   574→                </div>
   575→                {group.project.contractValue > 0 && (
   576→                  <div className="mt-3 pt-2 border-t text-xs text-muted-foreground">
   577→                    {t('قيمة العقد', 'Contract Value', lang)}: <MoneyDisplay value={group.project.contractValue} lang={lang} size="sm" inline />
   578→                  </div>
   579→                )}
   580→              </CardContent>
   581→            </Card>
   582→          ))}
   583→        </div>
   584→      )}
   585→
   586→      {/* Table View */}
   587→      <Card><CardContent className="p-0">
   588→        {isLoading ? (<div className="p-6"><TableSkeleton /></div>) : isError ? (
   589→          <div className="flex flex-col items-center gap-3 py-10"><p className="text-rose-600">{t('حدث خطأ', 'Error', lang)}</p><Button variant="outline" onClick={() => refetch()}>{t('إعادة المحاولة', 'Retry', lang)}</Button></div>
   590→        ) : filtered.length === 0 ? (
   591→          <div className="flex flex-col items-center gap-3 py-10"><LayoutGrid className="size-12 text-gray-300" /><p className="text-muted-foreground">{t('لا يوجد توزيع موارد', 'No resource allocations', lang)}</p>
   592→            <Button className="bg-emerald-600 hover:bg-emerald-700" onClick={() => setDialogOpen(true)}><Plus className="size-4 mr-1" />{t('توزيع مورد', 'Allocate Resource', lang)}</Button></div>
   593→        ) : (
   594→          <div className="overflow-x-auto">
   595→            <Table>
   596→              <TableHeader><TableRow>
   597→                <TableHead className="text-right">{t('المشروع', 'Project', lang)}</TableHead>
   598→                <TableHead className="text-right">{t('نوع المورد', 'Resource Type', lang)}</TableHead>
   599→                <TableHead className="text-right">{t('المورد', 'Resource', lang)}</TableHead>
   600→                <TableHead className="text-right">{t('تاريخ البداية', 'Start Date', lang)}</TableHead>
   601→                <TableHead className="text-right">{t('تاريخ النهاية', 'End Date', lang)}</TableHead>
   602→                <TableHead className="text-right">{t('الإجراءات', 'Actions', lang)}</TableHead>
   603→              </TableRow></TableHeader>
   604→              <TableBody>
   605→                {filtered.map(d => (
   606→                  <TableRow key={d.id}>
   607→                    <TableCell className="font-medium">
   608→                      <div className="flex items-center gap-1">
   609→                        {d.project.name}
   610→                        {d.project.projectType && <ProjectTypeBadge projectType={d.project.projectType} lang={lang} />}
   611→                      </div>
   612→                    </TableCell>
   613→                    <TableCell><ResourceTypeBadge type={d.resourceType} lang={lang} /></TableCell>
   614→                    <TableCell>{(d.resource as Record<string, unknown>)?.name as string || '—'}</TableCell>
   615→                    <TableCell>{formatDate(d.startDate, lang)}</TableCell>
   616→                    <TableCell>{d.endDate ? formatDate(d.endDate, lang) : '—'}</TableCell>
   617→                    <TableCell>
   618→                      <div className="flex items-center gap-1">
   619→                        <Button variant="ghost" size="icon" className="size-8 text-emerald-600 hover:text-emerald-700" onClick={() => setSelectedProject(d.project)} title={t('كشف التكاليف', 'Cost Sheet', lang)}><BarChart3 className="size-4" /></Button>
   620→                        <Button variant="ghost" size="icon" className="size-8 text-rose-600 hover:text-rose-700" onClick={() => { if (confirm(t('هل أنت متأكد من إلغاء التوزيع؟', 'Are you sure you want to remove this allocation?', lang))) deleteMutation.mutate(d.id) }}><Trash2 className="size-4" /></Button>
   621→                      </div>
   622→                    </TableCell>
   623→                  </TableRow>
   624→                ))}
   625→              </TableBody>
   626→            </Table>
   627→          </div>
   628→        )}
   629→      </CardContent></Card>
   630→
   631→      <DistributionFormDialog open={dialogOpen} onOpenChange={setDialogOpen} projects={projects} employees={employees} equipment={equipment} teams={teams} />
   632→    </ModuleLayout>
   633→  )
   634→}

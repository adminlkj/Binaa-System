   351→// ============ Tab 1: Chart of Accounts ============
   352→function ChartOfAccountsTab({ accounts, isLoading, onInitialize, onReInitialize, isInitializing }: {
   353→  accounts: Account[]; isLoading: boolean; onInitialize: () => void; onReInitialize: () => void; isInitializing: boolean
   354→}) {
   355→  const { lang } = useAppStore()
   356→  const [activityFilter, setActivityFilter] = useState<string>('all')
   357→  const [typeFilter, setTypeFilter] = useState<string>('all')
   358→  const [searchTerm, setSearchTerm] = useState('')
   359→  const [expandedIds, setExpandedIds] = useState<Set<string>>(() => {
   360→    const initial = new Set<string>()
   361→    accounts.forEach(a => { if (!a.parentId || a.level === 0) initial.add(a.id) })
   362→    return initial
   363→  })
   364→  const [selectedAccount, setSelectedAccount] = useState<Account | null>(null)
   365→  const [detailOpen, setDetailOpen] = useState(false)
   366→
   367→  const allParentIds = useMemo(() => {
   368→    const ids = new Set<string>()
   369→    accounts.forEach(a => {
   370→      if (a.children && a.children.length > 0) ids.add(a.id)
   371→    })
   372→    return ids
   373→  }, [accounts])
   374→
   375→  const rootAccounts = useMemo(() => {
   376→    return accounts.filter(a => !a.parentId).sort((a, b) => a.code.localeCompare(b.code))
   377→  }, [accounts])
   378→
   379→  const childMap = useMemo(() => {
   380→    const map = new Map<string, Account[]>()
   381→    accounts.forEach(a => {
   382→      if (a.parentId) {
   383→        const siblings = map.get(a.parentId) || []
   384→        siblings.push(a)
   385→        map.set(a.parentId, siblings)
   386→      }
   387→    })
   388→    map.forEach((children) => children.sort((a, b) => a.code.localeCompare(b.code)))
   389→    return map
   390→  }, [accounts])
   391→
   392→  const filteredAccounts = useMemo(() => {
   393→    let filtered = accounts
   394→    if (activityFilter !== 'all') {
   395→      if (activityFilter === 'NONE') {
   396→        filtered = filtered.filter(a => !a.activityType)
   397→      } else if (activityFilter === 'BOTH') {
   398→        filtered = filtered.filter(a => a.activityType === 'BOTH' || !a.activityType)
   399→      } else {
   400→        filtered = filtered.filter(a => a.activityType === activityFilter)
   401→      }
   402→    }
   403→    if (typeFilter !== 'all') {
   404→      filtered = filtered.filter(a => a.type === typeFilter)
   405→    }
   406→    if (searchTerm) {
   407→      const term = searchTerm.toLowerCase()
   408→      filtered = filtered.filter(a =>
   409→        a.code.toLowerCase().includes(term) ||
   410→        a.name.toLowerCase().includes(term) ||
   411→        (a.nameAr && a.nameAr.toLowerCase().includes(term))
   412→      )
   413→    }
   414→    return filtered
   415→  }, [accounts, activityFilter, typeFilter, searchTerm])
   416→
   417→  const flatAccounts = useMemo(() => {
   418→    function flatten(roots: Account[], level: number): (Account & { displayLevel: number })[] {
   419→      const result: (Account & { displayLevel: number })[] = []
   420→      for (const root of roots) {
   421→        if (!filteredAccounts.find(a => a.id === root.id)) continue
   422→        result.push({ ...root, displayLevel: level })
   423→        const children = childMap.get(root.id) || []
   424→        if (children.length > 0 && expandedIds.has(root.id)) {
   425→          result.push(...flatten(children, level + 1))
   426→        }
   427→      }
   428→      return result
   429→    }
   430→    return flatten(rootAccounts, 0)
   431→  }, [rootAccounts, filteredAccounts, childMap, expandedIds])
   432→
   433→  const toggleExpand = useCallback((id: string) => {
   434→    setExpandedIds(prev => {
   435→      const next = new Set(prev)
   436→      if (next.has(id)) next.delete(id)
   437→      else next.add(id)

   271→export async function ensureAccountExists(template: AccountTemplate, tx?: PrismaTransaction) {
   272→  const client = tx || db
   273→  const existing = await client.account.findUnique({ where: { code: template.code } })
   274→  if (existing) {
   275→    // Update existing account with new fields if they are missing
   276→    if (
   277→      existing.activityType !== (template.activityType || null) ||
   278→      existing.isSystem !== (template.isSystem || false) ||
   279→      existing.allowPosting !== (template.allowPosting || false) ||
   280→      existing.level !== (template.level || 0)
   281→    ) {
   282→      await client.account.update({
   283→        where: { code: template.code },
   284→        data: {
   285→          name: template.name,
   286→          nameAr: template.nameAr,
   287→          type: template.type,
   288→          activityType: template.activityType || null,
   289→          isSystem: template.isSystem || false,
   290→          allowPosting: template.allowPosting || false,
   291→          level: template.level || 0,
   292→        },
   293→      })
   294→    }
   295→    return client.account.findUnique({ where: { code: template.code } })
   296→  }
   297→
   298→  let parentId: string | undefined
   299→  if (template.parentId) {
   300→    const parent = await client.account.findUnique({ where: { code: template.parentId } })
   301→    if (parent) parentId = parent.id
   302→  }
   303→
   304→  return client.account.create({
   305→    data: {
   306→      code: template.code,
   307→      name: template.name,
   308→      nameAr: template.nameAr,
   309→      type: template.type,
   310→      parentId,
   311→      isActive: true,
   312→      activityType: template.activityType || null,
   313→      isSystem: template.isSystem || false,
   314→      allowPosting: template.allowPosting || false,
   315→      level: template.level || 0,
   316→    },
   317→  })
   318→}

   322→export async function initializeChartOfAccounts() {
   323→  let created = 0
   324→  let updated = 0
   325→
   326→  // Create parent accounts first, then children (sorted by code length then code)
   327→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   328→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   329→    return a.code.localeCompare(b.code)
   330→  })
   331→
   332→  for (const tmpl of sorted) {
   333→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   334→    if (existing) {
   335→      // Update existing account with new fields
   336→      await db.account.update({
   337→        where: { code: tmpl.code },
   338→        data: {
   339→          name: tmpl.name,
   340→          nameAr: tmpl.nameAr,
   341→          type: tmpl.type,
   342→          activityType: tmpl.activityType || null,
   343→          isSystem: tmpl.isSystem || false,
   344→          allowPosting: tmpl.allowPosting || false,
   345→          level: tmpl.level || 0,
   346→          isActive: true,
   347→        },
   348→      })
   349→      // Update parent reference if needed
   350→      if (tmpl.parentId) {
   351→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   352→        if (parent && existing.parentId !== parent.id) {
   353→          await db.account.update({
   354→            where: { code: tmpl.code },
   355→            data: { parentId: parent.id },
   356→          })
   357→        }
   358→      }
   359→      updated++
   360→    } else {
   361→      await ensureAccountExists(tmpl)
   362→      created++
   363→    }
   364→  }
   365→
   366→  const total = await db.account.count()
   367→  return { created, updated, total }
   368→}

   372→export async function createJournalEntry(template: JournalEntryTemplate, tx?: PrismaTransaction) {
   373→  const client = tx || db
   374→  // Validate: total debits must equal total credits
   375→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   376→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   377→
   378→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   379→    throw new Error(
   380→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   381→    )
   382→  }
   383→
   384→  // Ensure all referenced accounts exist
   385→  const accountMap = new Map<string, { id: string; code: string; nameAr: string }>()
   386→  for (const line of template.lines) {
   387→    const account = await getAccountByCode(line.accountCode, tx)
   388→    if (!account) {
   389→      throw new Error(`Account not found: ${line.accountCode}`)
   390→    }
   391→    accountMap.set(line.accountCode, { id: account.id, code: account.code, nameAr: account.nameAr || '' })
   392→  }
   393→
   394→  const now = new Date()
   395→
   396→  // Create the journal entry with lines and audit log
   397→  const entry = await client.journalEntry.create({
   398→    data: {
   399→      entryNo: template.entryNo,
   400→      date: template.date,
   401→      description: template.description,
   402→      descriptionAr: template.descriptionAr,
   403→      status: 'POSTED', // Auto-entries are posted immediately
   404→      sourceType: template.sourceType,
   405→      sourceId: template.sourceId,
   406→      sourceDocumentNo: template.sourceDocumentNo,
   407→      activityType: template.activityType,
   408→      projectId: template.projectId,
   409→      clientId: template.clientId,
   410→      supplierId: template.supplierId,
   411→      isReversal: template.isReversal || false,
   412→      reversedEntryId: template.reversedEntryId,
   413→      reversalReason: template.reversalReason,
   414→      postedAt: now,
   415→      lines: {
   416→        create: template.lines.map((line) => {
   417→          const acct = accountMap.get(line.accountCode)!
   418→          return {
   419→            accountId: acct.id,
   420→            accountCode: acct.code,
   421→            accountNameAr: acct.nameAr,
   422→            costCenterId: line.costCenterId,
   423→            debit: line.debit,
   424→            credit: line.credit,
   425→            description: line.description,
   426→          };
   427→        }),
   428→      },
   429→      auditLogs: {
   430→        create: [
   431→          {
   432→            action: 'CREATED',
   433→            actionAr: 'إنشاء القيد',
   434→            performedBy: 'System',
   435→            details: `Auto-created for ${template.sourceType}`,
   436→          },
   437→          {
   438→            action: 'POSTED',
   439→            actionAr: 'ترحيل القيد',
   440→            performedBy: 'System',
   441→            details: `Auto-posted at ${now.toISOString()}`,
   442→          },
   443→        ],
   444→      },
   445→    },
   446→    include: { lines: true, auditLogs: true },
   447→  })
   448→
   449→  return entry
   450→}

   455→/**
   456→ * فاتورة مبيعات - Sales Invoice (from Extract)
   457→ * Dr: Clients Receivable (1210) - totalAmount
   458→ * Cr: Progress Claims Revenue (6110) - subtotal
   459→ * Cr: VAT Payable (3200) - vatAmount
   460→ */
   461→export async function autoEntrySalesInvoice(data: {
   462→  invoiceNo: string
   463→  clientId: string
   464→  subtotal: number
   465→  vatRate: number
   466→  vatAmount: number
   467→  totalAmount: number
   468→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   469→  date: Date
   470→  projectId?: string
   471→  costCenterId?: string
   472→}, tx?: PrismaTransaction) {
   473→  // Determine revenue account based on invoice type
   474→  let revenueAccountCode: string
   475→  switch (data.invoiceType) {
   476→    case 'RENTAL':
   477→      revenueAccountCode = '6210' // Equipment Rental Revenue
   478→      break
   479→    case 'PROGRESS_CLAIM':
   480→      revenueAccountCode = '6110' // Progress Claims Revenue
   481→      break
   482→    default:
   483→      revenueAccountCode = '6340' // Service Revenue
   484→  }
   485→
   486→  const lines = [
   487→    { accountCode: '1210', debit: data.totalAmount, credit: 0, costCenterId: data.costCenterId }, // Clients Receivable
   488→    { accountCode: revenueAccountCode, debit: 0, credit: data.subtotal, costCenterId: data.costCenterId }, // Revenue
   489→  ]
   490→
   491→  // Add VAT line only if VAT > 0 (Output VAT)
   492→  if (data.vatAmount > 0) {
   493→    lines.push({ accountCode: '3110', debit: 0, credit: data.vatAmount, costCenterId: data.costCenterId }) // Output VAT (ضريبة مخرجات)
   494→  }
   495→
   496→  return createJournalEntry({
   497→    entryNo: `JE-SI-${Date.now()}`,
   498→    date: data.date,
   499→    description: `Sales Invoice ${data.invoiceNo}`,
   500→    descriptionAr: `فاتورة مبيعات ${data.invoiceNo}`,
   501→    lines,
   502→    sourceType: 'SALES_INVOICE',
   503→    sourceId: data.invoiceNo,
   504→  }, tx)
   505→}

   930→/**
   931→ * رواتب - Salary Entry (Accrual + Payment)
   932→ *
   933→ * Supports two modes via the `accrualOnly` parameter:
   934→ *
   935→ * ACCRUAL MODE (accrualOnly = true):
   936→ *   Used during payroll run to record the expense and liability.
   937→ *   Dr: Salaries & Wages (8110) - grossSalary
   938→ *   Dr: GOSI Expense (8210) - gosiEmployerContribution
   939→ *   Cr: Salaries Payable (3310) - netSalary (gross - gosiEmployeeDeduction)
   940→ *   Cr: GOSI Payable (3830) - total GOSI (employee + employer)
   941→ *
   942→ * PAYMENT MODE (accrualOnly = false, default):
   943→ *   Used when salary is actually paid to clear the payable.
   944→ *   Dr: Salaries Payable (3310) - netSalary
   945→ *   Cr: Cash/Bank (1110/1120) - netCashPaid (netSalary)
   946→ *
   947→ * FULL COMBINED MODE (for backward compatibility, use accrualOnly = undefined):
   948→ *   Creates a single entry that records both expense and payment.
   949→ *   Dr: Salaries & Wages (8110) - grossSalary
   950→ *   Dr: GOSI Expense (8210) - gosiEmployerContribution
   951→ *   Cr: Cash/Bank (1110/1120) - netCashPaid
   952→ *   Cr: GOSI Payable (3830) - total GOSI
   953→ *   Cr: Salaries & Wages (8110) - gosiEmployeeDeduction (contra-entry)
   954→ */
   955→export async function autoEntrySalary(data: {
   956→  employeeName: string
   957→  grossSalary: number
   958→  gosiEmployeeDeduction: number
   959→  gosiEmployerContribution: number
   960→  date: Date
   961→  payFrom: 'TREASURY' | 'BANK'
   962→  costCenterId?: string
   963→  accrualOnly?: boolean // true = accrual entry only, false/undefined = payment entry
   964→}, tx?: PrismaTransaction) {
   965→  const cashAccountCode = data.payFrom === 'BANK' ? '1120' : '1110'
   966→  const netSalary = data.grossSalary - data.gosiEmployeeDeduction
   967→
   968→  // ACCRUAL MODE: Record expense and liability without cash payment
   969→  if (data.accrualOnly === true) {
   970→    const lines = [
   971→      { accountCode: '8110', debit: data.grossSalary, credit: 0, costCenterId: data.costCenterId }, // Salaries & Wages (gross)
   972→      { accountCode: '8210', debit: data.gosiEmployerContribution, credit: 0, costCenterId: data.costCenterId }, // GOSI Expense (employer)
   973→    ]
   974→
   975→    // Credit Salaries Payable for the net amount
   976→    lines.push({ accountCode: '3310', debit: 0, credit: netSalary, costCenterId: data.costCenterId }) // Salaries Payable (رواتب مستحقة)
   977→
   978→    // Credit GOSI Payable for total GOSI (employee deduction + employer contribution)
   979→    const totalGOSI = data.gosiEmployeeDeduction + data.gosiEmployerContribution
   980→    if (totalGOSI > 0) {
   981→      lines.push({ accountCode: '3830', debit: 0, credit: totalGOSI, costCenterId: data.costCenterId }) // GOSI Payable
   982→    }
   983→
   984→    return createJournalEntry({
   985→      entryNo: `JE-SAL-ACC-${Date.now()}`,
   986→      date: data.date,
   987→      description: `Salary accrual - ${data.employeeName}`,
   988→      descriptionAr: `استحقاق راتب - ${data.employeeName}`,
   989→      lines,
   990→      sourceType: 'SALARY_ACCRUAL',
   991→      sourceId: `SAL-ACC-${Date.now()}`,
   992→    }, tx)
   993→  }
   994→
   995→  // PAYMENT MODE: Clear the payable with cash
   996→  // When accrualOnly is explicitly false, this is the payment step
   997→  if (data.accrualOnly === false) {
   998→    return createJournalEntry({
   999→      entryNo: `JE-SAL-PAY-${Date.now()}`,
  1000→      date: data.date,
  1001→      description: `Salary payment - ${data.employeeName}`,
  1002→      descriptionAr: `صرف راتب - ${data.employeeName}`,
  1003→      lines: [
  1004→        { accountCode: '3310', debit: netSalary, credit: 0 }, // Dr: Salaries Payable
  1005→        { accountCode: cashAccountCode, debit: 0, credit: netSalary }, // Cr: Cash/Bank
  1006→      ],
  1007→      sourceType: 'SALARY_PAYMENT',
  1008→      sourceId: `SAL-PAY-${Date.now()}`,
  1009→    }, tx)
  1010→  }
  1011→
  1012→  // FULL COMBINED MODE (backward compatible): Record expense + payment in one entry
  1013→  const lines = [
  1014→    { accountCode: '8110', debit: data.grossSalary, credit: 0, costCenterId: data.costCenterId }, // Salaries & Wages (gross)
  1015→    { accountCode: '8210', debit: data.gosiEmployerContribution, credit: 0, costCenterId: data.costCenterId }, // GOSI Expense (employer)
  1016→  ]
  1017→
  1018→  if (data.gosiEmployeeDeduction > 0) {
  1019→    lines.push({ accountCode: '8110', debit: 0, credit: data.gosiEmployeeDeduction, costCenterId: data.costCenterId }) // Reduce salary by GOSI employee share
  1020→  }
  1021→
  1022→  lines.push({ accountCode: cashAccountCode, debit: 0, credit: netSalary, costCenterId: data.costCenterId }) // Cash/Bank paid
  1023→  lines.push({ accountCode: '3830', debit: 0, credit: data.gosiEmployeeDeduction + data.gosiEmployerContribution, costCenterId: data.costCenterId }) // GOSI Payable (total)
  1024→
  1025→  return createJournalEntry({
  1026→    entryNo: `JE-SAL-${Date.now()}`,
  1027→    date: data.date,
  1028→    description: `Salary payment - ${data.employeeName}`,
  1029→    descriptionAr: `صرف راتب - ${data.employeeName}`,
  1030→    lines,
  1031→    sourceType: 'SALARY',
  1032→    sourceId: `SAL-${Date.now()}`,
  1033→  }, tx)
  1034→}

  1372→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1373→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1374→  if (dateFrom || dateTo) {
  1375→    dateFilter.date = {
  1376→      ...(dateFrom && { gte: dateFrom }),
  1377→      ...(dateTo && { lte: dateTo }),
  1378→    }
  1379→  }
  1380→  const entries = await db.journalEntry.findMany({
  1381→    where: {
  1382→      status: 'POSTED',
  1383→      ...dateFilter,
  1384→    },
  1385→    include: {
  1386→      lines: {
  1387→        include: {
  1388→          account: true,
  1389→        },
  1390→      },
  1391→    },
  1392→  })
  1393→
  1394→  // Aggregate by account
  1395→  const accountBalances = new Map<string, {
  1396→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1397→    totalDebit: number
  1398→    totalCredit: number
  1399→  }>()
  1400→
  1401→  for (const entry of entries) {
  1402→    for (const line of entry.lines) {
  1403→      const key = line.accountId
  1404→      if (!accountBalances.has(key)) {
  1405→        accountBalances.set(key, {
  1406→          account: line.account,
  1407→          totalDebit: 0,
  1408→          totalCredit: 0,
  1409→        })
  1410→      }
  1411→      const bal = accountBalances.get(key)!
  1412→      bal.totalDebit += toNumber(line.debit)
  1413→      bal.totalCredit += toNumber(line.credit)
  1414→    }
  1415→  }
  1416→
  1417→  // Calculate net balance for each account
  1418→  const results = Array.from(accountBalances.values()).map(bal => {
  1419→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1420→    let netDebit = 0
  1421→    let netCredit = 0
  1422→
  1423→    if (normalBalance === 'DEBIT') {
  1424→      const net = bal.totalDebit - bal.totalCredit
  1425→      if (net >= 0) netDebit = net
  1426→      else netCredit = Math.abs(net)
  1427→    } else {
  1428→      const net = bal.totalCredit - bal.totalDebit
  1429→      if (net >= 0) netCredit = net
  1430→      else netDebit = Math.abs(net)
  1431→    }
  1432→
  1433→    return {
  1434→      ...bal,
  1435→      netDebit,
  1436→      netCredit,
  1437→    }
  1438→  })
  1439→
  1440→  // Sort by account code
  1441→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1442→
  1443→  return results
  1444→}

  1525→    return {

  1527→      entryNo: line.journalEntry.entryNo,
  1528→      description: line.journalEntry.description,

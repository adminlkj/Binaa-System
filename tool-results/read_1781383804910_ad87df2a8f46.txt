    90→function PRCreateView({ onBack }: { onBack: () => void }) {
    91→  const { lang } = useAppStore()
    92→  const queryClient = useQueryClient()
    93→  const [form, setForm] = useState<PRFormData>(defaultForm)
    94→
    95→  const { data: projects = [] } = useQuery<Project[]>({
    96→    queryKey: ['projects-list'],
    97→    queryFn: async () => { const res = await fetch('/api/projects/list'); if (!res.ok) return []; return res.json() },
    98→  })
    99→
   100→  const addItem = () => setForm(f => ({ ...f, items: [...f.items, { description: '', quantity: '1', unit: '', notes: '' }] }))
   101→  const removeItem = (idx: number) => setForm(f => ({ ...f, items: f.items.filter((_, i) => i !== idx) }))
   102→  const updateItem = (idx: number, field: string, value: string) => setForm(f => ({ ...f, items: f.items.map((item, i) => i === idx ? { ...item, [field]: value } : item) }))
   103→
   104→  const createMutation = useMutation({
   105→    mutationFn: (data: Record<string, unknown>) => fetch('/api/purchase-requests', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   106→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['purchase-requests'] }); toast.success(t('تم إنشاء طلب الشراء بنجاح', 'Purchase request created successfully', lang)); onBack() },
   107→    onError: () => toast.error(t('فشل في إنشاء طلب الشراء', 'Failed to create purchase request', lang)),
   108→  })
   109→
   110→  const handleSubmit = (e: React.FormEvent) => {
   111→    e.preventDefault()
   112→    createMutation.mutate({
   113→      projectId: form.projectId || null,
   114→      source: form.source,
   115→      date: form.date,
   116→      description: form.description || null,
   117→      requestedBy: form.requestedBy || null,
   118→      items: form.items.map(i => ({ ...i, quantity: parseFloat(i.quantity) || 1 })),
   119→    })
   120→  }
   121→
   122→  return (
   123→    <div className="space-y-6">
   124→      <div className="flex items-center gap-3">
   125→        <Button variant="outline" size="icon" onClick={onBack}><ArrowRight className="size-4" /></Button>

   129→        </div>
   130→      </div>
   131→
   132→      <form onSubmit={handleSubmit} className="space-y-6">
   133→        <Card>
   134→          <CardHeader><CardTitle className="text-lg">{t('معلومات أساسية', 'Basic Information', lang)}</CardTitle></CardHeader>
   135→          <CardContent className="space-y-4">
   136→            <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   137→              <div className="space-y-2">
   138→                <Label>{t('المشروع', 'Project', lang)}</Label>
   139→                <Select value={form.projectId} onValueChange={v => setForm(f => ({ ...f, projectId: v }))}>
   140→                  <SelectTrigger><SelectValue placeholder={t('اختر المشروع', 'Select project', lang)} /></SelectTrigger>
   141→                  <SelectContent>
   142→                    {projects.map(p => <SelectItem key={p.id} value={p.id}>{p.name}</SelectItem>)}
   143→                  </SelectContent>
   144→                </Select>
   145→              </div>
   146→              <div className="space-y-2">
   147→                <Label>{t('المصدر', 'Source', lang)}</Label>
   148→                <Select value={form.source} onValueChange={v => setForm(f => ({ ...f, source: v }))}>
   149→                  <SelectTrigger><SelectValue /></SelectTrigger>
   150→                  <SelectContent>
   151→                    <SelectItem value="PROJECT">{t('مشروع', 'Project', lang)}</SelectItem>
   152→                    <SelectItem value="INVENTORY">{t('مخزون', 'Inventory', lang)}</SelectItem>
   153→                    <SelectItem value="WORKSHOP">{t('ورشة', 'Workshop', lang)}</SelectItem>
   154→                    <SelectItem value="ADMIN">{t('إدارة', 'Admin', lang)}</SelectItem>

   160→                <Input type="date" value={form.date} onChange={e => setForm(f => ({ ...f, date: e.target.value }))} required />
   161→              </div>
   162→              <div className="space-y-2">
   163→                <Label>{t('مقدم الطلب', 'Requested By', lang)}</Label>
   164→                <Input value={form.requestedBy} onChange={e => setForm(f => ({ ...f, requestedBy: e.target.value }))} placeholder={t('اسم مقدم الطلب', 'Requester name', lang)} />
   165→              </div>
   166→            </div>
   167→            <div className="space-y-2">
   168→              <Label>{t('الوصف', 'Description', lang)}</Label>

   170→            </div>

   182→            </div>

   193→                    <Label className="text-xs">{t('الكمية', 'Qty', lang)}</Label>

   224→// ============ Detail View ============
   225→function PRDetailView({ id, onBack, onCreatePO }: { id: string; onBack: () => void; onCreatePO: (prId: string) => void }) {
   226→  const { lang } = useAppStore()
   227→  const queryClient = useQueryClient()
   228→
   229→  const { data: pr, isLoading, isError } = useQuery<PurchaseRequest>({
   230→    queryKey: ['purchase-requests', id],
   231→    queryFn: async () => {
   232→      const res = await fetch(`/api/purchase-requests/${id}`)
   233→      if (!res.ok) throw new Error()
   234→      return res.json()
   235→    },
   236→    enabled: !!id,
   237→  })
   238→
   239→  const approveMutation = useMutation({
   240→    mutationFn: () => fetch(`/api/purchase-requests/${id}`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ status: 'APPROVED' }) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   241→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['purchase-requests', id] }); queryClient.invalidateQueries({ queryKey: ['purchase-requests'] }); toast.success(t('تم اعتماد طلب الشراء', 'Purchase request approved', lang)) },
   242→    onError: () => toast.error(t('فشل في اعتماد الطلب', 'Failed to approve request', lang)),
   243→  })
   244→
   245→  const cancelMutation = useMutation({
   246→    mutationFn: () => fetch(`/api/purchase-requests/${id}`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ status: 'CANCELLED' }) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   247→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['purchase-requests', id] }); queryClient.invalidateQueries({ queryKey: ['purchase-requests'] }); toast.success(t('تم إلغاء الطلب', 'Request cancelled', lang)) },
   248→    onError: () => toast.error(t('فشل في إلغاء الطلب', 'Failed to cancel request', lang)),
   249→  })
   250→
   251→  if (isLoading) return <div className="p-6"><TableSkeleton /></div>
   252→  if (isError || !pr) return (
   253→    <div className="flex flex-col items-center gap-3 py-10">
   254→      <p className="text-rose-600">{t('حدث خطأ أثناء تحميل البيانات', 'Error loading data', lang)}</p>
   255→      <Button variant="outline" onClick={onBack}>{t('رجوع', 'Go Back', lang)}</Button>
   256→    </div>
   257→  )
   258→
   259→  const cfg = statusConfig[pr.status] || statusConfig.NEW
   260→  const srcCfg = sourceConfig[pr.source || 'PROJECT'] || sourceConfig.PROJECT
   261→
   262→  return (
   263→    <div className="space-y-6">
   264→      <div className="flex items-center gap-3">
   265→        <Button variant="outline" size="icon" onClick={onBack}><ArrowRight className="size-4" /></Button>
   266→        <div className="flex-1">
   267→          <div className="flex items-center gap-3">
   268→            <h2 className="text-xl font-bold">{t('طلب شراء', 'Purchase Request', lang)} {pr.requestNo}</h2>
   269→            <Badge className={`${cfg.bg} ${cfg.color} border-0`}>{cfg.label[lang]}</Badge>
   270→          </div>
   271→          <p className="text-sm text-muted-foreground">{pr.project?.name || '—'}</p>
   272→        </div>
   273→        {pr.status === 'NEW' && (
   274→          <Button className="gap-2 bg-emerald-600 hover:bg-emerald-700" onClick={() => approveMutation.mutate()} disabled={approveMutation.isPending}>
   275→            <CheckCircle className="size-4" /> {t('اعتماد', 'Approve', lang)}
   276→          </Button>
   277→        )}
   278→        {pr.status === 'APPROVED' && (
   279→          <Button className="gap-2 bg-blue-600 hover:bg-blue-700" onClick={() => onCreatePO(pr.id)}>
   280→            <ShoppingCart className="size-4" /> {t('إنشاء أمر شراء', 'Create PO', lang)}
   281→          </Button>
   282→        )}
   283→        {(pr.status === 'NEW' || pr.status === 'APPROVED') && (
   284→          <Button variant="outline" className="gap-2 text-rose-600 border-rose-300 hover:bg-rose-50" onClick={() => { if (confirm(t('هل أنت متأكد من الإلغاء؟', 'Are you sure you want to cancel?', lang))) cancelMutation.mutate() }} disabled={cancelMutation.isPending}>
   285→            <XCircle className="size-4" /> {t('إلغاء الطلب', 'Cancel Request', lang)}
   286→          </Button>
   287→        )}
   288→      </div>
   289→
   290→      {/* Info Cards */}
   291→      <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   292→        <Card><CardContent className="p-3">
   293→          <p className="text-xs text-muted-foreground">{t('المشروع', 'Project', lang)}</p>
   294→          <p className="text-sm font-medium truncate">{pr.project?.name || '—'}</p>
   295→        </CardContent></Card>
   296→        <Card><CardContent className="p-3">
   297→          <p className="text-xs text-muted-foreground">{t('المصدر', 'Source', lang)}</p>
   298→          <Badge className={`${srcCfg.bg} ${srcCfg.color} border-0 text-xs`}>{srcCfg.label[lang]}</Badge>
   299→        </CardContent></Card>
   300→        <Card><CardContent className="p-3">
   301→          <p className="text-xs text-muted-foreground">{t('التاريخ', 'Date', lang)}</p>
   302→          <p className="text-sm font-medium">{formatDate(pr.date, lang)}</p>
   303→        </CardContent></Card>
   304→        <Card><CardContent className="p-3">
   305→          <p className="text-xs text-muted-foreground">{t('مقدم الطلب', 'Requested By', lang)}</p>
   306→          <p className="text-sm font-medium">{pr.requestedBy || '—'}</p>
   307→        </CardContent></Card>
   308→      </div>
   309→
   310→      {/* Workflow Status Indicator */}
   311→      <Card className="bg-gray-50 border-dashed">
   312→        <CardContent className="p-4">
   313→          <div className="flex items-center justify-center gap-2 text-sm flex-wrap">
   314→            <Badge className={pr.status === 'NEW' ? 'bg-yellow-100 text-yellow-700 border-0' : 'bg-emerald-100 text-emerald-700 border-0'}>1. {t('طلب شراء', 'PR', lang)}</Badge>
   315→            <span className="text-gray-400">→</span>
   316→            <Badge className={pr.status === 'APPROVED' || pr.status === 'CONVERTED_TO_PO' ? 'bg-emerald-100 text-emerald-700 border-0' : 'bg-gray-100 text-gray-400 border-0'}>2. {t('اعتماد', 'Approved', lang)}</Badge>
   317→            <span className="text-gray-400">→</span>
   318→            <Badge className={pr.status === 'CONVERTED_TO_PO' ? 'bg-blue-100 text-blue-700 border-0' : 'bg-gray-100 text-gray-400 border-0'}>3. {t('أمر شراء', 'PO', lang)}</Badge>
   319→            <span className="text-gray-400">→</span>
   320→            <Badge className="bg-gray-100 text-gray-400 border-0">4. {t('استلام', 'GR', lang)}</Badge>
   321→            <span className="text-gray-400">→</span>
   322→            <Badge className="bg-gray-100 text-gray-400 border-0">5. {t('فاتورة', 'Invoice', lang)}</Badge>
   323→          </div>
   324→        </CardContent>
   325→      </Card>
   326→
   327→      {/* Items Table */}
   328→      <Card>
   329→        <CardHeader>
   330→          <div className="flex items-center gap-2">
   331→            <CardTitle className="text-lg">{t('بنود الطلب', 'Request Items', lang)}</CardTitle>
   332→            <Badge className="bg-emerald-100 text-emerald-700 border-0">{pr.items?.length || 0}</Badge>
   333→          </div>
   334→        </CardHeader>
   335→        <CardContent className="px-0 pb-2">
   336→          <div className="overflow-x-auto">
   337→            <Table>
   338→              <TableHeader><TableRow>
   339→                <TableHead className="text-right">#</TableHead>
   340→                <TableHead className="text-right">{t('الوصف', 'Description', lang)}</TableHead>
   341→                <TableHead className="text-right">{t('الكمية', 'Qty', lang)}</TableHead>
   342→                <TableHead className="text-right">{t('الوحدة', 'Unit', lang)}</TableHead>
   343→                <TableHead className="text-right">{t('ملاحظات', 'Notes', lang)}</TableHead>
   344→              </TableRow></TableHeader>
   345→              <TableBody>
   346→                {pr.items?.map((item, idx) => (
   347→                  <TableRow key={item.id}>
   348→                    <TableCell>{idx + 1}</TableCell>
   349→                    <TableCell>{item.description}</TableCell>
   350→                    <TableCell>{formatNumber(item.quantity)}</TableCell>
   351→                    <TableCell>{item.unit || '—'}</TableCell>
   352→                    <TableCell className="text-muted-foreground">{item.notes || '—'}</TableCell>
   353→                  </TableRow>
   354→                ))}
   355→              </TableBody>
   356→            </Table>
   357→          </div>
   358→        </CardContent>
   359→      </Card>
   360→
   361→      {/* Linked POs */}
   362→      {pr.purchaseOrders && pr.purchaseOrders.length > 0 && (
   363→        <Card>
   364→          <CardHeader>
   365→            <div className="flex items-center gap-2">
   366→              <CardTitle className="text-lg">{t('أوامر الشراء المرتبطة', 'Linked Purchase Orders', lang)}</CardTitle>
   367→              <Badge className="bg-blue-100 text-blue-700 border-0">{pr.purchaseOrders.length}</Badge>
   368→            </div>
   369→          </CardHeader>
   370→          <CardContent>
   371→            <div className="flex flex-wrap gap-2">
   372→              {pr.purchaseOrders.map(po => (
   373→                <Badge key={po.id} className="bg-blue-100 text-blue-700 border-0 gap-1">

   116→   116→function numberToArabicWords(amount: number): string {
   117→   117→  if (amount === 0) return 'صفر ريالاً سعودياً فقط لا غير'
   118→   118→  const riyals = Math.floor(amount)
   119→   119→  const halalas = Math.round((amount - riyals) * 100)
   120→   120→  const ones = ['', 'واحد', 'اثنان', 'ثلاثة', 'أربعة', 'خمسة', 'ستة', 'سبعة', 'ثمانية', 'تسعة', 'عشرة', 'أحد عشر', 'اثنا عشر', 'ثلاثة عشر', 'أربعة عشر', 'خمسة عشر', 'ستة عشر', 'سبعة عشر', 'ثمانية عشر', 'تسعة عشر']
   121→   121→  const tens = ['', '', 'عشرون', 'ثلاثون', 'أربعون', 'خمسون', 'ستون', 'سبعون', 'ثمانون', 'تسعون']
   122→   122→  const hundreds = ['', 'مائة', 'مئتان', 'ثلاثمائة', 'أربعمائة', 'خمسمائة', 'ستمائة', 'سبعمائة', 'ثمانمائة', 'تسعمائة']
   123→   123→  function below1000(n: number): string {
   124→   124→    if (n === 0) return ''
   125→   125→    if (n < 20) return ones[n]
   126→   126→    if (n < 100) { const t = Math.floor(n / 10); const o = n % 10; return o === 0 ? tens[t] : ones[o] + ' و' + tens[t] }
   127→   127→    const h = Math.floor(n / 100); const r = n % 100; return r === 0 ? hundreds[h] : hundreds[h] + ' و' + below1000(r)
   128→   128→  }
   129→   129→  function convert(n: number): string {
   130→   130→    if (n === 0) return ''
   131→   131→    if (n < 1000) return below1000(n)
   132→   132→    if (n < 1000000) { const th = Math.floor(n / 1000); const r = n % 1000; let w = th === 1 ? 'ألف' : th === 2 ? 'ألفان' : th <= 10 ? below1000(th) + ' آلاف' : below1000(th) + ' ألفاً'; return r === 0 ? w : w + ' و' + below1000(r) }
   133→   133→    if (n < 1000000000) { const m = Math.floor(n / 1000000); const r = n % 1000000; let w = m === 1 ? 'مليون' : m === 2 ? 'مليونان' : m <= 10 ? below1000(m) + ' ملايين' : below1000(m) + ' مليوناً'; return r === 0 ? w : w + ' و' + convert(r) }
   134→   134→    const b = Math.floor(n / 1000000000); const r = n % 1000000000; let w = b === 1 ? 'مليار' : b === 2 ? 'ملياران' : below1000(b) + ' ملياراً'; return r === 0 ? w : w + ' و' + convert(r)
   135→   135→  }
   136→   136→  let result = ''
   137→   137→  if (riyals > 0) result = convert(riyals) + ' ريالاً سعودياً'
   138→   138→  if (halalas > 0) { if (riyals > 0) result += ' و'; result += convert(halalas) + ' هللة' }
   139→   139→  return result + ' فقط لا غير'
   140→   140→}

   142→   142→function numberToEnglishWords(amount: number): string {
   143→   143→  if (amount === 0) return 'Zero Saudi Riyals only'
   144→   144→  const riyals = Math.floor(amount)
   145→   145→  const halalas = Math.round((amount - riyals) * 100)
   146→   146→  const ones = ['', 'One', 'Two', 'Three', 'Four', 'Five', 'Six', 'Seven', 'Eight', 'Nine', 'Ten', 'Eleven', 'Twelve', 'Thirteen', 'Fourteen', 'Fifteen', 'Sixteen', 'Seventeen', 'Eighteen', 'Nineteen']
   147→   147→  const tens = ['', '', 'Twenty', 'Thirty', 'Forty', 'Fifty', 'Sixty', 'Seventy', 'Eighty', 'Ninety']
   148→   148→  function below1000(n: number): string {
   149→   149→    if (n === 0) return ''
   150→   150→    if (n < 20) return ones[n]
   151→   151→    if (n < 100) { const t = Math.floor(n / 10); const o = n % 10; return o === 0 ? tens[t] : tens[t] + '-' + ones[o] }
   152→   152→    const h = Math.floor(n / 100); const r = n % 100; return r === 0 ? ones[h] + ' Hundred' : ones[h] + ' Hundred and ' + below1000(r)
   153→   153→  }
   154→   154→  function convert(n: number): string {
   155→   155→    if (n === 0) return ''
   156→   156→    if (n < 1000) return below1000(n)
   157→   157→    if (n < 1000000) { const th = Math.floor(n / 1000); const r = n % 1000; return r === 0 ? below1000(th) + ' Thousand' : below1000(th) + ' Thousand ' + below1000(r) }
   158→   158→    if (n < 1000000000) { const m = Math.floor(n / 1000000); const r = n % 1000000; return r === 0 ? below1000(m) + ' Million' : below1000(m) + ' Million ' + convert(r) }
   159→   159→    const b = Math.floor(n / 1000000000); const r = n % 1000000000; return r === 0 ? below1000(b) + ' Billion' : below1000(b) + ' Billion ' + convert(r)
   160→   160→  }
   161→   161→  let result = ''
   162→   162→  if (riyals > 0) result = convert(riyals) + ' Saudi Riyals'
   163→   163→  if (halalas > 0) { if (riyals > 0) result += ' and '; result += convert(halalas) + ' Halalas' }
   164→   164→  return result + ' only'
   165→   165→}

   168→   168→  return lang === 'ar' ? numberToArabicWords(amount) : numberToEnglishWords(amount)

   942→   942→// ============ ZATCA TLV Encoding (Inline for print service) ============
   943→   943→function encodeZATCATLV(sellerName: string, vatNumber: string, date: string, total: string, vatTotal: string): string {
   944→   944→  // TLV encoding: Tag (1 byte) + Length (1 byte) + Value
   945→   945→  const encodeTag = (tag: number, value: string): Buffer => {
   946→   946→    const buf = Buffer.from(value, 'utf8')
   947→   947→    return Buffer.concat([Buffer.from([tag, buf.length]), buf])
   948→   948→  }
   949→   949→  const tlv = Buffer.concat([
   950→   950→    encodeTag(0x01, sellerName),
   951→   951→    encodeTag(0x02, vatNumber),
   952→   952→    encodeTag(0x03, date),
   953→   953→    encodeTag(0x04, total),
   954→   954→    encodeTag(0x05, vatTotal),
   955→   955→  ])
   956→   956→  return tlv.toString('base64')
   957→   957→}

   960→   960→type FontSizePreset = { base: number; h1: number; h2: number; small: number; xsmall: number; table: number; tableHead: number }
   961→   961→function getFontSizePreset(size: string): FontSizePreset {
   962→   962→  switch (size) {
   963→   963→    case 'small':  return { base: 9, h1: 12, h2: 10, small: 7.5, xsmall: 6.5, table: 7.5, tableHead: 7 }
   964→   964→    case 'large':  return { base: 12, h1: 16, h2: 13, small: 9.5, xsmall: 8.5, table: 10, tableHead: 9 }
   965→   965→    default:       return { base: 10, h1: 14, h2: 11, small: 8.5, xsmall: 7.5, table: 8.5, tableHead: 7.5 } // medium
   966→   966→  }
   967→   967→}

   969→   969→function hexToRgba(hex: string, alpha: number): string {
   970→   970→  const r = parseInt(hex.slice(1, 3), 16)
   971→   971→  const g = parseInt(hex.slice(3, 5), 16)
   972→   972→  const b = parseInt(hex.slice(5, 7), 16)
   973→   973→  return `rgba(${r},${g},${b},${alpha})`
   974→   974→}
   975→   975→
   976→   976→function lightenHex(hex: string, factor: number): string {
   977→   977→  let r = parseInt(hex.slice(1, 3), 16)
   978→   978→  let g = parseInt(hex.slice(3, 5), 16)
   979→   979→  let b = parseInt(hex.slice(5, 7), 16)
   980→   980→  r = Math.min(255, Math.round(r + (255 - r) * factor))
   981→   981→  g = Math.min(255, Math.round(g + (255 - g) * factor))
   982→   982→  b = Math.min(255, Math.round(b + (255 - b) * factor))
   983→   983→  return `#${r.toString(16).padStart(2, '0')}${g.toString(16).padStart(2, '0')}${b.toString(16).padStart(2, '0')}`
   984→   984→}

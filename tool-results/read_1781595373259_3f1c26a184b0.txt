   264→export async function ensureAccountExists(template: AccountTemplate, tx?: PrismaTransaction) {
   265→  const client = tx || db
   266→  const existing = await client.account.findUnique({ where: { code: template.code } })
   267→  if (existing) {
   268→    // Update existing account with new fields if they are missing
   269→    if (
   270→      existing.activityType !== (template.activityType || null) ||
   271→      existing.isSystem !== (template.isSystem || false) ||
   272→      existing.allowPosting !== (template.allowPosting || false) ||
   273→      existing.level !== (template.level || 0) ||
   274→      (template.accountRole && existing.accountRole !== template.accountRole) ||
   275→      (template.parentId && existing.parentCode !== template.parentId)
   276→    ) {
   277→      await client.account.update({
   278→        where: { code: template.code },
   279→        data: {
   280→          name: template.name,
   281→          nameAr: template.nameAr,
   282→          type: template.type,
   283→          activityType: template.activityType || null,
   284→          isSystem: template.isSystem || false,
   285→          allowPosting: template.allowPosting || false,
   286→          level: template.level || 0,
   287→          accountRole: template.accountRole || existing.accountRole,
   288→          parentCode: template.parentId || existing.parentCode,
   289→        },
   290→      })
   291→    }
   292→    return client.account.findUnique({ where: { code: template.code } })
   293→  }
   294→
   295→  let parentId: string | undefined
   296→  if (template.parentId) {
   297→    const parent = await client.account.findUnique({ where: { code: template.parentId } })
   298→    if (parent) parentId = parent.id
   299→  }
   300→
   301→  return client.account.create({
   302→    data: {
   303→      code: template.code,
   304→      name: template.name,
   305→      nameAr: template.nameAr,
   306→      type: template.type,
   307→      parentId,
   308→      parentCode: template.parentId || null,
   309→      isActive: true,
   310→      activityType: template.activityType || null,
   311→      accountRole: template.accountRole || null,
   312→      isSystem: template.isSystem || false,
   313→      allowPosting: template.allowPosting || false,
   314→      level: template.level || 0,
   315→    },
   316→  })
   317→}

   321→export async function initializeChartOfAccounts() {
   322→  let created = 0
   323→  let updated = 0
   324→
   325→  // Create parent accounts first, then children (sorted by code length then code)
   326→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   327→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   328→    return a.code.localeCompare(b.code)
   329→  })
   330→
   331→  for (const tmpl of sorted) {
   332→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   333→    if (existing) {
   334→      // Update existing account with new fields
   335→      await db.account.update({
   336→        where: { code: tmpl.code },
   337→        data: {
   338→          name: tmpl.name,
   339→          nameAr: tmpl.nameAr,
   340→          type: tmpl.type,
   341→          activityType: tmpl.activityType || null,
   342→          accountRole: tmpl.accountRole || null,
   343→          parentCode: tmpl.parentId || null,
   344→          isSystem: tmpl.isSystem || false,
   345→          allowPosting: tmpl.allowPosting || false,
   346→          level: tmpl.level || 0,
   347→          isActive: true,
   348→        },
   349→      })
   350→      // Update parent reference if needed
   351→      if (tmpl.parentId) {
   352→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   353→        if (parent && existing.parentId !== parent.id) {
   354→          await db.account.update({
   355→            where: { code: tmpl.code },
   356→            data: { parentId: parent.id },
   357→          })
   358→        }
   359→      }
   360→      updated++
   361→    } else {
   362→      await ensureAccountExists(tmpl)
   363→      created++
   364→    }
   365→  }
   366→
   367→  const total = await db.account.count()
   368→  return { created, updated, total }
   369→}

   373→/**
   374→ * عكس قيد محاسبي - Reverse a posted journal entry
   375→ *
   376→ * Creates a reversal entry with flipped debit/credit on all lines,
   377→ * marks the original entry as CANCELLED, and links them together.
   378→ *
   379→ * IMPORTANT: This function MUST be called inside a $transaction callback.
   380→ * The `tx` parameter is required — no standalone calls allowed.
   381→ *
   382→ * @param journalEntryId - ID of the journal entry to reverse
   383→ * @param tx - Prisma transaction client (required)
   384→ * @returns The reversal journal entry with its lines
   385→ * @throws Error if entry not found, not POSTED, or already reversed
   386→ */
   387→export async function reverseEntry(journalEntryId: string, tx: PrismaTransaction) {
   388→  // 1. Find the original entry with its lines
   389→  const originalEntry = await tx.journalEntry.findUnique({
   390→    where: { id: journalEntryId },
   391→    include: { lines: true },
   392→  })
   393→
   394→  if (!originalEntry) {
   395→    throw new Error(`القيد المحاسبي غير موجود: ${journalEntryId}`)
   396→  }
   397→
   398→  // 2. Check that the entry is POSTED (not DRAFT or CANCELLED)
   399→  if (originalEntry.status !== 'POSTED') {
   400→    throw new Error(
   401→      `لا يمكن عكس قيد بحالة ${originalEntry.status} - يجب أن يكون القيد مرحّلاً (POSTED)`
   402→    )
   403→  }
   404→
   405→  // 3. Check that it hasn't already been reversed
   406→  if (originalEntry.isReversal) {
   407→    throw new Error('لا يمكن عكس قيد عكسي - القيود العكسية لا يمكن عكسها')
   408→  }
   409→
   410→  const existingReversal = await tx.journalEntry.findFirst({
   411→    where: { reversedEntryId: journalEntryId },
   412→  })
   413→  if (existingReversal) {
   414→    throw new Error('القيد محسوب عكسه مسبقاً - يوجد قيد عكسي مرتبط بهذا القيد')
   415→  }
   416→
   417→  // 4. Generate next entry number
   418→  const last = await tx.journalEntry.findFirst({ orderBy: { entryNo: 'desc' } })
   419→  const next = last ? parseInt(last.entryNo.replace('JE-', '')) + 1 : 1
   420→  const entryNo = `JE-${next.toString().padStart(6, '0')}`
   421→
   422→  // 5. Build reversal description with "عكس" prefix
   423→  const reversalDescription = originalEntry.description
   424→    ? `عكس - ${originalEntry.description}`
   425→    : 'عكس قيد محاسبي'
   426→
   427→  // 6. Create the reversal entry with flipped debit/credit lines
   428→  const reversalEntry = await tx.journalEntry.create({
   429→    data: {
   430→      entryNo,
   431→      date: originalEntry.date,
   432→      description: reversalDescription,
   433→      status: 'POSTED',
   434→      sourceType: originalEntry.sourceType,
   435→      sourceId: originalEntry.sourceId,
   436→      isReversal: true,
   437→      reversedEntryId: originalEntry.id,
   438→      isSystem: true,
   439→      lines: {
   440→        create: originalEntry.lines.map((line) => ({
   441→          accountId: line.accountId,
   442→          costCenterId: line.costCenterId,
   443→          // Flip debit and credit
   444→          debit: line.credit,
   445→          credit: line.debit,
   446→          description: line.description
   447→            ? `عكس - ${line.description}`
   448→            : 'عكس بند قيد',
   449→        })),
   450→      },
   451→    },
   452→    include: { lines: true },
   453→  })
   454→
   455→  // 7. Update the original entry status to CANCELLED
   456→  await tx.journalEntry.update({
   457→    where: { id: originalEntry.id },
   458→    data: { status: 'CANCELLED' },
   459→  })
   460→
   461→  return reversalEntry
   462→}

   464→// ============ JOURNAL ENTRY CREATION ============
   465→
   466→export async function createJournalEntry(template: JournalEntryTemplate, tx?: PrismaTransaction) {
   467→  const client = tx || db
   468→  // Validate: total debits must equal total credits
   469→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   470→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   471→
   472→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   473→    throw new Error(
   474→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   475→    )
   476→  }
   477→
   478→  // Ensure all referenced accounts exist
   479→  for (const line of template.lines) {
   480→    const account = await getAccountByCode(line.accountCode, tx)
   481→    if (!account) {
   482→      throw new Error(`Account not found: ${line.accountCode}`)
   483→    }
   484→  }
   485→
   486→  // Create the journal entry with lines
   487→  const entry = await client.journalEntry.create({
   488→    data: {
   489→      entryNo: template.entryNo,
   490→      date: template.date,
   491→      description: template.description,
   492→      status: 'POSTED', // Auto-entries are posted immediately
   493→      lines: {
   494→        create: await Promise.all(
   495→          template.lines.map(async (line) => {
   496→            const account = await getAccountByCode(line.accountCode, tx)
   497→            return {
   498→              accountId: account!.id,
   499→              costCenterId: line.costCenterId,
   500→              debit: line.debit,
   501→              credit: line.credit,
   502→              description: line.description,
   503→            };
   504→          })
   505→        ),
   506→      },
   507→    },
   508→    include: { lines: true },
   509→  })
   510→
   511→  return entry
   512→}

   517→/**
   518→ * فاتورة مبيعات - Sales Invoice (from Extract)
   519→ * Dr: Clients Receivable (1210) - totalAmount
   520→ * Cr: Progress Claims Revenue (6110) - subtotal
   521→ * Cr: VAT Payable (3200) - vatAmount
   522→ */
   523→export async function autoEntrySalesInvoice(data: {
   524→  invoiceNo: string
   525→  clientId: string
   526→  subtotal: number
   527→  vatRate: number
   528→  vatAmount: number
   529→  totalAmount: number
   530→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   531→  date: Date
   532→  projectId?: string
   533→  costCenterId?: string
   534→}, tx?: PrismaTransaction) {
   535→  // Determine revenue account based on invoice type
   536→  let revenueAccountCode: string
   537→  switch (data.invoiceType) {
   538→    case 'RENTAL':
   539→      revenueAccountCode = '6210' // Equipment Rental Revenue

  1247→ * Dr: Cash/Bank (1110/1120) - sale price
  1248→ * Cr: Asset account (2110/2120/2130/2140) - original cost (or net book value)
  1249→ * Dr/Cr: Gain/Loss on disposal (6310 gain / 8610 loss)
  1250→ */
  1251→export async function autoEntryAssetDisposal(data: {
  1252→  assetAccountCode: string // e.g., '2110', '2120', '2130', '2140'
  1253→  accumulatedDepAccountCode: string // e.g., '2210', '2220', '2230', '2240'
  1254→  originalCost: number
  1255→  accumulatedDepreciation: number
  1256→  salePrice: number
  1257→  date: Date
  1258→  receivedIn: 'TREASURY' | 'BANK'
  1259→}, tx?: PrismaTransaction) {

  1265→    { accountCode: cashAccountCode, debit: data.salePrice, credit: 0 }, // Cash received
  1266→    { accountCode: data.accumulatedDepAccountCode, debit: data.accumulatedDepreciation, credit: 0 }, // Remove accumulated depreciation

  1285→  }, tx)

  1288→/**
  1289→ * إقرار ضريبي - VAT Declaration
  1290→ * Dr: Output VAT (3110) - total output VAT
  1291→ * Cr: Input VAT (3120) - total input VAT
  1292→ * Cr: VAT Due (3130) - net VAT payable (if output > input)
  1293→ * OR
  1294→ * Dr: Input VAT (3120) - if input > output (refund scenario)
  1295→ * Cr: Output VAT (3110)
  1296→ * Dr: VAT Refund Receivable (1410) - if refund due
  1297→ */
  1298→export async function autoEntryVATDeclaration(data: {
  1299→  period: string
  1300→  outputVat: number
  1301→  inputVat: number
  1302→  netVat: number // positive = payable, negative = refundable
  1303→  date: Date
  1304→}, tx?: PrismaTransaction) {

  1309→
  1310→  // Close Input VAT (3120) - credit to zero it out

  1316→    lines.push({ accountCode: '3130', debit: 0, credit: data.netVat }) // VAT Due

  1318→    // More input than output → refund due
  1319→    lines.push({ accountCode: '1410', debit: Math.abs(data.netVat), credit: 0 }) // VAT Refund Receivable
  1320→  }
  1321→
  1322→  return createJournalEntry({
  1323→    entryNo: `JE-VAT-${Date.now()}`,
  1324→    date: data.date,
  1325→    description: `VAT Declaration - ${data.period}`,
  1326→    descriptionAr: `إقرار ضريبي - ${data.period}`,
  1327→    lines,
  1328→    sourceType: 'VAT_DECLARATION',
  1329→    sourceId: `VAT-${data.period}`,
  1330→  }, tx)
  1331→}

  1360→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1361→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1362→  if (dateFrom || dateTo) {
  1363→    dateFilter.date = {
  1364→      ...(dateFrom && { gte: dateFrom }),
  1365→      ...(dateTo && { lte: dateTo }),
  1366→    }
  1367→  }
  1368→  const entries = await db.journalEntry.findMany({
  1369→    where: {
  1370→      status: 'POSTED',
  1371→      ...dateFilter,
  1372→    },
  1373→    include: {
  1374→      lines: {
  1375→        include: {
  1376→          account: true,
  1377→        },
  1378→      },
  1379→    },
  1380→  })
  1381→
  1382→  // Aggregate by account
  1383→  const accountBalances = new Map<string, {
  1384→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1385→    totalDebit: number
  1386→    totalCredit: number
  1387→  }>()
  1388→
  1389→  for (const entry of entries) {
  1390→    for (const line of entry.lines) {
  1391→      const key = line.accountId
  1392→      if (!accountBalances.has(key)) {
  1393→        accountBalances.set(key, {
  1394→          account: line.account,
  1395→          totalDebit: 0,
  1396→          totalCredit: 0,
  1397→        })
  1398→      }
  1399→      const bal = accountBalances.get(key)!
  1400→      bal.totalDebit += toNumber(line.debit)
  1401→      bal.totalCredit += toNumber(line.credit)
  1402→    }
  1403→  }
  1404→
  1405→  // Calculate net balance for each account
  1406→  const results = Array.from(accountBalances.values()).map(bal => {
  1407→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1408→    let netDebit = 0
  1409→    let netCredit = 0
  1410→
  1411→    if (normalBalance === 'DEBIT') {
  1412→      const net = bal.totalDebit - bal.totalCredit
  1413→      if (net >= 0) netDebit = net
  1414→      else netCredit = Math.abs(net)
  1415→    } else {
  1416→      const net = bal.totalCredit - bal.totalDebit
  1417→      if (net >= 0) netCredit = net
  1418→      else netDebit = Math.abs(net)
  1419→    }
  1420→
  1421→    return {
  1422→      ...bal,
  1423→      netDebit,
  1424→      netCredit,
  1425→    }
  1426→  })
  1427→
  1428→  // Sort by account code
  1429→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1430→
  1431→  return results
  1432→}

  1478→export async function getGeneralLedger(accountCode: string, dateFrom?: Date, dateTo?: Date) {
  1479→  const account = await getAccountByCode(accountCode)

  1483→  if (dateFrom || dateTo) {

  1485→      ...(dateFrom && { gte: dateFrom }),
  1486→      ...(dateTo && { lte: dateTo }),
  1487→    }
  1488→  }
  1489→  const lines = await db.journalLine.findMany({
  1490→    where: {
  1491→      accountId: account.id,
  1492→      journalEntry: {
  1493→        status: 'POSTED',
  1494→        ...glDateFilter,
  1495→      },
  1496→    },
  1497→    include: { journalEntry: true },
  1498→    orderBy: { journalEntry: { date: 'asc' } },
  1499→  })
  1500→
  1501→  let runningBalance = 0
  1502→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1503→
  1504→  return lines.map(line => {
  1505→    const debit = toNumber(line.debit)
  1506→    const credit = toNumber(line.credit)
  1507→    if (normalBalance === 'DEBIT') {
  1508→      runningBalance += debit - credit
  1509→    } else {
  1510→      runningBalance += credit - debit
  1511→    }
  1512→
  1513→    return {
  1514→      date: line.journalEntry.date,
  1515→      entryNo: line.journalEntry.entryNo,
  1516→      description: line.journalEntry.description,
  1517→      debit,
  1518→      credit,
  1519→      balance: runningBalance,
  1520→    }
  1521→  })
  1522→}

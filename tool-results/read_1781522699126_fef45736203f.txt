   263→export async function ensureAccountExists(template: AccountTemplate, tx?: PrismaTransaction) {
   264→  const client = tx || db
   265→  const existing = await client.account.findUnique({ where: { code: template.code } })
   266→  if (existing) {
   267→    // Update existing account with new fields if they are missing
   268→    if (
   269→      existing.activityType !== (template.activityType || null) ||
   270→      existing.isSystem !== (template.isSystem || false) ||
   271→      existing.allowPosting !== (template.allowPosting || false) ||
   272→      existing.level !== (template.level || 0)
   273→    ) {
   274→      await client.account.update({
   275→        where: { code: template.code },
   276→        data: {
   277→          name: template.name,
   278→          nameAr: template.nameAr,
   279→          type: template.type,
   280→          activityType: template.activityType || null,
   281→          isSystem: template.isSystem || false,
   282→          allowPosting: template.allowPosting || false,
   283→          level: template.level || 0,
   284→        },
   285→      })
   286→    }
   287→    return client.account.findUnique({ where: { code: template.code } })
   288→  }
   289→
   290→  let parentId: string | undefined
   291→  if (template.parentId) {
   292→    const parent = await client.account.findUnique({ where: { code: template.parentId } })
   293→    if (parent) parentId = parent.id
   294→  }
   295→
   296→  return client.account.create({
   297→    data: {
   298→      code: template.code,
   299→      name: template.name,
   300→      nameAr: template.nameAr,
   301→      type: template.type,
   302→      parentId,
   303→      isActive: true,
   304→      activityType: template.activityType || null,
   305→      isSystem: template.isSystem || false,
   306→      allowPosting: template.allowPosting || false,
   307→      level: template.level || 0,
   308→    },
   309→  })
   310→}

   312→// ============ INITIALIZE CHART OF ACCOUNTS ============
   313→
   314→export async function initializeChartOfAccounts() {
   315→  let created = 0
   316→  let updated = 0
   317→
   318→  // Create parent accounts first, then children (sorted by code length then code)
   319→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   320→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   321→    return a.code.localeCompare(b.code)
   322→  })
   323→
   324→  for (const tmpl of sorted) {
   325→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   326→    if (existing) {
   327→      // Update existing account with new fields
   328→      await db.account.update({
   329→        where: { code: tmpl.code },
   330→        data: {
   331→          name: tmpl.name,
   332→          nameAr: tmpl.nameAr,
   333→          type: tmpl.type,
   334→          activityType: tmpl.activityType || null,
   335→          isSystem: tmpl.isSystem || false,
   336→          allowPosting: tmpl.allowPosting || false,
   337→          level: tmpl.level || 0,
   338→          isActive: true,
   339→        },
   340→      })
   341→      // Update parent reference if needed
   342→      if (tmpl.parentId) {
   343→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   344→        if (parent && existing.parentId !== parent.id) {
   345→          await db.account.update({
   346→            where: { code: tmpl.code },
   347→            data: { parentId: parent.id },
   348→          })
   349→        }
   350→      }
   351→      updated++
   352→    } else {
   353→      await ensureAccountExists(tmpl)
   354→      created++
   355→    }
   356→  }
   357→
   358→  const total = await db.account.count()
   359→  return { created, updated, total }
   360→}

   364→export async function createJournalEntry(template: JournalEntryTemplate, tx?: PrismaTransaction) {
   365→  const client = tx || db
   366→  // Validate: total debits must equal total credits
   367→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   368→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   369→
   370→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   371→    throw new Error(
   372→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   373→    )
   374→  }
   375→
   376→  // Ensure all referenced accounts exist
   377→  for (const line of template.lines) {
   378→    const account = await getAccountByCode(line.accountCode, tx)
   379→    if (!account) {
   380→      throw new Error(`Account not found: ${line.accountCode}`)
   381→    }
   382→  }
   383→
   384→  // Create the journal entry with lines
   385→  const entry = await client.journalEntry.create({
   386→    data: {
   387→      entryNo: template.entryNo,
   388→      date: template.date,
   389→      description: template.description,
   390→      status: 'POSTED', // Auto-entries are posted immediately
   391→      lines: {
   392→        create: await Promise.all(
   393→          template.lines.map(async (line) => {
   394→            const account = await getAccountByCode(line.accountCode, tx)
   395→            return {
   396→              accountId: account!.id,
   397→              costCenterId: line.costCenterId,
   398→              debit: line.debit,
   399→              credit: line.credit,
   400→              description: line.description,
   401→            };
   402→          })
   403→        ),
   404→      },
   405→    },
   406→    include: { lines: true },
   407→  })
   408→
   409→  return entry
   410→}

   415→/**
   416→ * فاتورة مبيعات - Sales Invoice (from Extract)
   417→ * Dr: Clients Receivable (1210) - totalAmount
   418→ * Cr: Progress Claims Revenue (6110) - subtotal
   419→ * Cr: VAT Payable (3200) - vatAmount
   420→ */
   421→export async function autoEntrySalesInvoice(data: {
   422→  invoiceNo: string
   423→  clientId: string
   424→  subtotal: number
   425→  vatRate: number
   426→  vatAmount: number
   427→  totalAmount: number
   428→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   429→  date: Date
   430→  projectId?: string
   431→  costCenterId?: string
   432→}, tx?: PrismaTransaction) {
   433→  // Determine revenue account based on invoice type
   434→  let revenueAccountCode: string
   435→  switch (data.invoiceType) {
   436→    case 'RENTAL':
   437→      revenueAccountCode = '6210' // Equipment Rental Revenue
   438→      break
   439→    case 'PROGRESS_CLAIM':
   440→      revenueAccountCode = '6110' // Progress Claims Revenue
   441→      break
   442→    default:
   443→      revenueAccountCode = '6340' // Service Revenue
   444→  }
   445→
   446→  const lines = [
   447→    { accountCode: '1210', debit: data.totalAmount, credit: 0, costCenterId: data.costCenterId }, // Clients Receivable
   448→    { accountCode: revenueAccountCode, debit: 0, credit: data.subtotal, costCenterId: data.costCenterId }, // Revenue
   449→  ]
   450→
   451→  // Add VAT line only if VAT > 0 (Output VAT)
   452→  if (data.vatAmount > 0) {
   453→    lines.push({ accountCode: '3110', debit: 0, credit: data.vatAmount, costCenterId: data.costCenterId }) // Output VAT (ضريبة مخرجات)
   454→  }
   455→
   456→  return createJournalEntry({
   457→    entryNo: `JE-SI-${Date.now()}`,
   458→    date: data.date,
   459→    description: `Sales Invoice ${data.invoiceNo}`,
   460→    descriptionAr: `فاتورة مبيعات ${data.invoiceNo}`,
   461→    lines,
   462→    sourceType: 'SALES_INVOICE',
   463→    sourceId: data.invoiceNo,
   464→  }, tx)
   465→}

   570→/**
   571→ * مصروف - Expense
   572→ * Dr: Expense account - amount
   573→ * Dr: Input VAT (3120) - vatAmount (if applicable)
   574→ * Cr: Cash (1110/1120/1130) - total (when payLater = false)
   575→ * Cr: Suppliers Payable (3210) - total (when payLater = true)
   576→ *
   577→ * When payLater is true, the expense is recorded as a payable to a supplier
   578→ * instead of an immediate cash outflow. This supports accrual-based accounting

   581→export async function autoEntryExpense(data: {

   589→  costCenterId?: string
   590→}, tx?: PrismaTransaction) {
   591→  const categoryMap: Record<string, string> = {
   592→    'CONSUMABLES': '7110',
   593→    'SERVICES': '7130',
   594→    'MAINTENANCE': '7220',
   595→    'FUEL': '7210',
   596→    'DRIVERS': '7230',
   597→    'TRANSPORT': '7240',

   628→  lines.push({ accountCode: creditAccountCode, debit: 0, credit: totalOut, costCenterId: data.costCenterId })
   629→

   631→    entryNo: `JE-EXP-${Date.now()}`,
   632→    date: data.date,

   890→/**
   891→ * رواتب - Salary Entry (Accrual + Payment)
   892→ *
   893→ * Supports two modes via the `accrualOnly` parameter:
   894→ *
   895→ * ACCRUAL MODE (accrualOnly = true):
   896→ *   Used during payroll run to record the expense and liability.
   897→ *   Dr: Salaries & Wages (8110) - grossSalary
   898→ *   Dr: GOSI Expense (8210) - gosiEmployerContribution
   899→ *   Cr: Salaries Payable (3310) - netSalary (gross - gosiEmployeeDeduction)
   900→ *   Cr: GOSI Payable (3830) - total GOSI (employee + employer)
   901→ *
   902→ * PAYMENT MODE (accrualOnly = false, default):
   903→ *   Used when salary is actually paid to clear the payable.
   904→ *   Dr: Salaries Payable (3310) - netSalary
   905→ *   Cr: Cash/Bank (1110/1120) - netCashPaid (netSalary)
   906→ *
   907→ * FULL COMBINED MODE (for backward compatibility, use accrualOnly = undefined):
   908→ *   Creates a single entry that records both expense and payment.
   909→ *   Dr: Salaries & Wages (8110) - grossSalary
   910→ *   Dr: GOSI Expense (8210) - gosiEmployerContribution
   911→ *   Cr: Cash/Bank (1110/1120) - netCashPaid
   912→ *   Cr: GOSI Payable (3830) - total GOSI
   913→ *   Cr: Salaries & Wages (8110) - gosiEmployeeDeduction (contra-entry)
   914→ */
   915→export async function autoEntrySalary(data: {
   916→  employeeName: string
   917→  grossSalary: number
   918→  gosiEmployeeDeduction: number
   919→  gosiEmployerContribution: number
   920→  date: Date
   921→  payFrom: 'TREASURY' | 'BANK'
   922→  costCenterId?: string
   923→  accrualOnly?: boolean // true = accrual entry only, false/undefined = payment entry
   924→}, tx?: PrismaTransaction) {
   925→  const cashAccountCode = data.payFrom === 'BANK' ? '1120' : '1110'
   926→  const netSalary = data.grossSalary - data.gosiEmployeeDeduction
   927→
   928→  // ACCRUAL MODE: Record expense and liability without cash payment
   929→  if (data.accrualOnly === true) {
   930→    const lines = [
   931→      { accountCode: '8110', debit: data.grossSalary, credit: 0, costCenterId: data.costCenterId }, // Salaries & Wages (gross)
   932→      { accountCode: '8210', debit: data.gosiEmployerContribution, credit: 0, costCenterId: data.costCenterId }, // GOSI Expense (employer)
   933→    ]
   934→
   935→    // Credit Salaries Payable for the net amount
   936→    lines.push({ accountCode: '3310', debit: 0, credit: netSalary, costCenterId: data.costCenterId }) // Salaries Payable (رواتب مستحقة)
   937→
   938→    // Credit GOSI Payable for total GOSI (employee deduction + employer contribution)
   939→    const totalGOSI = data.gosiEmployeeDeduction + data.gosiEmployerContribution
   940→    if (totalGOSI > 0) {
   941→      lines.push({ accountCode: '3830', debit: 0, credit: totalGOSI, costCenterId: data.costCenterId }) // GOSI Payable
   942→    }
   943→
   944→    return createJournalEntry({
   945→      entryNo: `JE-SAL-ACC-${Date.now()}`,
   946→      date: data.date,
   947→      description: `Salary accrual - ${data.employeeName}`,
   948→      descriptionAr: `استحقاق راتب - ${data.employeeName}`,
   949→      lines,
   950→      sourceType: 'SALARY_ACCRUAL',
   951→      sourceId: `SAL-ACC-${Date.now()}`,
   952→    }, tx)
   953→  }
   954→
   955→  // PAYMENT MODE: Clear the payable with cash
   956→  // When accrualOnly is explicitly false, this is the payment step
   957→  if (data.accrualOnly === false) {
   958→    return createJournalEntry({
   959→      entryNo: `JE-SAL-PAY-${Date.now()}`,
   960→      date: data.date,
   961→      description: `Salary payment - ${data.employeeName}`,
   962→      descriptionAr: `صرف راتب - ${data.employeeName}`,
   963→      lines: [
   964→        { accountCode: '3310', debit: netSalary, credit: 0 }, // Dr: Salaries Payable
   965→        { accountCode: cashAccountCode, debit: 0, credit: netSalary }, // Cr: Cash/Bank
   966→      ],
   967→      sourceType: 'SALARY_PAYMENT',
   968→      sourceId: `SAL-PAY-${Date.now()}`,
   969→    }, tx)
   970→  }
   971→
   972→  // FULL COMBINED MODE (backward compatible): Record expense + payment in one entry
   973→  const lines = [
   974→    { accountCode: '8110', debit: data.grossSalary, credit: 0, costCenterId: data.costCenterId }, // Salaries & Wages (gross)
   975→    { accountCode: '8210', debit: data.gosiEmployerContribution, credit: 0, costCenterId: data.costCenterId }, // GOSI Expense (employer)
   976→  ]
   977→
   978→  if (data.gosiEmployeeDeduction > 0) {
   979→    lines.push({ accountCode: '8110', debit: 0, credit: data.gosiEmployeeDeduction, costCenterId: data.costCenterId }) // Reduce salary by GOSI employee share
   980→  }
   981→
   982→  lines.push({ accountCode: cashAccountCode, debit: 0, credit: netSalary, costCenterId: data.costCenterId }) // Cash/Bank paid
   983→  lines.push({ accountCode: '3830', debit: 0, credit: data.gosiEmployeeDeduction + data.gosiEmployerContribution, costCenterId: data.costCenterId }) // GOSI Payable (total)
   984→
   985→  return createJournalEntry({
   986→    entryNo: `JE-SAL-${Date.now()}`,
   987→    date: data.date,
   988→    description: `Salary payment - ${data.employeeName}`,
   989→    descriptionAr: `صرف راتب - ${data.employeeName}`,
   990→    lines,
   991→    sourceType: 'SALARY',
   992→    sourceId: `SAL-${Date.now()}`,
   993→  }, tx)
   994→}

  1217→/**
  1218→ * التخلص من أصل - Asset Disposal
  1219→ * Dr: Cash/Bank (1110/1120) - sale price
  1220→ * Cr: Asset account (2110/2120/2130/2140) - original cost (or net book value)
  1221→ * Dr/Cr: Gain/Loss on disposal (6310 gain / 8610 loss)
  1222→ */
  1223→export async function autoEntryAssetDisposal(data: {
  1224→  assetAccountCode: string // e.g., '2110', '2120', '2130', '2140'
  1225→  accumulatedDepAccountCode: string // e.g., '2210', '2220', '2230', '2240'
  1226→  originalCost: number
  1227→  accumulatedDepreciation: number
  1228→  salePrice: number
  1229→  date: Date
  1230→  receivedIn: 'TREASURY' | 'BANK'
  1231→}, tx?: PrismaTransaction) {
  1232→  const cashAccountCode = data.receivedIn === 'BANK' ? '1120' : '1110'
  1233→  const netBookValue = data.originalCost - data.accumulatedDepreciation
  1234→  const gainLoss = data.salePrice - netBookValue
  1235→
  1236→  const lines = [
  1237→    { accountCode: cashAccountCode, debit: data.salePrice, credit: 0 }, // Cash received
  1238→    { accountCode: data.accumulatedDepAccountCode, debit: data.accumulatedDepreciation, credit: 0 }, // Remove accumulated depreciation
  1239→    { accountCode: data.assetAccountCode, debit: 0, credit: data.originalCost }, // Remove asset
  1240→  ]
  1241→
  1242→  // If gain, credit gain account. If loss, debit loss account.
  1243→  if (gainLoss > 0) {
  1244→    lines.push({ accountCode: '6310', debit: 0, credit: gainLoss }) // Gain on Sale of Used Equipment
  1245→  } else if (gainLoss < 0) {
  1246→    lines.push({ accountCode: '8610', debit: Math.abs(gainLoss), credit: 0 }) // Loss on Asset Disposal

  1249→  return createJournalEntry({
  1250→    entryNo: `JE-DSP-${Date.now()}`,
  1251→    date: data.date,

  1255→    sourceType: 'ASSET_DISPOSAL',
  1256→    sourceId: `DSP-${Date.now()}`,
  1257→  }, tx)

  1260→/**
  1261→ * إقرار ضريبي - VAT Declaration
  1262→ * Dr: Output VAT (3110) - total output VAT
  1263→ * Cr: Input VAT (3120) - total input VAT
  1264→ * Cr: VAT Due (3130) - net VAT payable (if output > input)
  1265→ * OR
  1266→ * Dr: Input VAT (3120) - if input > output (refund scenario)
  1267→ * Cr: Output VAT (3110)
  1268→ * Dr: VAT Refund Receivable (1410) - if refund due
  1269→ */
  1270→export async function autoEntryVATDeclaration(data: {
  1271→  period: string
  1272→  outputVat: number
  1273→  inputVat: number
  1274→  netVat: number // positive = payable, negative = refundable
  1275→  date: Date
  1276→}, tx?: PrismaTransaction) {

  1278→
  1279→  // Close Output VAT (3110) - debit to zero it out
  1280→  lines.push({ accountCode: '3110', debit: data.outputVat, credit: 0 })
  1281→
  1282→  // Close Input VAT (3120) - credit to zero it out
  1283→  lines.push({ accountCode: '3120', debit: 0, credit: data.inputVat })
  1284→
  1285→  // Net VAT position
  1286→  if (data.netVat > 0) {
  1287→    // More output than input → owe VAT
  1288→    lines.push({ accountCode: '3130', debit: 0, credit: data.netVat }) // VAT Due
  1289→  } else if (data.netVat < 0) {
  1290→    // More input than output → refund due
  1291→    lines.push({ accountCode: '1410', debit: Math.abs(data.netVat), credit: 0 }) // VAT Refund Receivable
  1292→  }
  1293→
  1294→  return createJournalEntry({
  1295→    entryNo: `JE-VAT-${Date.now()}`,
  1296→    date: data.date,
  1297→    description: `VAT Declaration - ${data.period}`,
  1298→    descriptionAr: `إقرار ضريبي - ${data.period}`,
  1299→    lines,
  1300→    sourceType: 'VAT_DECLARATION',
  1301→    sourceId: `VAT-${data.period}`,
  1302→  }, tx)
  1303→}

  1332→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1333→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1334→  if (dateFrom || dateTo) {
  1335→    dateFilter.date = {
  1336→      ...(dateFrom && { gte: dateFrom }),
  1337→      ...(dateTo && { lte: dateTo }),
  1338→    }
  1339→  }
  1340→  const entries = await db.journalEntry.findMany({
  1341→    where: {
  1342→      status: 'POSTED',
  1343→      ...dateFilter,
  1344→    },
  1345→    include: {
  1346→      lines: {
  1347→        include: {
  1348→          account: true,
  1349→        },
  1350→      },
  1351→    },
  1352→  })
  1353→
  1354→  // Aggregate by account
  1355→  const accountBalances = new Map<string, {
  1356→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1357→    totalDebit: number
  1358→    totalCredit: number
  1359→  }>()
  1360→
  1361→  for (const entry of entries) {
  1362→    for (const line of entry.lines) {
  1363→      const key = line.accountId
  1364→      if (!accountBalances.has(key)) {
  1365→        accountBalances.set(key, {
  1366→          account: line.account,
  1367→          totalDebit: 0,
  1368→          totalCredit: 0,
  1369→        })
  1370→      }
  1371→      const bal = accountBalances.get(key)!
  1372→      bal.totalDebit += toNumber(line.debit)
  1373→      bal.totalCredit += toNumber(line.credit)
  1374→    }
  1375→  }
  1376→
  1377→  // Calculate net balance for each account
  1378→  const results = Array.from(accountBalances.values()).map(bal => {
  1379→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1380→    let netDebit = 0
  1381→    let netCredit = 0
  1382→
  1383→    if (normalBalance === 'DEBIT') {
  1384→      const net = bal.totalDebit - bal.totalCredit
  1385→      if (net >= 0) netDebit = net
  1386→      else netCredit = Math.abs(net)
  1387→    } else {
  1388→      const net = bal.totalCredit - bal.totalDebit
  1389→      if (net >= 0) netCredit = net
  1390→      else netDebit = Math.abs(net)
  1391→    }
  1392→
  1393→    return {
  1394→      ...bal,
  1395→      netDebit,
  1396→      netCredit,
  1397→    }
  1398→  })
  1399→
  1400→  // Sort by account code
  1401→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1402→
  1403→  return results
  1404→}

  1450→export async function getGeneralLedger(accountCode: string, dateFrom?: Date, dateTo?: Date) {
  1451→  const account = await getAccountByCode(accountCode)
  1452→  if (!account) return []
  1453→
  1454→  const glDateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1455→  if (dateFrom || dateTo) {
  1456→    glDateFilter.date = {
  1457→      ...(dateFrom && { gte: dateFrom }),
  1458→      ...(dateTo && { lte: dateTo }),
  1459→    }
  1460→  }
  1461→  const lines = await db.journalLine.findMany({
  1462→    where: {
  1463→      accountId: account.id,
  1464→      journalEntry: {
  1465→        status: 'POSTED',
  1466→        ...glDateFilter,
  1467→      },
  1468→    },
  1469→    include: { journalEntry: true },
  1470→    orderBy: { journalEntry: { date: 'asc' } },
  1471→  })
  1472→
  1473→  let runningBalance = 0
  1474→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1475→
  1476→  return lines.map(line => {
  1477→    const debit = toNumber(line.debit)
  1478→    const credit = toNumber(line.credit)
  1479→    if (normalBalance === 'DEBIT') {
  1480→      runningBalance += debit - credit
  1481→    } else {
  1482→      runningBalance += credit - debit
  1483→    }
  1484→
  1485→    return {
  1486→      date: line.journalEntry.date,
  1487→      entryNo: line.journalEntry.entryNo,
  1488→      description: line.journalEntry.description,
  1489→      debit,
  1490→      credit,
  1491→      balance: runningBalance,
  1492→    }
  1493→  })
  1494→}

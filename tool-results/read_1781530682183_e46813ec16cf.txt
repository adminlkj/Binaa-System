   126→// ============ Add Payment Dialog ============
   127→
   128→function AddPaymentDialog({
   129→  open,
   130→  onClose,
   131→}: {
   132→  open: boolean
   133→  onClose: () => void
   134→}) {
   135→  const { lang } = useAppStore()
   136→  const queryClient = useQueryClient()
   137→  const { toast } = useToast()
   138→  const tt = (ar: string, en: string) => t(ar, en, lang)
   139→
   140→  const [clientId, setClientId] = useState('')
   141→  const [invoiceId, setInvoiceId] = useState('')
   142→  const [amount, setAmount] = useState('')
   143→  const [date, setDate] = useState(new Date().toISOString().split('T')[0])
   144→  const [receivedIn, setReceivedIn] = useState('TREASURY')
   145→  const [reference, setReference] = useState('')
   146→  const [notes, setNotes] = useState('')
   147→
   148→  // Fetch clients
   149→  const { data: clients = [] } = useQuery<ClientSummary[]>({
   150→    queryKey: ['clients-list'],
   151→    queryFn: async () => {
   152→      const res = await fetch('/api/clients')
   153→      if (!res.ok) return []
   154→      return res.json()
   155→    },
   156→  })
   157→
   158→  // Fetch unpaid invoices for selected client
   159→  const { data: invoices = [] } = useQuery<InvoiceSummary[]>({
   160→    queryKey: ['unpaid-invoices', clientId],
   161→    queryFn: async () => {
   162→      if (!clientId) return []
   163→      const res = await fetch(`/api/sales-invoices?clientId=${clientId}&status=unpaid`)
   164→      if (!res.ok) return []
   165→      const data = await res.json()
   166→      return data.filter((inv: InvoiceSummary) =>
   167→        inv.status !== 'PAID' && inv.status !== 'CANCELLED' && inv.totalAmount - inv.paidAmount > 0
   168→      )
   169→    },
   170→    enabled: !!clientId,
   171→  })
   172→
   173→  // Selected invoice
   174→  const selectedInvoice = invoices.find(inv => inv.id === invoiceId)
   175→  const remainingBalance = selectedInvoice
   176→    ? (selectedInvoice.totalAmount ?? 0) - (selectedInvoice.paidAmount ?? 0)
   177→    : 0
   178→
   179→  // Auto-fill amount when invoice is selected
   180→  React.useEffect(() => {
   181→    if (selectedInvoice && remainingBalance > 0) {
   182→      setAmount((remainingBalance ?? 0).toFixed(2))
   183→    }
   184→  }, [invoiceId, selectedInvoice, remainingBalance])
   185→
   186→  // Reset form when dialog opens/closes
   187→  React.useEffect(() => {
   188→    if (!open) {
   189→      setClientId('')
   190→      setInvoiceId('')
   191→      setAmount('')
   192→      setDate(new Date().toISOString().split('T')[0])
   193→      setReceivedIn('TREASURY')
   194→      setReference('')
   195→      setNotes('')
   196→    }
   197→  }, [open])
   198→
   199→  const createMutation = useMutation({
   200→    mutationFn: (data: Record<string, unknown>) =>
   201→      fetch('/api/client-payments', {
   202→        method: 'POST',
   203→        headers: { 'Content-Type': 'application/json' },
   204→        body: JSON.stringify(data),
   205→      }).then(r => {
   206→        if (!r.ok) throw new Error('Failed')
   207→        return r.json()
   208→      }),
   209→    onSuccess: () => {
   210→      queryClient.invalidateQueries({ queryKey: ['client-payments'] })
   211→      toast({ title: tt('تم تسجيل التحصيل', 'Payment recorded'), description: tt('تم تسجيل التحصيل بنجاح', 'Payment has been recorded successfully') })
   212→      onClose()
   213→    },
   214→    onError: () => {
   215→      toast({ title: tt('خطأ', 'Error'), description: tt('فشل في تسجيل التحصيل', 'Failed to record payment'), variant: 'destructive' })
   216→    },
   217→  })
   218→
   219→  const handleSubmit = (e: React.FormEvent) => {
   220→    e.preventDefault()
   221→    const parsedAmount = parseFloat(amount) || 0
   222→    if (!clientId || parsedAmount <= 0 || !date) return
   223→
   224→    createMutation.mutate({
   225→      clientId,
   226→      invoiceId: invoiceId || null,
   227→      amount: parsedAmount,
   228→      date,
   229→      receivedIn,
   230→      reference: reference || null,
   231→      notes: notes || null,
   232→    })
   233→  }
   234→
   235→  return (
   236→    <Dialog open={open} onOpenChange={onClose}>
   237→      <DialogContent className="max-w-lg">
   238→        <DialogHeader>
   239→          <DialogTitle className="flex items-center gap-2">
   240→            <CreditCard className="size-5 text-emerald-600" />
   241→            {tt(labels.newPayment.ar, labels.newPayment.en)}
   242→          </DialogTitle>
   243→        </DialogHeader>
   244→
   245→        <form onSubmit={handleSubmit} className="space-y-4">
   246→          {/* Client Selection */}
   247→          <div className="space-y-2">
   248→            <Label>{tt(labels.client.ar, labels.client.en)} *</Label>
   249→            <Select value={clientId} onValueChange={v => { setClientId(v); setInvoiceId(''); setAmount('') }}>
   250→              <SelectTrigger className="w-full">
   251→                <SelectValue placeholder={tt(labels.selectClient.ar, labels.selectClient.en)} />
   252→              </SelectTrigger>
   253→              <SelectContent>
   254→                {clients.map(c => (
   255→                  <SelectItem key={c.id} value={c.id}>{c.name} ({c.code})</SelectItem>
   256→                ))}
   257→              </SelectContent>
   258→            </Select>
   259→          </div>
   260→
   261→          {/* Invoice Selection */}
   262→          <div className="space-y-2">
   263→            <Label>{tt(labels.invoice.ar, labels.invoice.en)}</Label>
   264→            <Select value={invoiceId} onValueChange={setInvoiceId} disabled={!clientId}>
   265→              <SelectTrigger className="w-full">
   266→                <SelectValue placeholder={
   267→                  !clientId
   268→                    ? tt(labels.selectClient.ar, labels.selectClient.en)
   269→                    : invoices.length === 0
   270→                      ? tt(labels.noInvoices.ar, labels.noInvoices.en)
   271→                      : tt(labels.selectInvoice.ar, labels.selectInvoice.en)
   272→                } />
   273→              </SelectTrigger>
   274→              <SelectContent>
   275→                {invoices.map(inv => {
   276→                  const remaining = inv.totalAmount - inv.paidAmount
   277→                  return (
   278→                    <SelectItem key={inv.id} value={inv.id}>
   279→                      {inv.invoiceNo} — {tt('متبقي', 'Remaining')}: <MoneyDisplay value={remaining} lang={lang} size="sm" inline />
   280→                    </SelectItem>
   281→                  )
   282→                })}
   283→              </SelectContent>
   284→            </Select>
   285→          </div>
   286→
   287→          {/* Invoice Info */}
   288→          {selectedInvoice && (
   289→            <div className="p-3 rounded-lg border bg-emerald-50 text-sm space-y-1">
   290→              <div className="flex justify-between">
   291→                <span className="text-muted-foreground">{tt('إجمالي الفاتورة', 'Invoice Total')}</span>
   292→                <MoneyDisplay value={selectedInvoice.totalAmount} lang={lang} size="sm" inline bold />
   293→              </div>
   294→              <div className="flex justify-between">
   295→                <span className="text-muted-foreground">{tt('المدفوع', 'Paid')}</span>
   296→                <MoneyDisplay value={selectedInvoice.paidAmount} lang={lang} size="sm" inline />
   297→              </div>
   298→              <Separator />
   299→              <div className="flex justify-between font-bold text-emerald-700">
   300→                <span>{tt(labels.remaining.ar, labels.remaining.en)}</span>
   301→                <MoneyDisplay value={remainingBalance} lang={lang} size="sm" inline bold />
   302→              </div>
   303→            </div>
   304→          )}
   305→
   306→          {/* Amount */}
   307→          <div className="space-y-2">
   308→            <Label>
   309→              {tt(labels.amount.ar, labels.amount.en)} (ر.س / SAR) *
   310→              {selectedInvoice && (
   311→                <span className="text-xs text-emerald-600 ml-2">({tt(labels.autoFill.ar, labels.autoFill.en)})</span>
   312→              )}
   313→            </Label>
   314→            <Input
   315→              type="number"
   316→              step="0.01"
   317→              min="0.01"
   318→              max={remainingBalance || undefined}
   319→              value={amount}
   320→              onChange={e => setAmount(e.target.value)}
   321→              placeholder="0.00"
   322→              required
   323→            />
   324→          </div>
   325→
   326→          {/* Date & Received In */}
   327→          <div className="grid grid-cols-2 gap-4">
   328→            <div className="space-y-2">
   329→              <Label>{tt(labels.date.ar, labels.date.en)} *</Label>
   330→              <Input type="date" value={date} onChange={e => setDate(e.target.value)} required />
   331→            </div>
   332→            <div className="space-y-2">
   333→              <Label>{tt(labels.receivedIn.ar, labels.receivedIn.en)}</Label>
   334→              <Select value={receivedIn} onValueChange={setReceivedIn}>
   335→                <SelectTrigger className="w-full">
   336→                  <SelectValue />
   337→                </SelectTrigger>
   338→                <SelectContent>
   339→                  <SelectItem value="TREASURY">
   340→                    <span className="flex items-center gap-2">
   341→                      <Landmark className="size-3.5" />
   342→                      {tt(receivedInLabels.TREASURY.ar, receivedInLabels.TREASURY.en)}
   343→                    </span>
   344→                  </SelectItem>
   345→                  <SelectItem value="BANK">
   346→                    <span className="flex items-center gap-2">
   347→                      <Building2 className="size-3.5" />
   348→                      {tt(receivedInLabels.BANK.ar, receivedInLabels.BANK.en)}
   349→                    </span>
   350→                  </SelectItem>
   351→                </SelectContent>
   352→              </Select>
   353→            </div>
   354→          </div>
   355→
   356→          {/* Reference */}
   357→          <div className="space-y-2">
   358→            <Label>{tt(labels.reference.ar, labels.reference.en)}</Label>
   359→            <Input value={reference} onChange={e => setReference(e.target.value)} placeholder={tt('رقم الإيصال', 'Receipt No.')} />
   360→          </div>
   361→
   362→          {/* Notes */}
   363→          <div className="space-y-2">
   364→            <Label>{tt(labels.notes.ar, labels.notes.en)}</Label>
   365→            <Input value={notes} onChange={e => setNotes(e.target.value)} placeholder={tt('ملاحظات', 'Notes')} />
   366→          </div>
   367→
   368→          <DialogFooter>
   369→            <Button type="button" variant="outline" onClick={onClose}>
   370→              {commonText.cancel[lang]}
   371→            </Button>
   372→            <Button
   373→              type="submit"
   374→              disabled={createMutation.isPending || !clientId || !amount || !date}
   375→              className="bg-emerald-600 hover:bg-emerald-700 min-w-[140px]"
   376→            >
   377→              {createMutation.isPending ? tt('جاري الحفظ...', 'Saving...') : tt('تسجيل التحصيل', 'Record Payment')}
   378→            </Button>
   379→          </DialogFooter>
   380→        </form>
   381→      </DialogContent>
   382→    </Dialog>
   383→  )
   384→}

   386→// ============ Edit Payment Dialog ============
   387→
   388→function EditPaymentDialog({
   389→  payment,
   390→  open,
   391→  onClose,
   392→}: {
   393→  payment: ClientPaymentItem | null
   394→  open: boolean
   395→  onClose: () => void
   396→}) {
   397→  const { lang } = useAppStore()
   398→  const queryClient = useQueryClient()
   399→  const { toast } = useToast()
   400→  const tt = (ar: string, en: string) => t(ar, en, lang)
   401→
   402→  const [amount, setAmount] = useState('')
   403→  const [date, setDate] = useState('')
   404→  const [receivedIn, setReceivedIn] = useState('TREASURY')
   405→  const [reference, setReference] = useState('')
   406→  const [notes, setNotes] = useState('')
   407→
   408→  React.useEffect(() => {
   409→    if (payment && open) {
   410→      setAmount(String(payment.amount))
   411→      setDate(payment.date ? new Date(payment.date).toISOString().split('T')[0] : '')
   412→      setReceivedIn(payment.receivedIn || 'TREASURY')
   413→      setReference(payment.reference || '')
   414→      setNotes(payment.notes || '')
   415→    }
   416→  }, [payment, open])
   417→
   418→  const editMutation = useMutation({
   419→    mutationFn: (data: Record<string, unknown>) =>
   420→      fetch(`/api/client-payments/${payment?.id}`, {
   421→        method: 'PATCH',
   422→        headers: { 'Content-Type': 'application/json' },
   423→        body: JSON.stringify(data),
   424→      }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   425→    onSuccess: () => {
   426→      queryClient.invalidateQueries({ queryKey: ['client-payments'] })
   427→      toast({ title: tt('تم تحديث التحصيل', 'Payment updated'), description: tt('تم تحديث التحصيل بنجاح', 'Payment has been updated successfully') })
   428→      onClose()
   429→    },
   430→    onError: () => {
   431→      toast({ title: tt('خطأ', 'Error'), description: tt('فشل في تحديث التحصيل', 'Failed to update payment'), variant: 'destructive' })
   432→    },
   433→  })
   434→
   435→  const handleSubmit = (e: React.FormEvent) => {
   436→    e.preventDefault()
   437→    editMutation.mutate({
   438→      amount: parseFloat(amount) || 0,
   439→      date,
   440→      receivedIn,
   441→      reference: reference || null,
   442→      notes: notes || null,
   443→    })
   444→  }
   445→
   446→  if (!payment) return null
   447→
   448→  const isPosted = !!payment.journalEntryId
   449→
   450→  return (
   451→    <Dialog open={open} onOpenChange={onClose}>
   452→      <DialogContent className="max-w-lg">
   453→        <DialogHeader>
   454→          <DialogTitle className="flex items-center gap-2">
   455→            <Pencil className="size-5 text-emerald-600" />
   456→            {tt(labels.editPayment.ar, labels.editPayment.en)}
   457→          </DialogTitle>
   458→        </DialogHeader>
   459→
   460→        {isPosted && (
   461→          <div className="p-3 rounded-lg border bg-amber-50 text-amber-700 text-sm">
   462→            {tt('هذا التحصيل مرحّل محاسبياً - التعديل محدود', 'This payment is posted - editing is limited')}
   463→          </div>
   464→        )}
   465→
   466→        <form onSubmit={handleSubmit} className="space-y-4">
   467→          <div className="p-3 rounded-lg bg-gray-50 text-sm">
   468→            <span className="text-muted-foreground">{tt('العميل:', 'Client:')}</span>{' '}
   469→            <span className="font-medium">{payment.client.name}</span>
   470→            {payment.invoice && (
   471→              <><span className="text-muted-foreground mx-2">|</span>
   472→              <span className="text-muted-foreground">{tt('الفاتورة:', 'Invoice:')}</span>{' '}
   473→              <span className="font-medium">{payment.invoice.invoiceNo}</span></>
   474→            )}
   475→          </div>
   476→
   477→          <div className="space-y-2">
   478→            <Label>{tt(labels.amount.ar, labels.amount.en)} (ر.س / SAR) *</Label>
   479→            <Input type="number" step="0.01" min="0.01" value={amount} onChange={e => setAmount(e.target.value)} dir="ltr" required disabled={isPosted} />
   480→          </div>
   481→
   482→          <div className="grid grid-cols-2 gap-4">
   483→            <div className="space-y-2">
   484→              <Label>{tt(labels.date.ar, labels.date.en)} *</Label>
   485→              <Input type="date" value={date} onChange={e => setDate(e.target.value)} required disabled={isPosted} />
   486→            </div>
   487→            <div className="space-y-2">
   488→              <Label>{tt(labels.receivedIn.ar, labels.receivedIn.en)}</Label>
   489→              <Select value={receivedIn} onValueChange={setReceivedIn} disabled={isPosted}>
   490→                <SelectTrigger className="w-full"><SelectValue /></SelectTrigger>
   491→                <SelectContent>
   492→                  <SelectItem value="TREASURY">{tt(receivedInLabels.TREASURY.ar, receivedInLabels.TREASURY.en)}</SelectItem>
   493→                  <SelectItem value="BANK">{tt(receivedInLabels.BANK.ar, receivedInLabels.BANK.en)}</SelectItem>
   494→                </SelectContent>
   495→              </Select>
   496→            </div>
   497→          </div>
   498→
   499→          <div className="space-y-2">
   500→            <Label>{tt(labels.reference.ar, labels.reference.en)}</Label>
   501→            <Input value={reference} onChange={e => setReference(e.target.value)} disabled={isPosted} />
   502→          </div>

   633→export function ClientPaymentsModule() {
   634→  const { lang } = useAppStore()
   635→  const queryClient = useQueryClient()
   636→  const { toast } = useToast()
   637→  const tt = (ar: string, en: string) => t(ar, en, lang)
   638→
   639→  const [search, setSearch] = useState('')
   640→  const [clientFilter, setClientFilter] = useState<string>('all')
   641→  const [receivedInFilter, setReceivedInFilter] = useState<string>('all')
   642→  const [showAddDialog, setShowAddDialog] = useState(false)
   643→  const [detailPayment, setDetailPayment] = useState<ClientPaymentItem | null>(null)
   644→  const [editPayment, setEditPayment] = useState<ClientPaymentItem | null>(null)
   645→  const [deleteId, setDeleteId] = useState<string | null>(null)
   646→
   647→  // Fetch payments
   648→  const { data: payments = [], isLoading, isError, refetch } = useQuery<ClientPaymentItem[]>({
   649→    queryKey: ['client-payments'],
   650→    queryFn: async () => {
   651→      const res = await fetch('/api/client-payments')
   652→      if (!res.ok) throw new Error('Failed')
   653→      return res.json()
   654→    },
   655→    staleTime: 30000,
   656→  })
   657→
   658→  // Fetch clients for filter
   659→  const { data: clients = [] } = useQuery<ClientSummary[]>({
   660→    queryKey: ['clients-list-for-payments'],
   661→    queryFn: async () => {
   662→      const res = await fetch('/api/clients')
   663→      if (!res.ok) return []
   664→      return res.json()
   665→    },
   666→  })
   667→
   668→  // Delete mutation
   669→  const deleteMutation = useMutation({
   670→    mutationFn: (id: string) =>

   672→        if (!r.ok) throw new Error()

   676→      queryClient.invalidateQueries({ queryKey: ['client-payments'] })

   682→    },
   683→  })
   684→
   685→  // Filter
   686→  const filtered = useMemo(() => {
   687→    return payments.filter(p => {
   688→      const matchSearch = !search ||
   689→        p.client.name.toLowerCase().includes(search.toLowerCase()) ||
   690→        (p.reference && p.reference.toLowerCase().includes(search.toLowerCase())) ||
   691→        (p.invoice?.invoiceNo && p.invoice.invoiceNo.toLowerCase().includes(search.toLowerCase()))

   693→      const matchReceivedIn = receivedInFilter === 'all' || p.receivedIn === receivedInFilter
   694→      return matchSearch && matchClient && matchReceivedIn
   695→    })

   697→

   699→  const totalPayments = filtered.reduce((s, p) => s + p.amount, 0)
   700→  const treasuryPayments = filtered.filter(p => p.receivedIn === 'TREASURY').reduce((s, p) => s + p.amount, 0)
   701→  const bankPayments = filtered.filter(p => p.receivedIn === 'BANK').reduce((s, p) => s + p.amount, 0)
   702→

   704→  const handleExport = () => {
   705→    const columns: CSVColumn[] = [
   706→      { key: 'date', label: tt('التاريخ', 'Date') },
   707→      { key: 'clientName', label: tt('العميل', 'Client') },
   708→      { key: 'invoiceNo', label: tt('الفاتورة', 'Invoice') },
   709→      { key: 'projectName', label: tt('المشروع', 'Project') },
   710→      { key: 'amount', label: tt('المبلغ', 'Amount'), format: (v) => (Number(v) || 0).toFixed(2) },
   711→      { key: 'receivedIn', label: tt('عن طريق', 'Via'), format: (v) => receivedInLabels[v as string]?.[lang] || String(v) },
   712→      { key: 'reference', label: tt('المرجع', 'Reference') },
   713→      { key: 'notes', label: tt('ملاحظات', 'Notes') },
   714→    ]
   715→    const rows = filtered.map(p => ({
   716→      date: formatDate(p.date, lang),
   717→      clientName: p.client.name,
   718→      invoiceNo: p.invoice?.invoiceNo || '',
   719→      projectName: p.invoice?.project?.name || '',
   720→      amount: p.amount,
   721→      receivedIn: p.receivedIn,
   722→      reference: p.reference || '',
   723→      notes: p.notes || '',
   724→    }))

   749→        <Card className="bg-emerald-50 border-emerald-200">
   750→          <CardContent className="p-3">
   751→            <p className="text-xs text-muted-foreground">{tt(labels.totalPayments.ar, labels.totalPayments.en)}</p>

   759→          </CardContent>
   760→        </Card>
   761→        <Card className="bg-sky-50 border-sky-200">
   762→          <CardContent className="p-3">
   763→            <p className="text-xs text-muted-foreground">{tt(labels.bank.ar, labels.bank.en)}</p>

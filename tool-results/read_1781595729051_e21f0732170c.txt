   270→   270→export async function ensureAccountExists(template: AccountTemplate, tx?: PrismaTransaction) {
   271→   271→  const client = tx || db
   272→   272→  const existing = await client.account.findUnique({ where: { code: template.code } })
   273→   273→  if (existing) {
   274→   274→    // Update existing account with new fields if they are missing
   275→   275→    if (
   276→   276→      existing.activityType !== (template.activityType || null) ||
   277→   277→      existing.isSystem !== (template.isSystem || false) ||
   278→   278→      existing.allowPosting !== (template.allowPosting || false) ||
   279→   279→      existing.level !== (template.level || 0) ||
   280→   280→      (template.accountRole && existing.accountRole !== template.accountRole) ||
   281→   281→      (template.parentId && existing.parentCode !== template.parentId)
   282→   282→    ) {
   283→   283→      await client.account.update({
   284→   284→        where: { code: template.code },
   285→   285→        data: {
   286→   286→          name: template.name,
   287→   287→          nameAr: template.nameAr,
   288→   288→          type: template.type,
   289→   289→          activityType: template.activityType || null,
   290→   290→          isSystem: template.isSystem || false,
   291→   291→          allowPosting: template.allowPosting || false,
   292→   292→          level: template.level || 0,
   293→   293→          accountRole: template.accountRole || existing.accountRole,
   294→   294→          parentCode: template.parentId || existing.parentCode,
   295→   295→        },
   296→   296→      })
   297→   297→    }
   298→   298→    return client.account.findUnique({ where: { code: template.code } })
   299→   299→  }
   300→   300→
   301→   301→  let parentId: string | undefined
   302→   302→  if (template.parentId) {
   303→   303→    const parent = await client.account.findUnique({ where: { code: template.parentId } })
   304→   304→    if (parent) parentId = parent.id
   305→   305→  }
   306→   306→
   307→   307→  return client.account.create({
   308→   308→    data: {
   309→   309→      code: template.code,
   310→   310→      name: template.name,
   311→   311→      nameAr: template.nameAr,
   312→   312→      type: template.type,

   318→   318→      isSystem: template.isSystem || false,
   319→   319→      allowPosting: template.allowPosting || false,

   321→   321→    },
   322→   322→  })

   327→   327→export async function initializeChartOfAccounts() {
   328→   328→  let created = 0
   329→   329→  let updated = 0
   330→   330→
   331→   331→  // Create parent accounts first, then children (sorted by code length then code)
   332→   332→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   333→   333→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   334→   334→    return a.code.localeCompare(b.code)
   335→   335→  })
   336→   336→
   337→   337→  for (const tmpl of sorted) {
   338→   338→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   339→   339→    if (existing) {
   340→   340→      // Update existing account with new fields
   341→   341→      await db.account.update({
   342→   342→        where: { code: tmpl.code },
   343→   343→        data: {
   344→   344→          name: tmpl.name,
   345→   345→          nameAr: tmpl.nameAr,
   346→   346→          type: tmpl.type,
   347→   347→          activityType: tmpl.activityType || null,
   348→   348→          accountRole: tmpl.accountRole || null,
   349→   349→          parentCode: tmpl.parentId || null,
   350→   350→          isSystem: tmpl.isSystem || false,
   351→   351→          allowPosting: tmpl.allowPosting || false,
   352→   352→          level: tmpl.level || 0,
   353→   353→          isActive: true,
   354→   354→        },
   355→   355→      })
   356→   356→      // Update parent reference if needed
   357→   357→      if (tmpl.parentId) {
   358→   358→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   359→   359→        if (parent && existing.parentId !== parent.id) {
   360→   360→          await db.account.update({
   361→   361→            where: { code: tmpl.code },
   362→   362→            data: { parentId: parent.id },
   363→   363→          })
   364→   364→        }
   365→   365→      }
   366→   366→      updated++
   367→   367→    } else {
   368→   368→      await ensureAccountExists(tmpl)
   369→   369→      created++
   370→   370→    }
   371→   371→  }
   372→   372→
   373→   373→  const total = await db.account.count()
   374→   374→  return { created, updated, total }
   375→   375→}

   377→   377→// ============ JOURNAL ENTRY REVERSAL ============
   378→   378→
   379→   379→/**
   380→   380→ * عكس قيد محاسبي - Reverse a posted journal entry
   381→   381→ *
   382→   382→ * Creates a reversal entry with flipped debit/credit on all lines,
   383→   383→ * marks the original entry as CANCELLED, and links them together.
   384→   384→ *
   385→   385→ * IMPORTANT: This function MUST be called inside a $transaction callback.
   386→   386→ * The `tx` parameter is required — no standalone calls allowed.
   387→   387→ *
   388→   388→ * @param journalEntryId - ID of the journal entry to reverse
   389→   389→ * @param tx - Prisma transaction client (required)
   390→   390→ * @returns The reversal journal entry with its lines
   391→   391→ * @throws Error if entry not found, not POSTED, or already reversed
   392→   392→ */
   393→   393→export async function reverseEntry(journalEntryId: string, tx: PrismaTransaction) {
   394→   394→  // 1. Find the original entry with its lines
   395→   395→  const originalEntry = await tx.journalEntry.findUnique({
   396→   396→    where: { id: journalEntryId },
   397→   397→    include: { lines: true },
   398→   398→  })
   399→   399→
   400→   400→  if (!originalEntry) {
   401→   401→    throw new Error(`القيد المحاسبي غير موجود: ${journalEntryId}`)
   402→   402→  }
   403→   403→
   404→   404→  // 2. Check that the entry is POSTED (not DRAFT or CANCELLED)
   405→   405→  if (originalEntry.status !== 'POSTED') {
   406→   406→    throw new Error(
   407→   407→      `لا يمكن عكس قيد بحالة ${originalEntry.status} - يجب أن يكون القيد مرحّلاً (POSTED)`
   408→   408→    )
   409→   409→  }
   410→   410→
   411→   411→  // 3. Check that it hasn't already been reversed
   412→   412→  if (originalEntry.isReversal) {
   413→   413→    throw new Error('لا يمكن عكس قيد عكسي - القيود العكسية لا يمكن عكسها')
   414→   414→  }
   415→   415→
   416→   416→  const existingReversal = await tx.journalEntry.findFirst({
   417→   417→    where: { reversedEntryId: journalEntryId },
   418→   418→  })
   419→   419→  if (existingReversal) {
   420→   420→    throw new Error('القيد محسوب عكسه مسبقاً - يوجد قيد عكسي مرتبط بهذا القيد')
   421→   421→  }
   422→   422→
   423→   423→  // 4. Generate next entry number
   424→   424→  const last = await tx.journalEntry.findFirst({ orderBy: { entryNo: 'desc' } })
   425→   425→  const next = last ? parseInt(last.entryNo.replace('JE-', '')) + 1 : 1
   426→   426→  const entryNo = `JE-${next.toString().padStart(6, '0')}`
   427→   427→
   428→   428→  // 5. Build reversal description with "عكس" prefix
   429→   429→  const reversalDescription = originalEntry.description
   430→   430→    ? `عكس - ${originalEntry.description}`
   431→   431→    : 'عكس قيد محاسبي'
   432→   432→
   433→   433→  // 6. Create the reversal entry with flipped debit/credit lines
   434→   434→  const reversalEntry = await tx.journalEntry.create({
   435→   435→    data: {
   436→   436→      entryNo,
   437→   437→      date: originalEntry.date,
   438→   438→      description: reversalDescription,
   439→   439→      status: 'POSTED',
   440→   440→      sourceType: originalEntry.sourceType,
   441→   441→      sourceId: originalEntry.sourceId,
   442→   442→      isReversal: true,
   443→   443→      reversedEntryId: originalEntry.id,
   444→   444→      isSystem: true,
   445→   445→      lines: {
   446→   446→        create: originalEntry.lines.map((line) => ({
   447→   447→          accountId: line.accountId,
   448→   448→          costCenterId: line.costCenterId,
   449→   449→          // Flip debit and credit
   450→   450→          debit: line.credit,
   451→   451→          credit: line.debit,
   452→   452→          description: line.description
   453→   453→            ? `عكس - ${line.description}`
   454→   454→            : 'عكس بند قيد',
   455→   455→        })),
   456→   456→      },
   457→   457→    },
   458→   458→    include: { lines: true },
   459→   459→  })
   460→   460→
   461→   461→  // 7. Update the original entry status to CANCELLED
   462→   462→  await tx.journalEntry.update({
   463→   463→    where: { id: originalEntry.id },
   464→   464→    data: { status: 'CANCELLED' },
   465→   465→  })
   466→   466→
   467→   467→  return reversalEntry
   468→   468→}

   472→   472→export async function createJournalEntry(template: JournalEntryTemplate, tx?: PrismaTransaction) {
   473→   473→  const client = tx || db
   474→   474→  // Validate: total debits must equal total credits
   475→   475→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   476→   476→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   477→   477→
   478→   478→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   479→   479→    throw new Error(
   480→   480→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   481→   481→    )
   482→   482→  }
   483→   483→
   484→   484→  // Ensure all referenced accounts exist
   485→   485→  for (const line of template.lines) {
   486→   486→    const account = await getAccountByCode(line.accountCode, tx)
   487→   487→    if (!account) {
   488→   488→      throw new Error(`Account not found: ${line.accountCode}`)
   489→   489→    }
   490→   490→  }
   491→   491→
   492→   492→  // Create the journal entry with lines
   493→   493→  const entry = await client.journalEntry.create({
   494→   494→    data: {
   495→   495→      entryNo: template.entryNo,
   496→   496→      date: template.date,
   497→   497→      description: template.description,
   498→   498→      status: 'POSTED', // Auto-entries are posted immediately
   499→   499→      lines: {
   500→   500→        create: await Promise.all(
   501→   501→          template.lines.map(async (line) => {
   502→   502→            const account = await getAccountByCode(line.accountCode, tx)
   503→   503→            return {
   504→   504→              accountId: account!.id,
   505→   505→              costCenterId: line.costCenterId,
   506→   506→              debit: line.debit,
   507→   507→              credit: line.credit,
   508→   508→              description: line.description,
   509→   509→            };
   510→   510→          })
   511→   511→        ),
   512→   512→      },
   513→   513→    },
   514→   514→    include: { lines: true },
   515→   515→  })
   516→   516→
   517→   517→  return entry
   518→   518→}

  1315→  1315→export async function autoEntryVATDeclaration(data: {

  1318→  1318→  inputVat: number
  1319→  1319→  netVat: number // positive = payable, negative = refundable

  1340→  1340→    entryNo: `JE-VAT-${Date.now()}`,

  1377→  1377→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1378→  1378→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1379→  1379→  if (dateFrom || dateTo) {
  1380→  1380→    dateFilter.date = {
  1381→  1381→      ...(dateFrom && { gte: dateFrom }),
  1382→  1382→      ...(dateTo && { lte: dateTo }),
  1383→  1383→    }
  1384→  1384→  }
  1385→  1385→  const entries = await db.journalEntry.findMany({
  1386→  1386→    where: {
  1387→  1387→      status: 'POSTED',
  1388→  1388→      ...dateFilter,
  1389→  1389→    },
  1390→  1390→    include: {
  1391→  1391→      lines: {
  1392→  1392→        include: {
  1393→  1393→          account: true,
  1394→  1394→        },
  1395→  1395→      },
  1396→  1396→    },
  1397→  1397→  })
  1398→  1398→
  1399→  1399→  // Aggregate by account
  1400→  1400→  const accountBalances = new Map<string, {
  1401→  1401→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1402→  1402→    totalDebit: number
  1403→  1403→    totalCredit: number
  1404→  1404→  }>()
  1405→  1405→
  1406→  1406→  for (const entry of entries) {
  1407→  1407→    for (const line of entry.lines) {
  1408→  1408→      const key = line.accountId
  1409→  1409→      if (!accountBalances.has(key)) {
  1410→  1410→        accountBalances.set(key, {
  1411→  1411→          account: line.account,
  1412→  1412→          totalDebit: 0,
  1413→  1413→          totalCredit: 0,
  1414→  1414→        })
  1415→  1415→      }
  1416→  1416→      const bal = accountBalances.get(key)!
  1417→  1417→      bal.totalDebit += toNumber(line.debit)
  1418→  1418→      bal.totalCredit += toNumber(line.credit)
  1419→  1419→    }
  1420→  1420→  }
  1421→  1421→
  1422→  1422→  // Calculate net balance for each account
  1423→  1423→  const results = Array.from(accountBalances.values()).map(bal => {
  1424→  1424→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1425→  1425→    let netDebit = 0
  1426→  1426→    let netCredit = 0
  1427→  1427→
  1428→  1428→    if (normalBalance === 'DEBIT') {
  1429→  1429→      const net = bal.totalDebit - bal.totalCredit
  1430→  1430→      if (net >= 0) netDebit = net
  1431→  1431→      else netCredit = Math.abs(net)
  1432→  1432→    } else {
  1433→  1433→      const net = bal.totalCredit - bal.totalDebit
  1434→  1434→      if (net >= 0) netCredit = net
  1435→  1435→      else netDebit = Math.abs(net)
  1436→  1436→    }
  1437→  1437→
  1438→  1438→    return {
  1439→  1439→      ...bal,
  1440→  1440→      netDebit,
  1441→  1441→      netCredit,
  1442→  1442→    }
  1443→  1443→  })
  1444→  1444→
  1445→  1445→  // Sort by account code
  1446→  1446→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1447→  1447→
  1448→  1448→  return results
  1449→  1449→}

  1528→  1528→    }

  1530→  1530→    return {

   262→export async function ensureAccountExists(template: AccountTemplate, tx?: PrismaTransaction) {
   263→  const client = tx || db
   264→  const existing = await client.account.findUnique({ where: { code: template.code } })
   265→  if (existing) {
   266→    // Update existing account with new fields if they are missing
   267→    if (
   268→      existing.activityType !== (template.activityType || null) ||
   269→      existing.isSystem !== (template.isSystem || false) ||
   270→      existing.allowPosting !== (template.allowPosting || false) ||
   271→      existing.level !== (template.level || 0)
   272→    ) {
   273→      await client.account.update({
   274→        where: { code: template.code },
   275→        data: {
   276→          name: template.name,
   277→          nameAr: template.nameAr,
   278→          type: template.type,
   279→          activityType: template.activityType || null,
   280→          isSystem: template.isSystem || false,
   281→          allowPosting: template.allowPosting || false,
   282→          level: template.level || 0,
   283→        },
   284→      })
   285→    }
   286→    return client.account.findUnique({ where: { code: template.code } })
   287→  }
   288→
   289→  let parentId: string | undefined
   290→  if (template.parentId) {
   291→    const parent = await client.account.findUnique({ where: { code: template.parentId } })
   292→    if (parent) parentId = parent.id
   293→  }
   294→
   295→  return client.account.create({
   296→    data: {
   297→      code: template.code,
   298→      name: template.name,
   299→      nameAr: template.nameAr,
   300→      type: template.type,
   301→      parentId,
   302→      isActive: true,
   303→      activityType: template.activityType || null,
   304→      isSystem: template.isSystem || false,
   305→      allowPosting: template.allowPosting || false,
   306→      level: template.level || 0,
   307→    },
   308→  })
   309→}

   311→// ============ INITIALIZE CHART OF ACCOUNTS ============

   313→export async function initializeChartOfAccounts() {
   314→  let created = 0
   315→  let updated = 0
   316→
   317→  // Create parent accounts first, then children (sorted by code length then code)
   318→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   319→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   320→    return a.code.localeCompare(b.code)
   321→  })
   322→
   323→  for (const tmpl of sorted) {
   324→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   325→    if (existing) {
   326→      // Update existing account with new fields
   327→      await db.account.update({
   328→        where: { code: tmpl.code },
   329→        data: {
   330→          name: tmpl.name,
   331→          nameAr: tmpl.nameAr,
   332→          type: tmpl.type,
   333→          activityType: tmpl.activityType || null,
   334→          isSystem: tmpl.isSystem || false,
   335→          allowPosting: tmpl.allowPosting || false,
   336→          level: tmpl.level || 0,
   337→          isActive: true,
   338→        },
   339→      })
   340→      // Update parent reference if needed
   341→      if (tmpl.parentId) {
   342→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   343→        if (parent && existing.parentId !== parent.id) {
   344→          await db.account.update({
   345→            where: { code: tmpl.code },
   346→            data: { parentId: parent.id },
   347→          })
   348→        }
   349→      }
   350→      updated++
   351→    } else {
   352→      await ensureAccountExists(tmpl)
   353→      created++
   354→    }
   355→  }
   356→
   357→  const total = await db.account.count()
   358→  return { created, updated, total }
   359→}

   363→export async function createJournalEntry(template: JournalEntryTemplate, tx?: PrismaTransaction) {
   364→  const client = tx || db
   365→  // Validate: total debits must equal total credits
   366→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   367→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   368→
   369→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   370→    throw new Error(
   371→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   372→    )
   373→  }
   374→
   375→  // Ensure all referenced accounts exist
   376→  for (const line of template.lines) {
   377→    const account = await getAccountByCode(line.accountCode, tx)
   378→    if (!account) {
   379→      throw new Error(`Account not found: ${line.accountCode}`)
   380→    }
   381→  }
   382→
   383→  // Create the journal entry with lines
   384→  const entry = await client.journalEntry.create({
   385→    data: {
   386→      entryNo: template.entryNo,
   387→      date: template.date,
   388→      description: template.description,
   389→      status: 'POSTED', // Auto-entries are posted immediately
   390→      lines: {
   391→        create: await Promise.all(
   392→          template.lines.map(async (line) => {
   393→            const account = await getAccountByCode(line.accountCode, tx)
   394→            return {
   395→              accountId: account!.id,
   396→              costCenterId: line.costCenterId,
   397→              debit: line.debit,
   398→              credit: line.credit,
   399→              description: line.description,
   400→            };
   401→          })
   402→        ),
   403→      },
   404→    },
   405→    include: { lines: true },
   406→  })
   407→
   408→  return entry
   409→}

   414→/**

   416→ * Dr: Clients Receivable (1210) - totalAmount
   417→ * Cr: Progress Claims Revenue (6110) - subtotal
   418→ * Cr: VAT Payable (3200) - vatAmount

   420→export async function autoEntrySalesInvoice(data: {
   421→  invoiceNo: string
   422→  clientId: string
   423→  subtotal: number
   424→  vatRate: number
   425→  vatAmount: number
   426→  totalAmount: number
   427→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   428→  date: Date
   429→  projectId?: string
   430→  costCenterId?: string
   431→}, tx?: PrismaTransaction) {
   432→  // Determine revenue account based on invoice type
   433→  let revenueAccountCode: string
   434→  switch (data.invoiceType) {
   435→    case 'RENTAL':
   436→      revenueAccountCode = '6210' // Equipment Rental Revenue
   437→      break
   438→    case 'PROGRESS_CLAIM':
   439→      revenueAccountCode = '6110' // Progress Claims Revenue
   440→      break
   441→    default:
   442→      revenueAccountCode = '6340' // Service Revenue
   443→  }
   444→
   445→  const lines = [
   446→    { accountCode: '1210', debit: data.totalAmount, credit: 0, costCenterId: data.costCenterId }, // Clients Receivable
   447→    { accountCode: revenueAccountCode, debit: 0, credit: data.subtotal, costCenterId: data.costCenterId }, // Revenue
   448→  ]
   449→
   450→  // Add VAT line only if VAT > 0 (Output VAT)
   451→  if (data.vatAmount > 0) {
   452→    lines.push({ accountCode: '3110', debit: 0, credit: data.vatAmount }) // Output VAT (ضريبة مخرجات)
   453→  }
   454→
   455→  return createJournalEntry({
   456→    entryNo: `JE-SI-${Date.now()}`,
   457→    date: data.date,
   458→    description: `Sales Invoice ${data.invoiceNo}`,
   459→    descriptionAr: `فاتورة مبيعات ${data.invoiceNo}`,
   460→    lines,
   461→    sourceType: 'SALES_INVOICE',
   462→    sourceId: data.invoiceNo,
   463→  }, tx)
   464→}

  1142→/**
  1143→ * التخلص من أصل - Asset Disposal
  1144→ * Dr: Cash/Bank (1110/1120) - sale price
  1145→ * Cr: Asset account (2110/2120/2130/2140) - original cost (or net book value)
  1146→ * Dr/Cr: Gain/Loss on disposal (6310 gain / 8610 loss)
  1147→ */
  1148→export async function autoEntryAssetDisposal(data: {
  1149→  assetAccountCode: string // e.g., '2110', '2120', '2130', '2140'
  1150→  accumulatedDepAccountCode: string // e.g., '2210', '2220', '2230', '2240'
  1151→  originalCost: number
  1152→  accumulatedDepreciation: number
  1153→  salePrice: number
  1154→  date: Date
  1155→  receivedIn: 'TREASURY' | 'BANK'
  1156→}, tx?: PrismaTransaction) {
  1157→  const cashAccountCode = data.receivedIn === 'BANK' ? '1120' : '1110'
  1158→  const netBookValue = data.originalCost - data.accumulatedDepreciation
  1159→  const gainLoss = data.salePrice - netBookValue
  1160→
  1161→  const lines = [
  1162→    { accountCode: cashAccountCode, debit: data.salePrice, credit: 0 }, // Cash received
  1163→    { accountCode: data.accumulatedDepAccountCode, debit: data.accumulatedDepreciation, credit: 0 }, // Remove accumulated depreciation
  1164→    { accountCode: data.assetAccountCode, debit: 0, credit: data.originalCost }, // Remove asset
  1165→  ]
  1166→
  1167→  // If gain, credit gain account. If loss, debit loss account.
  1168→  if (gainLoss > 0) {
  1169→    lines.push({ accountCode: '6310', debit: 0, credit: gainLoss }) // Gain on Sale of Used Equipment
  1170→  } else if (gainLoss < 0) {
  1171→    lines.push({ accountCode: '8610', debit: Math.abs(gainLoss), credit: 0 }) // Loss on Asset Disposal
  1172→  }
  1173→
  1174→  return createJournalEntry({
  1175→    entryNo: `JE-DSP-${Date.now()}`,
  1176→    date: data.date,
  1177→    description: `Asset disposal - ${data.assetAccountCode}`,
  1178→    descriptionAr: `التخلص من أصل - ${data.assetAccountCode}`,
  1179→    lines,
  1180→    sourceType: 'ASSET_DISPOSAL',
  1181→    sourceId: `DSP-${Date.now()}`,
  1182→  }, tx)
  1183→}

  1185→/**
  1186→ * إقرار ضريبي - VAT Declaration
  1187→ * Dr: Output VAT (3110) - total output VAT
  1188→ * Cr: Input VAT (3120) - total input VAT
  1189→ * Cr: VAT Due (3130) - net VAT payable (if output > input)
  1190→ * OR
  1191→ * Dr: Input VAT (3120) - if input > output (refund scenario)
  1192→ * Cr: Output VAT (3110)
  1193→ * Dr: VAT Refund Receivable (1410) - if refund due
  1194→ */
  1195→export async function autoEntryVATDeclaration(data: {
  1196→  period: string
  1197→  outputVat: number
  1198→  inputVat: number
  1199→  netVat: number // positive = payable, negative = refundable
  1200→  date: Date
  1201→}, tx?: PrismaTransaction) {
  1202→  const lines: { accountCode: string; debit: number; credit: number }[] = []
  1203→
  1204→  // Close Output VAT (3110) - debit to zero it out
  1205→  lines.push({ accountCode: '3110', debit: data.outputVat, credit: 0 })
  1206→
  1207→  // Close Input VAT (3120) - credit to zero it out
  1208→  lines.push({ accountCode: '3120', debit: 0, credit: data.inputVat })
  1209→
  1210→  // Net VAT position
  1211→  if (data.netVat > 0) {
  1212→    // More output than input → owe VAT
  1213→    lines.push({ accountCode: '3130', debit: 0, credit: data.netVat }) // VAT Due
  1214→  } else if (data.netVat < 0) {
  1215→    // More input than output → refund due
  1216→    lines.push({ accountCode: '1410', debit: Math.abs(data.netVat), credit: 0 }) // VAT Refund Receivable
  1217→  }
  1218→
  1219→  return createJournalEntry({
  1220→    entryNo: `JE-VAT-${Date.now()}`,
  1221→    date: data.date,
  1222→    description: `VAT Declaration - ${data.period}`,
  1223→    descriptionAr: `إقرار ضريبي - ${data.period}`,
  1224→    lines,
  1225→    sourceType: 'VAT_DECLARATION',
  1226→    sourceId: `VAT-${data.period}`,
  1227→  }, tx)
  1228→}

  1257→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1258→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1259→  if (dateFrom || dateTo) {
  1260→    dateFilter.date = {
  1261→      ...(dateFrom && { gte: dateFrom }),
  1262→      ...(dateTo && { lte: dateTo }),
  1263→    }
  1264→  }
  1265→  const entries = await db.journalEntry.findMany({
  1266→    where: {
  1267→      status: 'POSTED',
  1268→      ...dateFilter,
  1269→    },
  1270→    include: {
  1271→      lines: {
  1272→        include: {
  1273→          account: true,
  1274→        },
  1275→      },
  1276→    },
  1277→  })
  1278→
  1279→  // Aggregate by account
  1280→  const accountBalances = new Map<string, {
  1281→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1282→    totalDebit: number
  1283→    totalCredit: number
  1284→  }>()
  1285→
  1286→  for (const entry of entries) {
  1287→    for (const line of entry.lines) {
  1288→      const key = line.accountId
  1289→      if (!accountBalances.has(key)) {
  1290→        accountBalances.set(key, {
  1291→          account: line.account,
  1292→          totalDebit: 0,
  1293→          totalCredit: 0,
  1294→        })
  1295→      }
  1296→      const bal = accountBalances.get(key)!
  1297→      bal.totalDebit += line.debit
  1298→      bal.totalCredit += line.credit
  1299→    }
  1300→  }
  1301→
  1302→  // Calculate net balance for each account
  1303→  const results = Array.from(accountBalances.values()).map(bal => {
  1304→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1305→    let netDebit = 0
  1306→    let netCredit = 0
  1307→
  1308→    if (normalBalance === 'DEBIT') {
  1309→      const net = bal.totalDebit - bal.totalCredit
  1310→      if (net >= 0) netDebit = net
  1311→      else netCredit = Math.abs(net)
  1312→    } else {
  1313→      const net = bal.totalCredit - bal.totalDebit
  1314→      if (net >= 0) netCredit = net
  1315→      else netDebit = Math.abs(net)
  1316→    }
  1317→
  1318→    return {
  1319→      ...bal,
  1320→      netDebit,
  1321→      netCredit,
  1322→    }
  1323→  })
  1324→
  1325→  // Sort by account code
  1326→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1327→
  1328→  return results
  1329→}

  1383→      ...(dateTo && { lte: dateTo }),

  1385→  }
  1386→  const lines = await db.journalLine.findMany({
  1387→    where: {
  1388→      accountId: account.id,

  1390→        status: 'POSTED',

  1394→    include: { journalEntry: true },
  1395→    orderBy: { journalEntry: { date: 'asc' } },
  1396→  })
  1397→
  1398→  let runningBalance = 0
  1399→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1400→
  1401→  return lines.map(line => {
  1402→    if (normalBalance === 'DEBIT') {
  1403→      runningBalance += line.debit - line.credit
  1404→    } else {
  1405→      runningBalance += line.credit - line.debit
  1406→    }
  1407→
  1408→    return {
  1409→      date: line.journalEntry.date,
  1410→      entryNo: line.journalEntry.entryNo,
  1411→      description: line.journalEntry.description,
  1412→      debit: line.debit,
  1413→      credit: line.credit,
  1414→      balance: runningBalance,
  1415→    }
  1416→  })

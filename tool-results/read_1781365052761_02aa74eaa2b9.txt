   373→// Helper function to get the correct salary account based on activity type
   374→export function getSalaryAccountCode(activityType: 'PROJECT' | 'RENTAL' | 'ADMIN' | 'EXECUTION' | 'GENERAL'): string {
   375→  switch (activityType) {
   376→    case 'PROJECT':
   377→    case 'EXECUTION':
   378→      return '8110' // رواتب مشاريع
   379→    case 'RENTAL':
   380→      return '8120' // رواتب تأجير
   381→    case 'ADMIN':
   382→    case 'GENERAL':
   383→    default:
   384→      return '8130' // رواتب إدارية
   385→  }
   386→}

   487→// ============ INITIALIZE CHART OF ACCOUNTS ============
   488→
   489→export async function initializeChartOfAccounts() {
   490→  let created = 0
   491→  let updated = 0
   492→
   493→  // Create parent accounts first, then children (sorted by code length then code)
   494→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   495→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   496→    return a.code.localeCompare(b.code)
   497→  })
   498→
   499→  for (const tmpl of sorted) {
   500→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   501→    if (existing) {
   502→      // Update existing account with new fields
   503→      await db.account.update({
   504→        where: { code: tmpl.code },
   505→        data: {
   506→          name: tmpl.name,
   507→          nameAr: tmpl.nameAr,
   508→          type: tmpl.type,
   509→          activityType: tmpl.activityType || null,
   510→          isSystem: tmpl.isSystem || false,
   511→          allowPosting: tmpl.allowPosting || false,
   512→          level: tmpl.level || 0,
   513→          isActive: true,
   514→        },
   515→      })
   516→      // Update parent reference if needed
   517→      if (tmpl.parentId) {
   518→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   519→        if (parent && existing.parentId !== parent.id) {
   520→          await db.account.update({
   521→            where: { code: tmpl.code },
   522→            data: { parentId: parent.id },
   523→          })
   524→        }
   525→      }
   526→      updated++
   527→    } else {
   528→      await ensureAccountExists(tmpl)
   529→      created++
   530→    }
   531→  }
   532→
   533→  const total = await db.account.count()
   534→  return { created, updated, total }
   535→}

   537→// ============ JOURNAL ENTRY CREATION ============
   538→
   539→export async function createJournalEntry(template: JournalEntryTemplate) {
   540→  // Validate: total debits must equal total credits
   541→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   542→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   543→
   544→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   545→    throw new Error(
   546→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   547→    )
   548→  }
   549→
   550→  // Ensure all referenced accounts exist
   551→  for (const line of template.lines) {
   552→    const account = await getAccountByCode(line.accountCode)
   553→    if (!account) {
   554→      throw new Error(`Account not found: ${line.accountCode}`)
   555→    }
   556→  }
   557→
   558→  // Create the journal entry with lines
   559→  const entry = await db.journalEntry.create({
   560→    data: {
   561→      entryNo: template.entryNo,
   562→      date: template.date,
   563→      description: template.description,
   564→      status: 'POSTED', // Auto-entries are posted immediately
   565→      lines: {
   566→        create: await Promise.all(
   567→          template.lines.map(async (line) => {
   568→            const account = await getAccountByCode(line.accountCode)
   569→            return {
   570→              accountId: account!.id,
   571→              costCenterId: line.costCenterId,
   572→              debit: line.debit,
   573→              credit: line.credit,
   574→              description: line.description,
   575→            }
   576→          })
   577→        ),
   578→      },
   579→    },
   580→    include: { lines: true },
   581→  })
   582→
   583→  return entry
   584→}

   595→export async function autoEntrySalesInvoice(data: {
   596→  invoiceNo: string
   597→  clientId: string
   598→  subtotal: number
   599→  vatRate: number
   600→  vatAmount: number
   601→  totalAmount: number
   602→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   603→  date: Date
   604→  projectId?: string
   605→  costCenterId?: string
   606→}) {
   607→  // Determine revenue and receivable accounts based on invoice type
   608→  let revenueAccountCode: string
   609→  let receivableAccountCode: string
   610→  switch (data.invoiceType) {
   611→    case 'RENTAL':
   612→      revenueAccountCode = '6210' // Equipment Rental Revenue
   613→      receivableAccountCode = '1220' // ذمم عملاء التأجير
   614→      break
   615→    case 'PROGRESS_CLAIM':
   616→      revenueAccountCode = '6110' // Progress Claims Revenue
   617→      receivableAccountCode = '1210' // ذمم عملاء المشاريع
   618→      break
   619→    default:
   620→      revenueAccountCode = '6340' // Service Revenue
   621→      receivableAccountCode = '1210' // ذمم عملاء المشاريع
   622→  }
   623→
   624→  const lines = [
   625→    { accountCode: receivableAccountCode, debit: data.totalAmount, credit: 0, costCenterId: data.costCenterId },
   626→    { accountCode: revenueAccountCode, debit: 0, credit: data.subtotal, costCenterId: data.costCenterId },
   627→  ]
   628→
   629→  // Add VAT line only if VAT > 0 (Output VAT)
   630→  if (data.vatAmount > 0) {
   631→    lines.push({ accountCode: '3110', debit: 0, credit: data.vatAmount }) // Output VAT (ضريبة مخرجات)
   632→  }
   633→
   634→  return createJournalEntry({
   635→    entryNo: `JE-SI-${Date.now()}`,
   636→    date: data.date,
   637→    description: `Sales Invoice ${data.invoiceNo}`,
   638→    descriptionAr: `فاتورة مبيعات ${data.invoiceNo}`,
   639→    lines,
   640→    sourceType: 'SALES_INVOICE',
   641→    sourceId: data.invoiceNo,
   642→  })
   643→}

  1337→ * Cr: Asset account (2110/2120/2130/2140) - original cost (or net book value)
  1338→ * Dr/Cr: Gain/Loss on disposal (6310 gain / 8610 loss)
  1339→ */
  1340→export async function autoEntryAssetDisposal(data: {
  1341→  assetAccountCode: string // e.g., '2110', '2120', '2130', '2140'
  1342→  accumulatedDepAccountCode: string // e.g., '2210', '2220', '2230', '2240'

  1353→  const lines = [
  1354→    { accountCode: cashAccountCode, debit: data.salePrice, credit: 0 }, // Cash received
  1355→    { accountCode: data.accumulatedDepAccountCode, debit: data.accumulatedDepreciation, credit: 0 }, // Remove accumulated depreciation

  1362→  } else if (gainLoss < 0) {
  1363→    lines.push({ accountCode: '8610', debit: Math.abs(gainLoss), credit: 0 }) // Loss on Asset Disposal
  1364→  }
  1365→
  1366→  return createJournalEntry({
  1367→    entryNo: `JE-DSP-${Date.now()}`,
  1368→    date: data.date,
  1369→    description: `Asset disposal - ${data.assetAccountCode}`,
  1370→    descriptionAr: `التخلص من أصل - ${data.assetAccountCode}`,
  1371→    lines,
  1372→    sourceType: 'ASSET_DISPOSAL',
  1373→    sourceId: `DSP-${Date.now()}`,
  1374→  })

  1477→// ============ VAT DECLARATION & PAYMENT ============
  1478→
  1479→/**
  1480→ * إقرار ضريبي - VAT Declaration
  1481→ * Dr: Output VAT (3110) - total output VAT
  1482→ * Cr: Input VAT (3120) - total input VAT
  1483→ * Cr: VAT Due (3130) - net VAT payable (if output > input)
  1484→ * OR
  1485→ * Dr: Input VAT (3120) - if input > output (refund scenario)
  1486→ * Cr: Output VAT (3110)
  1487→ * Dr: VAT Refund Receivable (1410) - if refund due
  1488→ */
  1489→export async function autoEntryVATDeclaration(data: {
  1490→  period: string
  1491→  outputVat: number
  1492→  inputVat: number
  1493→  netVat: number // positive = payable, negative = refundable
  1494→  date: Date
  1495→}) {
  1496→  const lines: { accountCode: string; debit: number; credit: number }[] = []
  1497→
  1498→  // Close Output VAT (3110) - debit to zero it out
  1499→  lines.push({ accountCode: '3110', debit: data.outputVat, credit: 0 })
  1500→
  1501→  // Close Input VAT (3120) - credit to zero it out
  1502→  lines.push({ accountCode: '3120', debit: 0, credit: data.inputVat })
  1503→
  1504→  // Net VAT position
  1505→  if (data.netVat > 0) {
  1506→    // More output than input → owe VAT
  1507→    lines.push({ accountCode: '3130', debit: 0, credit: data.netVat }) // VAT Due
  1508→  } else if (data.netVat < 0) {
  1509→    // More input than output → refund due
  1510→    lines.push({ accountCode: '1410', debit: Math.abs(data.netVat), credit: 0 }) // VAT Refund Receivable
  1511→  }
  1512→
  1513→  return createJournalEntry({
  1514→    entryNo: `JE-VAT-${Date.now()}`,
  1515→    date: data.date,
  1516→    description: `VAT Declaration - ${data.period}`,
  1517→    descriptionAr: `إقرار ضريبي - ${data.period}`,
  1518→    lines,
  1519→    sourceType: 'VAT_DECLARATION',
  1520→    sourceId: `VAT-${data.period}`,
  1521→  })
  1522→}
  1523→

  1525→ * سداد الضريبة - VAT Payment
  1526→ * Dr: VAT Due (3130)
  1527→ * Cr: Bank (1120)
  1528→ */
  1529→export async function autoEntryVATPayment(data: {
  1530→  period: string
  1531→  amount: number
  1532→  date: Date
  1533→  reference?: string
  1534→}) {

  1539→    descriptionAr: `سداد ضريبي - ${data.period}${data.reference ? ` - مرجع: ${data.reference}` : ''}`,
  1540→    lines: [
  1541→      { accountCode: '3130', debit: data.amount, credit: 0 }, // VAT Due (ضريبة مستحقة)
  1542→      { accountCode: '1121', debit: 0, credit: data.amount }, // Bank - بنك الراجحي (default)
  1543→    ],
  1544→    sourceType: 'VAT_PAYMENT',

  1551→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1552→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1553→  if (dateFrom || dateTo) {
  1554→    dateFilter.date = {
  1555→      ...(dateFrom && { gte: dateFrom }),
  1556→      ...(dateTo && { lte: dateTo }),
  1557→    }
  1558→  }
  1559→  const entries = await db.journalEntry.findMany({
  1560→    where: {
  1561→      status: 'POSTED',
  1562→      ...dateFilter,
  1563→    },
  1564→    include: {
  1565→      lines: {
  1566→        include: {
  1567→          account: true,
  1568→        },
  1569→      },
  1570→    },
  1571→  })
  1572→
  1573→  // Aggregate by account
  1574→  const accountBalances = new Map<string, {
  1575→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1576→    totalDebit: number
  1577→    totalCredit: number
  1578→  }>()
  1579→
  1580→  for (const entry of entries) {
  1581→    for (const line of entry.lines) {
  1582→      const key = line.accountId
  1583→      if (!accountBalances.has(key)) {
  1584→        accountBalances.set(key, {
  1585→          account: line.account,
  1586→          totalDebit: 0,
  1587→          totalCredit: 0,
  1588→        })
  1589→      }
  1590→      const bal = accountBalances.get(key)!
  1591→      bal.totalDebit += line.debit
  1592→      bal.totalCredit += line.credit
  1593→    }
  1594→  }
  1595→
  1596→  // Calculate net balance for each account
  1597→  const results = Array.from(accountBalances.values()).map(bal => {
  1598→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1599→    let netDebit = 0
  1600→    let netCredit = 0
  1601→
  1602→    if (normalBalance === 'DEBIT') {
  1603→      const net = bal.totalDebit - bal.totalCredit
  1604→      if (net >= 0) netDebit = net
  1605→      else netCredit = Math.abs(net)
  1606→    } else {
  1607→      const net = bal.totalCredit - bal.totalDebit
  1608→      if (net >= 0) netCredit = net
  1609→      else netDebit = Math.abs(net)
  1610→    }
  1611→
  1612→    return {
  1613→      ...bal,
  1614→      netDebit,
  1615→      netCredit,
  1616→    }
  1617→  })
  1618→
  1619→  // Sort by account code
  1620→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1621→
  1622→  return results
  1623→}

  1627→export async function getAccountBalance(accountCode: string): Promise<number> {
  1628→  const account = await getAccountByCode(accountCode)
  1629→  if (!account) return 0
  1630→
  1631→  const lines = await db.journalLine.findMany({
  1632→    where: {
  1633→      accountId: account.id,
  1634→      journalEntry: { status: 'POSTED' },
  1635→    },
  1636→  })
  1637→
  1638→  const totalDebit = lines.reduce((sum, l) => sum + l.debit, 0)
  1639→  const totalCredit = lines.reduce((sum, l) => sum + l.credit, 0)
  1640→
  1641→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1642→
  1643→  return normalBalance === 'DEBIT'
  1644→    ? totalDebit - totalCredit
  1645→    : totalCredit - totalDebit
  1646→}

  1648→/**
  1649→ * Get account balance including all child accounts (recursive)
  1650→ * Used for parent accounts like 1120 البنوك which now have sub-accounts 1121-1124
  1651→ */
  1652→export async function getAccountBalanceWithChildren(accountCode: string): Promise<number> {
  1653→  const account = await getAccountByCode(accountCode)
  1654→  if (!account) return 0
  1655→
  1656→  // Get direct balance of this account
  1657→  let totalBalance = await getAccountBalance(accountCode)
  1658→
  1659→  // Find all child accounts
  1660→  const children = await db.account.findMany({
  1661→    where: { parentId: account.id, isActive: true },
  1662→    select: { code: true },
  1663→  })
  1664→
  1665→  // Recursively add child balances
  1666→  for (const child of children) {
  1667→    totalBalance += await getAccountBalanceWithChildren(child.code)
  1668→  }
  1669→
  1670→  return totalBalance
  1671→}

  1673→/**
  1674→ * Get total balance for a group of account codes

  1677→export async function getAccountsBalanceSum(accountCodes: string[]): Promise<number> {

  1687→export async function getGeneralLedger(accountCode: string, dateFrom?: Date, dateTo?: Date) {
  1688→  const account = await getAccountByCode(accountCode)
  1689→  if (!account) return []
  1690→
  1691→  const glDateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1692→  if (dateFrom || dateTo) {
  1693→    glDateFilter.date = {
  1694→      ...(dateFrom && { gte: dateFrom }),
  1695→      ...(dateTo && { lte: dateTo }),
  1696→    }
  1697→  }
  1698→  const lines = await db.journalLine.findMany({
  1699→    where: {
  1700→      accountId: account.id,
  1701→      journalEntry: {
  1702→        status: 'POSTED',
  1703→        ...glDateFilter,
  1704→      },
  1705→    },
  1706→    include: { journalEntry: true },
  1707→    orderBy: { journalEntry: { date: 'asc' } },
  1708→  })
  1709→
  1710→  let runningBalance = 0
  1711→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1712→
  1713→  return lines.map(line => {
  1714→    if (normalBalance === 'DEBIT') {
  1715→      runningBalance += line.debit - line.credit
  1716→    } else {
  1717→      runningBalance += line.credit - line.debit
  1718→    }
  1719→
  1720→    return {
  1721→      date: line.journalEntry.date,
  1722→      entryNo: line.journalEntry.entryNo,
  1723→      description: line.journalEntry.description,
  1724→      debit: line.debit,
  1725→      credit: line.credit,
  1726→      balance: runningBalance,
  1727→    }
  1728→  })
  1729→}

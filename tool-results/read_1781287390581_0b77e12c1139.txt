   290→   290→export async function resolveCostCenterId(projectId: string | undefined | null): Promise<string | undefined> {

   293→   293→  if (!project) return undefined
   294→   294→  const costCenter = await db.costCenter.findFirst({ where: { code: project.code } })
   295→   295→  return costCenter?.id

   348→   348→export async function initializeChartOfAccounts() {
   349→   349→  let created = 0
   350→   350→  let updated = 0
   351→   351→
   352→   352→  // Create parent accounts first, then children (sorted by code length then code)
   353→   353→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   354→   354→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   355→   355→    return a.code.localeCompare(b.code)
   356→   356→  })
   357→   357→
   358→   358→  for (const tmpl of sorted) {
   359→   359→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   360→   360→    if (existing) {
   361→   361→      // Update existing account with new fields
   362→   362→      await db.account.update({
   363→   363→        where: { code: tmpl.code },
   364→   364→        data: {
   365→   365→          name: tmpl.name,
   366→   366→          nameAr: tmpl.nameAr,
   367→   367→          type: tmpl.type,
   368→   368→          activityType: tmpl.activityType || null,
   369→   369→          isSystem: tmpl.isSystem || false,
   370→   370→          allowPosting: tmpl.allowPosting || false,
   371→   371→          level: tmpl.level || 0,
   372→   372→          isActive: true,
   373→   373→        },
   374→   374→      })
   375→   375→      // Update parent reference if needed
   376→   376→      if (tmpl.parentId) {
   377→   377→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   378→   378→        if (parent && existing.parentId !== parent.id) {
   379→   379→          await db.account.update({
   380→   380→            where: { code: tmpl.code },
   381→   381→            data: { parentId: parent.id },
   382→   382→          })
   383→   383→        }
   384→   384→      }
   385→   385→      updated++
   386→   386→    } else {
   387→   387→      await ensureAccountExists(tmpl)
   388→   388→      created++
   389→   389→    }
   390→   390→  }
   391→   391→

   393→   393→  return { created, updated, total }

   396→   396→// ============ JOURNAL ENTRY CREATION ============

   398→   398→export async function createJournalEntry(template: JournalEntryTemplate) {
   399→   399→  // Validate: total debits must equal total credits
   400→   400→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   401→   401→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   402→   402→
   403→   403→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   404→   404→    throw new Error(
   405→   405→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   406→   406→    )
   407→   407→  }
   408→   408→
   409→   409→  // Ensure all referenced accounts exist
   410→   410→  for (const line of template.lines) {
   411→   411→    const account = await getAccountByCode(line.accountCode)
   412→   412→    if (!account) {
   413→   413→      throw new Error(`Account not found: ${line.accountCode}`)
   414→   414→    }
   415→   415→  }
   416→   416→
   417→   417→  // Create the journal entry with lines
   418→   418→  const entry = await db.journalEntry.create({
   419→   419→    data: {
   420→   420→      entryNo: template.entryNo,
   421→   421→      date: template.date,
   422→   422→      description: template.description,
   423→   423→      status: 'POSTED', // Auto-entries are posted immediately
   424→   424→      lines: {
   425→   425→        create: await Promise.all(
   426→   426→          template.lines.map(async (line) => {
   427→   427→            const account = await getAccountByCode(line.accountCode)
   428→   428→            return {
   429→   429→              accountId: account!.id,
   430→   430→              costCenterId: line.costCenterId,
   431→   431→              debit: line.debit,
   432→   432→              credit: line.credit,
   433→   433→              description: line.description,
   434→   434→            }
   435→   435→          })
   436→   436→        ),
   437→   437→      },
   438→   438→    },
   439→   439→    include: { lines: true },
   440→   440→  })
   441→   441→
   442→   442→  return entry
   443→   443→}

  1229→  1229→  const cashAccountCode = data.receivedIn === 'BANK' ? '1120' : '1110'
  1230→  1230→  const netBookValue = data.originalCost - data.accumulatedDepreciation
  1231→  1231→  const gainLoss = data.salePrice - netBookValue
  1232→  1232→
  1233→  1233→  const lines = [
  1234→  1234→    { accountCode: cashAccountCode, debit: data.salePrice, credit: 0 }, // Cash received
  1235→  1235→    { accountCode: data.accumulatedDepAccountCode, debit: data.accumulatedDepreciation, credit: 0 }, // Remove accumulated depreciation
  1236→  1236→    { accountCode: data.assetAccountCode, debit: 0, credit: data.originalCost }, // Remove asset
  1237→  1237→  ]
  1238→  1238→
  1239→  1239→  // If gain, credit gain account. If loss, debit loss account.
  1240→  1240→  if (gainLoss > 0) {
  1241→  1241→    lines.push({ accountCode: '6310', debit: 0, credit: gainLoss }) // Gain on Sale of Used Equipment
  1242→  1242→  } else if (gainLoss < 0) {
  1243→  1243→    lines.push({ accountCode: '8610', debit: Math.abs(gainLoss), credit: 0 }) // Loss on Asset Disposal
  1244→  1244→  }
  1245→  1245→
  1246→  1246→  return createJournalEntry({
  1247→  1247→    entryNo: `JE-DSP-${Date.now()}`,
  1248→  1248→    date: data.date,
  1249→  1249→    description: `Asset disposal - ${data.assetAccountCode}`,
  1250→  1250→    descriptionAr: `التخلص من أصل - ${data.assetAccountCode}`,
  1251→  1251→    lines,
  1252→  1252→    sourceType: 'ASSET_DISPOSAL',
  1253→  1253→    sourceId: `DSP-${Date.now()}`,
  1254→  1254→  })
  1255→  1255→}

  1257→  1257→// ============ VAT DECLARATION & PAYMENT ============
  1258→  1258→
  1259→  1259→/**
  1260→  1260→ * إقرار ضريبي - VAT Declaration
  1261→  1261→ * Dr: Output VAT (3110) - total output VAT
  1262→  1262→ * Cr: Input VAT (3120) - total input VAT
  1263→  1263→ * Cr: VAT Due (3130) - net VAT payable (if output > input)
  1264→  1264→ * OR
  1265→  1265→ * Dr: Input VAT (3120) - if input > output (refund scenario)
  1266→  1266→ * Cr: Output VAT (3110)
  1267→  1267→ * Dr: VAT Refund Receivable (1410) - if refund due
  1268→  1268→ */
  1269→  1269→export async function autoEntryVATDeclaration(data: {
  1270→  1270→  period: string
  1271→  1271→  outputVat: number
  1272→  1272→  inputVat: number
  1273→  1273→  netVat: number // positive = payable, negative = refundable
  1274→  1274→  date: Date
  1275→  1275→}) {
  1276→  1276→  const lines: { accountCode: string; debit: number; credit: number }[] = []
  1277→  1277→
  1278→  1278→  // Close Output VAT (3110) - debit to zero it out
  1279→  1279→  lines.push({ accountCode: '3110', debit: data.outputVat, credit: 0 })
  1280→  1280→
  1281→  1281→  // Close Input VAT (3120) - credit to zero it out
  1282→  1282→  lines.push({ accountCode: '3120', debit: 0, credit: data.inputVat })
  1283→  1283→
  1284→  1284→  // Net VAT position
  1285→  1285→  if (data.netVat > 0) {
  1286→  1286→    // More output than input → owe VAT
  1287→  1287→    lines.push({ accountCode: '3130', debit: 0, credit: data.netVat }) // VAT Due
  1288→  1288→  } else if (data.netVat < 0) {
  1289→  1289→    // More input than output → refund due
  1290→  1290→    lines.push({ accountCode: '1410', debit: Math.abs(data.netVat), credit: 0 }) // VAT Refund Receivable
  1291→  1291→  }
  1292→  1292→
  1293→  1293→  return createJournalEntry({
  1294→  1294→    entryNo: `JE-VAT-${Date.now()}`,
  1295→  1295→    date: data.date,
  1296→  1296→    description: `VAT Declaration - ${data.period}`,
  1297→  1297→    descriptionAr: `إقرار ضريبي - ${data.period}`,
  1298→  1298→    lines,
  1299→  1299→    sourceType: 'VAT_DECLARATION',
  1300→  1300→    sourceId: `VAT-${data.period}`,
  1301→  1301→  })
  1302→  1302→}
  1303→  1303→
  1304→  1304→/**
  1305→  1305→ * سداد الضريبة - VAT Payment
  1306→  1306→ * Dr: VAT Due (3130)
  1307→  1307→ * Cr: Bank (1120)
  1308→  1308→ */
  1309→  1309→export async function autoEntryVATPayment(data: {
  1310→  1310→  period: string
  1311→  1311→  amount: number
  1312→  1312→  date: Date
  1313→  1313→  reference?: string
  1314→  1314→}) {
  1315→  1315→  return createJournalEntry({
  1316→  1316→    entryNo: `JE-VTP-${Date.now()}`,
  1317→  1317→    date: data.date,
  1318→  1318→    description: `VAT Payment - ${data.period}${data.reference ? ` - Ref: ${data.reference}` : ''}`,
  1319→  1319→    descriptionAr: `سداد ضريبي - ${data.period}${data.reference ? ` - مرجع: ${data.reference}` : ''}`,
  1320→  1320→    lines: [
  1321→  1321→      { accountCode: '3130', debit: data.amount, credit: 0 }, // VAT Due (ضريبة مستحقة)
  1322→  1322→      { accountCode: '1120', debit: 0, credit: data.amount }, // Bank (البنك)
  1323→  1323→    ],
  1324→  1324→    sourceType: 'VAT_PAYMENT',
  1325→  1325→    sourceId: `VTP-${data.period}`,
  1326→  1326→  })
  1327→  1327→}

  1331→  1331→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1332→  1332→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1333→  1333→  if (dateFrom || dateTo) {
  1334→  1334→    dateFilter.date = {
  1335→  1335→      ...(dateFrom && { gte: dateFrom }),
  1336→  1336→      ...(dateTo && { lte: dateTo }),
  1337→  1337→    }
  1338→  1338→  }
  1339→  1339→  const entries = await db.journalEntry.findMany({
  1340→  1340→    where: {
  1341→  1341→      status: 'POSTED',
  1342→  1342→      ...dateFilter,
  1343→  1343→    },
  1344→  1344→    include: {
  1345→  1345→      lines: {
  1346→  1346→        include: {
  1347→  1347→          account: true,
  1348→  1348→        },
  1349→  1349→      },
  1350→  1350→    },
  1351→  1351→  })
  1352→  1352→
  1353→  1353→  // Aggregate by account
  1354→  1354→  const accountBalances = new Map<string, {
  1355→  1355→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1356→  1356→    totalDebit: number
  1357→  1357→    totalCredit: number
  1358→  1358→  }>()
  1359→  1359→
  1360→  1360→  for (const entry of entries) {
  1361→  1361→    for (const line of entry.lines) {
  1362→  1362→      const key = line.accountId
  1363→  1363→      if (!accountBalances.has(key)) {
  1364→  1364→        accountBalances.set(key, {
  1365→  1365→          account: line.account,
  1366→  1366→          totalDebit: 0,
  1367→  1367→          totalCredit: 0,
  1368→  1368→        })
  1369→  1369→      }
  1370→  1370→      const bal = accountBalances.get(key)!
  1371→  1371→      bal.totalDebit += line.debit
  1372→  1372→      bal.totalCredit += line.credit
  1373→  1373→    }
  1374→  1374→  }
  1375→  1375→
  1376→  1376→  // Calculate net balance for each account
  1377→  1377→  const results = Array.from(accountBalances.values()).map(bal => {
  1378→  1378→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1379→  1379→    let netDebit = 0
  1380→  1380→    let netCredit = 0
  1381→  1381→
  1382→  1382→    if (normalBalance === 'DEBIT') {
  1383→  1383→      const net = bal.totalDebit - bal.totalCredit
  1384→  1384→      if (net >= 0) netDebit = net
  1385→  1385→      else netCredit = Math.abs(net)
  1386→  1386→    } else {
  1387→  1387→      const net = bal.totalCredit - bal.totalDebit
  1388→  1388→      if (net >= 0) netCredit = net
  1389→  1389→      else netDebit = Math.abs(net)
  1390→  1390→    }
  1391→  1391→
  1392→  1392→    return {
  1393→  1393→      ...bal,
  1394→  1394→      netDebit,
  1395→  1395→      netCredit,
  1396→  1396→    }
  1397→  1397→  })
  1398→  1398→
  1399→  1399→  // Sort by account code
  1400→  1400→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1401→  1401→
  1402→  1402→  return results
  1403→  1403→}

  1435→  1435→  if (dateFrom || dateTo) {
  1436→  1436→    glDateFilter.date = {
  1437→  1437→      ...(dateFrom && { gte: dateFrom }),
  1438→  1438→      ...(dateTo && { lte: dateTo }),

  1461→  1461→    }
  1462→  1462→
  1463→  1463→    return {
  1464→  1464→      date: line.journalEntry.date,
  1465→  1465→      entryNo: line.journalEntry.entryNo,
  1466→  1466→      description: line.journalEntry.description,
  1467→  1467→      debit: line.debit,
  1468→  1468→      credit: line.credit,

  1470→  1470→    }
  1471→  1471→  })
  1472→  1472→}

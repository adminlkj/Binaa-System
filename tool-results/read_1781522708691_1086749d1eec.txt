   280→// ============ Tab 1: Chart of Accounts ============
   281→function ChartOfAccountsTab({ accounts, isLoading, onInitialize, onReInitialize, isInitializing }: {
   282→  accounts: Account[]; isLoading: boolean; onInitialize: () => void; onReInitialize: () => void; isInitializing: boolean
   283→}) {
   284→  const { lang } = useAppStore()
   285→  const [activityFilter, setActivityFilter] = useState<string>('all')
   286→  const [typeFilter, setTypeFilter] = useState<string>('all')
   287→  const [searchTerm, setSearchTerm] = useState('')
   288→  const [expandedIds, setExpandedIds] = useState<Set<string>>(() => {
   289→    const initial = new Set<string>()
   290→    accounts.forEach(a => { if (!a.parentId || a.level === 0) initial.add(a.id) })
   291→    return initial
   292→  })
   293→  const [selectedAccount, setSelectedAccount] = useState<Account | null>(null)
   294→  const [detailOpen, setDetailOpen] = useState(false)
   295→
   296→  const allParentIds = useMemo(() => {
   297→    const ids = new Set<string>()
   298→    accounts.forEach(a => { if (a.children && a.children.length > 0) ids.add(a.id) })
   299→    return ids
   300→  }, [accounts])
   301→
   302→  const rootAccounts = useMemo(() => {
   303→    return accounts.filter(a => !a.parentId).sort((a, b) => a.code.localeCompare(b.code))
   304→  }, [accounts])
   305→
   306→  const childMap = useMemo(() => {
   307→    const map = new Map<string, Account[]>()
   308→    accounts.forEach(a => { if (a.parentId) { const siblings = map.get(a.parentId) || []; siblings.push(a); map.set(a.parentId, siblings) } })
   309→    map.forEach((children) => children.sort((a, b) => a.code.localeCompare(b.code)))
   310→    return map
   311→  }, [accounts])
   312→
   313→  const filteredAccounts = useMemo(() => {
   314→    let filtered = accounts
   315→    if (activityFilter !== 'all') {
   316→      if (activityFilter === 'BOTH') filtered = filtered.filter(a => a.activityType === 'BOTH' || !a.activityType)
   317→      else filtered = filtered.filter(a => a.activityType === activityFilter)
   318→    }
   319→    if (typeFilter !== 'all') filtered = filtered.filter(a => a.type === typeFilter)
   320→    if (searchTerm) {
   321→      const term = searchTerm.toLowerCase()
   322→      filtered = filtered.filter(a => a.code.toLowerCase().includes(term) || a.name.toLowerCase().includes(term) || (a.nameAr && a.nameAr.toLowerCase().includes(term)))
   323→    }
   324→    return filtered
   325→  }, [accounts, activityFilter, typeFilter, searchTerm])
   326→
   327→  const flatAccounts = useMemo(() => {
   328→    function flatten(roots: Account[], level: number): (Account & { displayLevel: number })[] {
   329→      const result: (Account & { displayLevel: number })[] = []
   330→      for (const root of roots) {
   331→        if (!filteredAccounts.find(a => a.id === root.id)) continue
   332→        result.push({ ...root, displayLevel: level })
   333→        const children = childMap.get(root.id) || []
   334→        if (children.length > 0 && expandedIds.has(root.id)) result.push(...flatten(children, level + 1))
   335→      }
   336→      return result
   337→    }
   338→    return flatten(rootAccounts, 0)
   339→  }, [rootAccounts, filteredAccounts, childMap, expandedIds])
   340→
   341→  const toggleExpand = useCallback((id: string) => {
   342→    setExpandedIds(prev => { const next = new Set(prev); if (next.has(id)) next.delete(id); else next.add(id); return next })
   343→  }, [])
   344→
   345→  const typeSummary = useMemo(() => { const s: Record<string, number> = {}; accounts.forEach(a => { s[a.type] = (s[a.type] || 0) + 1 }); return s }, [accounts])
   346→
   347→  if (isLoading) return <TableSkeleton />
   348→  if (accounts.length === 0) {
   349→    return (
   350→      <Card className="border-emerald-200 bg-emerald-50">
   351→        <CardContent className="p-6 flex flex-col items-center gap-4 text-center">
   352→          <Database className="size-12 text-emerald-500" />
   353→          <div>
   354→            <h3 className="text-lg font-semibold text-emerald-800">{t('لا توجد حسابات', 'No Accounts Found', lang)}</h3>
   355→            <p className="text-sm text-emerald-600 mt-1">{t('قم بتهيئة دليل الحسابات الافتراضي', 'Initialize the default chart of accounts', lang)}</p>
   356→          </div>
   357→          <Button onClick={onInitialize} disabled={isInitializing} className="gap-2 bg-emerald-600 hover:bg-emerald-700">
   358→            {isInitializing ? <RefreshCw className="size-4 animate-spin" /> : <PlusCircle className="size-4" />}
   359→            {isInitializing ? t('جاري التهيئة...', 'Initializing...', lang) : t('تهيئة دليل الحسابات', 'Initialize Chart of Accounts', lang)}
   360→          </Button>
   361→        </CardContent>
   362→      </Card>
   363→    )
   364→  }
   365→
   366→  return (
   367→    <div className="space-y-4">
   368→      {/* Summary */}
   369→      <div className="grid grid-cols-2 sm:grid-cols-5 gap-2">
   370→        {Object.entries(typeSummary).map(([type, count]) => {
   371→          const cfg = typeConfig[type]
   372→          if (!cfg) return null
   373→          return <Card key={type} className={`${cfg.bg} border-0`}><CardContent className="p-3 text-center"><p className={`text-xs ${cfg.color}`}>{cfg.label[lang]}</p><p className={`text-lg font-bold ${cfg.color}`}>{count}</p></CardContent></Card>
   374→        })}
   375→      </div>
   376→
   377→      {/* Filters */}
   378→      <Card className="bg-gray-50/50">
   379→        <CardContent className="p-4">
   380→          <div className="flex flex-col sm:flex-row gap-3 items-end">
   381→            <div className="space-y-1 flex-1 min-w-[120px]">
   382→              <Label className="text-xs">{t('النشاط', 'Activity', lang)}</Label>
   383→              <Select value={activityFilter} onValueChange={setActivityFilter}>
   384→                <SelectTrigger className="h-9"><SelectValue /></SelectTrigger>
   385→                <SelectContent>

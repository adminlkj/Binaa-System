   240→// ============ Status Badge for Contracts ============
   241→function ContractStatusBadge({ status, lang }: { status: string; lang: 'ar' | 'en' }) {
   242→  const config = statusOptions.find(s => s.value === status)
   243→  if (!config) return <StatusBadge status={status} lang={lang} />
   244→  return (
   245→    <span className={`inline-flex items-center rounded-full px-2.5 py-0.5 text-xs font-medium ${config.color}`}>
   246→      {config.label[lang]}
   247→    </span>
   248→  )
   249→}

   279→// ============ Contract Form Page ============
   280→function ContractFormPage({
   281→  form, setForm, onSubmit, onBack, isEdit, isLoading, clients, projects,
   282→}: {
   283→  form: ContractFormData; setForm: React.Dispatch<React.SetStateAction<ContractFormData>>
   284→  onSubmit: (e: React.FormEvent) => void; onBack: () => void
   285→  isEdit: boolean; isLoading: boolean
   286→  clients: ClientOption[]; projects: ProjectOption[]
   287→}) {
   288→  const { lang } = useAppStore()
   289→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   290→
   291→  const val = parseFloat(form.value) || 0
   292→  const rate = parseFloat(form.vatRate) || 0
   293→  const vatAmount = Math.round(val * rate * 100) / 100
   294→  const totalValue = Math.round((val + vatAmount) * 100) / 100
   295→
   296→  // Filter projects by selected client
   297→  const filteredProjects = form.clientId
   298→    ? projects.filter(p => !p.clientId || p.clientId === form.clientId)
   299→    : projects
   300→
   301→  return (
   302→    <div className="space-y-6">
   303→      <div className="flex items-center gap-3">
   304→        <Button variant="outline" size="icon" onClick={onBack}>
   305→          <ArrowRight className="size-4" />
   306→        </Button>
   307→        <div className="flex-1">
   308→          <h1 className="text-2xl font-bold text-gray-900">
   309→            {isEdit ? t(labels.editContract.ar, labels.editContract.en) : t(labels.newContract.ar, labels.newContract.en)}
   310→          </h1>
   311→        </div>
   312→      </div>
   313→
   314→      <form onSubmit={onSubmit} className="space-y-6">
   315→        {/* القسم الأول: بيانات العقد الأساسية */}
   316→        <Card>
   317→          <CardContent className="pt-6">
   318→            <SectionHeader icon={FileCheck} title={labels.section1} lang={lang} />
   319→            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
   320→              <div className="space-y-2">
   321→                <Label>{t(labels.contractNo.ar, labels.contractNo.en)}</Label>
   322→                <Input value={form.contractNo} readOnly className="bg-gray-50" dir="ltr" placeholder="CTR-0001" />
   323→                <p className="text-xs text-muted-foreground">{t('يتم التوليد تلقائياً', 'Auto-generated')}</p>
   324→              </div>
   325→              <div className="space-y-2">
   326→                <Label>{t(labels.client.ar, labels.client.en)} *</Label>
   327→                <Select value={form.clientId} onValueChange={v => setForm(f => ({ ...f, clientId: v, projectId: '' }))}>
   328→                  <SelectTrigger className="w-full"><SelectValue placeholder={t(labels.selectClient.ar, labels.selectClient.en)} /></SelectTrigger>
   329→                  <SelectContent>
   330→                    {clients.map(c => <SelectItem key={c.id} value={c.id}>{c.name}</SelectItem>)}
   331→                  </SelectContent>
   332→                </Select>
   333→              </div>
   334→              <div className="space-y-2">
   335→                <Label>{t(labels.project.ar, labels.project.en)} *</Label>
   336→                <Select value={form.projectId} onValueChange={v => setForm(f => ({ ...f, projectId: v }))}>
   337→                  <SelectTrigger className="w-full"><SelectValue placeholder={t(labels.selectProject.ar, labels.selectProject.en)} /></SelectTrigger>
   338→                  <SelectContent>
   339→                    {filteredProjects.map(p => <SelectItem key={p.id} value={p.id}>{p.name}</SelectItem>)}
   340→                  </SelectContent>
   341→                </Select>
   342→              </div>
   343→              <div className="space-y-2">
   344→                <Label>{t(labels.contractType.ar, labels.contractType.en)}</Label>
   345→                <div className="flex items-center gap-2 h-10 px-3 rounded-md border bg-emerald-50 border-emerald-200">
   346→                  <CheckCircle2 className="size-4 text-emerald-600" />
   347→                  <span className="text-sm font-medium text-emerald-700">{t(labels.projectContract.ar, labels.projectContract.en)}</span>
   348→                </div>
   349→              </div>
   350→              <div className="space-y-2">
   351→                <Label>{t(labels.date.ar, labels.date.en)} *</Label>
   352→                <Input type="date" value={form.date} onChange={e => setForm(f => ({ ...f, date: e.target.value }))} required />
   353→              </div>
   354→              <div className="space-y-2">
   355→                <Label>{t(labels.startDate.ar, labels.startDate.en)} *</Label>
   356→                <Input type="date" value={form.startDate} onChange={e => setForm(f => ({ ...f, startDate: e.target.value }))} required />
   357→              </div>
   358→              <div className="space-y-2">
   359→                <Label>{t(labels.endDate.ar, labels.endDate.en)}</Label>
   360→                <Input type="date" value={form.endDate} onChange={e => setForm(f => ({ ...f, endDate: e.target.value }))} />
   361→              </div>
   362→            </div>
   363→          </CardContent>
   364→        </Card>
   365→
   366→        {/* القسم الثاني: البيانات المالية */}
   367→        <Card>
   368→          <CardContent className="pt-6">
   369→            <SectionHeader icon={DollarSign} title={labels.section2} lang={lang} />
   370→            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
   371→              <div className="space-y-2">
   372→                <Label>{t(labels.value.ar, labels.value.en)} *</Label>
   373→                <Input type="number" step="0.01" min="0" value={form.value} onChange={e => setForm(f => ({ ...f, value: e.target.value }))} placeholder="0.00" dir="ltr" required />
   374→              </div>
   375→              <div className="space-y-2">
   376→                <Label>{t(labels.vatRate.ar, labels.vatRate.en)}</Label>
   377→                <Input type="number" step="0.01" value={form.vatRate} onChange={e => setForm(f => ({ ...f, vatRate: e.target.value }))} dir="ltr" />
   378→                <p className="text-xs text-muted-foreground">({((rate ?? 0) * 100).toFixed(0)}%)</p>
   379→              </div>
   380→              <div className="space-y-2">
   381→                <Label>{t(labels.vatAmount.ar, labels.vatAmount.en)}</Label>
   382→                <Input value={(vatAmount ?? 0).toFixed(2)} readOnly className="bg-gray-50" dir="ltr" />
   383→                <p className="text-xs text-muted-foreground">{t('محسوب تلقائياً', 'Auto-calculated')}</p>
   384→              </div>
   385→            </div>
   386→
   387→            {/* VAT Preview */}
   388→            {val > 0 && (
   389→              <Card className="bg-emerald-50 border-emerald-200 mt-4">
   390→                <CardContent className="p-4">
   391→                  <div className="grid grid-cols-3 gap-4 text-center">
   392→                    <div>
   393→                      <p className="text-xs text-muted-foreground">{t('القيمة', 'Value')}</p>
   394→                      <MoneyDisplay value={val} lang={lang} bold size="md" className="justify-center" />
   395→                    </div>
   396→                    <div>
   397→                      <p className="text-xs text-muted-foreground">{t('الضريبة', 'VAT')} ({((rate ?? 0) * 100).toFixed(0)}%)</p>
   398→                      <MoneyDisplay value={vatAmount} lang={lang} bold size="md" className="justify-center" />
   399→                    </div>
   400→                    <div>
   401→                      <p className="text-xs text-muted-foreground">{t('الإجمالي', 'Total')}</p>
   402→                      <MoneyDisplay value={totalValue} lang={lang} bold size="lg" className="justify-center text-emerald-700" />
   403→                    </div>
   404→                  </div>
   405→                </CardContent>
   406→              </Card>

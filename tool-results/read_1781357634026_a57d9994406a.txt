    50→/** Format a DateTime string or time-only string to a localized time display (e.g., "08:00 ص") */
    51→function formatTime(time: string | null, lang: 'ar' | 'en'): string {
    52→  if (!time) return '—'
    53→  try {
    54→    const d = new Date(time)
    55→    if (isNaN(d.getTime())) {
    56→      // Might be a time-only string like "08:00"
    57→      if (/^\d{1,2}:\d{2}$/.test(time)) {
    58→        const [h, m] = time.split(':').map(Number)
    59→        if (lang === 'ar') {
    60→          const period = h >= 12 ? 'م' : 'ص'
    61→          const h12 = h === 0 ? 12 : h > 12 ? h - 12 : h
    62→          return `${String(h12).padStart(2, '0')}:${String(m).padStart(2, '0')} ${period}`
    63→        }
    64→        const period = h >= 12 ? 'PM' : 'AM'
    65→        const h12 = h === 0 ? 12 : h > 12 ? h - 12 : h
    66→        return `${String(h12).padStart(2, '0')}:${String(m).padStart(2, '0')} ${period}`
    67→      }
    68→      return time
    69→    }
    70→    return d.toLocaleTimeString(lang === 'ar' ? 'ar-SA' : 'en-US', { hour: '2-digit', minute: '2-digit', hour12: true })
    71→  } catch {
    72→    return time
    73→  }
    74→}

    82→// ============ Attendance Form Dialog ============
    83→function AttendanceFormDialog({ open, onOpenChange, employees }: {
    84→  open: boolean; onOpenChange: (open: boolean) => void; employees: Employee[]
    85→}) {
    86→  const queryClient = useQueryClient()
    87→  const [form, setForm] = useState<AttendanceFormData>(defaultForm)
    88→  const { lang } = useAppStore()
    89→
    90→  // Calculate work hours and auto-overtime from checkIn/checkOut
    91→  const calculatedWorkHours = React.useMemo(() => {
    92→    if (!form.checkIn || !form.checkOut) return null
    93→    const [inH, inM] = form.checkIn.split(':').map(Number)
    94→    const [outH, outM] = form.checkOut.split(':').map(Number)
    95→    if (isNaN(inH) || isNaN(outH)) return null
    96→    const inMin = inH * 60 + (inM || 0)
    97→    const outMin = outH * 60 + (outM || 0)
    98→    const diff = outMin - inMin
    99→    if (diff <= 0) return null
   100→    return Math.round((diff / 60) * 100) / 100
   101→  }, [form.checkIn, form.checkOut])
   102→
   103→  // Auto-calculate overtime (hours beyond 8)
   104→  const autoOvertime = React.useMemo(() => {
   105→    if (calculatedWorkHours === null) return 0
   106→    return Math.max(0, Math.round((calculatedWorkHours - 8) * 100) / 100)
   107→  }, [calculatedWorkHours])
   108→
   109→  React.useEffect(() => {
   110→    if (open) setForm(defaultForm)
   111→  }, [open])
   112→
   113→  const createMutation = useMutation({
   114→    mutationFn: (data: Record<string, unknown>) => fetch('/api/attendance', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   115→    onSuccess: () => {
   116→      queryClient.invalidateQueries({ queryKey: ['attendance'] })
   117→      onOpenChange(false)
   118→      toast.success(t('تم تسجيل الحضور', 'Attendance recorded', lang))
   119→    },
   120→  })

   122→  const handleSubmit = (e: React.FormEvent) => {
   123→    e.preventDefault()
   124→    // Combine date + time for proper server-side calculation
   125→    const checkInDT = form.checkIn && form.date ? `${form.date}T${form.checkIn}` : (form.checkIn || null)
   126→    const checkOutDT = form.checkOut && form.date ? `${form.date}T${form.checkOut}` : (form.checkOut || null)
   127→    const otValue = form.overtimeHours !== '' ? parseFloat(form.overtimeHours) : autoOvertime
   128→    createMutation.mutate({
   129→      ...form,
   130→      overtimeHours: isNaN(otValue) ? autoOvertime : otValue,
   131→      checkIn: checkInDT,
   132→      checkOut: checkOutDT,
   133→    })
   134→  }
   135→
   136→  return (
   137→    <Dialog open={open} onOpenChange={onOpenChange}>
   138→      <DialogContent className="max-w-lg max-h-[90vh] overflow-y-auto">
   139→        <DialogHeader>
   140→          <DialogTitle>{t('تسجيل حضور', 'Record Attendance', lang)}</DialogTitle>
   141→          <DialogDescription>{t('تسجيل حضور وانصراف موظف', 'Record employee attendance', lang)}</DialogDescription>
   142→        </DialogHeader>
   143→        <form onSubmit={handleSubmit} className="space-y-4">
   144→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   145→            <div className="space-y-2 sm:col-span-2">
   146→              <Label>{t('الموظف *', 'Employee *', lang)}</Label>
   147→              <Select value={form.employeeId} onValueChange={v => setForm(f => ({ ...f, employeeId: v }))}>
   148→                <SelectTrigger><SelectValue placeholder={t('اختر الموظف', 'Select employee', lang)} /></SelectTrigger>
   149→                <SelectContent>
   150→                  {employees.map(e => <SelectItem key={e.id} value={e.id}>{e.name} ({e.code})</SelectItem>)}
   151→                </SelectContent>
   152→              </Select>
   153→            </div>
   154→            <div className="space-y-2"><Label>{t('التاريخ *', 'Date *', lang)}</Label><Input type="date" value={form.date} onChange={e => setForm(f => ({ ...f, date: e.target.value }))} required /></div>
   155→            <div className="space-y-2"><Label>{t('وقت الحضور', 'Check In', lang)}</Label><Input type="time" value={form.checkIn} onChange={e => setForm(f => ({ ...f, checkIn: e.target.value }))} /></div>
   156→            <div className="space-y-2"><Label>{t('وقت الانصراف', 'Check Out', lang)}</Label><Input type="time" value={form.checkOut} onChange={e => setForm(f => ({ ...f, checkOut: e.target.value }))} /></div>
   157→            <div className="space-y-2">
   158→              <Label>{t('ساعات إضافية', 'Overtime Hours', lang)}</Label>
   159→              <Input
   160→                type="number" min="0" step="0.5"
   161→                value={form.overtimeHours !== '' ? form.overtimeHours : (autoOvertime > 0 ? String(autoOvertime) : '0')}
   162→                onChange={e => setForm(f => ({ ...f, overtimeHours: e.target.value }))}
   163→                dir="ltr"
   164→              />
   165→              {autoOvertime > 0 && <p className="text-xs text-amber-600">{t('إضافي محسوب تلقائياً', 'Auto-calculated overtime', lang)}: {autoOvertime} {t('ساعة', 'hrs', lang)}</p>}
   166→            </div>
   167→          </div>
   168→
   169→          {calculatedWorkHours !== null && (
   170→            <Card className="bg-emerald-50 border-emerald-200">

   172→                <div className="flex justify-between text-sm">

   177→                </div>
   178→              </CardContent>
   179→            </Card>
   180→          )}
   181→
   182→          <DialogFooter>
   183→            <Button type="button" variant="outline" onClick={() => onOpenChange(false)}>{t('إلغاء', 'Cancel', lang)}</Button>
   184→            <Button type="submit" disabled={createMutation.isPending || !form.employeeId || !form.date} className="bg-emerald-600 hover:bg-emerald-700">{createMutation.isPending ? t('جاري الحفظ...', 'Saving...', lang) : t('تسجيل', 'Record', lang)}</Button>
   185→          </DialogFooter>
   186→        </form>
   187→      </DialogContent>
   188→    </Dialog>
   189→  )
   190→}

   193→function BulkAttendanceDialog({ open, onOpenChange, employees }: {
   194→  open: boolean; onOpenChange: (open: boolean) => void; employees: Employee[]
   195→}) {
   196→  const queryClient = useQueryClient()
   197→  const { lang } = useAppStore()
   198→  const [date, setDate] = useState('')
   199→  const [checkIn, setCheckIn] = useState('08:00')
   200→  const [checkOut, setCheckOut] = useState('17:00')
   201→  const [selectedIds, setSelectedIds] = useState<string[]>([])
   202→
   203→  React.useEffect(() => {
   204→    if (open) {
   205→      setDate('')
   206→      setCheckIn('08:00')
   207→      setCheckOut('17:00')
   208→      setSelectedIds([])
   209→    }
   210→  }, [open])
   211→
   212→  const toggleEmployee = (id: string) => {
   213→    setSelectedIds(prev => prev.includes(id) ? prev.filter(i => i !== id) : [...prev, id])
   214→  }
   215→
   216→  const selectAll = () => {
   217→    setSelectedIds(employees.map(e => e.id))
   218→  }
   219→
   220→  const selectNone = () => {
   221→    setSelectedIds([])
   222→  }
   223→
   224→  // Calculate work hours from check-in/check-out
   225→  const calculatedHours = React.useMemo(() => {
   226→    if (!checkIn || !checkOut) return 0
   227→    const [inH, inM] = checkIn.split(':').map(Number)
   228→    const [outH, outM] = checkOut.split(':').map(Number)
   229→    if (isNaN(inH) || isNaN(outH)) return 0
   230→    const diff = (outH * 60 + (outM || 0)) - (inH * 60 + (inM || 0))
   231→    return diff > 0 ? Math.round((diff / 60) * 100) / 100 : 0
   232→  }, [checkIn, checkOut])
   233→
   234→  const autoOvertime = Math.max(0, calculatedHours - 8)
   235→
   236→  const bulkMutation = useMutation({
   237→    mutationFn: async () => {
   238→      const results = await Promise.allSettled(
   239→        selectedIds.map(employeeId =>
   240→          fetch('/api/attendance', {
   241→            method: 'POST',
   242→            headers: { 'Content-Type': 'application/json' },
   243→            body: JSON.stringify({
   244→              employeeId,
   245→              date,
   246→              checkIn: checkIn && date ? `${date}T${checkIn}` : checkIn,
   247→              checkOut: checkOut && date ? `${date}T${checkOut}` : checkOut,
   248→              overtimeHours: autoOvertime,
   249→            }),
   250→          }).then(r => { if (!r.ok) throw new Error(); return r.json() })
   251→        )
   252→      )
   253→      return results
   254→    },
   255→    onSuccess: () => {
   256→      queryClient.invalidateQueries({ queryKey: ['attendance'] })
   257→      onOpenChange(false)
   258→      toast.success(t('تم تسجيل الحضور الجماعي', 'Bulk attendance recorded', lang))
   259→    },
   260→  })
   261→
   262→  const handleSubmit = (e: React.FormEvent) => {
   263→    e.preventDefault()
   264→    if (!date || selectedIds.length === 0) return
   265→    bulkMutation.mutate()
   266→  }
   267→
   268→  return (
   269→    <Dialog open={open} onOpenChange={onOpenChange}>
   270→      <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
   271→        <DialogHeader>
   272→          <DialogTitle>{t('تسجيل حضور جماعي', 'Bulk Attendance', lang)}</DialogTitle>
   273→          <DialogDescription>{t('تسجيل حضور لعدة موظفين في نفس اليوم', 'Record attendance for multiple employees on the same day', lang)}</DialogDescription>
   274→        </DialogHeader>
   275→        <form onSubmit={handleSubmit} className="space-y-4">
   276→          <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   277→            <div className="space-y-2"><Label>{t('التاريخ *', 'Date *', lang)}</Label><Input type="date" value={date} onChange={e => setDate(e.target.value)} required /></div>
   278→            <div className="space-y-2"><Label>{t('وقت الحضور', 'Check In', lang)}</Label><Input type="time" value={checkIn} onChange={e => setCheckIn(e.target.value)} /></div>
   279→            <div className="space-y-2"><Label>{t('وقت الانصراف', 'Check Out', lang)}</Label><Input type="time" value={checkOut} onChange={e => setCheckOut(e.target.value)} /></div>
   280→          </div>
   281→
   282→          {calculatedHours > 0 && (
   283→            <Card className="bg-emerald-50 border-emerald-200">
   284→              <CardContent className="p-3 flex justify-between text-sm">
   285→                <p className="text-emerald-600">{t('ساعات العمل', 'Work Hours', lang)}: <span className="font-bold">{calculatedHours}</span></p>
   286→                {autoOvertime > 0 && <p className="text-amber-600">{t('إضافي', 'Overtime', lang)}: <span className="font-bold">{autoOvertime}</span></p>}
   287→              </CardContent>
   288→            </Card>
   289→          )}
   290→
   291→          <div className="space-y-2">
   292→            <div className="flex items-center justify-between">
   293→              <h4 className="font-semibold text-sm text-emerald-700">
   294→                {t('اختر الموظفين', 'Select Employees', lang)} ({selectedIds.length}/{employees.length})
   295→              </h4>
   296→              <div className="flex gap-2">
   297→                <Button type="button" variant="ghost" size="sm" onClick={selectAll}>{t('تحديد الكل', 'Select All', lang)}</Button>
   298→                <Button type="button" variant="ghost" size="sm" onClick={selectNone}>{t('إلغاء التحديد', 'Deselect All', lang)}</Button>
   299→              </div>
   300→            </div>
   301→            <div className="max-h-48 overflow-y-auto border rounded-lg p-2 space-y-1">
   302→              {employees.map(emp => (
   303→                <label key={emp.id} className="flex items-center gap-2 p-1.5 rounded hover:bg-gray-50 cursor-pointer">
   304→                  <Checkbox
   305→                    checked={selectedIds.includes(emp.id)}
   306→                    onCheckedChange={() => toggleEmployee(emp.id)}
   307→                  />
   308→                  <span className="text-sm">{emp.name} <span className="text-muted-foreground">({emp.code})</span></span>
   309→                </label>
   310→              ))}
   311→            </div>
   312→          </div>
   313→
   314→          <DialogFooter>
   315→            <Button type="button" variant="outline" onClick={() => onOpenChange(false)}>{t('إلغاء', 'Cancel', lang)}</Button>
   316→            <Button type="submit" disabled={bulkMutation.isPending || !date || selectedIds.length === 0} className="bg-emerald-600 hover:bg-emerald-700">
   317→              {bulkMutation.isPending ? t('جاري التسجيل...', 'Recording...', lang) : t('تسجيل حضور', 'Record Attendance', lang)}
   318→            </Button>
   319→          </DialogFooter>
   320→        </form>
   321→      </DialogContent>
   322→    </Dialog>
   323→  )
   324→}

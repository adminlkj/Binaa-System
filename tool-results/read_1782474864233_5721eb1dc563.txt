   199→  const mutation = useMutation({
   200→    mutationFn: async () => {
   201→      const payload = {

   203→        acquisitionCost: Number(form.acquisitionCost) || 0,
   204→        residualValue: Number(form.residualValue) || 0,
   205→        usefulLifeMonths: parseInt(form.usefulLifeMonths) || 0,

   226→    onError: (err: Error) => toast.error(err.message),

   369→          </Button>
   370→        </DialogFooter>
   371→      </DialogContent>
   372→    </Dialog>
   373→  )
   374→}

   842→function ReportTab() {
   843→  const { lang } = useAppStore()
   844→  const { data, isLoading } = useQuery({
   845→    queryKey: ['depreciation-report'],
   846→    queryFn: async () => {
   847→      const res = await fetch('/api/fixed-assets/report')
   848→      if (!res.ok) throw new Error()
   849→      return res.json()
   850→    },
   851→  })
   852→
   853→  const summary = data?.summary
   854→  const categorySummary = data?.categorySummary || []
   855→
   856→  return (
   857→    <div className="space-y-4">
   858→      {summary && (
   859→        <div className="grid grid-cols-2 md:grid-cols-4 gap-3">
   860→          <SummaryCard icon={Package} label={{ ar: 'إجمالي الأصول', en: 'Total Assets' }} value={summary.totalAssets} color="bg-teal-50 text-teal-700" lang={lang} />
   861→          <SummaryCard icon={Wallet} label={{ ar: 'إجمالي التكلفة', en: 'Total Cost' }} value={summary.totalCost} color="bg-sky-50 text-sky-700" lang={lang} />
   862→          <SummaryCard icon={TrendingDown} label={{ ar: 'إجمالي الإهلاك', en: 'Total Depreciation' }} value={summary.totalAccumDep} color="bg-amber-50 text-amber-700" lang={lang} />
   863→          <SummaryCard icon={BarChart3} label={{ ar: 'صافي القيمة', en: 'Net Book Value' }} value={summary.totalNBV} color="bg-emerald-50 text-emerald-700" lang={lang} />
   864→        </div>
   865→      )}
   866→
   867→      <Card>
   868→        <CardContent className="p-4">
   869→          <h3 className="font-semibold mb-3 flex items-center gap-2">
   870→            <BarChart3 className="size-4 text-teal-600" />
   871→            {t(lang, 'تقرير الإهلاك حسب الفئة', 'Depreciation by Category')}
   872→          </h3>
   873→          {isLoading ? (
   874→            <p className="text-center py-8 text-muted-foreground">{t(lang, 'جاري التحميل...', 'Loading...')}</p>
   875→          ) : categorySummary.length === 0 ? (
   876→            <p className="text-center py-8 text-muted-foreground">{t(lang, 'لا توجد بيانات', 'No data')}</p>
   877→          ) : (
   878→            <div className="space-y-3">
   879→              {categorySummary.map((cat: any) => {
   880→                const catInfo = categoryLabels[cat.category] || categoryLabels.OTHER
   881→                const CatIcon = catInfo.icon
   882→                const percentage = summary?.totalCost ? (cat.cost / summary.totalCost) * 100 : 0
   883→                return (
   884→                  <div key={cat.category} className="border rounded-lg p-3">
   885→                    <div className="flex items-center justify-between mb-2">
   886→                      <div className="flex items-center gap-2">
   887→                        <CatIcon className="size-4 text-muted-foreground" />
   888→                        <span className="font-medium">{lang === 'ar' ? catInfo.ar : catInfo.en}</span>
   889→                        <Badge variant="outline" className="text-xs">{cat.count} {t(lang, 'أصل', 'assets')}</Badge>
   890→                      </div>
   891→                      <span className="text-xs text-muted-foreground">{percentage.toFixed(1)}%</span>
   892→                    </div>
   893→                    <div className="w-full bg-gray-100 rounded-full h-2 mb-2">
   894→                      <div className="bg-teal-500 h-2 rounded-full" style={{ width: `${percentage}%` }} />
   895→                    </div>
   896→                    <div className="grid grid-cols-3 gap-2 text-sm">
   897→                      <div>
   898→                        <p className="text-xs text-muted-foreground">{t(lang, 'التكلفة', 'Cost')}</p>
   899→                        <MoneyDisplay value={cat.cost} lang={lang} size="sm" />
   900→                      </div>
   901→                      <div>
   902→                        <p className="text-xs text-muted-foreground">{t(lang, 'الإهلاك', 'Depreciation')}</p>
   903→                        <MoneyDisplay value={cat.accumDep} lang={lang} size="sm" className="text-amber-700" />
   904→                      </div>
   905→                      <div>
   906→                        <p className="text-xs text-muted-foreground">{t(lang, 'القيمة الدفترية', 'NBV')}</p>
   907→                        <MoneyDisplay value={cat.nbv} lang={lang} size="sm" className="text-emerald-700" />
   908→                      </div>
   909→                    </div>
   910→                  </div>
   911→                )
   912→              })}
   913→            </div>
   914→          )}
   915→        </CardContent>
   916→      </Card>
   917→
   918→      {/* Monthly trend */}
   919→      {data?.monthlyTrend?.length > 0 && (
   920→        <Card>
   921→          <CardContent className="p-4">
   922→            <h3 className="font-semibold mb-3 flex items-center gap-2">
   923→              <Calendar className="size-4 text-teal-600" />
   924→              {t(lang, 'اتجاه الإهلاك الشهري', 'Monthly Depreciation Trend')}
   925→            </h3>
   926→            <div className="space-y-2 max-h-64 overflow-y-auto">
   927→              {data.monthlyTrend.map((item: any) => (
   928→                <div key={item.period} className="flex items-center gap-3">
   929→                  <span className="text-xs font-mono w-20">{item.period}</span>
   930→                  <div className="flex-1 bg-gray-100 rounded-full h-4 relative">
   931→                    <div
   932→                      className="bg-teal-500 h-4 rounded-full"
   933→                      style={{ width: `${Math.min(100, (item.amount / Math.max(...data.monthlyTrend.map((m: any) => m.amount))) * 100)}%` }}
   934→                    />
   935→                  </div>
   936→                  <MoneyDisplay value={item.amount} lang={lang} size="sm" className="w-24 text-left" />
   937→                </div>
   938→              ))}
   939→            </div>
   940→          </CardContent>
   941→        </Card>
   942→      )}
   943→    </div>
   944→  )
   945→}

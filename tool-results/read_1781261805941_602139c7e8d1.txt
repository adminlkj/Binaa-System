   125→// ============ Add Payment Dialog ============

   127→function AddPaymentDialog({
   128→  open,
   129→  onClose,
   130→}: {
   131→  open: boolean
   132→  onClose: () => void
   133→}) {
   134→  const { lang } = useAppStore()
   135→  const queryClient = useQueryClient()
   136→  const tt = (ar: string, en: string) => t(ar, en, lang)
   137→
   138→  const [clientId, setClientId] = useState('')
   139→  const [invoiceId, setInvoiceId] = useState('')
   140→  const [amount, setAmount] = useState('')
   141→  const [date, setDate] = useState(new Date().toISOString().split('T')[0])
   142→  const [receivedIn, setReceivedIn] = useState('TREASURY')
   143→  const [reference, setReference] = useState('')
   144→  const [notes, setNotes] = useState('')
   145→
   146→  // Fetch clients
   147→  const { data: clients = [] } = useQuery<ClientSummary[]>({
   148→    queryKey: ['clients-list'],
   149→    queryFn: async () => {
   150→      const res = await fetch('/api/clients')
   151→      if (!res.ok) return []
   152→      return res.json()
   153→    },
   154→  })
   155→
   156→  // Fetch unpaid invoices for selected client
   157→  const { data: invoices = [] } = useQuery<InvoiceSummary[]>({
   158→    queryKey: ['unpaid-invoices', clientId],
   159→    queryFn: async () => {
   160→      if (!clientId) return []
   161→      const res = await fetch(`/api/sales-invoices?clientId=${clientId}&status=unpaid`)
   162→      if (!res.ok) return []
   163→      const data = await res.json()
   164→      return data.filter((inv: InvoiceSummary) =>
   165→        inv.status !== 'PAID' && inv.status !== 'CANCELLED' && inv.totalAmount - inv.paidAmount > 0
   166→      )
   167→    },
   168→    enabled: !!clientId,
   169→  })
   170→
   171→  // Selected invoice
   172→  const selectedInvoice = invoices.find(inv => inv.id === invoiceId)
   173→  const remainingBalance = selectedInvoice
   174→    ? (selectedInvoice.totalAmount ?? 0) - (selectedInvoice.paidAmount ?? 0)
   175→    : 0
   176→
   177→  // Auto-fill amount when invoice is selected
   178→  React.useEffect(() => {
   179→    if (selectedInvoice && remainingBalance > 0) {
   180→      setAmount((remainingBalance ?? 0).toFixed(2))
   181→    }
   182→  }, [invoiceId, selectedInvoice, remainingBalance])
   183→
   184→  // Reset form when dialog opens/closes
   185→  React.useEffect(() => {
   186→    if (!open) {
   187→      setClientId('')
   188→      setInvoiceId('')
   189→      setAmount('')
   190→      setDate(new Date().toISOString().split('T')[0])
   191→      setReceivedIn('TREASURY')
   192→      setReference('')
   193→      setNotes('')
   194→    }
   195→  }, [open])
   196→
   197→  const createMutation = useMutation({
   198→    mutationFn: (data: Record<string, unknown>) =>
   199→      fetch('/api/client-payments', {
   200→        method: 'POST',
   201→        headers: { 'Content-Type': 'application/json' },
   202→        body: JSON.stringify(data),
   203→      }).then(r => {
   204→        if (!r.ok) throw new Error('Failed')
   205→        return r.json()
   206→      }),
   207→    onSuccess: () => {
   208→      queryClient.invalidateQueries({ queryKey: ['client-payments'] })
   209→      onClose()
   210→    },
   211→  })
   212→
   213→  const handleSubmit = (e: React.FormEvent) => {
   214→    e.preventDefault()
   215→    const parsedAmount = parseFloat(amount) || 0
   216→    if (!clientId || parsedAmount <= 0 || !date) return
   217→
   218→    createMutation.mutate({
   219→      clientId,
   220→      invoiceId: invoiceId || null,
   221→      amount: parsedAmount,
   222→      date,
   223→      receivedIn,
   224→      reference: reference || null,
   225→      notes: notes || null,
   226→    })
   227→  }
   228→
   229→  return (
   230→    <Dialog open={open} onOpenChange={onClose}>
   231→      <DialogContent className="max-w-lg">
   232→        <DialogHeader>
   233→          <DialogTitle className="flex items-center gap-2">
   234→            <CreditCard className="size-5 text-emerald-600" />
   235→            {tt(labels.newPayment.ar, labels.newPayment.en)}
   236→          </DialogTitle>
   237→        </DialogHeader>
   238→
   239→        <form onSubmit={handleSubmit} className="space-y-4">
   240→          {/* Client Selection */}
   241→          <div className="space-y-2">
   242→            <Label>{tt(labels.client.ar, labels.client.en)} *</Label>
   243→            <Select value={clientId} onValueChange={v => { setClientId(v); setInvoiceId(''); setAmount('') }}>
   244→              <SelectTrigger className="w-full">
   245→                <SelectValue placeholder={tt(labels.selectClient.ar, labels.selectClient.en)} />
   246→              </SelectTrigger>
   247→              <SelectContent>
   248→                {clients.map(c => (
   249→                  <SelectItem key={c.id} value={c.id}>{c.name} ({c.code})</SelectItem>
   250→                ))}
   251→              </SelectContent>
   252→            </Select>
   253→          </div>
   254→
   255→          {/* Invoice Selection */}
   256→          <div className="space-y-2">
   257→            <Label>{tt(labels.invoice.ar, labels.invoice.en)}</Label>
   258→            <Select value={invoiceId} onValueChange={setInvoiceId} disabled={!clientId}>
   259→              <SelectTrigger className="w-full">
   260→                <SelectValue placeholder={
   261→                  !clientId
   262→                    ? tt(labels.selectClient.ar, labels.selectClient.en)
   263→                    : invoices.length === 0
   264→                      ? tt(labels.noInvoices.ar, labels.noInvoices.en)
   265→                      : tt(labels.selectInvoice.ar, labels.selectInvoice.en)
   266→                } />
   267→              </SelectTrigger>
   268→              <SelectContent>
   269→                {invoices.map(inv => {
   270→                  const remaining = inv.totalAmount - inv.paidAmount
   271→                  return (
   272→                    <SelectItem key={inv.id} value={inv.id}>
   273→                      {inv.invoiceNo} — {tt('متبقي', 'Remaining', lang)}: <MoneyDisplay value={remaining} lang={lang} size="sm" inline />
   274→                    </SelectItem>
   275→                  )
   276→                })}
   277→              </SelectContent>
   278→            </Select>
   279→          </div>
   280→
   281→          {/* Invoice Info */}
   282→          {selectedInvoice && (
   283→            <div className="p-3 rounded-lg border bg-emerald-50 text-sm space-y-1">
   284→              <div className="flex justify-between">
   285→                <span className="text-muted-foreground">{tt('إجمالي الفاتورة', 'Invoice Total', lang)}</span>
   286→                <MoneyDisplay value={selectedInvoice.totalAmount} lang={lang} size="sm" inline bold />
   287→              </div>
   288→              <div className="flex justify-between">
   289→                <span className="text-muted-foreground">{tt('المدفوع', 'Paid', lang)}</span>
   290→                <MoneyDisplay value={selectedInvoice.paidAmount} lang={lang} size="sm" inline />
   291→              </div>
   292→              <Separator />
   293→              <div className="flex justify-between font-bold text-emerald-700">
   294→                <span>{tt(labels.remaining.ar, labels.remaining.en)}</span>
   295→                <MoneyDisplay value={remainingBalance} lang={lang} size="sm" inline bold />
   296→              </div>
   297→            </div>
   298→          )}
   299→
   300→          {/* Amount */}
   301→          <div className="space-y-2">
   302→            <Label>
   303→              {tt(labels.amount.ar, labels.amount.en)} (ر.س / SAR) *
   304→              {selectedInvoice && (
   305→                <span className="text-xs text-emerald-600 ml-2">({tt(labels.autoFill.ar, labels.autoFill.en)})</span>
   306→              )}
   307→            </Label>
   308→            <Input
   309→              type="number"
   310→              step="0.01"
   311→              min="0.01"
   312→              max={remainingBalance || undefined}
   313→              value={amount}
   314→              onChange={e => setAmount(e.target.value)}
   315→              placeholder="0.00"
   316→              required
   317→            />
   318→          </div>
   319→
   320→          {/* Date & Received In */}
   321→          <div className="grid grid-cols-2 gap-4">
   322→            <div className="space-y-2">
   323→              <Label>{tt(labels.date.ar, labels.date.en)} *</Label>
   324→              <Input type="date" value={date} onChange={e => setDate(e.target.value)} required />
   325→            </div>
   326→            <div className="space-y-2">
   327→              <Label>{tt(labels.receivedIn.ar, labels.receivedIn.en)}</Label>
   328→              <Select value={receivedIn} onValueChange={setReceivedIn}>
   329→                <SelectTrigger className="w-full">
   330→                  <SelectValue />
   331→                </SelectTrigger>
   332→                <SelectContent>
   333→                  <SelectItem value="TREASURY">
   334→                    <span className="flex items-center gap-2">
   335→                      <Landmark className="size-3.5" />
   336→                      {tt(receivedInLabels.TREASURY.ar, receivedInLabels.TREASURY.en)}
   337→                    </span>
   338→                  </SelectItem>
   339→                  <SelectItem value="BANK">
   340→                    <span className="flex items-center gap-2">
   341→                      <Building2 className="size-3.5" />
   342→                      {tt(receivedInLabels.BANK.ar, receivedInLabels.BANK.en)}
   343→                    </span>
   344→                  </SelectItem>
   345→                </SelectContent>
   346→              </Select>
   347→            </div>
   348→          </div>
   349→
   350→          {/* Reference */}
   351→          <div className="space-y-2">
   352→            <Label>{tt(labels.reference.ar, labels.reference.en)}</Label>
   353→            <Input value={reference} onChange={e => setReference(e.target.value)} placeholder={tt('رقم الإيصال', 'Receipt No.', lang)} />
   354→          </div>
   355→
   356→          {/* Notes */}
   357→          <div className="space-y-2">
   358→            <Label>{tt(labels.notes.ar, labels.notes.en)}</Label>
   359→            <Textarea value={notes} onChange={e => setNotes(e.target.value)} placeholder={tt('ملاحظات', 'Notes', lang)} rows={2} />
   360→          </div>
   361→
   362→          <DialogFooter>
   363→            <Button type="button" variant="outline" onClick={onClose}>
   364→              {commonText.cancel[lang]}
   365→            </Button>
   366→            <Button
   367→              type="submit"
   368→              disabled={createMutation.isPending || !clientId || !amount || !date}
   369→              className="bg-emerald-600 hover:bg-emerald-700 min-w-[140px]"
   370→            >
   371→              {createMutation.isPending ? tt('جاري الحفظ...', 'Saving...') : tt('تسجيل التحصيل', 'Record Payment')}
   372→            </Button>
   373→          </DialogFooter>
   374→        </form>
   375→      </DialogContent>
   376→    </Dialog>
   377→  )
   378→}

   518→    queryFn: async () => {
   519→      const res = await fetch('/api/clients')
   520→      if (!res.ok) return []
   521→      return res.json()
   522→    },
   523→  })
   524→
   525→  // Delete mutation
   526→  const deleteMutation = useMutation({
   527→    mutationFn: (id: string) =>
   528→      fetch(`/api/client-payments/${id}`, { method: 'DELETE' }).then(r => {
   529→        if (!r.ok) throw new Error()
   530→        return r.json()
   531→      }),
   532→    onSuccess: () => {
   533→      queryClient.invalidateQueries({ queryKey: ['client-payments'] })
   534→      setDeleteId(null)
   535→    },
   536→  })
   537→
   538→  // Filter
   539→  const filtered = useMemo(() => {
   540→    return payments.filter(p => {
   541→      const matchSearch = !search ||
   542→        p.client.name.toLowerCase().includes(search.toLowerCase()) ||
   543→        (p.reference && p.reference.toLowerCase().includes(search.toLowerCase())) ||
   544→        (p.invoice?.invoiceNo && p.invoice.invoiceNo.toLowerCase().includes(search.toLowerCase()))
   545→      const matchClient = clientFilter === 'all' || p.clientId === clientFilter
   546→      const matchReceivedIn = receivedInFilter === 'all' || p.receivedIn === receivedInFilter
   547→      return matchSearch && matchClient && matchReceivedIn
   548→    })
   549→  }, [payments, search, clientFilter, receivedInFilter])
   550→
   551→  // Stats
   552→  const totalPayments = filtered.reduce((s, p) => s + (p.amount ?? 0), 0)
   553→  const treasuryPayments = filtered.filter(p => p.receivedIn === 'TREASURY').reduce((s, p) => s + (p.amount ?? 0), 0)
   554→  const bankPayments = filtered.filter(p => p.receivedIn === 'BANK').reduce((s, p) => s + (p.amount ?? 0), 0)
   555→
   556→  // Print data for client payments report
   557→  const printData = useMemo(() => ({
   558→    columns: [
   559→      { key: 'client', label: lang === 'ar' ? 'العميل' : 'Client' },
   560→      { key: 'date', label: lang === 'ar' ? 'التاريخ' : 'Date' },
   561→      { key: 'amount', label: lang === 'ar' ? 'المبلغ' : 'Amount', align: 'amount' as const },
   562→      { key: 'receivedIn', label: lang === 'ar' ? 'طريقة الاستلام' : 'Received In' },
   563→      { key: 'reference', label: lang === 'ar' ? 'المرجع' : 'Reference' },
   564→    ],
   565→    rows: filtered.map(p => ({
   566→      client: p.client?.name || '',
   567→      date: formatDate(p.date, lang),
   568→      amount: p.amount,
   569→      receivedIn: receivedInLabels[p.receivedIn]?.[lang] || p.receivedIn,
   570→      reference: p.reference || '',
   571→    })),
   572→    infoItems: [
   573→      { label: lang === 'ar' ? 'تاريخ الطباعة' : 'Print Date', value: new Date().toLocaleDateString() },
   574→      { label: lang === 'ar' ? 'إجمالي التحصيلات' : 'Total Payments', value: String(totalPayments) },
   575→    ],
   576→    totals: [
   577→      { label: lang === 'ar' ? 'إجمالي التحصيلات' : 'Total Payments', value: totalPayments, isGrand: true },
   578→    ],
   579→  }), [filtered, lang, totalPayments])
   580→
   581→  return (
   582→    <ModuleLayout
   583→      title={labels.title}
   584→      subtitle={labels.subtitle}
   585→      actions={
   586→        <>
   587→          <PrintButton type="generic-table" data={printData} size="icon" />
   588→          <Button variant="outline" size="icon" onClick={() => refetch()} title={tt('تحديث', 'Refresh')}>
   589→            <RefreshCw className="size-4" />
   590→          </Button>
   591→          <Button className="gap-2 bg-emerald-600 hover:bg-emerald-700" onClick={() => setShowAddDialog(true)}>
   592→            <Plus className="size-4" /> {tt(labels.newPayment.ar, labels.newPayment.en)}
   593→          </Button>
   594→        </>
   595→      }
   596→    >
   597→      {/* Summary Cards */}
   598→      <div className="grid grid-cols-1 sm:grid-cols-3 gap-3">
   599→        <Card className="bg-emerald-50 border-emerald-200">
   600→          <CardContent className="p-3">
   601→            <p className="text-xs text-muted-foreground">{tt(labels.totalPayments.ar, labels.totalPayments.en)}</p>
   602→            <MoneyDisplay value={totalPayments} lang={lang} size="xl" bold className="text-emerald-700" />
   603→          </CardContent>
   604→        </Card>
   605→        <Card className="bg-amber-50 border-amber-200">
   606→          <CardContent className="p-3">
   607→            <p className="text-xs text-muted-foreground">{tt(labels.treasury.ar, labels.treasury.en)}</p>
   608→            <MoneyDisplay value={treasuryPayments} lang={lang} size="xl" bold className="text-amber-700" />
   609→          </CardContent>
   610→        </Card>
   611→        <Card className="bg-sky-50 border-sky-200">
   612→          <CardContent className="p-3">
   613→            <p className="text-xs text-muted-foreground">{tt(labels.bank.ar, labels.bank.en)}</p>
   614→            <MoneyDisplay value={bankPayments} lang={lang} size="xl" bold className="text-sky-700" />
   615→          </CardContent>
   616→        </Card>
   617→      </div>
   618→
   619→      {/* Filters */}
   620→      <Card>
   621→        <CardContent className="p-4">
   622→          <div className="flex flex-col sm:flex-row gap-3">
   623→            <Select value={clientFilter} onValueChange={setClientFilter}>
   624→              <SelectTrigger className="w-full sm:w-52">
   625→                <SelectValue placeholder={tt(labels.allClients.ar, labels.allClients.en)} />
   626→              </SelectTrigger>
   627→              <SelectContent>
   628→                <SelectItem value="all">{tt(labels.allClients.ar, labels.allClients.en)}</SelectItem>
   629→                {clients.map(c => (
   630→                  <SelectItem key={c.id} value={c.id}>{c.name}</SelectItem>
   631→                ))}
   632→              </SelectContent>
   633→            </Select>
   634→            <div className="relative flex-1">
   635→              <Search className="absolute right-3 top-1/2 -translate-y-1/2 size-4 text-muted-foreground" />
   636→              <Input placeholder={tt(labels.search.ar, labels.search.en)} value={search} onChange={e => setSearch(e.target.value)} className="pr-9" />
   637→            </div>
   638→            <Select value={receivedInFilter} onValueChange={setReceivedInFilter}>
   639→              <SelectTrigger className="w-full sm:w-36">
   640→                <SelectValue placeholder={tt(labels.receivedIn.ar, labels.receivedIn.en)} />
   641→              </SelectTrigger>
   642→              <SelectContent>
   643→                <SelectItem value="all">{tt('الكل', 'All', lang)}</SelectItem>
   644→                <SelectItem value="TREASURY">{tt(receivedInLabels.TREASURY.ar, receivedInLabels.TREASURY.en)}</SelectItem>
   645→                <SelectItem value="BANK">{tt(receivedInLabels.BANK.ar, receivedInLabels.BANK.en)}</SelectItem>
   646→              </SelectContent>
   647→            </Select>
   648→          </div>
   649→        </CardContent>
   650→      </Card>
   651→
   652→      {/* Table */}
   653→      <Card>
   654→        <CardContent className="p-0">
   655→          {isLoading ? (
   656→            <TableSkeleton />
   657→          ) : isError ? (
   658→            <div className="flex flex-col items-center gap-3 py-10">
   659→              <p className="text-rose-600">{commonText.error[lang]}</p>
   660→              <Button variant="outline" onClick={() => refetch()}>{commonText.retry[lang]}</Button>
   661→            </div>
   662→          ) : filtered.length === 0 ? (
   663→            <div className="flex flex-col items-center gap-3 py-10">
   664→              <CreditCard className="size-12 text-gray-300" />
   665→              <p className="text-muted-foreground">{tt(labels.noPayments.ar, labels.noPayments.en)}</p>
   666→              <Button className="bg-emerald-600 hover:bg-emerald-700" onClick={() => setShowAddDialog(true)}>
   667→                <Plus className="size-4 mr-1" /> {tt(labels.newPayment.ar, labels.newPayment.en)}
   668→              </Button>
   669→            </div>
   670→          ) : (
   671→            <div className="overflow-x-auto">
   672→              <Table>
   673→                <TableHeader>
   674→                  <TableRow>
   675→                    <TableHead className="text-right">{tt(labels.date.ar, labels.date.en)}</TableHead>
   676→                    <TableHead className="text-right">{tt(labels.client.ar, labels.client.en)}</TableHead>
   677→                    <TableHead className="text-right">{tt(labels.invoice.ar, labels.invoice.en)}</TableHead>
   678→                    <TableHead className="text-right">{tt(labels.project.ar, labels.project.en)}</TableHead>
   679→                    <TableHead className="text-right">{tt(labels.amount.ar, labels.amount.en)}</TableHead>
   680→                    <TableHead className="text-right">{tt(labels.receivedIn.ar, labels.receivedIn.en)}</TableHead>
   681→                    <TableHead className="text-right">{tt(labels.reference.ar, labels.reference.en)}</TableHead>
   682→                    <TableHead className="text-right">{tt(labels.accountingEntry.ar, labels.accountingEntry.en)}</TableHead>
   683→                    <TableHead className="text-right">{tt(labels.actions.ar, labels.actions.en)}</TableHead>
   684→                  </TableRow>
   685→                </TableHeader>
   686→                <TableBody>
   687→                  {filtered.map(p => (
   688→                    <TableRow key={p.id} className="cursor-pointer hover:bg-emerald-50/50" onClick={() => setDetailPayment(p)}>
   689→                      <TableCell className="whitespace-nowrap">{formatDate(p.date, lang)}</TableCell>
   690→                      <TableCell className="font-medium">{p.client.name}</TableCell>
   691→                      <TableCell>
   692→                        {p.invoice ? (
   693→                          <span className="flex items-center gap-1 text-sm">
   694→                            <FileText className="size-3.5 text-muted-foreground" />
   695→                            {p.invoice.invoiceNo}
   696→                          </span>
   697→                        ) : (
   698→                          <span className="text-muted-foreground text-xs">—</span>
   699→                        )}
   700→                      </TableCell>
   701→                      <TableCell>
   702→                        {p.invoice?.project ? (
   703→                          <span className="text-xs text-muted-foreground">{p.invoice.project.name}</span>
   704→                        ) : (
   705→                          <span className="text-muted-foreground text-xs">—</span>
   706→                        )}
   707→                      </TableCell>
   708→                      <TableCell className="font-semibold">
   709→                        <MoneyDisplay value={p.amount} lang={lang} size="sm" inline bold />
   710→                      </TableCell>
   711→                      <TableCell>
   712→                        <Badge variant="outline" className={`text-[10px] ${receivedInColors[p.receivedIn] || ''}`}>
   713→                          {receivedInLabels[p.receivedIn]?.[lang] || p.receivedIn}
   714→                        </Badge>
   715→                      </TableCell>
   716→                      <TableCell className="text-xs text-muted-foreground">{p.reference || '—'}</TableCell>
   717→                      <TableCell>
   718→                        {p.journalEntryId ? (
   719→                          <Badge variant="outline" className="text-[10px] bg-sky-50 text-sky-700 border-sky-200">
   720→                            {tt('قيد', 'JE', lang)}
   721→                          </Badge>
   722→                        ) : (
   723→                          <span className="text-muted-foreground text-xs">—</span>
   724→                        )}
   725→                      </TableCell>
   726→                      <TableCell>
   727→                        <div className="flex items-center gap-1" onClick={e => e.stopPropagation()}>
   728→                          <PrintButton type="client-payment" documentId={p.id} size="icon" />
   729→                          <Button variant="ghost" size="icon" className="size-8" onClick={() => setDetailPayment(p)} title={tt('عرض', 'View')}>
   730→                            <Eye className="size-4" />
   731→                          </Button>
   732→                          <Button variant="ghost" size="icon" className="size-8 text-rose-500" onClick={() => setDeleteId(p.id)} title={tt('حذف', 'Delete')}>
   733→                            <Trash2 className="size-4" />
   734→                          </Button>
   735→                        </div>
   736→                      </TableCell>
   737→                    </TableRow>
   738→                  ))}
   739→                </TableBody>
   740→              </Table>
   741→            </div>
   742→          )}
   743→        </CardContent>
   744→      </Card>
   745→
   746→      {/* Add Payment Dialog */}
   747→      <AddPaymentDialog open={showAddDialog} onClose={() => setShowAddDialog(false)} />
   748→
   749→      {/* Detail Dialog */}
   750→      <PaymentDetailDialog payment={detailPayment} open={!!detailPayment} onClose={() => setDetailPayment(null)} />
   751→
   752→      {/* Delete Confirmation */}
   753→      <AlertDialog open={!!deleteId} onOpenChange={() => setDeleteId(null)}>
   754→        <AlertDialogContent>
   755→          <AlertDialogHeader>
   756→            <AlertDialogTitle>{tt(labels.deleteTitle.ar, labels.deleteTitle.en)}</AlertDialogTitle>
   757→            <AlertDialogDescription>{tt(labels.deleteConfirm.ar, labels.deleteConfirm.en)}</AlertDialogDescription>
   758→          </AlertDialogHeader>
   759→          <AlertDialogFooter>
   760→            <AlertDialogCancel>{commonText.cancel[lang]}</AlertDialogCancel>
   761→            <AlertDialogAction className="bg-rose-600 hover:bg-rose-700" onClick={() => deleteId && deleteMutation.mutate(deleteId)}>
   762→              {commonText.delete[lang]}
   763→            </AlertDialogAction>
   764→          </AlertDialogFooter>
   765→        </AlertDialogContent>
   766→      </AlertDialog>
   767→    </ModuleLayout>
   768→  )
   769→}

   263→export async function ensureAccountExists(template: AccountTemplate, tx?: PrismaTransaction) {
   264→  const client = tx || db
   265→  const existing = await client.account.findUnique({ where: { code: template.code } })
   266→  if (existing) {
   267→    // Update existing account with new fields if they are missing
   268→    if (
   269→      existing.activityType !== (template.activityType || null) ||
   270→      existing.isSystem !== (template.isSystem || false) ||
   271→      existing.allowPosting !== (template.allowPosting || false) ||
   272→      existing.level !== (template.level || 0)
   273→    ) {
   274→      await client.account.update({
   275→        where: { code: template.code },
   276→        data: {
   277→          name: template.name,
   278→          nameAr: template.nameAr,
   279→          type: template.type,
   280→          activityType: template.activityType || null,
   281→          isSystem: template.isSystem || false,
   282→          allowPosting: template.allowPosting || false,
   283→          level: template.level || 0,
   284→        },
   285→      })
   286→    }
   287→    return client.account.findUnique({ where: { code: template.code } })
   288→  }
   289→
   290→  let parentId: string | undefined
   291→  if (template.parentId) {
   292→    const parent = await client.account.findUnique({ where: { code: template.parentId } })
   293→    if (parent) parentId = parent.id
   294→  }
   295→
   296→  return client.account.create({
   297→    data: {
   298→      code: template.code,
   299→      name: template.name,
   300→      nameAr: template.nameAr,
   301→      type: template.type,
   302→      parentId,
   303→      isActive: true,
   304→      activityType: template.activityType || null,
   305→      isSystem: template.isSystem || false,
   306→      allowPosting: template.allowPosting || false,
   307→      level: template.level || 0,
   308→    },
   309→  })
   310→}

   312→// ============ INITIALIZE CHART OF ACCOUNTS ============

   314→export async function initializeChartOfAccounts() {
   315→  let created = 0
   316→  let updated = 0
   317→
   318→  // Create parent accounts first, then children (sorted by code length then code)
   319→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   320→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   321→    return a.code.localeCompare(b.code)
   322→  })
   323→
   324→  for (const tmpl of sorted) {
   325→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   326→    if (existing) {
   327→      // Update existing account with new fields
   328→      await db.account.update({
   329→        where: { code: tmpl.code },
   330→        data: {
   331→          name: tmpl.name,
   332→          nameAr: tmpl.nameAr,
   333→          type: tmpl.type,
   334→          activityType: tmpl.activityType || null,
   335→          isSystem: tmpl.isSystem || false,
   336→          allowPosting: tmpl.allowPosting || false,
   337→          level: tmpl.level || 0,
   338→          isActive: true,
   339→        },
   340→      })
   341→      // Update parent reference if needed
   342→      if (tmpl.parentId) {
   343→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   344→        if (parent && existing.parentId !== parent.id) {
   345→          await db.account.update({
   346→            where: { code: tmpl.code },
   347→            data: { parentId: parent.id },
   348→          })
   349→        }
   350→      }
   351→      updated++
   352→    } else {
   353→      await ensureAccountExists(tmpl)
   354→      created++
   355→    }
   356→  }
   357→
   358→  const total = await db.account.count()
   359→  return { created, updated, total }
   360→}

   364→export async function createJournalEntry(template: JournalEntryTemplate, tx?: PrismaTransaction) {
   365→  const client = tx || db
   366→  // Validate: total debits must equal total credits
   367→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   368→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   369→
   370→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   371→    throw new Error(
   372→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   373→    )
   374→  }
   375→
   376→  // Ensure all referenced accounts exist
   377→  for (const line of template.lines) {
   378→    const account = await getAccountByCode(line.accountCode, tx)
   379→    if (!account) {
   380→      throw new Error(`Account not found: ${line.accountCode}`)
   381→    }
   382→  }
   383→
   384→  // Validate: fiscal period must be open for this date
   385→  const entryDate = new Date(template.date)
   386→  const closedPeriod = await client.fiscalPeriod.findFirst({
   387→    where: {
   388→      startDate: { lte: entryDate },
   389→      endDate: { gte: entryDate },
   390→      status: 'CLOSED',
   391→    },
   392→  })
   393→  if (closedPeriod) {
   394→    throw new Error(`لا يمكن تسجيل قيد في فترة مالية مقفلة (الفترة ${closedPeriod.periodNo})`)
   395→  }
   396→
   397→  // Create the journal entry with lines
   398→  const entry = await client.journalEntry.create({
   399→    data: {
   400→      entryNo: template.entryNo,
   401→      date: template.date,
   402→      description: template.description,
   403→      status: 'POSTED', // Auto-entries are posted immediately
   404→      lines: {
   405→        create: await Promise.all(
   406→          template.lines.map(async (line) => {
   407→            const account = await getAccountByCode(line.accountCode, tx)
   408→            return {
   409→              accountId: account!.id,
   410→              costCenterId: line.costCenterId,
   411→              debit: line.debit,
   412→              credit: line.credit,
   413→              description: line.description,
   414→            };
   415→          })
   416→        ),
   417→      },
   418→    },
   419→    include: { lines: true },
   420→  })
   421→
   422→  return entry
   423→}

   428→/**
   429→ * فاتورة مبيعات - Sales Invoice (from Extract)
   430→ * Dr: Clients Receivable (1210) - totalAmount
   431→ * Cr: Progress Claims Revenue (6110) - subtotal
   432→ * Cr: VAT Payable (3200) - vatAmount
   433→ */
   434→export async function autoEntrySalesInvoice(data: {
   435→  invoiceNo: string
   436→  clientId: string
   437→  subtotal: number
   438→  vatRate: number
   439→  vatAmount: number
   440→  totalAmount: number
   441→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   442→  date: Date
   443→  projectId?: string
   444→  costCenterId?: string
   445→}, tx?: PrismaTransaction) {
   446→  // Determine revenue account based on invoice type
   447→  let revenueAccountCode: string
   448→  switch (data.invoiceType) {
   449→    case 'RENTAL':
   450→      revenueAccountCode = '6210' // Equipment Rental Revenue
   451→      break
   452→    case 'PROGRESS_CLAIM':
   453→      revenueAccountCode = '6110' // Progress Claims Revenue
   454→      break
   455→    default:
   456→      revenueAccountCode = '6340' // Service Revenue
   457→  }
   458→
   459→  const lines = [
   460→    { accountCode: '1210', debit: data.totalAmount, credit: 0, costCenterId: data.costCenterId }, // Clients Receivable
   461→    { accountCode: revenueAccountCode, debit: 0, credit: data.subtotal, costCenterId: data.costCenterId }, // Revenue
   462→  ]
   463→
   464→  // Add VAT line only if VAT > 0 (Output VAT)
   465→  if (data.vatAmount > 0) {
   466→    lines.push({ accountCode: '3110', debit: 0, credit: data.vatAmount }) // Output VAT (ضريبة مخرجات)
   467→  }
   468→
   469→  return createJournalEntry({
   470→    entryNo: `JE-SI-${Date.now()}`,
   471→    date: data.date,
   472→    description: `Sales Invoice ${data.invoiceNo}`,
   473→    descriptionAr: `فاتورة مبيعات ${data.invoiceNo}`,
   474→    lines,
   475→    sourceType: 'SALES_INVOICE',
   476→    sourceId: data.invoiceNo,
   477→  }, tx)
   478→}

  1156→/**
  1157→ * التخلص من أصل - Asset Disposal
  1158→ * Dr: Cash/Bank (1110/1120) - sale price
  1159→ * Cr: Asset account (2110/2120/2130/2140) - original cost (or net book value)
  1160→ * Dr/Cr: Gain/Loss on disposal (6310 gain / 8610 loss)
  1161→ */
  1162→export async function autoEntryAssetDisposal(data: {
  1163→  assetAccountCode: string // e.g., '2110', '2120', '2130', '2140'
  1164→  accumulatedDepAccountCode: string // e.g., '2210', '2220', '2230', '2240'
  1165→  originalCost: number
  1166→  accumulatedDepreciation: number
  1167→  salePrice: number
  1168→  date: Date
  1169→  receivedIn: 'TREASURY' | 'BANK'
  1170→}, tx?: PrismaTransaction) {
  1171→  const cashAccountCode = data.receivedIn === 'BANK' ? '1120' : '1110'
  1172→  const netBookValue = data.originalCost - data.accumulatedDepreciation
  1173→  const gainLoss = data.salePrice - netBookValue
  1174→
  1175→  const lines = [
  1176→    { accountCode: cashAccountCode, debit: data.salePrice, credit: 0 }, // Cash received
  1177→    { accountCode: data.accumulatedDepAccountCode, debit: data.accumulatedDepreciation, credit: 0 }, // Remove accumulated depreciation
  1178→    { accountCode: data.assetAccountCode, debit: 0, credit: data.originalCost }, // Remove asset
  1179→  ]
  1180→
  1181→  // If gain, credit gain account. If loss, debit loss account.
  1182→  if (gainLoss > 0) {
  1183→    lines.push({ accountCode: '6310', debit: 0, credit: gainLoss }) // Gain on Sale of Used Equipment
  1184→  } else if (gainLoss < 0) {
  1185→    lines.push({ accountCode: '8610', debit: Math.abs(gainLoss), credit: 0 }) // Loss on Asset Disposal
  1186→  }
  1187→
  1188→  return createJournalEntry({
  1189→    entryNo: `JE-DSP-${Date.now()}`,
  1190→    date: data.date,
  1191→    description: `Asset disposal - ${data.assetAccountCode}`,
  1192→    descriptionAr: `التخلص من أصل - ${data.assetAccountCode}`,
  1193→    lines,
  1194→    sourceType: 'ASSET_DISPOSAL',
  1195→    sourceId: `DSP-${Date.now()}`,
  1196→  }, tx)
  1197→}

  1199→/**
  1200→ * إقرار ضريبي - VAT Declaration
  1201→ * Dr: Output VAT (3110) - total output VAT
  1202→ * Cr: Input VAT (3120) - total input VAT
  1203→ * Cr: VAT Due (3130) - net VAT payable (if output > input)
  1204→ * OR
  1205→ * Dr: Input VAT (3120) - if input > output (refund scenario)
  1206→ * Cr: Output VAT (3110)
  1207→ * Dr: VAT Refund Receivable (1410) - if refund due
  1208→ */
  1209→export async function autoEntryVATDeclaration(data: {
  1210→  period: string
  1211→  outputVat: number
  1212→  inputVat: number
  1213→  netVat: number // positive = payable, negative = refundable
  1214→  date: Date
  1215→}, tx?: PrismaTransaction) {
  1216→  const lines: { accountCode: string; debit: number; credit: number }[] = []
  1217→
  1218→  // Close Output VAT (3110) - debit to zero it out
  1219→  lines.push({ accountCode: '3110', debit: data.outputVat, credit: 0 })
  1220→
  1221→  // Close Input VAT (3120) - credit to zero it out
  1222→  lines.push({ accountCode: '3120', debit: 0, credit: data.inputVat })
  1223→
  1224→  // Net VAT position
  1225→  if (data.netVat > 0) {
  1226→    // More output than input → owe VAT
  1227→    lines.push({ accountCode: '3130', debit: 0, credit: data.netVat }) // VAT Due
  1228→  } else if (data.netVat < 0) {
  1229→    // More input than output → refund due
  1230→    lines.push({ accountCode: '1410', debit: Math.abs(data.netVat), credit: 0 }) // VAT Refund Receivable
  1231→  }
  1232→
  1233→  return createJournalEntry({
  1234→    entryNo: `JE-VAT-${Date.now()}`,
  1235→    date: data.date,
  1236→    description: `VAT Declaration - ${data.period}`,
  1237→    descriptionAr: `إقرار ضريبي - ${data.period}`,
  1238→    lines,
  1239→    sourceType: 'VAT_DECLARATION',
  1240→    sourceId: `VAT-${data.period}`,
  1241→  }, tx)
  1242→}

  1271→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1272→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1273→  if (dateFrom || dateTo) {
  1274→    dateFilter.date = {
  1275→      ...(dateFrom && { gte: dateFrom }),
  1276→      ...(dateTo && { lte: dateTo }),
  1277→    }
  1278→  }
  1279→  const entries = await db.journalEntry.findMany({
  1280→    where: {
  1281→      status: 'POSTED',
  1282→      ...dateFilter,
  1283→    },
  1284→    include: {
  1285→      lines: {
  1286→        include: {
  1287→          account: true,
  1288→        },
  1289→      },
  1290→    },
  1291→  })
  1292→
  1293→  // Aggregate by account
  1294→  const accountBalances = new Map<string, {
  1295→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1296→    totalDebit: number
  1297→    totalCredit: number
  1298→  }>()
  1299→
  1300→  for (const entry of entries) {
  1301→    for (const line of entry.lines) {
  1302→      const key = line.accountId
  1303→      if (!accountBalances.has(key)) {
  1304→        accountBalances.set(key, {
  1305→          account: line.account,
  1306→          totalDebit: 0,
  1307→          totalCredit: 0,
  1308→        })
  1309→      }
  1310→      const bal = accountBalances.get(key)!
  1311→      bal.totalDebit += toNumber(line.debit)
  1312→      bal.totalCredit += toNumber(line.credit)
  1313→    }
  1314→  }
  1315→
  1316→  // Calculate net balance for each account
  1317→  const results = Array.from(accountBalances.values()).map(bal => {
  1318→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1319→    let netDebit = 0
  1320→    let netCredit = 0
  1321→
  1322→    if (normalBalance === 'DEBIT') {
  1323→      const net = bal.totalDebit - bal.totalCredit
  1324→      if (net >= 0) netDebit = net
  1325→      else netCredit = Math.abs(net)
  1326→    } else {
  1327→      const net = bal.totalCredit - bal.totalDebit
  1328→      if (net >= 0) netCredit = net
  1329→      else netDebit = Math.abs(net)
  1330→    }
  1331→
  1332→    return {
  1333→      ...bal,
  1334→      netDebit,
  1335→      netCredit,
  1336→    }
  1337→  })
  1338→
  1339→  // Sort by account code
  1340→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1341→
  1342→  return results
  1343→}

  1389→export async function getGeneralLedger(accountCode: string, dateFrom?: Date, dateTo?: Date) {
  1390→  const account = await getAccountByCode(accountCode)
  1391→  if (!account) return []
  1392→
  1393→  const glDateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1394→  if (dateFrom || dateTo) {
  1395→    glDateFilter.date = {
  1396→      ...(dateFrom && { gte: dateFrom }),
  1397→      ...(dateTo && { lte: dateTo }),
  1398→    }
  1399→  }
  1400→  const lines = await db.journalLine.findMany({
  1401→    where: {
  1402→      accountId: account.id,
  1403→      journalEntry: {
  1404→        status: 'POSTED',
  1405→        ...glDateFilter,
  1406→      },
  1407→    },
  1408→    include: { journalEntry: true },
  1409→    orderBy: { journalEntry: { date: 'asc' } },
  1410→  })
  1411→
  1412→  let runningBalance = 0
  1413→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1414→
  1415→  return lines.map(line => {
  1416→    const debit = toNumber(line.debit)
  1417→    const credit = toNumber(line.credit)
  1418→    if (normalBalance === 'DEBIT') {
  1419→      runningBalance += debit - credit
  1420→    } else {
  1421→      runningBalance += credit - debit
  1422→    }
  1423→
  1424→    return {
  1425→      date: line.journalEntry.date,
  1426→      entryNo: line.journalEntry.entryNo,
  1427→      description: line.journalEntry.description,
  1428→      debit,
  1429→      credit,
  1430→      balance: runningBalance,
  1431→    }
  1432→  })
  1433→}

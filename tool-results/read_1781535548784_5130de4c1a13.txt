   272→export async function ensureAccountExists(template: AccountTemplate, tx?: PrismaTransaction) {
   273→  const client = tx || db
   274→  const existing = await client.account.findUnique({ where: { code: template.code } })
   275→  if (existing) {
   276→    // Update existing account with new fields if they are missing
   277→    if (
   278→      existing.activityType !== (template.activityType || null) ||
   279→      existing.isSystem !== (template.isSystem || false) ||
   280→      existing.allowPosting !== (template.allowPosting || false) ||
   281→      existing.level !== (template.level || 0) ||
   282→      existing.accountRole !== (template.accountRole || null)
   283→    ) {
   284→      await client.account.update({
   285→        where: { code: template.code },
   286→        data: {
   287→          name: template.name,
   288→          nameAr: template.nameAr,
   289→          type: template.type,
   290→          activityType: template.activityType || null,
   291→          isSystem: template.isSystem || false,
   292→          allowPosting: template.allowPosting || false,
   293→          level: template.level || 0,
   294→          accountRole: template.accountRole || null,
   295→        },
   296→      })
   297→    }
   298→    return client.account.findUnique({ where: { code: template.code } })
   299→  }
   300→
   301→  let parentId: string | undefined
   302→  if (template.parentId) {
   303→    const parent = await client.account.findUnique({ where: { code: template.parentId } })
   304→    if (parent) parentId = parent.id
   305→  }
   306→
   307→  return client.account.create({
   308→    data: {
   309→      code: template.code,
   310→      name: template.name,
   311→      nameAr: template.nameAr,
   312→      type: template.type,
   313→      parentId,
   314→      isActive: true,
   315→      activityType: template.activityType || null,
   316→      isSystem: template.isSystem || false,
   317→      allowPosting: template.allowPosting || false,
   318→      level: template.level || 0,
   319→      accountRole: template.accountRole || null,
   320→    },
   321→  })
   322→}

   324→// ============ INITIALIZE CHART OF ACCOUNTS ============
   325→
   326→export async function initializeChartOfAccounts() {
   327→  let created = 0
   328→  let updated = 0
   329→
   330→  // Create parent accounts first, then children (sorted by code length then code)
   331→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   332→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   333→    return a.code.localeCompare(b.code)
   334→  })
   335→
   336→  for (const tmpl of sorted) {
   337→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   338→    if (existing) {
   339→      // Update existing account with new fields
   340→      await db.account.update({
   341→        where: { code: tmpl.code },
   342→        data: {
   343→          name: tmpl.name,
   344→          nameAr: tmpl.nameAr,
   345→          type: tmpl.type,
   346→          activityType: tmpl.activityType || null,
   347→          isSystem: tmpl.isSystem || false,
   348→          allowPosting: tmpl.allowPosting || false,
   349→          level: tmpl.level || 0,
   350→          accountRole: tmpl.accountRole || null,
   351→          isActive: true,
   352→        },
   353→      })
   354→      // Update parent reference if needed
   355→      if (tmpl.parentId) {
   356→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   357→        if (parent && existing.parentId !== parent.id) {
   358→          await db.account.update({
   359→            where: { code: tmpl.code },
   360→            data: { parentId: parent.id },
   361→          })
   362→        }
   363→      }
   364→      updated++
   365→    } else {
   366→      await ensureAccountExists(tmpl)
   367→      created++
   368→    }
   369→  }
   370→
   371→  const total = await db.account.count()
   372→  return { created, updated, total }
   373→}

   377→export async function createJournalEntry(template: JournalEntryTemplate, tx?: PrismaTransaction) {
   378→  const client = tx || db
   379→  // Validate: total debits must equal total credits
   380→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   381→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   382→
   383→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   384→    throw new Error(
   385→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   386→    )
   387→  }
   388→
   389→  // Ensure all referenced accounts exist
   390→  const accountMap = new Map<string, { id: string; code: string; nameAr: string }>()
   391→  for (const line of template.lines) {
   392→    const account = await getAccountByCode(line.accountCode, tx)
   393→    if (!account) {
   394→      throw new Error(`Account not found: ${line.accountCode}`)
   395→    }
   396→    accountMap.set(line.accountCode, { id: account.id, code: account.code, nameAr: account.nameAr || '' })
   397→  }
   398→
   399→  const now = new Date()
   400→
   401→  // Create the journal entry with lines and audit log
   402→  const entry = await client.journalEntry.create({
   403→    data: {
   404→      entryNo: template.entryNo,
   405→      date: template.date,
   406→      description: template.description,
   407→      descriptionAr: template.descriptionAr,
   408→      status: 'POSTED', // Auto-entries are posted immediately
   409→      sourceType: template.sourceType,
   410→      sourceId: template.sourceId,
   411→      sourceDocumentNo: template.sourceDocumentNo,
   412→      activityType: template.activityType,
   413→      projectId: template.projectId,
   414→      clientId: template.clientId,
   415→      supplierId: template.supplierId,
   416→      isReversal: template.isReversal || false,
   417→      reversedEntryId: template.reversedEntryId,
   418→      reversalReason: template.reversalReason,
   419→      postedAt: now,
   420→      lines: {
   421→        create: template.lines.map((line) => {
   422→          const acct = accountMap.get(line.accountCode)!
   423→          return {
   424→            accountId: acct.id,
   425→            accountCode: acct.code,
   426→            accountNameAr: acct.nameAr,
   427→            costCenterId: line.costCenterId,
   428→            debit: line.debit,
   429→            credit: line.credit,
   430→            description: line.description,
   431→          };
   432→        }),
   433→      },
   434→      auditLogs: {
   435→        create: [
   436→          {
   437→            action: 'CREATED',
   438→            actionAr: 'إنشاء القيد',
   439→            performedBy: 'System',
   440→            details: `Auto-created for ${template.sourceType}`,
   441→          },
   442→          {
   443→            action: 'POSTED',
   444→            actionAr: 'ترحيل القيد',
   445→            performedBy: 'System',
   446→            details: `Auto-posted at ${now.toISOString()}`,
   447→          },
   448→        ],
   449→      },
   450→    },
   451→    include: { lines: true, auditLogs: true },
   452→  })
   453→
   454→  return entry
   455→}

   460→/**
   461→ * فاتورة مبيعات - Sales Invoice (from Extract)
   462→ * Dr: Clients Receivable (1210) - totalAmount
   463→ * Cr: Progress Claims Revenue (6110) - subtotal
   464→ * Cr: VAT Payable (3200) - vatAmount
   465→ */
   466→export async function autoEntrySalesInvoice(data: {
   467→  invoiceNo: string
   468→  clientId: string
   469→  subtotal: number
   470→  vatRate: number
   471→  vatAmount: number
   472→  totalAmount: number
   473→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   474→  date: Date
   475→  projectId?: string
   476→  costCenterId?: string
   477→}, tx?: PrismaTransaction) {
   478→  // Determine revenue account based on invoice type
   479→  let revenueAccountCode: string
   480→  switch (data.invoiceType) {
   481→    case 'RENTAL':
   482→      revenueAccountCode = '6210' // Equipment Rental Revenue
   483→      break
   484→    case 'PROGRESS_CLAIM':
   485→      revenueAccountCode = '6110' // Progress Claims Revenue
   486→      break
   487→    default:
   488→      revenueAccountCode = '6340' // Service Revenue
   489→  }
   490→
   491→  const lines = [
   492→    { accountCode: '1210', debit: data.totalAmount, credit: 0, costCenterId: data.costCenterId }, // Clients Receivable
   493→    { accountCode: revenueAccountCode, debit: 0, credit: data.subtotal, costCenterId: data.costCenterId }, // Revenue
   494→  ]
   495→
   496→  // Add VAT line only if VAT > 0 (Output VAT)
   497→  if (data.vatAmount > 0) {
   498→    lines.push({ accountCode: '3110', debit: 0, credit: data.vatAmount, costCenterId: data.costCenterId }) // Output VAT (ضريبة مخرجات)
   499→  }
   500→
   501→  return createJournalEntry({
   502→    entryNo: `JE-SI-${Date.now()}`,
   503→    date: data.date,
   504→    description: `Sales Invoice ${data.invoiceNo}`,
   505→    descriptionAr: `فاتورة مبيعات ${data.invoiceNo}`,
   506→    lines,
   507→    sourceType: 'SALES_INVOICE',
   508→    sourceId: data.invoiceNo,
   509→  }, tx)
   510→}

   512→/**
   513→ * فاتورة مشتريات - Purchase Invoice
   514→ * Dr: Expense/Asset account - subtotal
   515→ * Dr: VAT Receivable (1400) - vatAmount
   516→ * Cr: Suppliers Payable (3110) - totalAmount
   517→ */
   518→export async function autoEntryPurchaseInvoice(data: {
   519→  invoiceNo: string
   520→  supplierId: string
   521→  subtotal: number
   522→  vatRate: number
   523→  vatAmount: number
   524→  totalAmount: number

   527→  costCenterId?: string
   528→  expenseCategory?: string
   529→  activityType?: 'CONSTRUCTION' | 'EQUIPMENT_RENTAL' | 'BOTH'
   530→}, tx?: PrismaTransaction) {
   531→  // Determine expense account based on context
   532→  let expenseAccountCode = '7110' // Default: Material Costs
   533→  if (data.expenseCategory) {
   534→    const categoryMap: Record<string, string> = {
   535→      'CONSUMABLES': '7110',

   537→      'MAINTENANCE': '7220',
   538→      'FUEL': '7210',

   935→/**
   936→ * رواتب - Salary Entry (Accrual + Payment)
   937→ *
   938→ * Supports two modes via the `accrualOnly` parameter:
   939→ *
   940→ * ACCRUAL MODE (accrualOnly = true):
   941→ *   Used during payroll run to record the expense and liability.
   942→ *   Dr: Salaries & Wages (8110) - grossSalary
   943→ *   Dr: GOSI Expense (8210) - gosiEmployerContribution
   944→ *   Cr: Salaries Payable (3310) - netSalary (gross - gosiEmployeeDeduction)
   945→ *   Cr: GOSI Payable (3830) - total GOSI (employee + employer)
   946→ *
   947→ * PAYMENT MODE (accrualOnly = false, default):
   948→ *   Used when salary is actually paid to clear the payable.
   949→ *   Dr: Salaries Payable (3310) - netSalary
   950→ *   Cr: Cash/Bank (1110/1120) - netCashPaid (netSalary)
   951→ *
   952→ * FULL COMBINED MODE (for backward compatibility, use accrualOnly = undefined):
   953→ *   Creates a single entry that records both expense and payment.
   954→ *   Dr: Salaries & Wages (8110) - grossSalary
   955→ *   Dr: GOSI Expense (8210) - gosiEmployerContribution
   956→ *   Cr: Cash/Bank (1110/1120) - netCashPaid
   957→ *   Cr: GOSI Payable (3830) - total GOSI
   958→ *   Cr: Salaries & Wages (8110) - gosiEmployeeDeduction (contra-entry)
   959→ */
   960→export async function autoEntrySalary(data: {
   961→  employeeName: string
   962→  grossSalary: number
   963→  gosiEmployeeDeduction: number
   964→  gosiEmployerContribution: number
   965→  date: Date
   966→  payFrom: 'TREASURY' | 'BANK'
   967→  costCenterId?: string
   968→  accrualOnly?: boolean // true = accrual entry only, false/undefined = payment entry
   969→}, tx?: PrismaTransaction) {
   970→  const cashAccountCode = data.payFrom === 'BANK' ? '1120' : '1110'
   971→  const netSalary = data.grossSalary - data.gosiEmployeeDeduction
   972→
   973→  // ACCRUAL MODE: Record expense and liability without cash payment
   974→  if (data.accrualOnly === true) {
   975→    const lines = [
   976→      { accountCode: '8110', debit: data.grossSalary, credit: 0, costCenterId: data.costCenterId }, // Salaries & Wages (gross)
   977→      { accountCode: '8210', debit: data.gosiEmployerContribution, credit: 0, costCenterId: data.costCenterId }, // GOSI Expense (employer)
   978→    ]
   979→
   980→    // Credit Salaries Payable for the net amount
   981→    lines.push({ accountCode: '3310', debit: 0, credit: netSalary, costCenterId: data.costCenterId }) // Salaries Payable (رواتب مستحقة)
   982→
   983→    // Credit GOSI Payable for total GOSI (employee deduction + employer contribution)
   984→    const totalGOSI = data.gosiEmployeeDeduction + data.gosiEmployerContribution
   985→    if (totalGOSI > 0) {
   986→      lines.push({ accountCode: '3830', debit: 0, credit: totalGOSI, costCenterId: data.costCenterId }) // GOSI Payable
   987→    }
   988→
   989→    return createJournalEntry({
   990→      entryNo: `JE-SAL-ACC-${Date.now()}`,
   991→      date: data.date,
   992→      description: `Salary accrual - ${data.employeeName}`,
   993→      descriptionAr: `استحقاق راتب - ${data.employeeName}`,
   994→      lines,
   995→      sourceType: 'SALARY_ACCRUAL',
   996→      sourceId: `SAL-ACC-${Date.now()}`,
   997→    }, tx)
   998→  }
   999→
  1000→  // PAYMENT MODE: Clear the payable with cash
  1001→  // When accrualOnly is explicitly false, this is the payment step
  1002→  if (data.accrualOnly === false) {
  1003→    return createJournalEntry({
  1004→      entryNo: `JE-SAL-PAY-${Date.now()}`,
  1005→      date: data.date,
  1006→      description: `Salary payment - ${data.employeeName}`,
  1007→      descriptionAr: `صرف راتب - ${data.employeeName}`,
  1008→      lines: [
  1009→        { accountCode: '3310', debit: netSalary, credit: 0 }, // Dr: Salaries Payable
  1010→        { accountCode: cashAccountCode, debit: 0, credit: netSalary }, // Cr: Cash/Bank
  1011→      ],
  1012→      sourceType: 'SALARY_PAYMENT',
  1013→      sourceId: `SAL-PAY-${Date.now()}`,
  1014→    }, tx)
  1015→  }
  1016→
  1017→  // FULL COMBINED MODE (backward compatible): Record expense + payment in one entry
  1018→  const lines = [
  1019→    { accountCode: '8110', debit: data.grossSalary, credit: 0, costCenterId: data.costCenterId }, // Salaries & Wages (gross)
  1020→    { accountCode: '8210', debit: data.gosiEmployerContribution, credit: 0, costCenterId: data.costCenterId }, // GOSI Expense (employer)
  1021→  ]
  1022→
  1023→  if (data.gosiEmployeeDeduction > 0) {
  1024→    lines.push({ accountCode: '8110', debit: 0, credit: data.gosiEmployeeDeduction, costCenterId: data.costCenterId }) // Reduce salary by GOSI employee share
  1025→  }
  1026→
  1027→  lines.push({ accountCode: cashAccountCode, debit: 0, credit: netSalary, costCenterId: data.costCenterId }) // Cash/Bank paid
  1028→  lines.push({ accountCode: '3830', debit: 0, credit: data.gosiEmployeeDeduction + data.gosiEmployerContribution, costCenterId: data.costCenterId }) // GOSI Payable (total)
  1029→
  1030→  return createJournalEntry({
  1031→    entryNo: `JE-SAL-${Date.now()}`,
  1032→    date: data.date,
  1033→    description: `Salary payment - ${data.employeeName}`,
  1034→    descriptionAr: `صرف راتب - ${data.employeeName}`,
  1035→    lines,
  1036→    sourceType: 'SALARY',
  1037→    sourceId: `SAL-${Date.now()}`,
  1038→  }, tx)

  1309→ * Cr: VAT Due (3130) - net VAT payable (if output > input)

  1315→export async function autoEntryVATDeclaration(data: {

  1377→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1378→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1379→  if (dateFrom || dateTo) {
  1380→    dateFilter.date = {
  1381→      ...(dateFrom && { gte: dateFrom }),
  1382→      ...(dateTo && { lte: dateTo }),
  1383→    }
  1384→  }
  1385→  const entries = await db.journalEntry.findMany({
  1386→    where: {
  1387→      status: 'POSTED',
  1388→      ...dateFilter,
  1389→    },
  1390→    include: {
  1391→      lines: {
  1392→        include: {
  1393→          account: true,
  1394→        },
  1395→      },
  1396→    },
  1397→  })
  1398→
  1399→  // Aggregate by account
  1400→  const accountBalances = new Map<string, {
  1401→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1402→    totalDebit: number
  1403→    totalCredit: number
  1404→  }>()
  1405→
  1406→  for (const entry of entries) {
  1407→    for (const line of entry.lines) {
  1408→      const key = line.accountId
  1409→      if (!accountBalances.has(key)) {
  1410→        accountBalances.set(key, {
  1411→          account: line.account,
  1412→          totalDebit: 0,
  1413→          totalCredit: 0,
  1414→        })
  1415→      }
  1416→      const bal = accountBalances.get(key)!
  1417→      bal.totalDebit += toNumber(line.debit)
  1418→      bal.totalCredit += toNumber(line.credit)
  1419→    }
  1420→  }
  1421→
  1422→  // Calculate net balance for each account
  1423→  const results = Array.from(accountBalances.values()).map(bal => {
  1424→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1425→    let netDebit = 0
  1426→    let netCredit = 0
  1427→
  1428→    if (normalBalance === 'DEBIT') {
  1429→      const net = bal.totalDebit - bal.totalCredit
  1430→      if (net >= 0) netDebit = net
  1431→      else netCredit = Math.abs(net)
  1432→    } else {
  1433→      const net = bal.totalCredit - bal.totalDebit
  1434→      if (net >= 0) netCredit = net
  1435→      else netDebit = Math.abs(net)
  1436→    }
  1437→
  1438→    return {
  1439→      ...bal,
  1440→      netDebit,
  1441→      netCredit,
  1442→    }
  1443→  })
  1444→
  1445→  // Sort by account code
  1446→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1447→
  1448→  return results
  1449→}

  1495→export async function getGeneralLedger(accountCode: string, dateFrom?: Date, dateTo?: Date) {
  1496→  const account = await getAccountByCode(accountCode)
  1497→  if (!account) return []
  1498→
  1499→  const glDateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1500→  if (dateFrom || dateTo) {
  1501→    glDateFilter.date = {
  1502→      ...(dateFrom && { gte: dateFrom }),
  1503→      ...(dateTo && { lte: dateTo }),
  1504→    }
  1505→  }
  1506→  const lines = await db.journalLine.findMany({
  1507→    where: {
  1508→      accountId: account.id,
  1509→      journalEntry: {
  1510→        status: 'POSTED',
  1511→        ...glDateFilter,
  1512→      },
  1513→    },
  1514→    include: { journalEntry: true },
  1515→    orderBy: { journalEntry: { date: 'asc' } },
  1516→  })
  1517→
  1518→  let runningBalance = 0
  1519→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1520→
  1521→  return lines.map(line => {
  1522→    const debit = toNumber(line.debit)
  1523→    const credit = toNumber(line.credit)
  1524→    if (normalBalance === 'DEBIT') {
  1525→      runningBalance += debit - credit
  1526→    } else {
  1527→      runningBalance += credit - debit
  1528→    }
  1529→
  1530→    return {
  1531→      date: line.journalEntry.date,
  1532→      entryNo: line.journalEntry.entryNo,
  1533→      description: line.journalEntry.description,
  1534→      debit,
  1535→      credit,
  1536→      balance: runningBalance,
  1537→    }
  1538→  })
  1539→}

    78→// ============ Attendance Form Dialog ============
    79→function AttendanceFormDialog({ open, onOpenChange, employees }: {
    80→  open: boolean; onOpenChange: (open: boolean) => void; employees: Employee[]
    81→}) {
    82→  const queryClient = useQueryClient()
    83→  const [form, setForm] = useState<AttendanceFormData>(defaultForm)
    84→  const { lang } = useAppStore()
    85→
    86→  // Calculate work hours and auto-overtime from checkIn/checkOut
    87→  const calculatedWorkHours = React.useMemo(() => {
    88→    if (!form.checkIn || !form.checkOut) return null
    89→    const [inH, inM] = form.checkIn.split(':').map(Number)
    90→    const [outH, outM] = form.checkOut.split(':').map(Number)
    91→    if (isNaN(inH) || isNaN(outH)) return null
    92→    const inMin = inH * 60 + (inM || 0)
    93→    const outMin = outH * 60 + (outM || 0)
    94→    const diff = outMin - inMin
    95→    if (diff <= 0) return null
    96→    return Math.round((diff / 60) * 100) / 100
    97→  }, [form.checkIn, form.checkOut])
    98→
    99→  // Auto-calculate overtime (hours beyond 8)
   100→  const autoOvertime = React.useMemo(() => {
   101→    if (calculatedWorkHours === null) return 0
   102→    return Math.max(0, Math.round((calculatedWorkHours - 8) * 100) / 100)
   103→  }, [calculatedWorkHours])
   104→
   105→  // When status changes to ABSENT, auto-clear hours
   106→  React.useEffect(() => {
   107→    if (form.status === 'ABSENT') {
   108→      setForm(f => ({ ...f, overtimeHours: '0', checkIn: '', checkOut: '' }))
   109→    }
   110→  }, [form.status])
   111→
   112→  React.useEffect(() => {
   113→    if (open) setForm(defaultForm)
   114→  }, [open])
   115→
   116→  const createMutation = useMutation({
   117→    mutationFn: (data: Record<string, unknown>) => fetch('/api/attendance', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   118→    onSuccess: () => {
   119→      queryClient.invalidateQueries({ queryKey: ['attendance'] })
   120→      onOpenChange(false)
   121→      toast.success(t('تم تسجيل الحضور', 'Attendance recorded', lang))
   122→    },
   123→  })
   124→
   125→  const handleSubmit = (e: React.FormEvent) => {
   126→    e.preventDefault()
   127→    const payload: Record<string, unknown> = {
   128→      employeeId: form.employeeId,
   129→      date: form.date,
   130→      status: form.status,
   131→      checkIn: form.status === 'ABSENT' ? null : (form.checkIn || null),
   132→      checkOut: form.status === 'ABSENT' ? null : (form.checkOut || null),
   133→      overtimeHours: form.status === 'ABSENT' ? 0 : (form.overtimeHours ? parseFloat(form.overtimeHours) : autoOvertime),
   134→      leaveType: form.status === 'LEAVE' ? (form.leaveType || null) : null,
   135→    }
   136→    createMutation.mutate(payload)
   137→  }
   138→
   139→  const isAbsent = form.status === 'ABSENT'
   140→  const isLeave = form.status === 'LEAVE'
   141→
   142→  return (
   143→    <Dialog open={open} onOpenChange={onOpenChange}>
   144→      <DialogContent className="max-w-lg max-h-[90vh] overflow-y-auto">
   145→        <DialogHeader>
   146→          <DialogTitle>{t('تسجيل حضور', 'Record Attendance', lang)}</DialogTitle>
   147→          <DialogDescription>{t('تسجيل حضور وانصراف موظف', 'Record employee attendance', lang)}</DialogDescription>
   148→        </DialogHeader>
   149→        <form onSubmit={handleSubmit} className="space-y-4">
   150→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   151→            <div className="space-y-2 sm:col-span-2">
   152→              <Label>{t('الموظف *', 'Employee *', lang)}</Label>
   153→              <Select value={form.employeeId} onValueChange={v => setForm(f => ({ ...f, employeeId: v }))}>
   154→                <SelectTrigger><SelectValue placeholder={t('اختر الموظف', 'Select employee', lang)} /></SelectTrigger>
   155→                <SelectContent>
   156→                  {employees.map(e => <SelectItem key={e.id} value={e.id}>{e.name} ({e.code})</SelectItem>)}
   157→                </SelectContent>
   158→              </Select>
   159→            </div>
   160→            <div className="space-y-2"><Label>{t('التاريخ *', 'Date *', lang)}</Label><Input type="date" value={form.date} onChange={e => setForm(f => ({ ...f, date: e.target.value }))} required /></div>
   161→            <div className="space-y-2">
   162→              <Label>{t('الحالة *', 'Status *', lang)}</Label>
   163→              <Select value={form.status} onValueChange={v => setForm(f => ({ ...f, status: v, leaveType: '' }))}>
   164→                <SelectTrigger><SelectValue /></SelectTrigger>
   165→                <SelectContent>
   166→                  <SelectItem value="PRESENT">{t('حاضر', 'Present', lang)}</SelectItem>
   167→                  <SelectItem value="ABSENT">{t('غائب', 'Absent', lang)}</SelectItem>
   168→                  <SelectItem value="LEAVE">{t('إجازة', 'Leave', lang)}</SelectItem>
   169→                  <SelectItem value="HOLIDAY">{t('عطلة', 'Holiday', lang)}</SelectItem>
   170→                </SelectContent>
   171→              </Select>
   172→            </div>
   173→
   174→            {/* Leave Type - shown only when status is LEAVE */}
   175→            {isLeave && (
   176→              <div className="space-y-2 sm:col-span-2">
   177→                <Label>{t('نوع الإجازة', 'Leave Type', lang)}</Label>
   178→                <Select value={form.leaveType || '_none'} onValueChange={v => setForm(f => ({ ...f, leaveType: v === '_none' ? '' : v }))}>
   179→                  <SelectTrigger><SelectValue placeholder={t('اختر نوع الإجازة', 'Select leave type', lang)} /></SelectTrigger>
   180→                  <SelectContent>
   181→                    <SelectItem value="_none">{t('بدون تحديد', 'Not specified', lang)}</SelectItem>
   182→                    <SelectItem value="ANNUAL">{t('سنوية', 'Annual', lang)}</SelectItem>
   183→                    <SelectItem value="SICK">{t('مرضية', 'Sick', lang)}</SelectItem>
   184→                    <SelectItem value="UNPAID">{t('بدون راتب', 'Unpaid', lang)}</SelectItem>
   185→                  </SelectContent>
   186→                </Select>
   187→              </div>
   188→            )}
   189→
   190→            {/* Check-in/out - hidden when absent */}
   191→            {!isAbsent && (
   192→              <>
   193→                <div className="space-y-2"><Label>{t('وقت الحضور', 'Check In', lang)}</Label><Input type="time" value={form.checkIn} onChange={e => setForm(f => ({ ...f, checkIn: e.target.value }))} /></div>
   194→                <div className="space-y-2"><Label>{t('وقت الانصراف', 'Check Out', lang)}</Label><Input type="time" value={form.checkOut} onChange={e => setForm(f => ({ ...f, checkOut: e.target.value }))} /></div>
   195→                <div className="space-y-2">
   196→                  <Label>{t('ساعات إضافية', 'Overtime Hours', lang)}</Label>
   197→                  <Input
   198→                    type="number" min="0" step="0.5"
   199→                    value={form.overtimeHours || (autoOvertime > 0 ? String(autoOvertime) : '0')}
   200→                    onChange={e => setForm(f => ({ ...f, overtimeHours: e.target.value }))}
   201→                    dir="ltr"
   202→                  />
   203→                  {autoOvertime > 0 && <p className="text-xs text-amber-600">{t('محسوب تلقائياً', 'Auto-calculated', lang)}: {autoOvertime} {t('ساعة', 'hrs', lang)}</p>}
   204→                </div>
   205→              </>
   206→            )}
   207→          </div>
   208→
   209→          {calculatedWorkHours !== null && !isAbsent && (
   210→            <Card className="bg-emerald-50 border-emerald-200">
   211→              <CardContent className="p-3">
   212→                <div className="flex justify-between text-sm">
   213→                  <p className="text-emerald-600">{t('ساعات العمل المحسوبة', 'Calculated Work Hours', lang)}: <span className="font-bold text-emerald-700">{calculatedWorkHours} {t('ساعة', 'hrs', lang)}</span></p>
   214→                  {autoOvertime > 0 && (
   215→                    <p className="text-amber-600">{t('إضافي', 'Overtime', lang)}: <span className="font-bold text-amber-700">{autoOvertime} {t('ساعة', 'hrs', lang)}</span></p>
   216→                  )}

   218→              </CardContent>
   219→            </Card>
   220→          )}
   221→
   222→          <DialogFooter>
   223→            <Button type="button" variant="outline" onClick={() => onOpenChange(false)}>{t('إلغاء', 'Cancel', lang)}</Button>
   224→            <Button type="submit" disabled={createMutation.isPending || !form.employeeId || !form.date} className="bg-emerald-600 hover:bg-emerald-700">{createMutation.isPending ? t('جاري الحفظ...', 'Saving...', lang) : t('تسجيل', 'Record', lang)}</Button>
   225→          </DialogFooter>
   226→        </form>
   227→      </DialogContent>
   228→    </Dialog>
   229→  )
   230→}

   232→// ============ Bulk Attendance Dialog ============
   233→function BulkAttendanceDialog({ open, onOpenChange, employees }: {
   234→  open: boolean; onOpenChange: (open: boolean) => void; employees: Employee[]
   235→}) {
   236→  const queryClient = useQueryClient()
   237→  const { lang } = useAppStore()
   238→  const [date, setDate] = useState('')
   239→  const [checkIn, setCheckIn] = useState('08:00')
   240→  const [checkOut, setCheckOut] = useState('17:00')
   241→  const [selectedIds, setSelectedIds] = useState<string[]>([])
   242→
   243→  React.useEffect(() => {
   244→    if (open) {
   245→      setDate('')
   246→      setCheckIn('08:00')
   247→      setCheckOut('17:00')
   248→      setSelectedIds([])
   249→    }
   250→  }, [open])
   251→
   252→  const toggleEmployee = (id: string) => {
   253→    setSelectedIds(prev => prev.includes(id) ? prev.filter(i => i !== id) : [...prev, id])
   254→  }
   255→
   256→  const selectAll = () => {
   257→    setSelectedIds(employees.map(e => e.id))
   258→  }
   259→
   260→  const selectNone = () => {
   261→    setSelectedIds([])
   262→  }
   263→
   264→  // Calculate work hours from check-in/check-out
   265→  const calculatedHours = React.useMemo(() => {
   266→    if (!checkIn || !checkOut) return 0
   267→    const [inH, inM] = checkIn.split(':').map(Number)
   268→    const [outH, outM] = checkOut.split(':').map(Number)
   269→    if (isNaN(inH) || isNaN(outH)) return 0
   270→    const diff = (outH * 60 + (outM || 0)) - (inH * 60 + (inM || 0))
   271→    return diff > 0 ? Math.round((diff / 60) * 100) / 100 : 0
   272→  }, [checkIn, checkOut])
   273→
   274→  const autoOvertime = Math.max(0, calculatedHours - 8)
   275→
   276→  const bulkMutation = useMutation({
   277→    mutationFn: async () => {
   278→      const results = await Promise.allSettled(
   279→        selectedIds.map(employeeId =>
   280→          fetch('/api/attendance', {
   281→            method: 'POST',
   282→            headers: { 'Content-Type': 'application/json' },
   283→            body: JSON.stringify({
   284→              employeeId,
   285→              date,
   286→              checkIn,
   287→              checkOut,
   288→              overtimeHours: autoOvertime,
   289→              status: 'PRESENT',
   290→            }),
   291→          }).then(r => { if (!r.ok) throw new Error(); return r.json() })
   292→        )
   293→      )
   294→      return results
   295→    },
   296→    onSuccess: () => {
   297→      queryClient.invalidateQueries({ queryKey: ['attendance'] })
   298→      onOpenChange(false)
   299→      toast.success(t('تم تسجيل الحضور الجماعي', 'Bulk attendance recorded', lang))
   300→    },
   301→  })
   302→
   303→  const handleSubmit = (e: React.FormEvent) => {
   304→    e.preventDefault()
   305→    if (!date || selectedIds.length === 0) return
   306→    bulkMutation.mutate()
   307→  }
   308→
   309→  return (
   310→    <Dialog open={open} onOpenChange={onOpenChange}>
   311→      <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
   312→        <DialogHeader>
   313→          <DialogTitle>{t('تسجيل حضور جماعي', 'Bulk Attendance', lang)}</DialogTitle>
   314→          <DialogDescription>{t('تسجيل حضور لعدة موظفين في نفس اليوم', 'Record attendance for multiple employees on the same day', lang)}</DialogDescription>
   315→        </DialogHeader>
   316→        <form onSubmit={handleSubmit} className="space-y-4">
   317→          <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   318→            <div className="space-y-2"><Label>{t('التاريخ *', 'Date *', lang)}</Label><Input type="date" value={date} onChange={e => setDate(e.target.value)} required /></div>
   319→            <div className="space-y-2"><Label>{t('وقت الحضور', 'Check In', lang)}</Label><Input type="time" value={checkIn} onChange={e => setCheckIn(e.target.value)} /></div>
   320→            <div className="space-y-2"><Label>{t('وقت الانصراف', 'Check Out', lang)}</Label><Input type="time" value={checkOut} onChange={e => setCheckOut(e.target.value)} /></div>
   321→          </div>
   322→
   323→          {calculatedHours > 0 && (
   324→            <Card className="bg-emerald-50 border-emerald-200">
   325→              <CardContent className="p-3 flex justify-between text-sm">
   326→                <p className="text-emerald-600">{t('ساعات العمل', 'Work Hours', lang)}: <span className="font-bold">{calculatedHours}</span></p>
   327→                {autoOvertime > 0 && <p className="text-amber-600">{t('إضافي', 'Overtime', lang)}: <span className="font-bold">{autoOvertime}</span></p>}
   328→              </CardContent>
   329→            </Card>
   330→          )}
   331→
   332→          <div className="space-y-2">
   333→            <div className="flex items-center justify-between">
   334→              <h4 className="font-semibold text-sm text-emerald-700">
   335→                {t('اختر الموظفين', 'Select Employees', lang)} ({selectedIds.length}/{employees.length})
   336→              </h4>
   337→              <div className="flex gap-2">
   338→                <Button type="button" variant="ghost" size="sm" onClick={selectAll}>{t('تحديد الكل', 'Select All', lang)}</Button>
   339→                <Button type="button" variant="ghost" size="sm" onClick={selectNone}>{t('إلغاء التحديد', 'Deselect All', lang)}</Button>
   340→              </div>
   341→            </div>
   342→            <div className="max-h-48 overflow-y-auto border rounded-lg p-2 space-y-1">
   343→              {employees.map(emp => (
   344→                <label key={emp.id} className="flex items-center gap-2 p-1.5 rounded hover:bg-gray-50 cursor-pointer">
   345→                  <Checkbox
   346→                    checked={selectedIds.includes(emp.id)}
   347→                    onCheckedChange={() => toggleEmployee(emp.id)}
   348→                  />
   349→                  <span className="text-sm">{emp.name} <span className="text-muted-foreground">({emp.code})</span></span>
   350→                </label>
   351→              ))}
   352→            </div>
   353→          </div>
   354→
   355→          <DialogFooter>
   356→            <Button type="button" variant="outline" onClick={() => onOpenChange(false)}>{t('إلغاء', 'Cancel', lang)}</Button>
   357→            <Button type="submit" disabled={bulkMutation.isPending || !date || selectedIds.length === 0} className="bg-emerald-600 hover:bg-emerald-700">
   358→              {bulkMutation.isPending ? t('جاري التسجيل...', 'Recording...', lang) : t('تسجيل حضور', 'Record Attendance', lang)}
   359→            </Button>
   360→          </DialogFooter>
   361→        </form>
   362→      </DialogContent>
   363→    </Dialog>
   364→  )
   365→}

   367→// ============ Daily Grid View Dialog ============
   368→function DailyGridViewDialog({ open, onOpenChange, employees, attendanceRecords }: {
   369→  open: boolean; onOpenChange: (open: boolean) => void
   370→  employees: Employee[]; attendanceRecords: AttendanceRecord[]
   371→}) {
   372→  const { lang } = useAppStore()
   373→  const [selectedDate, setSelectedDate] = useState('')
   374→
   375→  React.useEffect(() => {
   376→    if (open) {
   377→      setSelectedDate(new Date().toISOString().split('T')[0])
   378→    }
   379→  }, [open])
   380→
   381→  // Build attendance map for the selected date
   382→  const dateAttendanceMap = useMemo(() => {
   383→    const map: Record<string, AttendanceRecord> = {}
   384→    if (!selectedDate) return map
   385→    attendanceRecords.forEach(r => {
   386→      const recordDate = r.date ? r.date.split('T')[0] : ''
   387→      if (recordDate === selectedDate) {
   388→        map[r.employeeId] = r
   389→      }
   390→    })
   391→    return map
   392→  }, [selectedDate, attendanceRecords])
   393→
   394→  // Stats for the selected date
   395→  const dateStats = useMemo(() => {
   396→    let present = 0, absent = 0, leave = 0, holiday = 0, noRecord = 0
   397→    employees.forEach(emp => {
   398→      const record = dateAttendanceMap[emp.id]
   399→      if (!record) {
   400→        noRecord++
   401→      } else {
   402→        switch (record.status) {
   403→          case 'PRESENT': present++; break
   404→          case 'ABSENT': absent++; break
   405→          case 'LEAVE': leave++; break
   406→          case 'HOLIDAY': holiday++; break
   407→          default: present++; break
   408→        }
   409→      }
   410→    })
   411→    return { present, absent, leave, holiday, noRecord }
   412→  }, [employees, dateAttendanceMap])
   413→
   414→  return (
   415→    <Dialog open={open} onOpenChange={onOpenChange}>
   416→      <DialogContent className="max-w-3xl max-h-[90vh] overflow-y-auto">
   417→        <DialogHeader>
   418→          <DialogTitle className="flex items-center gap-2">
   419→            <CalendarDays className="size-5" />
   420→            {t('عرض الحضور اليومي', 'Daily Attendance View', lang)}
   421→          </DialogTitle>
   422→          <DialogDescription>{t('عرض حالة حضور جميع الموظفين ليوم محدد', 'View all employees attendance status for a specific date', lang)}</DialogDescription>
   423→        </DialogHeader>
   424→
   425→        <div className="space-y-4">
   426→          <div className="space-y-2">
   427→            <Label>{t('التاريخ', 'Date', lang)}</Label>
   428→            <Input type="date" value={selectedDate} onChange={e => setSelectedDate(e.target.value)} />
   429→          </div>
   430→
   431→          {selectedDate && (
   432→            <>
   433→              {/* Summary */}
   434→              <div className="grid grid-cols-2 sm:grid-cols-5 gap-2">
   435→                <Card className="bg-emerald-50 border-emerald-200"><CardContent className="p-2 text-center">

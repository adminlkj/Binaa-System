   206вҶ’   206вҶ’function InvoiceCurrencySymbol({
   207вҶ’   207вҶ’  company,

   217вҶ’   217вҶ’    img ? invoiceSymbolCache.get(img) ?? null : null
   218вҶ’   218вҶ’  )
   219вҶ’   219вҶ’
   220вҶ’   220вҶ’  useEffect(() => {
   221вҶ’   221вҶ’    if (!img) return
   222вҶ’   222вҶ’    // If already processed (from initial cache or a prior fetch), no work to do.
   223вҶ’   223вҶ’    if (processed) return

   236вҶ’   236вҶ’      })
   237вҶ’   237вҶ’      .catch(() => {})
   238вҶ’   238вҶ’    return () => { cancelled = true }

   261вҶ’   261вҶ’      <span

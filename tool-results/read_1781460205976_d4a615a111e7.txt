     8→/**
     9→ * Generate and store ZATCA QR for a sales invoice after creation.
    10→ * Uses company settings for seller name and VAT number.
    11→ */
    12→async function storeZatcaQR(invoiceId: string, invoiceData: { date: Date; totalAmount: number; vatAmount: number }) {
    13→  try {
    14→    const settings = await db.companySetting.findFirst()
    15→    if (!settings?.taxNumber) return
    16→
    17→    const zatcaQr = generateZatcaQRForInvoice(invoiceData, {
    18→      nameAr: settings.nameAr,
    19→      nameEn: settings.nameEn,
    20→      taxNumber: settings.taxNumber,
    21→    })
    22→
    23→    if (zatcaQr) {
    24→      await db.salesInvoice.update({
    25→        where: { id: invoiceId },
    26→        data: { zatcaQr },
    27→      })
    28→    }
    29→  } catch (error) {
    30→    console.error('Failed to generate ZATCA QR for sales invoice:', error)
    31→    // Don't fail the invoice creation
    32→  }
    33→}

    35→export async function GET(request: Request) {
    36→  try {
    37→    const { searchParams } = new URL(request.url)
    38→    const clientId = searchParams.get('clientId')
    39→    const projectId = searchParams.get('projectId')
    40→    const status = searchParams.get('status')
    41→    const invoiceType = searchParams.get('invoiceType')
    42→    const sourceType = searchParams.get('sourceType')
    43→    const pageParam = searchParams.get('page')
    44→    const page = pageParam ? Math.max(1, parseInt(pageParam) || 1) : null
    45→    const pageSize = Math.max(1, parseInt(searchParams.get('pageSize') || '50') || 50)
    46→    const search = searchParams.get('search') || ''
    47→
    48→    const where: Record<string, unknown> = {}
    49→    if (clientId) where.clientId = clientId
    50→    if (projectId) where.projectId = projectId
    51→    if (status) where.status = status
    52→    if (invoiceType) where.invoiceType = invoiceType
    53→    if (sourceType) where.sourceType = sourceType
    54→    if (search) {
    55→      where.OR = [
    56→        { invoiceNo: { contains: search } },
    57→        { notes: { contains: search } },
    58→      ]
    59→    }
    60→
    61→    const include = {
    62→      client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
    63→      project: { select: { id: true, name: true, nameAr: true, code: true, projectType: true } },
    64→      contract: { select: { id: true, contractNo: true } },
    65→      timesheet: {
    66→        select: {
    67→          id: true, operatingHours: true, month: true, year: true, status: true,
    68→          project: { select: { id: true, name: true, code: true, projectType: true, client: { select: { id: true, name: true } } } },
    69→          equipment: { select: { id: true, name: true, code: true, nameAr: true } },
    70→          rental: { select: { id: true, hourlyRate: true, deliveryFees: true, deliveryFeesTaxable: true } },
    71→          contract: { select: { id: true, contractNo: true, hourlyRate: true, paymentTerms: true } },
    72→        },
    73→      },
    74→      progressClaim: {
    75→        select: {
    76→          id: true, claimNo: true, date: true, amount: true, vatAmount: true,
    77→          totalAmount: true, status: true, invoiced: true,
    78→          project: { select: { id: true, name: true, code: true, client: { select: { id: true, name: true } } } },
    79→          contract: { select: { id: true, contractNo: true } },
    80→        },
    81→      },
    82→      items: true,
    83→    }
    84→
    85→    const whereClause = Object.keys(where).length > 0 ? where : undefined
    86→
    87→    // Backward compatibility: return array if no page param, paginated object if page provided
    88→    if (page === null) {
    89→      const invoices = await db.salesInvoice.findMany({
    90→        where: whereClause,
    91→        include,
    92→        orderBy: { date: 'desc' },
    93→      })
    94→      return NextResponse.json(invoices)
    95→    }
    96→
    97→    const [data, total] = await Promise.all([
    98→      db.salesInvoice.findMany({
    99→        where: whereClause,
   100→        include,
   101→        orderBy: { date: 'desc' },
   102→        skip: (page - 1) * pageSize,
   103→        take: pageSize,
   104→      }),
   105→      db.salesInvoice.count({ where: whereClause }),
   106→    ])
   107→
   108→    return NextResponse.json({ data, total, page, pageSize, totalPages: Math.ceil(total / pageSize) })
   109→  } catch (error) {
   110→    console.error('[API] Failed to fetch sales invoices:', error)
   111→    return NextResponse.json({ error: 'Failed to fetch sales invoices', details: error instanceof Error ? error.message : 'Unknown error' }, { status: 500 })
   112→  }
   113→}

   115→export async function POST(request: Request) {
   116→  try {
   117→    const body = await request.json()
   118→    const { sourceType } = body
   119→
   120→    // ===== MODE A: Create from ProgressClaim (Extract) =====

   122→      return await createInvoiceFromExtract(body)

   125→    // ===== MODE B: Create from TimeSheet =====
   126→    if (sourceType === 'TIMESHEET') {
   127→      return await createInvoiceFromTimesheet(body)
   128→    }
   129→
   130→    // ===== LEGACY MODE: Manual creation with items array =====
   131→    return await createInvoiceManual(body)
   132→  } catch (error) {
   133→    console.error('[API] Failed to create sales invoice:', error)
   134→    return NextResponse.json({ error: 'Failed to create sales invoice', details: error instanceof Error ? error.message : 'Unknown error' }, { status: 500 })
   135→  }
   136→}

   139→// MODE A: Create invoice from ProgressClaim (Extract)

   141→async function createInvoiceFromExtract(body: Record<string, unknown>) {
   142→  const { progressClaimId, date, dueDate, notes } = body
   143→
   144→  if (!progressClaimId || typeof progressClaimId !== 'string') {
   145→    return NextResponse.json({ error: 'معرف المستخلص مطلوب' }, { status: 400 })
   146→  }
   147→
   148→  if (!date || !dueDate) {
   149→    return NextResponse.json({ error: 'التاريخ وتاريخ الاستحقاق مطلوبان' }, { status: 400 })
   150→  }
   151→
   152→  // 1. Fetch the ProgressClaim and verify it exists
   153→  const claim = await db.progressClaim.findUnique({
   154→    where: { id: progressClaimId },
   155→    include: {
   156→      project: { select: { id: true, name: true, code: true, clientId: true, client: { select: { id: true, name: true } } } },
   157→      contract: { select: { id: true, contractNo: true } },
   158→    },
   159→  })
   160→
   161→  if (!claim) {
   162→    return NextResponse.json({ error: 'المستخلص غير موجود' }, { status: 404 })
   163→  }
   164→
   165→  // 2. Verify the claim is APPROVED
   166→  if (claim.status !== 'APPROVED') {
   167→    return NextResponse.json({ error: 'يجب اعتماد المستخلص أولاً قبل إنشاء الفاتورة' }, { status: 400 })
   168→  }
   169→
   170→  // 3. DUPLICATE PREVENTION: Check that the claim is NOT already invoiced
   171→  if (claim.invoiced) {
   172→    return NextResponse.json({ error: 'تم إصدار فاتورة لهذا المستخلص بالفعل' }, { status: 400 })
   173→  }
   174→
   175→  // Also check via SalesInvoice table
   176→  const existingInvoice = await db.salesInvoice.findFirst({
   177→    where: { progressClaimId: claim.id },
   178→  })
   179→  if (existingInvoice) {
   180→    return NextResponse.json({ error: 'يوجد فاتورة مرتبطة بهذا المستخلص بالفعل' }, { status: 400 })
   181→  }
   182→
   183→  // 4. Auto-populate from claim
   184→  const clientId = claim.project.clientId
   185→  if (!clientId) {
   186→    return NextResponse.json({ error: 'العميل غير محدد للمشروع' }, { status: 400 })
   187→  }
   188→  const projectId = claim.projectId
   189→  const contractId = claim.contractId
   190→  const subtotal = claim.amount
   191→  const vatAmount = claim.vatAmount
   192→  const totalAmount = claim.totalAmount
   193→
   194→  // 5. Generate invoice number: PCL-YEAR-SEQ
   195→  const prefix = 'PCL'
   196→  const year = new Date().getFullYear()
   197→  const yearStr = String(year)
   198→  const likePattern = `${prefix}-${yearStr}-`
   199→
   200→  const lastInvoice = await db.salesInvoice.findFirst({
   201→    where: { invoiceNo: { startsWith: likePattern } },
   202→    orderBy: { invoiceNo: 'desc' },
   203→    select: { invoiceNo: true },
   204→  })
   205→
   206→  let seq = 1
   207→  if (lastInvoice) {
   208→    const parts = lastInvoice.invoiceNo.split('-')
   209→    const parsedSeq = parseInt(parts[2])
   210→    if (!isNaN(parsedSeq)) {
   211→      seq = parsedSeq + 1
   212→    }
   213→  }
   214→  const invoiceNo = `${prefix}-${yearStr}-${String(seq).padStart(4, '0')}`
   215→
   216→  const invoiceDate = new Date(date as string)
   217→  const invoiceDueDate = new Date(dueDate as string)
   218→
   219→  // 6-8. Create invoice + mark claim invoiced + create accounting entry (all in transaction)
   220→  const result = await db.$transaction(async (tx: PrismaTransaction) => {
   221→    // Create the invoice
   222→    const invoice = await tx.salesInvoice.create({
   223→      data: {
   224→        invoiceNo,
   225→        clientId,
   226→        projectId,
   227→        contractId,
   228→        date: invoiceDate,
   229→        dueDate: invoiceDueDate,
   230→        subtotal,
   231→        discountRate: 0,
   232→        discountAmount: 0,
   233→        netAmount: subtotal,
   234→        vatRate: claim.vatRate,
   235→        vatAmount,
   236→        totalAmount,
   237→        paidAmount: 0,
   238→        status: 'DRAFT',
   239→        invoiceType: 'PROGRESS_CLAIM',
   240→        sourceType: 'EXTRACT',
   241→        progressClaimId: claim.id,
   242→        contractNo: claim.contract.contractNo,
   243→        notes: (notes as string) || `فاتورة مستخلص رقم ${claim.claimNo}`,
   244→        items: {
   245→          create: [{
   246→            description: `مستخلص رقم ${claim.claimNo} - ${claim.project.name}`,
   247→            descriptionEn: `Progress Claim No. ${claim.claimNo} - ${claim.project.name}`,
   248→            quantity: 1,
   249→            unit: 'مستخلص',
   250→            unitPrice: subtotal,
   251→            totalPrice: subtotal,
   252→            itemType: 'PROGRESS_CLAIM',
   253→          }],
   254→        },
   255→      },
   256→      include: {
   257→        client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   258→        project: { select: { id: true, name: true, nameAr: true, code: true } },
   259→        contract: { select: { id: true, contractNo: true } },
   260→        timesheet: { select: { id: true, month: true, year: true, operatingHours: true, status: true } },
   261→        progressClaim: { select: { id: true, claimNo: true, status: true, invoiced: true } },
   262→        items: true,
   263→      },
   264→    })
   265→
   266→    // Mark the ProgressClaim as invoiced
   267→    await tx.progressClaim.update({
   268→      where: { id: claim.id },
   269→      data: { invoiced: true },
   270→    })
   271→
   272→    // Create auto accounting entry
   273→    try {
   274→      await createSalesInvoiceJournalEntry(invoice.id, tx)
   275→    } catch (accountingError) {
   276→      console.error('[API] Accounting entry failed for sales invoice from extract:', accountingError)
   277→    }
   278→
   279→    // Re-fetch to include journalEntryId
   280→    return await tx.salesInvoice.findUnique({
   281→      where: { id: invoice.id },
   282→      include: {
   283→        client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   284→        project: { select: { id: true, name: true, nameAr: true, code: true } },
   285→        contract: { select: { id: true, contractNo: true } },
   286→        timesheet: { select: { id: true, month: true, year: true, operatingHours: true, status: true } },
   287→        progressClaim: { select: { id: true, claimNo: true, status: true, invoiced: true } },
   288→        items: true,
   289→      },
   290→    })
   291→  })
   292→
   293→  // Generate and store ZATCA QR code
   294→  if (result?.id) {
   295→    await storeZatcaQR(result.id, {
   296→      date: result.date,
   297→      totalAmount: Number(result.totalAmount),
   298→      vatAmount: Number(result.vatAmount),
   299→    })
   300→  }
   301→
   302→  return NextResponse.json(result, { status: 201 })
   303→}

   305→// ============================================================================
   306→// MODE B: Create invoice from TimeSheet

   308→async function createInvoiceFromTimesheet(body: Record<string, unknown>) {
   309→  const { timesheetId, date, dueDate, notes } = body
   310→
   311→  if (!timesheetId || typeof timesheetId !== 'string') {
   312→    return NextResponse.json({ error: 'معرف التايم شيت مطلوب' }, { status: 400 })
   313→  }
   314→
   315→  if (!date || !dueDate) {
   316→    return NextResponse.json({ error: 'التاريخ وتاريخ الاستحقاق مطلوبان' }, { status: 400 })
   317→  }
   318→
   319→  // 1. Fetch the Timesheet with its rental/contract/equipment relations
   320→  const timesheet = await db.timesheet.findUnique({
   321→    where: { id: timesheetId },
   322→    include: {
   323→      contract: {
   324→        include: {
   325→          rental: {
   326→            select: { id: true, hourlyRate: true, pricingType: true, deliveryFees: true, deliveryFeesTaxable: true, clientId: true, status: true, salesOrderNo: true, paymentDuration: true },
   327→          },
   328→        },
   329→      },
   330→      equipment: { select: { id: true, code: true, name: true, nameAr: true } },
   331→      project: { select: { id: true, name: true, code: true, clientId: true } },
   332→      rental: {
   333→        select: { id: true, hourlyRate: true, pricingType: true, deliveryFees: true, deliveryFeesTaxable: true, clientId: true, status: true, salesOrderNo: true, paymentDuration: true },
   334→      },
   335→    },
   336→  })
   337→
   338→  if (!timesheet) {
   339→    return NextResponse.json({ error: 'التايم شيت غير موجود' }, { status: 404 })
   340→  }
   341→
   342→  // 2. Verify the Timesheet status is APPROVED (not DRAFT or INVOICED)
   343→  if (timesheet.status !== 'APPROVED') {
   344→    if (timesheet.status === 'INVOICED') {
   345→      return NextResponse.json({ error: 'تم إصدار فاتورة لهذا التايم شيت بالفعل' }, { status: 400 })
   346→    }
   347→    return NextResponse.json({ error: 'يجب اعتماد التايم شيت أولاً قبل إنشاء الفاتورة' }, { status: 400 })
   348→  }
   349→
   350→  // 3. Check if already invoiced via invoiced flag
   351→  if (timesheet.invoiced) {
   352→    return NextResponse.json({ error: 'تم إصدار فاتورة لهذا التايم شيت بالفعل' }, { status: 400 })
   353→  }
   354→
   355→  // Also check if there's already an ACTIVE invoice linked (not cancelled/draft)
   356→  const existingInvoice = await db.salesInvoice.findUnique({
   357→    where: { timesheetId: timesheet.id },
   358→  })
   359→  if (existingInvoice && existingInvoice.status !== 'CANCELLED' && existingInvoice.status !== 'DRAFT') {
   360→    return NextResponse.json({ error: 'تم إصدار فاتورة لهذا التايم شيت بالفعل' }, { status: 400 })
   361→  }
   362→  // If there's a cancelled/draft invoice, unlink it from the timesheet to allow re-invoicing
   363→  if (existingInvoice && (existingInvoice.status === 'CANCELLED' || existingInvoice.status === 'DRAFT')) {
   364→    await db.salesInvoice.update({
   365→      where: { id: existingInvoice.id },
   366→      data: { timesheetId: null },
   367→    })
   368→  }
   369→
   370→  // 4. Calculate: subtotal = operatingHours × hourlyRate (from contract)
   371→  const rental = timesheet.rental || timesheet.contract.rental
   372→  const hourlyRate = toNumber(rental?.hourlyRate || timesheet.contract.hourlyRate || 0)
   373→  const operatingHours = toNumber(timesheet.operatingHours)
   374→  const subtotal = Math.round(operatingHours * hourlyRate * 100) / 100
   375→
   376→  // 5. Add delivery fees if applicable (from rental contract)
   377→  const deliveryFees = toNumber(rental?.deliveryFees || timesheet.contract.deliveryFees || 0)
   378→  const deliveryFeesTaxable = rental?.deliveryFeesTaxable ?? timesheet.contract.deliveryFeesTaxable ?? true
   379→  const includeDelivery = deliveryFees > 0
   380→
   381→  const vatRate = 0.15
   382→  let deliveryVat = 0
   383→  if (includeDelivery && deliveryFeesTaxable) {
   384→    deliveryVat = Math.round(deliveryFees * vatRate * 100) / 100
   385→  }
   386→  const rentalVat = Math.round(subtotal * vatRate * 100) / 100
   387→  const vatAmount = Math.round((rentalVat + deliveryVat) * 100) / 100
   388→  const totalAmount = Math.round((subtotal + deliveryFees + vatAmount) * 100) / 100
   389→
   390→  // Get clientId from rental or project
   391→  const clientId = rental?.clientId || timesheet.project.clientId
   392→
   393→  // Month names for description
   394→  const monthNamesAr = [
   395→    'يناير', 'فبراير', 'مارس', 'أبريل', 'مايو', 'يونيو',
   396→    'يوليو', 'أغسطس', 'سبتمبر', 'أكتوبر', 'نوفمبر', 'ديسمبر',
   397→  ]
   398→  const monthLabel = `${monthNamesAr[timesheet.month - 1]} ${timesheet.year}`
   399→  const equipmentName = timesheet.equipment.nameAr || timesheet.equipment.name
   400→
   401→  // 6. Generate invoice number: RNT-YEAR-SEQ
   402→  const prefix = 'RNT'
   403→  const year = new Date().getFullYear()
   404→  const yearStr = String(year)
   405→  const likePattern = `${prefix}-${yearStr}-`
   406→
   407→  const lastInvoice = await db.salesInvoice.findFirst({
   408→    where: { invoiceNo: { startsWith: likePattern } },
   409→    orderBy: { invoiceNo: 'desc' },
   410→    select: { invoiceNo: true },
   411→  })
   412→
   413→  let seq = 1
   414→  if (lastInvoice) {
   415→    const parts = lastInvoice.invoiceNo.split('-')
   416→    const parsedSeq = parseInt(parts[2])
   417→    if (!isNaN(parsedSeq)) {
   418→      seq = parsedSeq + 1
   419→    }
   420→  }
   421→  const invoiceNo = `${prefix}-${yearStr}-${String(seq).padStart(4, '0')}`
   422→
   423→  const invoiceDate = new Date(date as string)
   424→  const invoiceDueDate = new Date(dueDate as string)
   425→
   426→  // 7. Create the invoice items
   427→  const invoiceItems = [
   428→    {
   429→      description: `تأجير ${equipmentName} - ${monthLabel} - ${operatingHours} ساعة`,
   430→      descriptionEn: `Equipment Rental - ${timesheet.equipment.name} - ${monthLabel} - ${operatingHours} hours`,
   431→      quantity: operatingHours,
   432→      unit: 'ساعة',
   433→      unitPrice: hourlyRate,
   434→      totalPrice: subtotal,
   435→      itemType: 'RENTAL',
   436→    },
   437→  ]
   438→
   439→  // Add delivery fee as separate item if applicable
   440→  if (includeDelivery && deliveryFees > 0) {
   441→    invoiceItems.push({
   442→      description: `رسوم نقل وتنزيل - ${equipmentName}`,
   443→      descriptionEn: `Delivery Fees - ${timesheet.equipment.name}`,
   444→      quantity: 1,
   445→      unit: 'خدمة',
   446→      unitPrice: deliveryFees,
   447→      totalPrice: deliveryFees,
   448→      itemType: 'DELIVERY',
   449→    })
   450→  }
   451→
   452→  // 8-9. Create invoice + mark timesheet + create accounting entry (all in transaction)
   453→  const result = await db.$transaction(async (tx: PrismaTransaction) => {
   454→    const invoice = await tx.salesInvoice.create({
   455→      data: {
   456→        invoiceNo,
   457→        clientId,
   458→        projectId: timesheet.projectId,
   459→        contractId: timesheet.contractId,
   460→        date: invoiceDate,
   461→        dueDate: invoiceDueDate,
   462→        subtotal,
   463→        discountRate: 0,
   464→        discountAmount: 0,
   465→        netAmount: subtotal,
   466→        vatRate,
   467→        vatAmount,
   468→        totalAmount,
   469→        paidAmount: 0,
   470→        status: 'DRAFT',
   471→        invoiceType: 'RENTAL',
   472→        sourceType: 'TIMESHEET',
   473→        timesheetId: timesheet.id,
   474→        contractNo: timesheet.contract.contractNo,
   475→        contractType: timesheet.contract.contractType,
   476→        contractPeriodStart: timesheet.contract.startDate,
   477→        contractPeriodEnd: timesheet.contract.endDate,
   478→        salesOrderNo: rental?.salesOrderNo || null,
   479→        equipmentName: timesheet.equipment.name,
   480→        operatingHours,
   481→        hourlyRate,
   482→        includeDelivery,
   483→        deliveryAmount: deliveryFees,
   484→        deliveryFeesTaxable,
   485→        includeVat: true,
   486→        deliveryMonth: `${timesheet.year}-${String(timesheet.month).padStart(2, '0')}`,
   487→        paymentTerms: rental?.paymentDuration || timesheet.contract.paymentTerms || null,
   488→        notes: (notes as string) || `فاتورة تأجير معدات - عقد ${timesheet.contract.contractNo} - ${monthLabel}`,
   489→        items: {
   490→          create: invoiceItems,
   491→        },
   492→      },
   493→      include: {
   494→        client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   495→        project: { select: { id: true, name: true, nameAr: true, code: true } },
   496→        contract: { select: { id: true, contractNo: true } },
   497→        timesheet: { select: { id: true, month: true, year: true, operatingHours: true, status: true } },
   498→        progressClaim: { select: { id: true, claimNo: true, status: true, invoiced: true } },
   499→        items: true,
   500→      },
   501→    })
   502→
   503→    // Mark the Timesheet as INVOICED, set invoiced=true and invoiceId
   504→    await tx.timesheet.update({
   505→      where: { id: timesheet.id },
   506→      data: { status: 'INVOICED', invoiced: true, invoiceId: invoice.id, approvedDate: timesheet.approvedDate || new Date() },
   507→    })
   508→
   509→    // Create auto accounting entry
   510→    try {

   530→  // Generate and store ZATCA QR code

   545→async function createInvoiceManual(body: Record<string, unknown>) {
   546→  const {
   547→    clientId, projectId, contractId, date, dueDate, notes, items,
   548→    vatRate = 0.15, discountRate = 0, discountAmount = 0,
   549→    invoiceType = 'TAX_INVOICE', paymentTerms,
   550→    referenceNo, contractNo, contractType, contractPeriodStart, contractPeriodEnd,
   551→    deliveryMonth, includeDelivery = false, deliveryAmount = 0, includeVat = true,
   552→  } = body as Record<string, unknown>
   553→
   554→  const typedItems = items as Array<{ description: string; descriptionEn?: string; quantity: number; unit?: string; unitPrice: number; itemType?: string }> | undefined
   555→
   556→  if (!clientId || !date || !dueDate || !typedItems?.length) {
   557→    return NextResponse.json({ error: 'البيانات المطلوبة غير مكتملة' }, { status: 400 })
   558→  }
   559→
   560→  // منع إنشاء فاتورة تأجير يدوياً - يجب إنشاؤها من التايم شيت فقط
   561→  // بما أن هذه الدالة تُستدعى فقط عندما لا يكون sourceType هو TIMESHEET أو EXTRACT،
   562→  // فإن أي فاتورة تأجير تصل هنا هي حتماً يدوية ويجب رفضها
   563→  if (invoiceType === 'RENTAL') {
   564→    return NextResponse.json({ error: 'لا يمكن إنشاء فاتورة تأجير يدوياً. يجب إنشاؤها من ساعات التشغيل (Time Sheet) فقط.' }, { status: 400 })
   565→  }
   566→
   567→  // Calculate totals from items
   568→  const subtotal = typedItems.reduce((sum: number, item: { quantity: number; unitPrice: number }) => {
   569→    return sum + (item.quantity * item.unitPrice)
   570→  }, 0)
   571→
   572→  const finalDiscountAmount = (discountAmount as number) || (subtotal * (discountRate as number))
   573→  const netAmount = subtotal - finalDiscountAmount
   574→  const deliveryTotal = includeDelivery ? (deliveryAmount as number) : 0
   575→  const vatAmount = includeVat ? (netAmount + deliveryTotal) * (vatRate as number) : 0
   576→  const totalAmount = netAmount + deliveryTotal + vatAmount
   577→
   578→  // Auto-generate invoice number: TYPE-YEAR-SEQ format
   579→  const typePrefixMap: Record<string, string> = {
   580→    TAX_INVOICE: 'SRV',
   581→    PROGRESS_CLAIM: 'PCL',
   582→    RENTAL: 'RNT',
   583→    SERVICE: 'SVC',
   584→  }
   585→  const prefix = typePrefixMap[invoiceType as string] || 'INV'
   586→  const year = new Date().getFullYear()
   587→  const yearStr = String(year)
   588→  const likePattern = `${prefix}-${yearStr}-`
   589→
   590→  const lastInvoice = await db.salesInvoice.findFirst({
   591→    where: { invoiceNo: { startsWith: likePattern } },
   592→    orderBy: { invoiceNo: 'desc' },
   593→    select: { invoiceNo: true },
   594→  })
   595→
   596→  let seq = 1
   597→  if (lastInvoice) {
   598→    const parts = lastInvoice.invoiceNo.split('-')
   599→    const parsedSeq = parseInt(parts[2])
   600→    if (!isNaN(parsedSeq)) {
   601→      seq = parsedSeq + 1
   602→    }
   603→  }
   604→  const invoiceNo = `${prefix}-${yearStr}-${String(seq).padStart(4, '0')}`
   605→
   606→  // Create invoice + accounting entry in transaction
   607→  const result = await db.$transaction(async (tx: PrismaTransaction) => {
   608→    const invoice = await tx.salesInvoice.create({
   609→      data: {
   610→        invoiceNo,
   611→        clientId: clientId as string,
   612→        projectId: (projectId as string) || null,
   613→        contractId: (contractId as string) || null,
   614→        date: new Date(date as string),
   615→        dueDate: new Date(dueDate as string),
   616→        subtotal,
   617→        discountRate: discountRate as number,
   618→        discountAmount: finalDiscountAmount,
   619→        netAmount,
   620→        vatRate: vatRate as number,
   621→        vatAmount,
   622→        totalAmount,
   623→        paidAmount: 0,
   624→        status: 'DRAFT',
   625→        invoiceType: invoiceType as string,
   626→        notes: (notes as string) || null,
   627→        paymentTerms: (paymentTerms as string) || null,
   628→        referenceNo: (referenceNo as string) || null,
   629→        contractNo: (contractNo as string) || null,
   630→        contractType: (contractType as string) || null,
   631→        contractPeriodStart: contractPeriodStart ? new Date(contractPeriodStart as string) : null,
   632→        contractPeriodEnd: contractPeriodEnd ? new Date(contractPeriodEnd as string) : null,
   633→        deliveryMonth: (deliveryMonth as string) || null,
   634→        includeDelivery: includeDelivery as boolean,
   635→        deliveryAmount: includeDelivery ? (deliveryAmount as number) : 0,
   636→        includeVat: includeVat as boolean,
   637→        items: {
   638→          create: typedItems.map((item: { description: string; descriptionEn?: string; quantity: number; unit?: string; unitPrice: number; itemType?: string }) => ({
   639→            description: item.description,
   640→            descriptionEn: item.descriptionEn || null,
   641→            quantity: item.quantity,
   642→            unit: item.unit || null,
   643→            unitPrice: item.unitPrice,
   644→            totalPrice: item.quantity * item.unitPrice,
   645→            itemType: item.itemType || 'PRODUCT',
   646→          })),
   647→        },
   648→      },
   649→      include: {
   650→        client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   651→        project: { select: { id: true, name: true, nameAr: true, code: true } },
   652→        contract: { select: { id: true, contractNo: true } },
   653→        items: true,
   654→      },
   655→    })
   656→
   657→    // Auto-create accounting journal entry
   658→    try {
   659→      await createSalesInvoiceJournalEntry(invoice.id, tx)
   660→    } catch (accountingError) {
   661→      console.error('[API] Accounting entry failed for sales invoice:', accountingError)
   662→    }
   663→
   664→    // Re-fetch to include journalEntryId
   665→    return await tx.salesInvoice.findUnique({
   666→      where: { id: invoice.id },
   667→      include: {
   668→        client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   669→        project: { select: { id: true, name: true, nameAr: true, code: true } },
   670→        contract: { select: { id: true, contractNo: true } },
   671→        items: true,
   672→      },
   673→    })
   674→  })
   675→
   676→  // Generate and store ZATCA QR code
   677→  if (result?.id) {
   678→    await storeZatcaQR(result.id, {
   679→      date: result.date,
   680→      totalAmount: Number(result.totalAmount),
   681→      vatAmount: Number(result.vatAmount),
   682→    })
   683→  }
   684→
   685→  return NextResponse.json(result, { status: 201 })
   686→}

   688→// PUT: Update a sales invoice (with reversal for approved/posted invoices)
   689→export async function PUT(request: Request) {
   690→  try {
   691→    const body = await request.json()
   692→    const { id, ...updateData } = body
   693→
   694→    if (!id) {
   695→      return NextResponse.json({ error: 'معرف الفاتورة مطلوب' }, { status: 400 })
   696→    }
   697→
   698→    const existing = await db.salesInvoice.findUnique({ where: { id } })
   699→    if (!existing) {
   700→      return NextResponse.json({ error: 'الفاتورة غير موجودة' }, { status: 404 })
   701→    }
   702→
   703→    // Cannot modify CANCELLED invoices
   704→    if (existing.status === 'CANCELLED') {
   705→      return NextResponse.json({ error: 'لا يمكن تعديل فاتورة ملغاة' }, { status: 400 })
   706→    }
   707→
   708→    // If the invoice has been approved/posted (has a journal entry), create a reversal + new entry
   709→    if (existing.journalEntryId && (updateData.subtotal !== undefined || updateData.totalAmount !== undefined || updateData.vatAmount !== undefined)) {
   710→      // Perform reversal + new entry + update in a transaction
   711→      await db.$transaction(async (tx: PrismaTransaction) => {
   712→        // Create reversal entry for the original
   713→        const originalEntry = await tx.journalEntry.findUnique({
   714→          where: { id: existing.journalEntryId! },
   715→          include: { lines: true },
   716→        })
   717→
   718→        if (originalEntry) {
   719→          // Create reversal entry (swap debit/credit)
   720→          const accountIds = originalEntry.lines.map(l => l.accountId)
   721→          const accounts = await tx.account.findMany({ where: { id: { in: accountIds } } })
   722→          const accountMap = new Map(accounts.map(a => [a.id, a.code]))
   723→
   724→          const resolvedReversalLines = originalEntry.lines.map(line => ({
   725→            accountCode: accountMap.get(line.accountId) || '',
   726→            debit: toNumber(line.credit),
   727→            credit: toNumber(line.debit),
   728→            costCenterId: line.costCenterId || undefined,
   729→            description: `Reversal: ${line.description || ''}`,
   730→          }))
   731→
   732→          await createJournalEntry({
   733→            entryNo: `JE-REV-SI-${Date.now()}`,
   734→            date: new Date(),
   735→            description: `Reversal for Sales Invoice ${existing.invoiceNo}`,
   736→            descriptionAr: `قيد عكسي لفاتورة مبيعات ${existing.invoiceNo}`,
   737→            lines: resolvedReversalLines,
   738→            sourceType: 'SALES_INVOICE_REVERSAL',
   739→            sourceId: existing.invoiceNo,
   740→          }, tx)
   741→
   742→          // Cancel the original entry
   743→          await tx.journalEntry.update({
   744→            where: { id: existing.journalEntryId! },
   745→            data: { status: 'CANCELLED' },
   746→          })
   747→        }
   748→
   749→        // Create new entry with updated values
   750→        const newSubtotal = updateData.subtotal ?? existing.subtotal
   751→        const newVatAmount = updateData.vatAmount ?? existing.vatAmount
   752→        const newDate = updateData.date ? new Date(updateData.date) : existing.date
   753→
   754→        // Use auto-journal for the new entry
   755→        // First update the invoice temporarily so createSalesInvoiceJournalEntry reads the new values
   756→        await tx.salesInvoice.update({
   757→          where: { id: existing.id },
   758→          data: {
   759→            subtotal: newSubtotal,
   760→            vatAmount: newVatAmount,
   761→            totalAmount: updateData.totalAmount ?? existing.totalAmount,
   762→            date: newDate,
   763→          },
   764→        })
   765→
   766→        try {
   767→          await createSalesInvoiceJournalEntry(existing.id, tx)
   768→        } catch (journalError) {
   769→          console.error('[API] Failed to create replacement journal entry:', journalError)
   770→        }
   771→      })
   772→    }
   773→
   774→    // Update the invoice
   775→    const updated = await db.salesInvoice.update({
   776→      where: { id },
   777→      data: {
   778→        ...updateData,
   779→        ...(updateData.date && { date: new Date(updateData.date) }),
   780→        ...(updateData.dueDate && { dueDate: new Date(updateData.dueDate) }),
   781→      },
   782→      include: {
   783→        client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   784→        project: { select: { id: true, name: true, nameAr: true, code: true } },
   785→        contract: { select: { id: true, contractNo: true } },
   786→        timesheet: {
   787→          select: {
   788→            id: true, operatingHours: true, month: true, year: true, status: true,
   789→            project: { select: { id: true, name: true, code: true, client: { select: { id: true, name: true } } } },
   790→            equipment: { select: { id: true, name: true, code: true, nameAr: true } },
   791→            rental: { select: { id: true, hourlyRate: true, deliveryFees: true, deliveryFeesTaxable: true } },
   792→            contract: { select: { id: true, contractNo: true, hourlyRate: true, paymentTerms: true } },
   793→          },
   794→        },
   795→        progressClaim: {
   796→          select: {
   797→            id: true, claimNo: true, date: true, amount: true, vatAmount: true,
   798→            totalAmount: true, status: true, invoiced: true,
   799→            project: { select: { id: true, name: true, code: true, client: { select: { id: true, name: true } } } },
   800→            contract: { select: { id: true, contractNo: true } },
   801→          },
   802→        },
   803→        items: true,
   804→      },
   805→    })
   806→
   807→    return NextResponse.json(updated)
   808→  } catch (error) {
   809→    console.error('[API] Failed to update sales invoice:', error)
   810→    return NextResponse.json({ error: 'Failed to update sales invoice', details: error instanceof Error ? error.message : 'Unknown error' }, { status: 500 })
   811→  }
   812→}

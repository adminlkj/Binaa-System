   317→  })

   320→    mutationFn: (id: string) => fetch(`/api/employees/${id}`, { method: 'DELETE' }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   321→    onSuccess: () => queryClient.invalidateQueries({ queryKey: ['employees'] }),
   322→  })

   324→  const filtered = employees.filter(e => {
   325→    if (!search) return true
   326→    const s = search.toLowerCase()
   327→    return e.name.toLowerCase().includes(s) || e.code.toLowerCase().includes(s) || (e.nameAr?.toLowerCase().includes(s)) || (e.phone?.includes(s)) || (e.profession?.toLowerCase().includes(s)) || (e.department?.toLowerCase().includes(s))
   328→  })
   329→
   330→  // Build a project lookup map
   331→  const projectMap = useMemo(() => {
   332→    const map: Record<string, Project> = {}
   333→    projects.forEach(p => { map[p.id] = p })
   334→    employees.forEach(e => {
   335→      if (e.project && e.projectId) map[e.projectId] = e.project
   336→    })
   337→    return map
   338→  }, [projects, employees])
   339→
   340→  const printData = useMemo(() => ({
   341→    columns: [
   342→      { key: 'code', label: lang === 'ar' ? 'الكود' : 'Code' },
   343→      { key: 'name', label: lang === 'ar' ? 'الاسم' : 'Name' },
   344→      { key: 'nationality', label: lang === 'ar' ? 'الجنسية' : 'Nationality' },
   345→      { key: 'profession', label: lang === 'ar' ? 'المهنة' : 'Profession' },
   346→      { key: 'department', label: lang === 'ar' ? 'القسم' : 'Department' },

   348→      { key: 'salaryType', label: lang === 'ar' ? 'النوع' : 'Type' },
   349→      { key: 'status', label: lang === 'ar' ? 'الحالة' : 'Status' },

   351→    rows: filtered.map(e => ({

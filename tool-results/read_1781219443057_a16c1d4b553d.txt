   126→function HubMetricCard({ title, value, icon: Icon, theme }: {
   127→  title: string; value: React.ReactNode; icon: React.ElementType; theme: 'emerald' | 'cyan'
   128→}) {
   129→  const themeClasses = theme === 'emerald'
   130→    ? 'bg-emerald-50 border-emerald-100'
   131→    : 'bg-cyan-50 border-cyan-100'
   132→  const iconClasses = theme === 'emerald'
   133→    ? 'bg-emerald-100 text-emerald-600'
   134→    : 'bg-cyan-100 text-cyan-600'
   135→
   136→  return (
   137→    <Card className={`border ${themeClasses}`}>
   138→      <CardContent className="p-3">
   139→        <div className="flex items-center gap-2">
   140→          <div className={`flex size-8 shrink-0 items-center justify-center rounded-lg ${iconClasses}`}>
   141→            <Icon className="size-4" />
   142→          </div>
   143→          <div className="min-w-0 flex-1">
   144→            <p className="text-[11px] font-medium text-muted-foreground leading-tight">{title}</p>
   145→            <div className="mt-0.5">{value}</div>
   146→          </div>
   147→        </div>
   148→      </CardContent>
   149→    </Card>
   150→  )

   153→function WorkflowChain({ steps, theme, lang }: {
   154→  steps: typeof CONSTRUCTION_WORKFLOW
   155→  theme: 'emerald' | 'cyan'
   156→  lang: 'ar' | 'en'
   157→}) {
   158→  const activeColor = theme === 'emerald'
   159→    ? 'bg-emerald-100 text-emerald-700 border-emerald-300'
   160→    : 'bg-cyan-100 text-cyan-700 border-cyan-300'
   161→  const pendingColor = 'bg-gray-50 text-gray-400 border-gray-200'
   162→  const arrowColor = theme === 'emerald' ? 'text-emerald-400' : 'text-cyan-400'
   163→
   164→  return (
   165→    <div className="flex items-center gap-1 overflow-x-auto pb-2 scrollbar-thin" dir="ltr">
   166→      {steps.map((step, i) => (
   167→        <React.Fragment key={step.step}>
   168→          <div
   169→            className={`shrink-0 rounded-full border px-2.5 py-1 text-[10px] font-medium whitespace-nowrap transition-colors ${activeColor}`}
   170→          >
   171→            {lang === 'ar' ? step.label.ar : step.label.en}
   172→          </div>
   173→          {i < steps.length - 1 && (
   174→            <ArrowRight className={`size-3 shrink-0 ${arrowColor}`} />
   175→          )}
   176→        </React.Fragment>
   177→      ))}
   178→    </div>
   179→  )

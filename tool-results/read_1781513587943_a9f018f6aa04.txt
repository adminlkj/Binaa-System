   405→// ============ Payroll Run Detail View ============
   406→function PayrollRunDetail({ payrollRun, onBack }: {
   407→  payrollRun: PayrollRun; onBack: () => void
   408→}) {
   409→  const queryClient = useQueryClient()
   410→  const { lang } = useAppStore()
   411→  const lines = payrollRun.lines || []
   412→
   413→  // Calculate totals from lines
   414→  const totalGosi = lines.reduce((sum, l) => sum + (l.gosiDeduction || 0), 0)
   415→  const totalOvertime = lines.reduce((sum, l) => sum + (l.overtimeAmount || 0), 0)
   416→
   417→  const statusMutation = useMutation({
   418→    mutationFn: (status: PayrollRunStatus) => fetch(`/api/payroll-runs/${payrollRun.id}`, {
   419→      method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ status }),
   420→    }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   421→    onSuccess: () => queryClient.invalidateQueries({ queryKey: ['payroll-runs'] }),
   422→  })
   423→
   424→  const printData = useMemo(() => ({
   425→    columns: [
   426→      { key: 'code', label: lang === 'ar' ? 'كود الموظف' : 'Emp. Code' },
   427→      { key: 'name', label: lang === 'ar' ? 'الموظف' : 'Employee' },
   428→      { key: 'salaryType', label: lang === 'ar' ? 'النوع' : 'Type' },
   429→      { key: 'hours', label: lang === 'ar' ? 'الساعات' : 'Hours' },
   430→      { key: 'totalEntitlement', label: lang === 'ar' ? 'الاستحقاق' : 'Entitlement' },
   431→      { key: 'deductions', label: lang === 'ar' ? 'الخصومات' : 'Deductions' },
   432→      { key: 'gosi', label: lang === 'ar' ? 'التأمينات' : 'GOSI' },
   433→      { key: 'netSalary', label: lang === 'ar' ? 'الصافي' : 'Net' },
   434→      { key: 'project', label: lang === 'ar' ? 'المشروع' : 'Project' },
   435→    ],
   436→    rows: lines.map(l => ({

   152→// ============ Purchase Request Form Dialog ============
   153→function PurchaseRequestFormDialog({
   154→  open, onOpenChange, projects,
   155→}: {
   156→  open: boolean; onOpenChange: (open: boolean) => void; projects: ProjectOption[]
   157→}) {
   158→  const { lang } = useAppStore()
   159→  const queryClient = useQueryClient()
   160→  const [projectId, setProjectId] = useState('')
   161→  const [date, setDate] = useState('')
   162→  const [description, setDescription] = useState('')
   163→  const [requestedBy, setRequestedBy] = useState('')
   164→  const [items, setItems] = useState<{ description: string; quantity: string; unit: string; notes: string }[]>([
   165→    { description: '', quantity: '1', unit: '', notes: '' },
   166→  ])
   167→
   168→  React.useEffect(() => {
   169→    if (open) { setProjectId(''); setDate(''); setDescription(''); setRequestedBy(''); setItems([{ description: '', quantity: '1', unit: '', notes: '' }]) }
   170→  }, [open])
   171→
   172→  const addItem = () => setItems([...items, { description: '', quantity: '1', unit: '', notes: '' }])
   173→  const removeItem = (idx: number) => { if (items.length > 1) setItems(items.filter((_, i) => i !== idx)) }
   174→  const updateItem = (idx: number, field: string, value: string) => {
   175→    setItems(items.map((item, i) => i === idx ? { ...item, [field]: value } : item))
   176→  }
   177→
   178→  const createMutation = useMutation({
   179→    mutationFn: (data: Record<string, unknown>) =>
   180→      fetch('/api/purchase-requests', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   181→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['purchase-requests'] }); onOpenChange(false) },
   182→  })
   183→
   184→  const handleSubmit = (e: React.FormEvent) => {
   185→    e.preventDefault()
   186→    createMutation.mutate({
   187→      projectId: projectId || null, date, description: description || null,
   188→      requestedBy: requestedBy || null,
   189→      items: items.map(i => ({ description: i.description, quantity: parseFloat(i.quantity) || 1, unit: i.unit || null, notes: i.notes || null })),
   190→    })
   191→  }
   192→
   193→  return (
   194→    <Dialog open={open} onOpenChange={onOpenChange}>
   195→      <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
   196→        <DialogHeader>
   197→          <DialogTitle>{t(lang, 'طلب شراء جديد', 'New Purchase Request')}</DialogTitle>
   198→          <DialogDescription>{t(lang, 'إنشاء طلب شراء جديد', 'Create a new purchase request')}</DialogDescription>
   199→        </DialogHeader>
   200→        <form onSubmit={handleSubmit} className="space-y-4">
   201→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   202→            <div className="space-y-2">
   203→              <Label>{t(lang, 'المشروع (اختياري)', 'Project (optional)')}</Label>
   204→              <Select value={projectId} onValueChange={setProjectId}>
   205→                <SelectTrigger className="w-full"><SelectValue placeholder={t(lang, 'اختر المشروع', 'Select project')} /></SelectTrigger>
   206→                <SelectContent>
   207→                  {projects.map(p => <SelectItem key={p.id} value={p.id}>{p.name}</SelectItem>)}
   208→                </SelectContent>
   209→              </Select>
   210→            </div>
   211→            <div className="space-y-2">
   212→              <Label>{t(lang, 'التاريخ *', 'Date *')}</Label>
   213→              <Input type="date" value={date} onChange={e => setDate(e.target.value)} required />
   214→            </div>
   215→            <div className="space-y-2 sm:col-span-2">
   216→              <Label>{t(lang, 'الوصف', 'Description')}</Label>
   217→              <Input value={description} onChange={e => setDescription(e.target.value)} placeholder={t(lang, 'وصف الطلب', 'Request description')} />
   218→            </div>
   219→            <div className="space-y-2 sm:col-span-2">
   220→              <Label>{t(lang, 'مقدم الطلب', 'Requested By')}</Label>
   221→              <Input value={requestedBy} onChange={e => setRequestedBy(e.target.value)} placeholder={t(lang, 'اسم مقدم الطلب', 'Requester name')} />
   222→            </div>
   223→          </div>
   224→
   225→          {/* Items */}
   226→          <div className="space-y-3">
   227→            <div className="flex items-center justify-between">
   228→              <Label className="text-base font-semibold">{t(lang, 'بنود الطلب', 'Request Items')}</Label>
   229→              <Button type="button" variant="outline" size="sm" className="gap-1" onClick={addItem}>
   230→                <Plus className="size-3" /> {t(lang, 'إضافة بند', 'Add Item')}
   231→              </Button>
   232→            </div>
   233→            <div className="space-y-2 max-h-48 overflow-y-auto">
   234→              {items.map((item, idx) => (
   235→                <div key={idx} className="flex items-end gap-2 p-2 rounded-lg border bg-gray-50">
   236→                  <div className="flex-1 min-w-0">
   237→                    <Label className="text-xs">{t(lang, 'الوصف', 'Description')}</Label>
   238→                    <Input value={item.description} onChange={e => updateItem(idx, 'description', e.target.value)} placeholder={t(lang, 'وصف البند', 'Item description')} className="h-9" />
   239→                  </div>
   240→                  <div className="w-20">
   241→                    <Label className="text-xs">{t(lang, 'الكمية', 'Qty')}</Label>
   242→                    <Input type="number" min="0" step="0.01" value={item.quantity} onChange={e => updateItem(idx, 'quantity', e.target.value)} className="h-9" dir="ltr" />
   243→                  </div>
   244→                  <div className="w-20">
   245→                    <Label className="text-xs">{t(lang, 'الوحدة', 'Unit')}</Label>
   246→                    <Input value={item.unit} onChange={e => updateItem(idx, 'unit', e.target.value)} placeholder={t(lang, 'وحدة', 'Unit')} className="h-9" />
   247→                  </div>
   248→                  <Button type="button" variant="ghost" size="icon" className="size-9 shrink-0 text-rose-500" onClick={() => removeItem(idx)} disabled={items.length <= 1}>
   249→                    <X className="size-4" />
   250→                  </Button>
   251→                </div>
   252→              ))}
   253→            </div>
   254→          </div>
   255→
   256→          <DialogFooter>
   257→            <Button type="button" variant="outline" onClick={() => onOpenChange(false)}>{t(lang, 'إلغاء', 'Cancel')}</Button>
   258→            <Button type="submit" disabled={createMutation.isPending || !date} className="bg-emerald-600 hover:bg-emerald-700">
   259→              {createMutation.isPending ? t(lang, 'جاري الإنشاء...', 'Creating...') : t(lang, 'إنشاء طلب الشراء', 'Create Purchase Request')}
   260→            </Button>
   261→          </DialogFooter>
   262→        </form>
   263→      </DialogContent>
   264→    </Dialog>
   265→  )
   266→}

   269→function PurchaseOrderFormDialog({

   272→  open: boolean; onOpenChange: (open: boolean) => void
   273→  suppliers: SupplierOption[]; projects: ProjectOption[]
   274→}) {
   275→  const { lang } = useAppStore()
   276→  const queryClient = useQueryClient()
   277→
   278→  const [supplierId, setSupplierId] = useState('')
   279→  const [projectId, setProjectId] = useState('')
   280→  const [date, setDate] = useState('')
   281→  const [deliveryDate, setDeliveryDate] = useState('')
   282→  const [notes, setNotes] = useState('')
   283→  const [lineItems, setLineItems] = useState<LineItemForm[]>([{ ...defaultLineItem }])
   284→
   285→  React.useEffect(() => {
   286→    if (open) {
   287→      setSupplierId(''); setProjectId(''); setDate(''); setDeliveryDate('')
   288→      setNotes(''); setLineItems([{ ...defaultLineItem }])
   289→    }
   290→  }, [open])
   291→
   292→  const subtotal = useMemo(() => lineItems.reduce((s, i) => s + (i.quantity * i.unitPrice), 0), [lineItems])
   293→  const vatRate = 0.15
   294→  const vatAmount = subtotal * vatRate
   295→  const totalAmount = subtotal + vatAmount
   296→
   297→  const addLine = () => setLineItems([...lineItems, { ...defaultLineItem }])

   300→    setLineItems(lineItems.map((item, i) => i === idx ? { ...item, [field]: value } : item))

   306→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['purchase-orders'] }); onOpenChange(false) },

   310→    e.preventDefault()
   311→    createMutation.mutate({
   312→      supplierId, projectId: projectId || null, date, deliveryDate: deliveryDate || null, notes,

   318→    <Dialog open={open} onOpenChange={onOpenChange}>
   319→      <DialogContent className="max-w-3xl max-h-[90vh] overflow-y-auto">
   320→        <DialogHeader>
   321→          <DialogTitle>{t(lang, 'أمر شراء جديد', 'New Purchase Order')}</DialogTitle>

   325→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   326→            <div className="space-y-2">
   327→              <Label>{t(lang, 'المورد *', 'Supplier *')}</Label>

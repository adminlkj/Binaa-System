    63→    63→// ============ ACCOUNT MAPPING INTERFACE ============
    64→    64→
    65→    65→export interface AccountMapping {
    66→    66→  /** Operation type */
    67→    67→  operationType: OperationType
    68→    68→  /** Arabic label for the operation */
    69→    69→  labelAr: string
    70→    70→  /** English label for the operation */
    71→    71→  labelEn: string
    72→    72→  /** Activity type this operation belongs to */
    73→    73→  activityType: 'CONSTRUCTION' | 'EQUIPMENT_RENTAL' | 'BOTH' | 'ADMIN'
    74→    74→  /** Accounts affected by this operation */
    75→    75→  accounts: AccountMappingEntry[]
    76→    76→  /** Human-readable description of the journal entry pattern (Arabic) */
    77→    77→  entryPatternAr: string
    78→    78→  /** Human-readable description of the journal entry pattern (English) */
    79→    79→  entryPatternEn: string
    80→    80→}
    81→    81→
    82→    82→export interface AccountMappingEntry {
    83→    83→  /** Account code in the chart of accounts */
    84→    84→  accountCode: string
    85→    85→  /** Arabic name of the account */
    86→    86→  accountNameAr: string
    87→    87→  /** English name of the account */
    88→    88→  accountNameEn: string
    89→    89→  /** Whether this account is typically debited */
    90→    90→  side: 'DEBIT' | 'CREDIT' | 'BOTH'
    91→    91→  /** What this account represents in this operation */
    92→    92→  role: string // e.g., 'receivable', 'revenue', 'vat', 'cash', 'payable', 'cost'
    93→    93→  /** Whether this is a variable account (depends on sub-type or user selection) */
    94→    94→  isVariable?: boolean
    95→    95→  /** Description of when this account varies */
    96→    96→  variableNote?: string
    97→    97→}

   574→   574→/**
   575→   575→ * Generate expected journal entry preview for an operation
   576→   576→ */
   577→   577→export function generateEntryPreview(
   578→   578→  operationType: OperationType,
   579→   579→  amounts: Record<string, number>,
   580→   580→  options?: {
   581→   581→    activityType?: 'CONSTRUCTION' | 'EQUIPMENT_RENTAL'
   582→   582→    paymentMethod?: string
   583→   583→  }
   584→   584→): Array<{ accountCode: string; accountNameAr: string; side: 'DEBIT' | 'CREDIT'; amount: number; role: string }> {
   585→   585→  const mapping = ACCOUNT_MAPPINGS[operationType]
   586→   586→  if (!mapping) return []
   587→   587→
   588→   588→  return mapping.accounts.map(account => {
   589→   589→    let amount = 0
   590→   590→    switch (account.role) {
   591→   591→      case 'receivable': amount = amounts.totalAmount || amounts.subtotal || 0; break
   592→   592→      case 'revenue': amount = amounts.subtotal || amounts.amount || 0; break
   593→   593→      case 'vat': amount = amounts.vatAmount || 0; break
   594→   594→      case 'cost': amount = amounts.amount || amounts.subtotal || 0; break
   595→   595→      case 'cash': amount = amounts.totalAmount || amounts.amount || 0; break
   596→   596→      case 'payable': amount = amounts.totalAmount || 0; break
   597→   597→      case 'advance': amount = amounts.amount || 0; break
   598→   598→      case 'gosi': amount = amounts.gosiAmount || 0; break
   599→   599→      case 'gosi_payable': amount = amounts.gosiAmount || 0; break
   600→   600→      case 'depreciation': amount = amounts.depreciation || 0; break
   601→   601→      case 'accum_depreciation': amount = amounts.depreciation || 0; break
   602→   602→      case 'retention': amount = amounts.retention || 0; break
   603→   603→      case 'provision': amount = amounts.amount || 0; break
   604→   604→      case 'output_vat': amount = amounts.outputVat || 0; break
   605→   605→      case 'input_vat': amount = amounts.inputVat || 0; break
   606→   606→      case 'vat_due': amount = amounts.vatDue || 0; break
   607→   607→      case 'gain': amount = amounts.gain || 0; break
   608→   608→      default: amount = amounts[account.role] || 0
   609→   609→    }
   610→   610→    return {
   611→   611→      accountCode: account.accountCode,
   612→   612→      accountNameAr: account.accountNameAr,
   613→   613→      side: account.side,
   614→   614→      amount,
   615→   615→      role: account.role,
   616→   616→    }
   617→   617→  }).filter(line => line.amount > 0)
   618→   618→}

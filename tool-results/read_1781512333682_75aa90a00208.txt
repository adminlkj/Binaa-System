   227→const arabicMonths = ['يناير', 'فبراير', 'مارس', 'أبريل', 'مايو', 'يونيو', 'يوليو', 'أغسطس', 'سبتمبر', 'أكتوبر', 'نوفمبر', 'ديسمبر']

   229→function formatMonthYear(month: number, year: number, lang: 'ar' | 'en'): string {

   231→  return `${englishMonths[month - 1]} ${year}`

    30→// ============ Arabic/English Month Names ============
    31→const arabicMonths = ['يناير', 'فبراير', 'مارس', 'أبريل', 'مايو', 'يونيو', 'يوليو', 'أغسطس', 'سبتمبر', 'أكتوبر', 'نوفمبر', 'ديسمبر']
    32→const englishMonths = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December']
    33→
    34→function formatMonthYear(month: number, year: number, lang: 'ar' | 'en'): string {
    35→  if (lang === 'ar') return `${arabicMonths[month - 1]} ${year}`
    36→  return `${englishMonths[month - 1]} ${year}`
    37→}

   140→// ============ Create Rental Invoice Page ============
   141→function CreateRentalInvoicePage({
   142→  approvedTimesheets, onBack,
   143→}: {
   144→  approvedTimesheets: TimesheetOption[]; onBack: () => void
   145→}) {
   146→  const queryClient = useQueryClient()
   147→  const { lang } = useAppStore()
   148→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   149→
   150→  const [timesheetId, setTimesheetId] = useState('')
   151→  const [date, setDate] = useState('')
   152→  const [dueDate, setDueDate] = useState('')
   153→  const [notes, setNotes] = useState('')
   154→
   155→  // Selected timesheet - RULE: User selects ONLY a Timesheet
   156→  const selectedTimesheet = approvedTimesheets.find(ts => ts.id === timesheetId)
   157→
   158→  // RULE: System auto-fills from contract and timesheet
   159→  // Auto-filled from contract: client, project, equipment, contract no, sales order no, hourly rate, delivery fees, payment terms
   160→  // Auto-filled from timesheet: month, year, operating hours
   161→  // RULE: Rate is read-only (from contract)
   162→  const contract = selectedTimesheet?.contract
   163→  const rental = selectedTimesheet?.rental
   164→  const hourlyRate = rental?.rate || contract?.hourlyRate || 0
   165→  const operatingHours = selectedTimesheet?.operatingHours || 0
   166→  const deliveryFees = contract?.deliveryFees || 0
   167→  const deliveryFeesTaxable = contract?.deliveryFeesTaxable ?? true
   168→  const clientId = rental?.client?.id || ''
   169→  const clientName = rental?.client?.name || ''
   170→  const projectId = selectedTimesheet?.projectId || ''
   171→  const projectName = selectedTimesheet?.project?.name || ''
   172→  const equipmentName = selectedTimesheet?.equipment?.name || ''
   173→  const contractNo = contract?.contractNo || ''
   174→  const salesOrderNo = contract?.salesOrderNo || ''
   175→  const paymentTerms = contract?.paymentTerms || '30 days'
   176→
   177→  // RULE: System auto-calculates: hours × rate = subtotal
   178→  const subtotal = operatingHours * hourlyRate
   179→  const vatRate = 0.15
   180→  const rentalVat = Math.round(subtotal * vatRate * 100) / 100
   181→  // Delivery fees (with/without VAT based on contract setting)
   182→  const deliveryVat = deliveryFeesTaxable && deliveryFees > 0 ? Math.round(deliveryFees * vatRate * 100) / 100 : 0
   183→  const totalVat = rentalVat + deliveryVat
   184→  const totalAmount = subtotal + deliveryFees + totalVat
   185→
   186→  const createMutation = useMutation({
   187→    mutationFn: (data: Record<string, unknown>) =>
   188→      fetch('/api/sales-invoices', {
   189→        method: 'POST',
   190→        headers: { 'Content-Type': 'application/json' },
   191→        body: JSON.stringify(data),
   192→      }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   193→    onSuccess: () => {
   194→      queryClient.invalidateQueries({ queryKey: ['rental-invoices'] })
   195→      queryClient.invalidateQueries({ queryKey: ['equipment-timesheets'] })
   196→      onBack()
   197→    },
   198→  })
   199→
   200→  const handleSubmit = (e: React.FormEvent) => {
   201→    e.preventDefault()
   202→    if (!timesheetId || !date || !dueDate || !clientId) return
   203→
   204→    // Accounting Integration: When rental invoice is created,
   205→    // the accounting engine should create journal entries:
   206→    // Debit: Accounts Receivable (totalAmount)
   207→    // Credit: Rental Revenue (subtotal)
   208→    // Credit: Delivery Revenue (deliveryFees) if applicable
   209→    // Credit: Output VAT (totalVat)
   210→    // Also, timesheet status should be updated to INVOICED
   211→    // This is handled via POST /api/journal-entries with sourceType: SALES_INVOICE
   212→    createMutation.mutate({
   213→      clientId,
   214→      projectId: projectId || null,
   215→      contractId: contract?.id || null,
   216→      date,
   217→      dueDate,
   218→      notes,
   219→      paymentTerms,
   220→      invoiceType: 'RENTAL',
   221→      discountRate: 0,
   222→      discountAmount: 0,
   223→      timesheetId, // RULE: After invoice creation, timesheet status = INVOICED
   224→      deliveryMonth: selectedTimesheet ? formatMonthYear(selectedTimesheet.month, selectedTimesheet.year, 'ar') : null,
   225→      includeDelivery: deliveryFees > 0,
   226→      deliveryAmount: deliveryFees,
   227→      deliveryFeesTaxable,
   228→      contractNo,
   229→      salesOrderNo,
   230→      equipmentName,
   231→      operatingHours,
   232→      hourlyRate,
   233→      items: [{
   234→        description: `${t('إيجار', 'Rental')} ${equipmentName} - ${selectedTimesheet ? formatMonthYear(selectedTimesheet.month, selectedTimesheet.year, lang) : ''}`,
   235→        quantity: operatingHours,
   236→        unit: t('ساعة', 'hour'),
   237→        unitPrice: hourlyRate,
   238→        itemType: 'RENTAL',
   239→      }],
   240→    })
   241→  }
   242→
   243→  // RULE: Cannot create invoice without an approved timesheet
   244→  if (approvedTimesheets.length === 0) {
   245→    return (
   246→      <div className="space-y-6">
   247→        <div className="flex items-center gap-3">
   248→          <Button variant="outline" size="icon" onClick={onBack}>
   249→            <ArrowRight className="size-4" />
   250→          </Button>
   251→          <div className="flex-1">
   252→            <h1 className="text-2xl font-bold text-gray-900">{t('فاتورة إيجار جديدة', 'New Rental Invoice')}</h1>
   253→          </div>
   254→        </div>
   255→        <Card className="bg-amber-50 border-amber-200">
   256→          <CardContent className="p-6 text-center">
   257→            <p className="text-amber-700 font-medium">{t(labels.noApprovedTimesheets.ar, labels.noApprovedTimesheets.en)}</p>
   258→            <p className="text-sm text-amber-600 mt-1">{t('يجب اعتماد سجل ساعات عمل أولاً من وحدة ساعات العمل', 'An approved timesheet must be created first from the Timesheets module')}</p>
   259→          </CardContent>
   260→        </Card>
   261→      </div>
   262→    )
   263→  }
   264→
   265→  return (
   266→    <div className="space-y-6">
   267→      <div className="flex items-center gap-3">
   268→        <Button variant="outline" size="icon" onClick={onBack}>
   269→          <ArrowRight className="size-4" />
   270→        </Button>
   271→        <div className="flex-1">
   272→          <h1 className="text-2xl font-bold text-gray-900">{t('فاتورة إيجار جديدة', 'New Rental Invoice')}</h1>
   273→          <p className="text-sm text-muted-foreground">{t('إنشاء فاتورة إيجار من سجل ساعات عمل معتمد', 'Create rental invoice from an approved timesheet')}</p>
   274→        </div>
   275→      </div>
   276→
   277→      <form onSubmit={handleSubmit} className="space-y-6">
   278→        {/* RULE: User selects ONLY a Timesheet */}
   279→        <Card>
   280→          <CardHeader className="pb-4">
   281→            <CardTitle className="text-lg">{t('اختيار سجل الساعات', 'Select Timesheet')}</CardTitle>
   282→          </CardHeader>
   283→          <CardContent className="space-y-4">
   284→            <div className="space-y-2">
   285→              <Label>{t(labels.selectTimesheet.ar, labels.selectTimesheet.en)} *</Label>
   286→              <Select value={timesheetId} onValueChange={setTimesheetId}>
   287→                <SelectTrigger className="w-full"><SelectValue placeholder={t(labels.selectTimesheet.ar, labels.selectTimesheet.en)} /></SelectTrigger>
   288→                <SelectContent>
   289→                  {approvedTimesheets.map(ts => (
   290→                    <SelectItem key={ts.id} value={ts.id}>
   291→                      {ts.equipment?.name || '—'} - {formatMonthYear(ts.month, ts.year, lang)} ({formatNumber(ts.operatingHours)} {t('ساعة', 'hrs')}) [{ts.contract?.contractNo}]
   292→                    </SelectItem>
   293→                  ))}
   294→                </SelectContent>
   295→              </Select>
   296→            </div>
   297→
   298→            {/* Auto-filled info from timesheet & contract */}
   299→            {selectedTimesheet && (
   300→              <div className="p-4 rounded-lg border bg-blue-50 space-y-3">
   301→                <div className="flex items-center gap-2 text-sm font-medium text-blue-700 mb-2">
   302→                  <FileText className="size-4" />
   303→                  {t(labels.autoFilledFromContract.ar, labels.autoFilledFromContract.en)} / {t(labels.autoFilledFromTimesheet.ar, labels.autoFilledFromTimesheet.en)}
   304→                </div>
   305→                <div className="grid grid-cols-2 sm:grid-cols-4 gap-3 text-sm">
   306→                  <div><span className="text-muted-foreground">{t(labels.client.ar, labels.client.en)}:</span><p className="font-medium">{clientName}</p></div>
   307→                  <div><span className="text-muted-foreground">{t(labels.project.ar, labels.project.en)}:</span><p className="font-medium">{projectName}</p></div>
   308→                  <div><span className="text-muted-foreground">{t(labels.equipment.ar, labels.equipment.en)}:</span><p className="font-medium">{equipmentName}</p></div>
   309→                  <div><span className="text-muted-foreground">{t(labels.deliveryMonth.ar, labels.deliveryMonth.en)}:</span><p className="font-medium">{formatMonthYear(selectedTimesheet.month, selectedTimesheet.year, lang)}</p></div>
   310→                  <div><span className="text-muted-foreground">{t(labels.contractNo.ar, labels.contractNo.en)}:</span><p className="font-medium font-mono">{contractNo}</p></div>
   311→                  <div><span className="text-muted-foreground">{t(labels.salesOrderNo.ar, labels.salesOrderNo.en)}:</span><p className="font-medium font-mono">{salesOrderNo || '—'}</p></div>
   312→                  <div><span className="text-muted-foreground">{t(labels.paymentTerms.ar, labels.paymentTerms.en)}:</span><p className="font-medium">{paymentTerms}</p></div>
   313→                </div>
   314→              </div>
   315→            )}
   316→          </CardContent>
   317→        </Card>
   318→
   319→        {/* Operating Hours & Rate - RULE: No manual editing of rate or hours */}
   320→        <Card>
   321→          <CardHeader className="pb-4">
   322→            <CardTitle className="text-lg">{t('ساعات التشغيل والسعر', 'Operating Hours & Rate')}</CardTitle>
   323→          </CardHeader>
   324→          <CardContent>
   325→            <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   326→              <div className="space-y-2">
   327→                <Label>{t(labels.operatingHours.ar, labels.operatingHours.en)}</Label>
   328→                {/* Auto-filled from timesheet - read only */}
   329→                <Input type="number" value={operatingHours || ''} readOnly className="bg-gray-100" dir="ltr" />
   330→                <p className="text-xs text-muted-foreground">{t('من سجل الساعات - للقراءة فقط', 'From timesheet - read only')}</p>
   331→              </div>
   332→              <div className="space-y-2">
   333→                <Label>{t(labels.hourlyRate.ar, labels.hourlyRate.en)}</Label>
   334→                {/* RULE: Rate is read-only (from contract) */}
   335→                <Input type="number" value={hourlyRate || ''} readOnly className="bg-gray-100" dir="ltr" />
   336→                <p className="text-xs text-muted-foreground">{t(labels.readOnlyFromContract.ar, labels.readOnlyFromContract.en)}</p>
   337→              </div>
   338→              <div className="space-y-2">
   339→                <Label>{t(labels.subtotal.ar, labels.subtotal.en)} ({t('ساعات × سعر', 'hrs × rate')})</Label>
   340→                <div className="h-9 flex items-center">
   341→                  <MoneyDisplay value={subtotal} lang={lang} size="md" bold />
   342→                </div>
   343→              </div>
   344→            </div>
   345→          </CardContent>
   346→        </Card>
   347→
   348→        {/* Delivery Fees (from contract) */}
   349→        {deliveryFees > 0 && (
   350→          <Card>
   351→            <CardHeader className="pb-4">
   352→              <CardTitle className="text-lg">{t(labels.deliveryFees.ar, labels.deliveryFees.en)}</CardTitle>
   353→            </CardHeader>
   354→            <CardContent>
   355→              <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   356→                <div className="space-y-2">
   357→                  <Label>{t(labels.deliveryFees.ar, labels.deliveryFees.en)}</Label>
   358→                  <Input type="number" value={deliveryFees} readOnly className="bg-gray-100" dir="ltr" />
   359→                </div>
   360→                <div className="space-y-2 flex items-center gap-2 pt-6">
   361→                  <Badge variant="outline" className={deliveryFeesTaxable ? 'bg-emerald-100 text-emerald-700' : 'bg-gray-100 text-gray-700'}>
   362→                    {deliveryFeesTaxable ? t('خاضعة للضريبة', 'Taxable') : t('غير خاضعة للضريبة', 'Not Taxable')}
   363→                  </Badge>
   364→                </div>
   365→              </div>
   366→            </CardContent>
   367→          </Card>
   368→        )}
   369→
   370→        {/* Dates */}
   371→        <Card>
   372→          <CardHeader className="pb-4">
   373→            <CardTitle className="text-lg">{t('التواريخ', 'Dates')}</CardTitle>
   374→          </CardHeader>
   375→          <CardContent>
   376→            <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   377→              <div className="space-y-2">
   378→                <Label>{t(labels.date.ar, labels.date.en)} *</Label>

   380→              </div>

   386→                <Label>{t(labels.notes.ar, labels.notes.en)}</Label>

   389→            </div>

   392→
   393→        {/* Invoice Summary */}
   394→        {selectedTimesheet && subtotal > 0 && (
   395→          <Card className="bg-gray-50 border-dashed">
   396→            <CardContent className="p-6 space-y-3">

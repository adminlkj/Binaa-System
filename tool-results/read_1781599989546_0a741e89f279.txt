   261→      if (!account) return null

   264→      if (dateTo) params.set('dateTo', dateTo)
   265→      const res = await fetch(`/api/accounts/statement?${params}`)

   409→  const [detailTab, setDetailTab] = useState('lines')

   414→    const uniqueAccounts = new Map<string, { code: string; name: string; nameAr: string | null; totalDebit: number; totalCredit: number }>()
   415→    for (const line of entry.lines) {
   416→      const existing = uniqueAccounts.get(line.accountId)
   417→      if (existing) {
   418→        existing.totalDebit += line.debit
   419→        existing.totalCredit += line.credit
   420→      } else {
   421→        uniqueAccounts.set(line.accountId, {
   422→          code: line.account.code,
   423→          name: line.account.name,
   424→          nameAr: line.account.nameAr,
   425→          totalDebit: line.debit,
   426→          totalCredit: line.credit,
   427→        })
   428→      }
   429→    }
   430→
   431→    // For each unique account, compute before/after balance
   432→    const impactItems: {
   433→      accountId: string; code: string; name: string; nameAr: string | null
   434→      beforeBalance: number; movementDebit: number; movementCredit: number; afterBalance: number
   435→    }[] = []
   436→
   437→    for (const [accountId, info] of uniqueAccounts) {
   438→      const accountData = accounts.find(a => a.id === accountId)
   439→      const currentBalance = accountData?.balance || 0
   440→      const normalBalance = accountData?.normalBalance || 'DEBIT'
   441→      const isPosted = entry.status === 'POSTED'
   442→
   443→      // Net movement effect on this account
   444→      // For DEBIT-normal accounts: balance increases with debit, decreases with credit
   445→      // For CREDIT-normal accounts: balance increases with credit, decreases with debit
   446→      const netMovement = normalBalance === 'DEBIT'
   447→        ? info.totalDebit - info.totalCredit
   448→        : info.totalCredit - info.totalDebit
   449→
   450→      let beforeBalance: number
   451→      let afterBalance: number
   452→
   453→      if (isPosted) {
   454→        // Current balance includes this JE's effect
   455→        beforeBalance = currentBalance - netMovement
   456→        afterBalance = currentBalance
   457→      } else {
   458→        // Current balance does NOT include this JE's effect (DRAFT)
   459→        beforeBalance = currentBalance
   460→        afterBalance = currentBalance + netMovement
   461→      }
   462→
   463→      impactItems.push({
   464→        accountId,
   465→        code: info.code,
   466→        name: info.name,
   467→        nameAr: info.nameAr,
   468→        beforeBalance,
   469→        movementDebit: info.totalDebit,
   470→        movementCredit: info.totalCredit,
   471→        afterBalance,
   472→      })
   473→    }
   474→
   475→    return impactItems.sort((a, b) => a.code.localeCompare(b.code))
   476→  }, [entry, accounts])
   477→
   478→  return (
   479→    <div className="space-y-4">
   480→      <div className="flex items-center gap-3">
   481→        <Button variant="outline" size="icon" onClick={onBack}><ChevronLeft className="size-4" /></Button>
   482→        <div>
   483→          <h2 className="text-xl font-bold">{entry.entryNo}</h2>
   484→          <p className="text-sm text-muted-foreground">{entry.description || ''}</p>
   485→        </div>
   486→        <div className="mr-auto flex items-center gap-2">
   487→          <JEStatusBadge status={entry.status} lang={lang} />
   488→          {entry.sourceType && <Badge variant="outline" className="bg-gray-50">{sourceTypeLabels[entry.sourceType]?.[lang] || entry.sourceType}</Badge>}
   489→        </div>
   490→      </div>
   491→      <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   492→        <Card className="bg-gray-50"><CardContent className="p-3 text-center"><p className="text-xs text-gray-600">{t('التاريخ', 'Date', lang)}</p><p className="font-semibold">{formatDate(entry.date, lang)}</p></CardContent></Card>
   493→        <Card className="bg-emerald-50 border-emerald-200"><CardContent className="p-3 text-center"><p className="text-xs text-emerald-600">{t('مدين', 'Debit', lang)}</p><MoneyDisplay value={entry.totalDebit} lang={lang} bold className="text-emerald-700" /></CardContent></Card>
   494→        <Card className="bg-rose-50 border-rose-200"><CardContent className="p-3 text-center"><p className="text-xs text-rose-600">{t('دائن', 'Credit', lang)}</p><MoneyDisplay value={entry.totalCredit} lang={lang} bold className="text-rose-700" /></CardContent></Card>
   495→        <Card className="bg-gray-50"><CardContent className="p-3 text-center"><p className="text-xs text-gray-600">{t('عدد البنود', 'Lines', lang)}</p><p className="font-semibold">{formatNumber(entry.lines.length)}</p></CardContent></Card>
   496→      </div>

   498→      {/* Sub-tabs for Lines and Account Impact */}
   499→      <Tabs value={detailTab} onValueChange={setDetailTab}>
   500→        <TabsList>
   501→          <TabsTrigger value="lines" className="gap-1 text-xs">

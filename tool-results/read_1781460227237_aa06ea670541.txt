    88→// ============ Salary Form Dialog ============
    89→function SalaryFormDialog({ open, onOpenChange, employees }: {
    90→  open: boolean; onOpenChange: (open: boolean) => void; employees: Employee[]
    91→}) {
    92→  const queryClient = useQueryClient()
    93→  const [form, setForm] = useState<SalaryFormData>(defaultForm)
    94→  const [autoCalcResult, setAutoCalcResult] = useState<AutoCalcResult | null>(null)
    95→  const { lang } = useAppStore()
    96→
    97→  React.useEffect(() => {
    98→    if (open) {
    99→      setForm(defaultForm)
   100→      setAutoCalcResult(null)
   101→    }
   102→  }, [open])
   103→
   104→  // Auto-calculate mutation
   105→  const autoCalcMutation = useMutation({
   106→    mutationFn: (data: { employeeId: string; month: number; year: number }) =>
   107→      fetch('/api/salaries/auto-calculate', {
   108→        method: 'POST',
   109→        headers: { 'Content-Type': 'application/json' },
   110→        body: JSON.stringify(data),
   111→      }).then(r => { if (!r.ok) throw new Error(); return r.json() }) as Promise<AutoCalcResult>,
   112→    onSuccess: (data) => {
   113→      setAutoCalcResult(data)
   114→      setForm(f => ({
   115→        ...f,
   116→        basicSalary: String(data.basicSalary),
   117→        housingAllowance: String(data.housingAllowance),
   118→        transportAllowance: String(data.transportAllowance),
   119→        otherAllowances: String(data.otherAllowances),
   120→        overtimeAmount: String(data.overtimeAmount),
   121→        deductions: String(data.deductions),
   122→      }))
   123→      toast.success(t('تم حساب الراتب تلقائياً', 'Salary auto-calculated', lang))
   124→    },
   125→    onError: () => {
   126→      toast.error(t('فشل في حساب الراتب تلقائياً', 'Failed to auto-calculate salary', lang))
   127→    },
   128→  })
   129→
   130→  const handleAutoCalculate = () => {
   131→    if (form.employeeId && form.month && form.year) {
   132→      autoCalcMutation.mutate({
   133→        employeeId: form.employeeId,
   134→        month: parseInt(form.month),
   135→        year: parseInt(form.year),
   136→      })
   137→    }
   138→  }
   139→
   140→  const totalAllowances = (parseFloat(form.housingAllowance) || 0) + (parseFloat(form.transportAllowance) || 0) + (parseFloat(form.otherAllowances) || 0)
   141→  const netSalary = (parseFloat(form.basicSalary) || 0) + totalAllowances + (parseFloat(form.overtimeAmount) || 0) - (parseFloat(form.deductions) || 0)
   142→
   143→  const createMutation = useMutation({
   144→    mutationFn: (data: Record<string, unknown>) => fetch('/api/salaries', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   145→    onSuccess: (data: SalaryRecord) => {
   146→      queryClient.invalidateQueries({ queryKey: ['salaries'] })
   147→      onOpenChange(false)
   148→      if (data.projectCostCreated) {
   149→        toast.success(t('تم إنشاء كشف الراتب وإضافة التكلفة للمشروع', 'Salary created and cost added to project', lang))
   150→      } else {
   151→        toast.success(t('تم إنشاء كشف الراتب', 'Salary record created', lang))
   152→      }
   153→    },
   154→  })
   155→
   156→  const handleSubmit = (e: React.FormEvent) => {
   157→    e.preventDefault()
   158→    createMutation.mutate({
   159→      employeeId: form.employeeId,
   160→      month: parseInt(form.month),
   161→      year: parseInt(form.year),
   162→      basicSalary: parseFloat(form.basicSalary) || 0,
   163→      housingAllowance: parseFloat(form.housingAllowance) || 0,
   164→      transportAllowance: parseFloat(form.transportAllowance) || 0,
   165→      otherAllowances: parseFloat(form.otherAllowances) || 0,
   166→      overtimeAmount: parseFloat(form.overtimeAmount) || 0,
   167→      deductions: parseFloat(form.deductions) || 0,
   168→      status: 'DRAFT',
   169→    })
   170→  }
   171→
   172→  return (
   173→    <Dialog open={open} onOpenChange={onOpenChange}>
   174→      <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
   175→        <DialogHeader>
   176→          <DialogTitle>{t('إعداد راتب', 'Prepare Salary', lang)}</DialogTitle>
   177→          <DialogDescription>{t('إعداد كشف راتب للموظف', 'Prepare salary statement for employee', lang)}</DialogDescription>
   178→        </DialogHeader>
   179→        <form onSubmit={handleSubmit} className="space-y-4">
   180→          <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   181→            <div className="space-y-2 sm:col-span-3">
   182→              <Label>{t('الموظف *', 'Employee *', lang)}</Label>
   183→              <Select value={form.employeeId} onValueChange={v => setForm(f => ({ ...f, employeeId: v }))}>
   184→                <SelectTrigger><SelectValue placeholder={t('اختر الموظف', 'Select employee', lang)} /></SelectTrigger>
   185→                <SelectContent>
   186→                  {employees.map(e => <SelectItem key={e.id} value={e.id}>{e.name} ({e.code})</SelectItem>)}
   187→                </SelectContent>
   188→              </Select>
   189→            </div>
   190→            <div className="space-y-2">
   191→              <Label>{t('الشهر *', 'Month *', lang)}</Label>
   192→              <Select value={form.month} onValueChange={v => setForm(f => ({ ...f, month: v }))}>
   193→                <SelectTrigger><SelectValue placeholder={t('اختر الشهر', 'Select month', lang)} /></SelectTrigger>
   194→                <SelectContent>
   195→                  {monthNames.map((m, i) => <SelectItem key={i + 1} value={String(i + 1)}>{m[lang]}</SelectItem>)}
   196→                </SelectContent>
   197→              </Select>
   198→            </div>
   199→            <div className="space-y-2">
   200→              <Label>{t('السنة *', 'Year *', lang)}</Label>
   201→              <Input type="number" min="2020" max="2050" value={form.year} onChange={e => setForm(f => ({ ...f, year: e.target.value }))} dir="ltr" required placeholder="2025" />
   202→            </div>
   203→            <div className="space-y-2 flex items-end">
   204→              <Button
   205→                type="button"
   206→                variant="outline"
   207→                className="gap-2 border-emerald-300 text-emerald-700 hover:bg-emerald-50 w-full"
   208→                disabled={!form.employeeId || !form.month || !form.year || autoCalcMutation.isPending}
   209→                onClick={handleAutoCalculate}
   210→              >
   211→                <Calculator className="size-4" />
   212→                {autoCalcMutation.isPending ? t('جاري الحساب...', 'Calculating...', lang) : t('حساب تلقائي', 'Auto-Calculate', lang)}
   213→              </Button>
   214→            </div>
   215→          </div>
   216→
   217→          {/* Auto-calc result summary */}
   218→          {autoCalcResult && (
   219→            <Card className="bg-blue-50 border-blue-200">
   220→              <CardContent className="p-3">
   221→                <p className="text-xs font-semibold text-blue-700 mb-1">{t('ملخص الحساب التلقائي', 'Auto-Calc Summary', lang)}</p>
   222→                <div className="grid grid-cols-3 gap-2 text-xs">
   223→                  <div><span className="text-blue-600">{t('أيام الحضور', 'Attendance Days', lang)}:</span> <span className="font-medium">{autoCalcResult.attendanceDays}</span></div>
   224→                  <div><span className="text-blue-600">{t('ساعات العمل', 'Work Hours', lang)}:</span> <span className="font-medium">{autoCalcResult.totalWorkHours}</span></div>
   225→                  <div><span className="text-blue-600">{t('ساعات إضافية', 'Overtime', lang)}:</span> <span className="font-medium">{autoCalcResult.totalOvertimeHours}</span></div>
   226→                </div>
   227→              </CardContent>
   228→            </Card>
   229→          )}
   230→
   231→          <div className="space-y-2">
   232→            <h4 className="font-semibold text-sm text-emerald-700 border-b border-emerald-200 pb-1">
   233→              {t('تفاصيل الراتب', 'Salary Details', lang)}
   234→            </h4>
   235→            <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   236→              <div className="space-y-2"><Label>{t('الراتب الأساسي', 'Basic Salary', lang)}</Label><Input type="number" min="0" step="0.01" value={form.basicSalary} onChange={e => setForm(f => ({ ...f, basicSalary: e.target.value }))} dir="ltr" /></div>
   237→              <div className="space-y-2"><Label>{t('بدل السكن', 'Housing Allowance', lang)}</Label><Input type="number" min="0" step="0.01" value={form.housingAllowance} onChange={e => setForm(f => ({ ...f, housingAllowance: e.target.value }))} dir="ltr" /></div>

   276→    queryKey: ['salaries', filterMonth, filterYear],
   277→    queryFn: async () => {

   280→      return res.json()
   281→    },
   282→  })

   287→  })

   289→  const approveMutation = useMutation({
   290→    mutationFn: ({ id, status }: { id: string; status: string }) => fetch(`/api/salaries/${id}`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ status }) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   291→    onSuccess: () => {
   292→      queryClient.invalidateQueries({ queryKey: ['salaries'] })
   293→      toast.success(t('تم تحديث حالة الراتب', 'Salary status updated', lang))

   375→      }
   376→    >
   377→      {/* Summary Cards */}
   378→      <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">

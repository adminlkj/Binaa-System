   198→// ============ Create Invoice Flow ============
   199→function CreateInvoiceFlow({
   200→  initialState,
   201→  onBack,
   202→  onComplete,
   203→}: {
   204→  initialState: ViewState & { type: 'create' }
   205→  onBack: () => void
   206→  onComplete: () => void
   207→}) {
   208→  const queryClient = useQueryClient()
   209→  const { lang } = useAppStore()
   210→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   211→
   212→  const [step, setStep] = useState(initialState.step || 1)
   213→  const [sourceType, setSourceType] = useState<'EXTRACT' | 'TIMESHEET' | null>(initialState.sourceType || null)
   214→  const [selectedClaimId, setSelectedClaimId] = useState<string>(initialState.selectedSourceId || '')
   215→  const [selectedTimesheetId, setSelectedTimesheetId] = useState<string>(initialState.selectedSourceId || '')
   216→  const [date, setDate] = useState(new Date().toISOString().split('T')[0])
   217→  const [dueDate, setDueDate] = useState('')
   218→  const [notes, setNotes] = useState('')
   219→
   220→  // Fetch uninvoiced approved claims
   221→  const { data: approvedClaims = [], isLoading: claimsLoading } = useQuery<ProgressClaimSource[]>({
   222→    queryKey: ['uninvoiced-claims'],
   223→    queryFn: async () => {
   224→      const res = await fetch('/api/progress-claims?status=APPROVED&invoiced=false')
   225→      if (!res.ok) return []
   226→      return res.json()
   227→    },
   228→    enabled: sourceType === 'EXTRACT' || step === 2,
   229→  })
   230→
   231→  // Fetch uninvoiced approved timesheets
   232→  const { data: approvedTimesheets = [], isLoading: timesheetsLoading } = useQuery<TimesheetSource[]>({
   233→    queryKey: ['uninvoiced-timesheets'],
   234→    queryFn: async () => {
   235→      const res = await fetch('/api/equipment/timesheets?status=APPROVED')
   236→      if (!res.ok) return []
   237→      const data = await res.json()
   238→      // Filter: only timesheets not yet invoiced (no invoice linked)
   239→      return (Array.isArray(data) ? data : []).filter(
   240→        (ts: { status: string; invoice: null }) => ts.status === 'APPROVED' && !ts.invoice
   241→      )
   242→    },
   243→    enabled: sourceType === 'TIMESHEET' || step === 2,
   244→  })
   245→
   246→  // Selected source
   247→  const selectedClaim = approvedClaims.find(c => c.id === selectedClaimId)
   248→  const selectedTimesheet = approvedTimesheets.find(ts => ts.id === selectedTimesheetId)
   249→
   250→  // Auto-calculations for EXTRACT source
   251→  const extractSubtotal = selectedClaim?.amount || 0
   252→  const extractVat = selectedClaim?.vatAmount || 0
   253→  const extractTotal = selectedClaim?.totalAmount || 0
   254→
   255→  // Auto-calculations for TIMESHEET source
   256→  const tsHourlyRate = selectedTimesheet?.rental?.hourlyRate || selectedTimesheet?.contract?.hourlyRate || 0
   257→  const tsOperatingHours = selectedTimesheet?.operatingHours || 0
   258→  const tsDeliveryFees = selectedTimesheet?.rental?.deliveryFees || 0
   259→  const tsDeliveryFeesTaxable = selectedTimesheet?.rental?.deliveryFeesTaxable ?? true
   260→  const tsSubtotal = tsOperatingHours * tsHourlyRate
   261→  const tsVatRate = 0.15
   262→  const tsRentalVat = Math.round(tsSubtotal * tsVatRate * 100) / 100
   263→  const tsDeliveryVat = tsDeliveryFeesTaxable && tsDeliveryFees > 0 ? Math.round(tsDeliveryFees * tsVatRate * 100) / 100 : 0
   264→  const tsTotalVat = tsRentalVat + tsDeliveryVat
   265→  const tsTotalAmount = tsSubtotal + tsDeliveryFees + tsTotalVat
   266→
   267→  const currentSubtotal = sourceType === 'EXTRACT' ? extractSubtotal : tsSubtotal
   268→  const currentVat = sourceType === 'EXTRACT' ? extractVat : tsTotalVat
   269→  const currentTotal = sourceType === 'EXTRACT' ? extractTotal : tsTotalAmount
   270→
   271→  // Create mutation
   272→  const createMutation = useMutation({
   273→    mutationFn: (data: Record<string, unknown>) =>
   274→      fetch('/api/sales-invoices', {
   275→        method: 'POST',
   276→        headers: { 'Content-Type': 'application/json' },
   277→        body: JSON.stringify(data),
   278→      }).then(async r => {
   279→        if (!r.ok) {
   280→          const err = await r.json().catch(() => ({}))
   281→          throw new Error((err as { error?: string }).error || 'Failed')
   282→        }
   283→        return r.json()
   284→      }),
   285→    onSuccess: () => {
   286→      queryClient.invalidateQueries({ queryKey: ['sales-invoices'] })
   287→      queryClient.invalidateQueries({ queryKey: ['uninvoiced-claims'] })
   288→      queryClient.invalidateQueries({ queryKey: ['uninvoiced-timesheets'] })
   289→      onComplete()
   290→    },
   291→  })
   292→
   293→  const handleNext = () => {
   294→    if (step === 1 && sourceType) {
   295→      setStep(2)
   296→    } else if (step === 2 && (selectedClaimId || selectedTimesheetId)) {
   297→      setStep(3)
   298→    }
   299→  }
   300→
   301→  const handleBack = () => {
   302→    if (step === 1) {
   303→      onBack()
   304→    } else {
   305→      setStep(step - 1)
   306→    }
   307→  }
   308→
   309→  const handleSubmit = () => {
   310→    if (sourceType === 'EXTRACT' && selectedClaimId) {
   311→      createMutation.mutate({
   312→        sourceType: 'EXTRACT',
   313→        progressClaimId: selectedClaimId,
   314→        date,
   315→        dueDate,
   316→        notes,
   317→      })
   318→    } else if (sourceType === 'TIMESHEET' && selectedTimesheetId) {
   319→      createMutation.mutate({
   320→        sourceType: 'TIMESHEET',
   321→        timesheetId: selectedTimesheetId,
   322→        date,
   323→        dueDate,
   324→        notes,
   325→      })
   326→    }
   327→  }
   328→
   329→  const canProceed = () => {
   330→    if (step === 1) return !!sourceType
   331→    if (step === 2) {
   332→      if (sourceType === 'EXTRACT') return !!selectedClaimId
   333→      if (sourceType === 'TIMESHEET') return !!selectedTimesheetId
   334→    }
   335→    if (step === 3) return !!date && !!dueDate
   336→    return false
   337→  }
   338→
   339→  // ============ STEP 1: Select Source Type ============
   340→  const renderStep1 = () => (
   341→    <div className="space-y-6">
   342→      <div className="text-center mb-6">
   343→        <h3 className="text-lg font-semibold mb-1">{t(labels.step1Title.ar, labels.step1Title.en)}</h3>
   344→        <p className="text-sm text-muted-foreground">
   345→          {t('اختر نوع المستند المصدر لإنشاء الفاتورة', 'Select the source document type to create an invoice')}
   346→        </p>
   347→      </div>
   348→
   349→      <div className="grid grid-cols-1 sm:grid-cols-2 gap-4 max-w-2xl mx-auto">
   350→        {/* Extract Source Card */}
   351→        <button
   352→          type="button"
   353→          onClick={() => setSourceType('EXTRACT')}
   354→          className={`relative p-6 rounded-xl border-2 transition-all text-start ${
   355→            sourceType === 'EXTRACT'
   356→              ? 'border-teal-500 bg-teal-50 shadow-md'
   357→              : 'border-gray-200 bg-white hover:border-teal-300 hover:bg-teal-50/30'
   358→          }`}
   359→        >
   360→          {sourceType === 'EXTRACT' && (
   361→            <div className="absolute top-3 end-3">
   362→              <CheckCircle className="size-5 text-teal-600" />
   363→            </div>
   364→          )}
   365→          <div className="flex items-center gap-3 mb-3">
   366→            <div className={`flex size-12 items-center justify-center rounded-xl ${sourceType === 'EXTRACT' ? 'bg-teal-100' : 'bg-gray-100'}`}>
   367→              <Building2 className={`size-6 ${sourceType === 'EXTRACT' ? 'text-teal-600' : 'text-gray-500'}`} />
   368→            </div>
   369→            <div>
   370→              <h4 className="font-semibold">{t(labels.extractSource.ar, labels.extractSource.en)}</h4>
   371→              <p className="text-xs text-muted-foreground">{t('للمشاريع الإنشائية', 'For construction projects')}</p>
   372→            </div>
   373→          </div>
   374→          <p className="text-sm text-muted-foreground">{t(labels.extractSourceDesc.ar, labels.extractSourceDesc.en)}</p>
   375→        </button>
   376→
   377→        {/* Timesheet Source Card */}
   378→        <button
   379→          type="button"
   380→          onClick={() => setSourceType('TIMESHEET')}
   381→          className={`relative p-6 rounded-xl border-2 transition-all text-start ${
   382→            sourceType === 'TIMESHEET'
   383→              ? 'border-purple-500 bg-purple-50 shadow-md'
   384→              : 'border-gray-200 bg-white hover:border-purple-300 hover:bg-purple-50/30'
   385→          }`}
   386→        >
   387→          {sourceType === 'TIMESHEET' && (
   388→            <div className="absolute top-3 end-3">
   389→              <CheckCircle className="size-5 text-purple-600" />
   390→            </div>
   391→          )}
   392→          <div className="flex items-center gap-3 mb-3">
   393→            <div className={`flex size-12 items-center justify-center rounded-xl ${sourceType === 'TIMESHEET' ? 'bg-purple-100' : 'bg-gray-100'}`}>
   394→              <HardHat className={`size-6 ${sourceType === 'TIMESHEET' ? 'text-purple-600' : 'text-gray-500'}`} />
   395→            </div>
   396→            <div>
   397→              <h4 className="font-semibold">{t(labels.timesheetSource.ar, labels.timesheetSource.en)}</h4>
   398→              <p className="text-xs text-muted-foreground">{t('لتأجير المعدات', 'For equipment rental')}</p>
   399→            </div>
   400→          </div>
   401→          <p className="text-sm text-muted-foreground">{t(labels.timesheetSourceDesc.ar, labels.timesheetSourceDesc.en)}</p>
   402→        </button>
   403→      </div>
   404→    </div>
   405→  )
   406→
   407→  // ============ STEP 2: Select Source Document ============
   408→  const renderStep2 = () => {
   409→    if (sourceType === 'EXTRACT') {
   410→      if (claimsLoading) {
   411→        return (
   412→          <div className="flex flex-col items-center py-10">
   413→            <div className="size-8 animate-spin rounded-full border-4 border-teal-200 border-t-teal-600" />
   414→            <p className="mt-3 text-sm text-muted-foreground">{commonText.loading[lang]}</p>
   415→          </div>
   416→        )
   417→      }
   418→
   419→      if (approvedClaims.length === 0) {
   420→        return (
   421→          <Card className="bg-amber-50 border-amber-200">
   422→            <CardContent className="p-6 text-center">
   423→              <Building2 className="size-10 text-amber-400 mx-auto mb-3" />

   836→// ============ Invoice Detail View ============
   837→function InvoiceDetailView({
   838→  invoice,
   839→  onBack,
   840→}: {
   841→  invoice: SalesInvoice
   842→  onBack: () => void
   843→}) {
   844→  const queryClient = useQueryClient()
   845→  const { lang } = useAppStore()
   846→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   847→
   848→  const statusMutation = useMutation({
   849→    mutationFn: ({ id, status }: { id: string; status: string }) =>
   850→      fetch(`/api/sales-invoices/${id}`, {
   851→        method: 'PATCH',
   852→        headers: { 'Content-Type': 'application/json' },
   853→        body: JSON.stringify({ status }),
   854→      }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   855→    onSuccess: () => {
   856→      queryClient.invalidateQueries({ queryKey: ['sales-invoices'] })
   857→    },
   858→  })
   859→
   860→  const isExtract = invoice.sourceType === 'EXTRACT'
   861→  const isTimesheet = invoice.sourceType === 'TIMESHEET'
   862→
   863→  return (
   864→    <div className="space-y-6">
   865→      {/* Header */}

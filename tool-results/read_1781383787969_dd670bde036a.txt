   271→// ============ Project Form Dialog ============
   272→interface ProjectFormData {
   273→  code: string; name: string; nameAr: string; clientId: string; branchId: string
   274→  location: string; startDate: string; endDate: string; status: string; description: string
   275→  contractValue: string; projectType: string

   277→
   278→function ProjectFormDialog({
   279→  open, onOpenChange, editingProject, clients, branches,
   280→}: {
   281→  open: boolean; onOpenChange: (open: boolean) => void
   282→  editingProject: ProjectListItem | null
   283→  clients: Client[]; branches: Branch[]
   284→}) {
   285→  const queryClient = useQueryClient()
   286→  const { lang } = useAppStore()
   287→  const isEdit = !!editingProject
   288→
   289→  const [form, setForm] = useState<ProjectFormData>({
   290→    code: '', name: '', nameAr: '', clientId: '', branchId: '',
   291→    location: '', startDate: '', endDate: '', status: 'PLANNING', description: '',
   292→    contractValue: '', projectType: 'CONSTRUCTION',
   293→  })
   294→
   295→  React.useEffect(() => {
   296→    if (open) {
   297→      if (editingProject) {
   298→        setForm({
   299→          code: editingProject.code,
   300→          name: editingProject.name,
   301→          nameAr: editingProject.nameAr || '',
   302→          clientId: editingProject.client.id,
   303→          branchId: editingProject.branch.id,
   304→          location: editingProject.location || '',
   305→          startDate: editingProject.startDate ? new Date(editingProject.startDate).toISOString().split('T')[0] : '',
   306→          endDate: editingProject.endDate ? new Date(editingProject.endDate).toISOString().split('T')[0] : '',
   307→          status: editingProject.status,
   308→          description: editingProject.description || '',
   309→          contractValue: editingProject.contractValue ? String(editingProject.contractValue) : '',
   310→          projectType: editingProject.projectType || 'CONSTRUCTION',
   311→        })
   312→      } else {
   313→        setForm({ code: '', name: '', nameAr: '', clientId: '', branchId: '', location: '', startDate: '', endDate: '', status: 'PLANNING', description: '', contractValue: '', projectType: 'CONSTRUCTION' })
   314→      }
   315→    }
   316→  }, [open, editingProject])
   317→
   318→  const createMutation = useMutation({
   319→    mutationFn: (data: ProjectFormData) =>
   320→      fetch('/api/projects', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   321→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['projects'] }); onOpenChange(false) },
   322→  })
   323→
   324→  const updateMutation = useMutation({
   325→    mutationFn: (data: ProjectFormData) =>
   326→      fetch(`/api/projects/${editingProject?.id}`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   327→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['projects'] }); onOpenChange(false) },
   328→  })
   329→
   330→  const handleSubmit = (e: React.FormEvent) => {
   331→    e.preventDefault()
   332→    if (isEdit) updateMutation.mutate(form)
   333→    else createMutation.mutate(form)
   334→  }
   335→
   336→  const isLoading = createMutation.isPending || updateMutation.isPending
   337→
   338→  return (
   339→    <Dialog open={open} onOpenChange={onOpenChange}>
   340→      <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
   341→        <DialogHeader>
   342→          <DialogTitle>{isEdit ? 'تعديل المشروع' : 'مشروع جديد'}</DialogTitle>
   343→          <DialogDescription>{isEdit ? 'تعديل بيانات المشروع' : 'إضافة مشروع جديد للنظام'}</DialogDescription>
   344→        </DialogHeader>
   345→        <form onSubmit={handleSubmit} className="space-y-4">
   346→          {/* Project Type Selector */}
   347→          <div className="space-y-2">
   348→            <Label>{lang === 'ar' ? 'نوع المشروع' : 'Project Type'}</Label>
   349→            <div className="grid grid-cols-2 gap-3">
   350→              <button
   351→                type="button"
   352→                onClick={() => setForm(f => ({ ...f, projectType: 'CONSTRUCTION' }))}
   353→                className={`flex items-center gap-3 rounded-lg border-2 p-4 transition-all ${
   354→                  form.projectType === 'CONSTRUCTION'
   355→                    ? 'border-emerald-500 bg-emerald-50 shadow-sm'
   356→                    : 'border-gray-200 bg-white hover:border-gray-300'
   357→                }`}
   358→              >
   359→                <div className={`flex size-10 items-center justify-center rounded-lg ${
   360→                  form.projectType === 'CONSTRUCTION' ? 'bg-emerald-100' : 'bg-gray-100'
   361→                }`}>
   362→                  <Building2 className={`size-5 ${form.projectType === 'CONSTRUCTION' ? 'text-emerald-600' : 'text-gray-400'}`} />
   363→                </div>
   364→                <div className="text-right">
   365→                  <p className={`text-sm font-bold ${form.projectType === 'CONSTRUCTION' ? 'text-emerald-700' : 'text-gray-700'}`}>
   366→                    {lang === 'ar' ? 'مشروع تنفيذي' : 'Construction'}
   367→                  </p>
   368→                  <p className="text-xs text-gray-500">
   369→                    {lang === 'ar' ? 'Construction Project' : 'Construction Project'}
   370→                  </p>
   371→                </div>
   372→              </button>
   373→              <button
   374→                type="button"
   375→                onClick={() => setForm(f => ({ ...f, projectType: 'EQUIPMENT_RENTAL' }))}
   376→                className={`flex items-center gap-3 rounded-lg border-2 p-4 transition-all ${
   377→                  form.projectType === 'EQUIPMENT_RENTAL'
   378→                    ? 'border-cyan-500 bg-cyan-50 shadow-sm'
   379→                    : 'border-gray-200 bg-white hover:border-gray-300'
   380→                }`}
   381→              >
   382→                <div className={`flex size-10 items-center justify-center rounded-lg ${
   383→                  form.projectType === 'EQUIPMENT_RENTAL' ? 'bg-cyan-100' : 'bg-gray-100'
   384→                }`}>
   385→                  <Truck className={`size-5 ${form.projectType === 'EQUIPMENT_RENTAL' ? 'text-cyan-600' : 'text-gray-400'}`} />
   386→                </div>
   387→                <div className="text-right">
   388→                  <p className={`text-sm font-bold ${form.projectType === 'EQUIPMENT_RENTAL' ? 'text-cyan-700' : 'text-gray-700'}`}>
   389→                    {lang === 'ar' ? 'مشروع تأجير معدات' : 'Equipment Rental'}
   390→                  </p>
   391→                  <p className="text-xs text-gray-500">
   392→                    {lang === 'ar' ? 'Equipment Rental Project' : 'Equipment Rental Project'}
   393→                  </p>
   394→                </div>
   395→              </button>
   396→            </div>
   397→          </div>
   398→
   399→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   400→            <div className="space-y-2">
   401→              <Label htmlFor="code">كود المشروع *</Label>
   402→              <Input id="code" value={form.code} onChange={e => setForm(f => ({ ...f, code: e.target.value }))} placeholder="PRJ-004" required />
   403→            </div>
   404→            <div className="space-y-2">
   405→              <Label htmlFor="name">اسم المشروع *</Label>
   406→              <Input id="name" value={form.name} onChange={e => setForm(f => ({ ...f, name: e.target.value }))} placeholder="اسم المشروع" required />
   407→            </div>
   408→            <div className="space-y-2">
   409→              <Label htmlFor="nameAr">الاسم بالعربي</Label>
   410→              <Input id="nameAr" value={form.nameAr} onChange={e => setForm(f => ({ ...f, nameAr: e.target.value }))} placeholder="الاسم بالعربي" />
   411→            </div>
   412→            <div className="space-y-2">
   413→              <Label htmlFor="location">الموقع</Label>
   414→              <Input id="location" value={form.location} onChange={e => setForm(f => ({ ...f, location: e.target.value }))} placeholder="المدينة - الحي" />
   415→            </div>
   416→            <div className="space-y-2">
   417→              <Label>العميل *</Label>
   418→              <Select value={form.clientId} onValueChange={v => setForm(f => ({ ...f, clientId: v }))}>
   419→                <SelectTrigger className="w-full"><SelectValue placeholder="اختر العميل" /></SelectTrigger>

    77→// ============ ImageUploadField Component ============
    78→function ImageUploadField({ 
    79→  value, 
    80→  onChange, 
    81→  label, 
    82→  labelAr,
    83→  lang,
    84→  hint,
    85→  hintAr,
    86→  previewHeight = 'h-24',
    87→  accept = 'image/svg+xml,image/png,image/jpeg,image/webp',
    88→}: {
    89→  value: string | null
    90→  onChange: (url: string | null) => void
    91→  label: string
    92→  labelAr: string
    93→  lang: 'ar' | 'en'
    94→  hint?: string
    95→  hintAr?: string
    96→  previewHeight?: string
    97→  accept?: string
    98→}) {
    99→  const [uploading, setUploading] = useState(false)
   100→  const [dragOver, setDragOver] = useState(false)
   101→  const fileInputRef = useRef<HTMLInputElement>(null)
   102→
   103→  const [error, setError] = useState<string | null>(null)
   104→
   105→  const uploadFile = useCallback(async (file: File) => {
   106→    if (file.size > 5 * 1024 * 1024) {
   107→      setError(lang === 'ar' ? 'حجم الملف يتجاوز 5 ميجابايت' : 'File size exceeds 5MB')
   108→      return
   109→    }
   110→    setUploading(true)
   111→    setError(null)
   112→    try {
   113→      const formData = new FormData()
   114→      formData.append('file', file)
   115→      const res = await fetch('/api/upload', { method: 'POST', body: formData })
   116→      if (!res.ok) {
   117→        const errData = await res.json().catch(() => ({}))
   118→        throw new Error(errData.error || 'Upload failed')
   119→      }
   120→      const data = await res.json()
   121→      onChange(data.url)
   122→    } catch (err) {
   123→      const msg = err instanceof Error ? err.message : 'Unknown error'
   124→      setError(lang === 'ar' ? `فشل في رفع الملف: ${msg}` : `Failed to upload file: ${msg}`)
   125→    } finally {
   126→      setUploading(false)
   127→    }
   128→  }, [lang, onChange])
   129→
   130→  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
   131→    const file = e.target.files?.[0]
   132→    if (file) {
   133→      setError(null)
   134→      uploadFile(file)
   135→    }
   136→    // Reset input so same file can be re-selected
   137→    setTimeout(() => { if (fileInputRef.current) fileInputRef.current.value = '' }, 100)
   138→  }
   139→
   140→  const handleDrop = (e: React.DragEvent) => {
   141→    e.preventDefault()
   142→    setDragOver(false)
   143→    const file = e.dataTransfer.files?.[0]
   144→    if (file) uploadFile(file)
   145→  }
   146→
   147→  const handleDragOver = (e: React.DragEvent) => {
   148→    e.preventDefault()
   149→    setDragOver(true)
   150→  }
   151→
   152→  const handleDragLeave = () => setDragOver(false)
   153→
   154→  const displayLabel = lang === 'ar' ? labelAr : label
   155→  const displayHint = lang === 'ar' ? hintAr : hint
   156→
   157→  return (
   158→    <div className="space-y-2">
   159→      <Label>{displayLabel}</Label>
   160→      {value ? (
   161→        <div className={`relative rounded-lg border bg-white p-2 ${previewHeight} flex items-center justify-center group overflow-hidden`}>
   162→          <img
   163→            src={value}
   164→            alt={displayLabel}
   165→            className="max-h-20 max-w-full object-contain"
   166→          />
   167→          <Button
   168→            variant="destructive"
   169→            size="icon"
   170→            className="absolute top-1 left-1 size-6 opacity-0 group-hover:opacity-100 transition-opacity z-10"
   171→            onClick={() => { setError(null); onChange(null) }}
   172→            type="button"
   173→          >
   174→            <X className="size-3" />
   175→          </Button>
   176→        </div>
   177→      ) : (
   178→        <div
   179→          className={`rounded-lg border-2 border-dashed transition-colors cursor-pointer ${
   180→            dragOver ? 'border-emerald-400 bg-emerald-50' : 'border-gray-300 bg-gray-50 hover:border-emerald-300 hover:bg-emerald-50/50'
   181→          } ${previewHeight} flex flex-col items-center justify-center gap-1`}
   182→          onClick={() => fileInputRef.current?.click()}
   183→          onDrop={handleDrop}
   184→          onDragOver={handleDragOver}
   185→          onDragLeave={handleDragLeave}
   186→        >
   187→          {uploading ? (
   188→            <>
   189→              <Loader2 className="size-5 text-emerald-500 animate-spin" />
   190→              <span className="text-xs text-muted-foreground">
   191→                {lang === 'ar' ? 'جاري الرفع...' : 'Uploading...'}
   192→              </span>
   193→            </>
   194→          ) : (
   195→            <>
   196→              <Upload className="size-5 text-gray-400" />
   197→              <span className="text-xs text-muted-foreground">
   198→                {lang === 'ar' ? 'اسحب الملف هنا أو انقر للاختيار' : 'Drag & drop or click to upload'}
   199→              </span>
   200→              <span className="text-[10px] text-muted-foreground">
   201→                {lang === 'ar' ? 'SVG, PNG, JPG - حد أقصى 5MB' : 'SVG, PNG, JPG - Max 5MB'}
   202→              </span>
   203→            </>
   204→          )}
   205→        </div>
   206→      )}
   207→      <input
   208→        ref={fileInputRef}
   209→        type="file"
   210→        accept={accept}
   211→        onChange={handleFileChange}
   212→        className="hidden"
   213→      />
   214→      {error && (
   215→        <p className="text-xs text-red-600">{error}</p>
   216→      )}
   217→      {displayHint && !error && (
   218→        <p className="text-xs text-muted-foreground">{displayHint}</p>
   219→      )}
   220→    </div>
   221→  )
   222→}

   871→function BranchDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (v: boolean) => void }) {
   872→  const { lang } = useAppStore()
   873→  const queryClient = useQueryClient()
   874→  const [name, setName] = useState('')
   875→  const [address, setAddress] = useState('')
   876→
   877→  React.useEffect(() => { if (open) { setName(''); setAddress('') } }, [open])
   878→
   879→  const createMutation = useMutation({
   880→    mutationFn: (data: Record<string, unknown>) =>
   881→      fetch('/api/branches', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) })
   882→        .then(r => { if (!r.ok) throw new Error(); return r.json() }),
   883→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['branches'] }); onOpenChange(false) },
   884→  })
   885→
   886→  const handleSubmit = (e: React.FormEvent) => {
   887→    e.preventDefault()
   888→    createMutation.mutate({ name, address: address || null })
   889→  }
   890→
   891→  return (
   892→    <Dialog open={open} onOpenChange={onOpenChange}>
   893→      <DialogContent className="max-w-md">
   894→        <DialogHeader>
   895→          <DialogTitle>{lang === 'ar' ? 'فرع جديد' : 'New Branch'}</DialogTitle>

   899→          <div className="space-y-2">

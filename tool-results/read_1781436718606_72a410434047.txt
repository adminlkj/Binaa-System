    88→// ============ Inventory Item Form Dialog (Create + Edit) ============
    89→function InventoryFormDialog({ open, onOpenChange, warehouses, editItem }: {
    90→  open: boolean; onOpenChange: (v: boolean) => void; warehouses: Warehouse[]
    91→  editItem?: InventoryItem | null
    92→}) {
    93→  const { lang } = useAppStore()
    94→  const queryClient = useQueryClient()
    95→  const { toast } = useToast()
    96→  const isEdit = !!editItem
    97→
    98→  const [name, setName] = useState('')
    99→  const [nameAr, setNameAr] = useState('')
   100→  const [itemType, setItemType] = useState('PRODUCT')
   101→  const [unit, setUnit] = useState('')
   102→  const [purchasePrice, setPurchasePrice] = useState('')
   103→  const [sellingPrice, setSellingPrice] = useState('')
   104→  const [quantity, setQuantity] = useState('')
   105→  const [minQuantity, setMinQuantity] = useState('')
   106→  const [warehouseId, setWarehouseId] = useState('')
   107→  const [category, setCategory] = useState('')
   108→
   109→  React.useEffect(() => {
   110→    if (open) {
   111→      if (editItem) {
   112→        setName(editItem.name)
   113→        setNameAr(editItem.nameAr || '')
   114→        setItemType(editItem.itemType || 'PRODUCT')
   115→        setUnit(editItem.unit)
   116→        setPurchasePrice(String(editItem.purchasePrice))
   117→        setSellingPrice(String(editItem.sellingPrice))
   118→        setQuantity(String(editItem.quantity))
   119→        setMinQuantity(String(editItem.minQuantity))
   120→        setWarehouseId(editItem.warehouseId || editItem.warehouse?.id || '')
   121→        setCategory(editItem.category || '')
   122→      } else {
   123→        setName(''); setNameAr(''); setItemType('PRODUCT'); setUnit('')
   124→        setPurchasePrice(''); setSellingPrice(''); setQuantity('')
   125→        setMinQuantity(''); setWarehouseId(''); setCategory('')
   126→      }
   127→    }
   128→  }, [open, editItem])
   129→
   130→  const saveMutation = useMutation({
   131→    mutationFn: (data: Record<string, unknown> & { id?: string }) => {
   132→      const payload = { name, nameAr, itemType, unit, purchasePrice, sellingPrice, quantity, minQuantity, warehouseId, category: category || null }
   133→      if (data.id) {
   134→        return fetch(`/api/inventory/${data.id}`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }).then(r => { if (!r.ok) throw new Error(); return r.json() })
   135→      }
   136→      return fetch('/api/inventory', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }).then(r => { if (!r.ok) throw new Error(); return r.json() })
   137→    },
   138→    onSuccess: () => {
   139→      queryClient.invalidateQueries({ queryKey: ['inventory'] })
   140→      toast({
   141→        title: t(isEdit ? 'تم التحديث' : 'تم الإنشاء', isEdit ? 'Updated' : 'Created', lang),
   142→        description: t(isEdit ? 'تم تحديث الصنف بنجاح' : 'تم إضافة الصنف بنجاح', isEdit ? 'Item updated successfully' : 'Item created successfully', lang),
   143→      })
   144→      onOpenChange(false)
   145→    },
   146→    onError: () => {
   147→      toast({ title: t('خطأ', 'Error', lang), description: t('فشل في حفظ الصنف', 'Failed to save item', lang), variant: 'destructive' })
   148→    },
   149→  })
   150→
   151→  const handleSubmit = (e: React.FormEvent) => {
   152→    e.preventDefault()
   153→    saveMutation.mutate({ name, nameAr, itemType, unit, purchasePrice, sellingPrice, quantity, minQuantity, warehouseId, category: category || null, id: editItem?.id })
   154→  }
   155→
   156→  const isService = itemType === 'SERVICE'
   157→
   158→  return (
   159→    <Dialog open={open} onOpenChange={onOpenChange}>
   160→      <DialogContent className="max-w-lg max-h-[90vh] overflow-y-auto">
   161→        <DialogHeader>
   162→          <DialogTitle>{t(isEdit ? 'تعديل صنف المخزون' : 'صنف مخزون جديد', isEdit ? 'Edit Inventory Item' : 'New Inventory Item', lang)}</DialogTitle>
   163→          <DialogDescription>{t(isEdit ? 'تعديل بيانات الصنف' : 'إضافة صنف جديد (منتج أو خدمة)', isEdit ? 'Edit item details' : 'Add new item (product or service)', lang)}</DialogDescription>
   164→        </DialogHeader>
   165→        <form onSubmit={handleSubmit} className="space-y-4">
   166→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   167→            <div className="space-y-2"><Label>{t('الاسم *', 'Name *', lang)}</Label><Input value={name} onChange={e => setName(e.target.value)} placeholder={t('اسم الصنف', 'Item name', lang)} required /></div>
   168→            <div className="space-y-2"><Label>{t('الاسم بالعربي', 'Arabic Name', lang)}</Label><Input value={nameAr} onChange={e => setNameAr(e.target.value)} placeholder={t('الاسم بالعربية', 'Arabic name', lang)} /></div>
   169→            <div className="space-y-2">
   170→              <Label>{t('النوع *', 'Type *', lang)}</Label>
   171→              <Select value={itemType} onValueChange={setItemType}>
   172→                <SelectTrigger><SelectValue /></SelectTrigger>
   173→                <SelectContent>
   174→                  {Object.entries(itemTypeConfig).map(([k, v]) => <SelectItem key={k} value={k}>{v.label[lang]}</SelectItem>)}
   175→                </SelectContent>
   176→              </Select>
   177→            </div>
   178→            <div className="space-y-2">
   179→              <Label>{t('الوحدة *', 'Unit *', lang)}</Label>
   180→              <Select value={unit} onValueChange={setUnit}>
   181→                <SelectTrigger><SelectValue placeholder={t('اختر الوحدة', 'Select unit', lang)} /></SelectTrigger>
   182→                <SelectContent>
   183→                  {['قطعة', 'كجم', 'طن', 'متر', 'متر مربع', 'لتر', 'كرتون', 'باكت', 'خدمة', 'ساعة', 'يوم'].map(u => <SelectItem key={u} value={u}>{u}</SelectItem>)}
   184→                </SelectContent>
   185→              </Select>
   186→            </div>
   187→            <div className="space-y-2">
   188→              <Label>{t('الفئة', 'Category', lang)}</Label>
   189→              <Select value={category} onValueChange={setCategory}>
   190→                <SelectTrigger><SelectValue placeholder={t('اختر الفئة', 'Select category', lang)} /></SelectTrigger>
   191→                <SelectContent>
   192→                  {categoryOptions.map(c => <SelectItem key={c.value} value={c.value}>{c.label[lang]}</SelectItem>)}
   193→                </SelectContent>
   194→              </Select>
   195→            </div>
   196→            <div className="space-y-2">
   197→              <Label>{t('المستودع *', 'Warehouse *', lang)}</Label>
   198→              <Select value={warehouseId} onValueChange={setWarehouseId}>
   199→                <SelectTrigger><SelectValue placeholder={t('اختر المستودع', 'Select warehouse', lang)} /></SelectTrigger>
   200→                <SelectContent>
   201→                  {warehouses.map(w => <SelectItem key={w.id} value={w.id}>{w.name}</SelectItem>)}
   202→                </SelectContent>
   203→              </Select>
   204→            </div>
   205→            <div className="space-y-2"><Label>{t('سعر الشراء', 'Purchase Price', lang)}</Label><Input type="number" min="0" step="0.01" value={purchasePrice} onChange={e => setPurchasePrice(e.target.value)} dir="ltr" placeholder="0.00" /></div>
   206→            <div className="space-y-2"><Label>{t('سعر البيع', 'Selling Price', lang)}</Label><Input type="number" min="0" step="0.01" value={sellingPrice} onChange={e => setSellingPrice(e.target.value)} dir="ltr" placeholder="0.00" /></div>
   207→            {!isService && (<>
   208→              <div className="space-y-2"><Label>{t('الكمية', 'Quantity', lang)}</Label><Input type="number" min="0" step="0.01" value={quantity} onChange={e => setQuantity(e.target.value)} dir="ltr" placeholder="0" /></div>
   209→              <div className="space-y-2"><Label>{t('الحد الأدنى', 'Min Quantity', lang)}</Label><Input type="number" min="0" step="0.01" value={minQuantity} onChange={e => setMinQuantity(e.target.value)} dir="ltr" placeholder="0" /></div>
   210→            </>)}
   211→          </div>
   212→          <DialogFooter>
   213→            <Button type="button" variant="outline" onClick={() => onOpenChange(false)}>{commonText.cancel[lang]}</Button>
   214→            <Button type="submit" disabled={saveMutation.isPending || !name || !unit || !warehouseId} className="bg-emerald-600 hover:bg-emerald-700">
   215→              {saveMutation.isPending ? t('جاري الحفظ...', 'Saving...', lang) : isEdit ? t('حفظ التعديلات', 'Save Changes', lang) : t('إضافة', 'Add', lang)}
   216→            </Button>
   217→          </DialogFooter>
   218→        </form>
   219→      </DialogContent>
   220→    </Dialog>
   221→  )
   222→}

   280→// ============ Items Tab ============
   281→function ItemsTab({ items, warehouses, isLoading, isError, refetch, onEdit, onDelete }: {
   282→  items: InventoryItem[]; warehouses: Warehouse[]
   283→  isLoading: boolean; isError: boolean; refetch: () => void
   284→  onEdit: (item: InventoryItem) => void; onDelete: (id: string) => void
   285→}) {
   286→  const { lang } = useAppStore()
   287→  const [search, setSearch] = useState('')
   288→  const [categoryFilter, setCategoryFilter] = useState<string>('all')
   289→  const [typeFilter, setTypeFilter] = useState<string>('all')
   290→
   291→  const filtered = items.filter(item => {
   292→    const matchCategory = categoryFilter === 'all' || item.category === categoryFilter
   293→    const matchType = typeFilter === 'all' || item.itemType === typeFilter
   294→    const matchSearch = !search || item.name.toLowerCase().includes(search.toLowerCase()) || item.code.toLowerCase().includes(search.toLowerCase())
   295→    return matchCategory && matchType && matchSearch
   296→  })
   297→
   298→  const productItems = items.filter(i => i.itemType === 'PRODUCT')
   299→  const lowStockItems = productItems.filter(i => i.quantity <= i.minQuantity)
   300→  const stockValue = productItems.reduce((s, i) => s + (i.quantity * i.purchasePrice), 0)
   301→
   302→  return (
   303→    <div className="space-y-4">
   304→      {/* Summary Cards */}
   305→      <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   306→        <Card className="bg-emerald-50 border-emerald-200"><CardContent className="p-4 flex items-center gap-3">
   307→          <div className="size-10 rounded-full bg-emerald-100 flex items-center justify-center"><Package className="size-5 text-emerald-600" /></div>
   308→          <div><p className="text-sm text-emerald-600">{t('إجمالي الأصناف', 'Total Items', lang)}</p><p className="text-xl font-bold text-emerald-700">{formatNumber(items.length)}</p></div>
   309→        </CardContent></Card>
   310→        <Card className="bg-teal-50 border-teal-200"><CardContent className="p-4 flex items-center gap-3">
   311→          <div className="size-10 rounded-full bg-teal-100 flex items-center justify-center"><Tag className="size-5 text-teal-600" /></div>
   312→          <div><p className="text-sm text-teal-600">{t('قيمة المخزون', 'Stock Value', lang)}</p><MoneyDisplay value={stockValue} lang={lang} size="sm" bold className="text-teal-700" /></div>
   313→        </CardContent></Card>
   314→        <Card className="bg-amber-50 border-amber-200"><CardContent className="p-4 flex items-center gap-3">
   315→          <div className="size-10 rounded-full bg-amber-100 flex items-center justify-center"><AlertTriangle className="size-5 text-amber-600" /></div>
   316→          <div><p className="text-sm text-amber-600">{t('أصناف منخفضة', 'Low Stock', lang)}</p><p className="text-xl font-bold text-amber-700">{formatNumber(lowStockItems.length)}</p></div>
   317→        </CardContent></Card>
   318→        <Card className="bg-purple-50 border-purple-200"><CardContent className="p-4 flex items-center gap-3">
   319→          <div className="size-10 rounded-full bg-purple-100 flex items-center justify-center"><Wrench className="size-5 text-purple-600" /></div>
   320→          <div><p className="text-sm text-purple-600">{t('خدمات', 'Services', lang)}</p><p className="text-xl font-bold text-purple-700">{formatNumber(items.filter(i => i.itemType === 'SERVICE').length)}</p></div>
   321→        </CardContent></Card>
   322→      </div>
   323→
   324→      {/* Low Stock Alert */}
   325→      {lowStockItems.length > 0 && (
   326→        <Card className="bg-amber-50 border-amber-300">
   327→          <CardContent className="p-4">
   328→            <div className="flex items-center gap-2 mb-2"><AlertTriangle className="size-5 text-amber-600" /><span className="font-semibold text-amber-700">{t(`تنبيه: ${lowStockItems.length} أصناف تحت الحد الأدنى`, `Alert: ${lowStockItems.length} items below minimum`, lang)}</span></div>
   329→            <div className="flex flex-wrap gap-2">
   330→              {lowStockItems.map(i => <Badge key={i.id} variant="outline" className="bg-amber-100 text-amber-700">{i.name} ({formatNumber(i.quantity)} / {formatNumber(i.minQuantity)})</Badge>)}
   331→            </div>
   332→          </CardContent>
   333→        </Card>
   334→      )}
   335→
   336→      {/* Filters */}
   337→      <Card><CardContent className="p-4">
   338→        <div className="flex flex-col sm:flex-row gap-3">
   339→          <div className="relative flex-1"><Search className="absolute right-3 top-1/2 -translate-y-1/2 size-4 text-muted-foreground" /><Input placeholder={t('بحث بالاسم أو الكود...', 'Search by name or code...', lang)} value={search} onChange={e => setSearch(e.target.value)} className="pr-9" /></div>
   340→          <Select value={typeFilter} onValueChange={setTypeFilter}>
   341→            <SelectTrigger className="w-full sm:w-40"><SelectValue placeholder={t('كل الأنواع', 'All Types', lang)} /></SelectTrigger>
   342→            <SelectContent>
   343→              <SelectItem value="all">{t('كل الأنواع', 'All Types', lang)}</SelectItem>
   344→              {Object.entries(itemTypeConfig).map(([k, v]) => <SelectItem key={k} value={k}>{v.label[lang]}</SelectItem>)}
   345→            </SelectContent>
   346→          </Select>
   347→          <Select value={categoryFilter} onValueChange={setCategoryFilter}>
   348→            <SelectTrigger className="w-full sm:w-48"><SelectValue placeholder={t('كل الفئات', 'All Categories', lang)} /></SelectTrigger>
   349→            <SelectContent>
   350→              <SelectItem value="all">{t('كل الفئات', 'All Categories', lang)}</SelectItem>
   351→              {categoryOptions.map(c => <SelectItem key={c.value} value={c.value}>{c.label[lang]}</SelectItem>)}
   352→            </SelectContent>
   353→          </Select>
   354→        </div>
   355→      </CardContent></Card>
   356→
   357→      {/* Table */}
   358→      <Card><CardContent className="p-0">
   359→        {isLoading ? (<div className="p-6"><TableSkeleton /></div>) : isError ? (
   360→          <div className="flex flex-col items-center gap-3 py-10"><p className="text-rose-600">{t('حدث خطأ', 'An error occurred', lang)}</p><Button variant="outline" onClick={() => refetch()}>{t('إعادة المحاولة', 'Retry', lang)}</Button></div>
   361→        ) : filtered.length === 0 ? (
   362→          <div className="flex flex-col items-center gap-3 py-10"><Package className="size-12 text-gray-300" /><p className="text-muted-foreground">{t('لا توجد أصناف', 'No items found', lang)}</p></div>
   363→        ) : (
   364→          <div className="overflow-x-auto">
   365→            <Table>
   366→              <TableHeader><TableRow>
   367→                <TableHead className="text-right">{t('الكود', 'Code', lang)}</TableHead>
   368→                <TableHead className="text-right">{t('الاسم', 'Name', lang)}</TableHead>

   371→                <TableHead className="text-right">{t('سعر الشراء', 'Purchase', lang)}</TableHead>

   393→                        <div className="flex items-center gap-1">

   457→// ============ Main Inventory Module ============
   458→export function InventoryModule() {
   459→  const { lang } = useAppStore()
   460→  const queryClient = useQueryClient()
   461→  const { toast } = useToast()
   462→  const [activeTab, setActiveTab] = useState('items')
   463→  const [itemDialogOpen, setItemDialogOpen] = useState(false)
   464→  const [editItem, setEditItem] = useState<InventoryItem | null>(null)
   465→  const [deleteId, setDeleteId] = useState<string | null>(null)
   466→
   467→  const { data: items = [], isLoading: loadingItems, isError: itemsError, refetch: refetchItems } = useQuery<InventoryItem[]>({
   468→    queryKey: ['inventory'],
   469→    queryFn: async () => { const res = await fetch('/api/inventory'); if (!res.ok) throw new Error(); return res.json() },
   470→  })
   471→
   472→  const { data: warehouses = [], isLoading: loadingWarehouses } = useQuery<Warehouse[]>({
   473→    queryKey: ['warehouses'],
   474→    queryFn: async () => { const res = await fetch('/api/warehouses'); if (!res.ok) return []; return res.json() },
   475→  })
   476→
   477→  const { data: branches = [] } = useQuery<Branch[]>({
   478→    queryKey: ['branches'],
   479→    queryFn: async () => { const res = await fetch('/api/branches'); if (!res.ok) return []; return res.json() },
   480→  })
   481→
   482→  // Delete mutation
   483→  const deleteMutation = useMutation({
   484→    mutationFn: (id: string) =>
   485→      fetch(`/api/inventory/${id}`, { method: 'DELETE' }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   486→    onSuccess: () => {
   487→      queryClient.invalidateQueries({ queryKey: ['inventory'] })
   488→      toast({ title: t('تم الحذف', 'Deleted', lang), description: t('تم حذف الصنف بنجاح', 'Item deleted successfully', lang) })
   489→      setDeleteId(null)
   490→    },
   491→    onError: () => {
   492→      toast({ title: t('خطأ', 'Error', lang), description: t('فشل في حذف الصنف', 'Failed to delete item', lang), variant: 'destructive' })
   493→    },
   494→  })
   495→
   496→  // CSV export
   497→  const handleExport = () => {
   498→    const columns: CSVColumn[] = [
   499→      { key: 'code', label: t('الكود', 'Code', lang) },
   500→      { key: 'name', label: t('الاسم', 'Name', lang) },
   501→      { key: 'itemType', label: t('النوع', 'Type', lang), format: (v) => itemTypeConfig[v as string]?.label[lang] || String(v) },
   502→      { key: 'unit', label: t('الوحدة', 'Unit', lang) },
   503→      { key: 'purchasePrice', label: t('سعر الشراء', 'Purchase', lang), format: (v) => (Number(v) || 0).toFixed(2) },
   504→      { key: 'sellingPrice', label: t('سعر البيع', 'Selling', lang), format: (v) => (Number(v) || 0).toFixed(2) },
   505→      { key: 'quantity', label: t('الكمية', 'Qty', lang), format: (v) => (Number(v) || 0).toFixed(2) },
   506→      { key: 'minQuantity', label: t('الحد الأدنى', 'Min', lang), format: (v) => (Number(v) || 0).toFixed(2) },
   507→      { key: 'category', label: t('الفئة', 'Category', lang) },
   508→      { key: 'warehouse', label: t('المستودع', 'Warehouse', lang) },
   509→    ]
   510→    const rows = items.map(item => ({
   511→      code: item.code,
   512→      name: item.name,
   513→      itemType: item.itemType,
   514→      unit: item.unit,
   515→      purchasePrice: item.purchasePrice,
   516→      sellingPrice: item.sellingPrice,
   517→      quantity: item.quantity,
   518→      minQuantity: item.minQuantity,
   519→      category: item.category || '',
   520→      warehouse: item.warehouse.name,
   521→    }))
   522→    exportToCSV(rows, `inventory-${new Date().toISOString().slice(0, 10)}`, columns)
   523→  }
   524→
   525→  // Handle edit
   526→  const handleEdit = (item: InventoryItem) => {
   527→    setEditItem(item)
   528→    setItemDialogOpen(true)
   529→  }
   530→
   531→  // Handle create (from header button)
   532→  const handleCreate = () => {
   533→    setEditItem(null)
   534→    setItemDialogOpen(true)
   535→  }
   536→
   537→  // Close item dialog
   538→  const handleItemDialogClose = (open: boolean) => {
   539→    setItemDialogOpen(open)
   540→    if (!open) setEditItem(null)
   541→  }
   542→
   543→  // Print data
   544→  const printData = useMemo(() => ({
   545→    columns: [
   546→      { key: 'code', label: lang === 'ar' ? 'الكود' : 'Code' },
   547→      { key: 'name', label: lang === 'ar' ? 'الاسم' : 'Name' },
   548→      { key: 'quantity', label: lang === 'ar' ? 'الكمية' : 'Qty' },
   549→      { key: 'warehouse', label: lang === 'ar' ? 'المستودع' : 'Warehouse' },
   550→    ],
   551→    rows: items.map(item => ({
   552→      code: item.code,
   553→      name: item.name,
   554→      quantity: item.quantity,
   555→      warehouse: item.warehouse.name,
   556→    })),
   557→    infoItems: [
   558→      { label: lang === 'ar' ? 'تاريخ الطباعة' : 'Print Date', value: new Date().toLocaleDateString() },
   559→      { label: lang === 'ar' ? 'عدد الأصناف' : 'Item Count', value: String(items.length) },
   560→    ],
   561→  }), [items, lang])
   562→
   563→  return (
   564→    <ModuleLayout
   565→      title={{ ar: 'المخزون', en: 'Inventory' }}
   566→      subtitle={{ ar: 'إدارة المنتجات والخدمات والمستودعات', en: 'Manage products, services & warehouses' }}
   567→      actions={
   568→        <div className="flex items-center gap-2">
   569→          <PrintButton type="generic-table" data={printData} size="icon" />
   570→          <Button variant="outline" size="icon" onClick={handleExport} title={t('تصدير CSV', 'Export CSV', lang)}>
   571→            <Download className="size-4" />
   572→          </Button>
   573→          <Button variant="outline" size="icon" onClick={() => refetchItems()} title={t('تحديث', 'Refresh', lang)}><RefreshCw className="size-4" /></Button>
   574→          {activeTab === 'items' && (
   575→            <Button className="gap-2 bg-emerald-600 hover:bg-emerald-700" onClick={handleCreate}>
   576→              <Plus className="size-4" />{t('صنف جديد', 'New Item', lang)}
   577→            </Button>
   578→          )}
   579→        </div>
   580→      }
   581→    >
   582→      <Tabs value={activeTab} onValueChange={setActiveTab}>
   583→        <TabsList className="grid grid-cols-2 w-full max-w-md">
   584→          <TabsTrigger value="items" className="gap-1"><Package className="size-3.5" />{t('الأصناف', 'Items', lang)}</TabsTrigger>
   585→          <TabsTrigger value="warehouses" className="gap-1"><WarehouseIcon className="size-3.5" />{t('المخازن', 'Warehouses', lang)}</TabsTrigger>

   601→      <AlertDialog open={!!deleteId} onOpenChange={() => setDeleteId(null)}>

   603→          <AlertDialogHeader>

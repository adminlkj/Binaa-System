   218→function t(ar: string, en: string, lang: Lang) {
   219→  return lang === 'ar' ? ar : en
   220→}
   221→
   222→function getCompletionEstimate(status: string, claimCount: number, contractValue: number): number {
   223→  if (contractValue <= 0) return 0
   224→  switch (status) {
   225→    case 'COMPLETED': return 100
   226→    case 'CANCELLED': return 0
   227→    case 'PLANNING': return Math.min(claimCount * 8, 15)
   228→    case 'ON_HOLD': return Math.min(15 + claimCount * 12, 60)
   229→    case 'ACTIVE': return Math.min(10 + claimCount * 15, 90)
   230→    default: return 0
   231→  }
   232→}
   233→
   234→function getActualCompletion(costSheet: CostSheet): number {
   235→  if (!costSheet || costSheet.contractValue <= 0) return 0
   236→  return Math.min(Math.round((costSheet.revenue / costSheet.contractValue) * 100), 100)
   237→}

   414→    queryKey: ['projects'],
   415→    queryFn: async () => {
   416→      const res = await fetch('/api/projects')
   417→      if (!res.ok) throw new Error('Failed to fetch projects')
   418→      return res.json()

   420→    staleTime: 30000,
   421→  })
   422→
   423→  const filtered = useMemo(() => {

   428→        p.client.name.toLowerCase().includes(search.toLowerCase()) ||
   429→        (p.nameAr && p.nameAr.includes(search))

   506→function ProjectDetailView() {
   507→  const { activeSubModule, setActiveSubModule, selectedProjectId, selectProject, lang } = useAppStore()
   508→
   509→  const { data: project, isLoading, isError, refetch } = useQuery<ProjectDetail>({
   510→    queryKey: ['project', selectedProjectId],
   511→    queryFn: async () => {
   512→      const res = await fetch(`/api/projects/${selectedProjectId}`)
   513→      if (!res.ok) throw new Error('Failed to fetch project')
   514→      return res.json()
   515→    },
   516→    enabled: !!selectedProjectId,
   517→    staleTime: 30000,
   518→  })
   519→
   520→  // Fetch sales invoices for this project
   521→  const { data: invoices = [] } = useQuery<SalesInvoiceItem[]>({
   522→    queryKey: ['project-invoices', selectedProjectId],
   523→    queryFn: async () => {
   524→      const res = await fetch(`/api/sales-invoices?projectId=${selectedProjectId}`)
   525→      if (!res.ok) return []
   526→      return res.json()
   527→    },
   528→    enabled: !!selectedProjectId && activeSubModule === 'project-invoices',
   529→  })
   530→
   531→  // Fetch client payments for this project's invoices
   532→  const { data: payments = [] } = useQuery<ClientPaymentItem[]>({
   533→    queryKey: ['project-payments', selectedProjectId],
   534→    queryFn: async () => {
   535→      const res = await fetch(`/api/client-payments?projectId=${selectedProjectId}`)
   536→      if (!res.ok) return []
   537→      return res.json()
   538→    },
   539→    enabled: !!selectedProjectId && activeSubModule === 'project-collections',
   540→  })
   541→
   542→  const handleBack = () => {
   543→    selectProject(null)
   544→    setActiveSubModule('project-list')
   545→  }
   546→
   547→  if (isLoading) return <DetailSkeleton />
   548→  if (isError || !project) return <ErrorState onRetry={() => refetch()} lang={lang} />
   549→
   550→  const completion = getActualCompletion(project.costSheet)
   551→  const statusLabel = statusLabels[project.status]?.[lang] || project.status
   552→
   553→  const renderTabContent = () => {
   554→    switch (activeSubModule) {
   555→      case 'project-overview':
   556→        return <OverviewTab project={project} completion={completion} invoices={invoices} payments={payments} lang={lang} />
   557→      case 'project-contracts':
   558→        return <ContractsTab project={project} lang={lang} />
   559→      case 'project-extracts':
   560→        return <ExtractsTab project={project} lang={lang} />
   561→      case 'project-invoices':
   562→        return <InvoicesTab invoices={invoices} lang={lang} />
   563→      case 'project-costs':
   564→        return <CostsTab project={project} lang={lang} />
   565→      case 'project-collections':
   566→        return <CollectionsTab payments={payments} lang={lang} />
   567→      case 'project-boq':
   568→        return <BOQTab project={project} lang={lang} />
   569→      case 'project-documents':
   570→        return <DocumentsPlaceholder lang={lang} />
   571→      default:
   572→        return <OverviewTab project={project} completion={completion} invoices={invoices} payments={payments} lang={lang} />
   573→    }
   574→  }
   575→
   576→  return (
   577→    <div className="space-y-0">
   578→      <div className="flex items-center gap-3 mb-4">
   579→        <Button variant="outline" size="icon" onClick={handleBack} className="shrink-0">
   580→          <ArrowRight className="size-4" />
   581→        </Button>
   582→        <div className="flex-1 min-w-0">
   583→          <div className="flex items-center gap-3 flex-wrap">
   584→            <h2 className="text-xl font-bold text-gray-900 truncate">{project.name}</h2>
   585→            <Badge variant="outline" className={statusColors[project.status]}>{statusLabel}</Badge>
   586→          </div>
   587→          <p className="text-sm text-muted-foreground mt-0.5">
   588→            {project.code} — {project.client.name}{project.location && ` — ${project.location}`}
   589→          </p>
   590→        </div>
   591→      </div>
   592→
   593→      <SectionLayout
   594→        title={{ ar: project.name, en: project.name }}
   595→        subtitle={{ ar: project.description || project.code, en: project.description || project.code }}
   596→        tabs={projectSubTabs}
   597→        showPrintExport={false}
   598→      >
   599→        {renderTabContent()}
   600→      </SectionLayout>
   601→    </div>
   602→  )
   603→}

   125→   125→// ============ Create Payroll Run Dialog ============
   126→   126→function CreatePayrollRunDialog({ open, onOpenChange }: {
   127→   127→  open: boolean; onOpenChange: (open: boolean) => void
   128→   128→}) {
   129→   129→  const queryClient = useQueryClient()
   130→   130→  const { lang } = useAppStore()
   131→   131→  const [form, setForm] = useState({
   132→   132→    month: String(new Date().getMonth() + 1),
   133→   133→    year: String(new Date().getFullYear()),
   134→   134→    selectionType: 'ALL' as SelectionType,
   135→   135→    selectionIds: [] as string[],
   136→   136→    notes: '',
   137→   137→  })
   138→   138→  const [employeeSearch, setEmployeeSearch] = useState('')
   139→   139→  const [selectedDepartment, setSelectedDepartment] = useState('')
   140→   140→
   141→   141→  const { data: teams = [] } = useQuery<WorkTeam[]>({
   142→   142→    queryKey: ['work-teams'],
   143→   143→    queryFn: async () => { const res = await fetch('/api/work-teams'); if (!res.ok) return []; return res.json() },
   144→   144→    enabled: open,
   145→   145→  })
   146→   146→
   147→   147→  const { data: projects = [] } = useQuery<Project[]>({
   148→   148→    queryKey: ['projects-list'],
   149→   149→    queryFn: async () => { const res = await fetch('/api/projects/list'); if (!res.ok) return []; return res.json() },
   150→   150→    enabled: open,
   151→   151→  })
   152→   152→
   153→   153→  const { data: employeesList = [] } = useQuery<EmployeeOption[]>({
   154→   154→    queryKey: ['employees-list'],
   155→   155→    queryFn: async () => { const res = await fetch('/api/employees?active=true'); if (!res.ok) return []; return res.json() },
   156→   156→    enabled: open && form.selectionType === 'EMPLOYEES',
   157→   157→  })
   158→   158→
   159→   159→  // Get unique departments from employees for DEPARTMENT selection
   160→   160→  const { data: allEmployeesForDepts = [] } = useQuery<EmployeeOption[]>({
   161→   161→    queryKey: ['employees-departments'],
   162→   162→    queryFn: async () => { const res = await fetch('/api/employees?active=true'); if (!res.ok) return []; return res.json() },
   163→   163→    enabled: open && form.selectionType === 'DEPARTMENT',
   164→   164→  })
   165→   165→
   166→   166→  const departments = useMemo(() => {
   167→   167→    const deptSet = new Set<string>()
   168→   168→    allEmployeesForDepts.forEach(e => {
   169→   169→      if (e.department) deptSet.add(e.department)
   170→   170→    })
   171→   171→    return Array.from(deptSet).sort()
   172→   172→  }, [allEmployeesForDepts])
   173→   173→
   174→   174→  // Filtered employees for searchable list
   175→   175→  const filteredEmployees = useMemo(() => {
   176→   176→    if (!employeeSearch) return employeesList
   177→   177→    const s = employeeSearch.toLowerCase()
   178→   178→    return employeesList.filter(e =>
   179→   179→      e.name.toLowerCase().includes(s) || e.code.toLowerCase().includes(s) || (e.department?.toLowerCase().includes(s))
   180→   180→    )
   181→   181→  }, [employeesList, employeeSearch])
   182→   182→
   183→   183→  const createMutation = useMutation({
   184→   184→    mutationFn: (data: Record<string, unknown>) => fetch('/api/payroll-runs', {
   185→   185→      method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data),
   186→   186→    }).then(r => { if (!r.ok) return r.json().then(e => { throw new Error(e.error || 'Error') }); return r.json() }),
   187→   187→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['payroll-runs'] }); onOpenChange(false) },
   188→   188→  })
   189→   189→
   190→   190→  const handleSubmit = (e: React.FormEvent) => {
   191→   191→    e.preventDefault()
   192→   192→    createMutation.mutate({
   193→   193→      month: parseInt(form.month),
   194→   194→      year: parseInt(form.year),
   195→   195→      selectionType: form.selectionType,
   196→   196→      selectionIds: form.selectionIds,
   197→   197→      notes: form.notes || null,
   198→   198→    })
   199→   199→  }
   200→   200→
   201→   201→  const toggleSelection = (id: string) => {
   202→   202→    setForm(f => ({
   203→   203→      ...f,
   204→   204→      selectionIds: f.selectionIds.includes(id)
   205→   205→        ? f.selectionIds.filter(x => x !== id)
   206→   206→        : [...f.selectionIds, id],
   207→   207→    }))
   208→   208→  }
   209→   209→
   210→   210→  // When DEPARTMENT is selected, set selectionIds to all employee IDs in that department
   211→   211→  React.useEffect(() => {
   212→   212→    if (form.selectionType === 'DEPARTMENT' && selectedDepartment) {
   213→   213→      const deptEmployeeIds = allEmployeesForDepts
   214→   214→        .filter(e => e.department === selectedDepartment)
   215→   215→        .map(e => e.id)
   216→   216→      setForm(f => ({ ...f, selectionIds: deptEmployeeIds }))
   217→   217→    }
   218→   218→  }, [form.selectionType, selectedDepartment, allEmployeesForDepts])
   219→   219→
   220→   220→  return (
   221→   221→    <Dialog open={open} onOpenChange={onOpenChange}>
   222→   222→      <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
   223→   223→        <DialogHeader>
   224→   224→          <DialogTitle>{t('إنشاء مسير رواتب', 'Create Payroll Run', lang)}</DialogTitle>
   225→   225→          <DialogDescription>{t('إنشاء مسير رواتب جديد للموظفين', 'Create a new payroll run for employees', lang)}</DialogDescription>
   226→   226→        </DialogHeader>
   227→   227→        <form onSubmit={handleSubmit} className="space-y-4">
   228→   228→          {/* Month & Year */}
   229→   229→          <div className="space-y-2">
   230→   230→            <h4 className="font-semibold text-sm text-emerald-700 border-b border-emerald-200 pb-1">
   231→   231→              {t('الفترة', 'Period', lang)}
   232→   232→            </h4>
   233→   233→            <div className="grid grid-cols-2 gap-4">
   234→   234→              <div className="space-y-2">
   235→   235→                <Label>{t('الشهر *', 'Month *', lang)}</Label>
   236→   236→                <Select value={form.month} onValueChange={v => setForm(f => ({ ...f, month: v }))}>
   237→   237→                  <SelectTrigger><SelectValue /></SelectTrigger>
   238→   238→                  <SelectContent>
   239→   239→                    {(lang === 'ar' ? arabicMonths : englishMonths).map((m, i) => (
   240→   240→                      <SelectItem key={i} value={String(i + 1)}>{m}</SelectItem>
   241→   241→                    ))}
   242→   242→                  </SelectContent>
   243→   243→                </Select>
   244→   244→              </div>
   245→   245→              <div className="space-y-2">
   246→   246→                <Label>{t('السنة *', 'Year *', lang)}</Label>
   247→   247→                <Input type="number" min="2020" max="2099" value={form.year} onChange={e => setForm(f => ({ ...f, year: e.target.value }))} dir="ltr" />
   248→   248→              </div>
   249→   249→            </div>
   250→   250→          </div>
   251→   251→
   252→   252→          {/* Selection Type */}
   253→   253→          <div className="space-y-2">
   254→   254→            <h4 className="font-semibold text-sm text-violet-700 border-b border-violet-200 pb-1 flex items-center gap-2">
   255→   255→              <Users className="size-4" />
   256→   256→              {t('اختيار الموظفين', 'Employee Selection', lang)}
   257→   257→            </h4>
   258→   258→            <div className="space-y-2">
   259→   259→              <Label>{t('نوع الاختيار *', 'Selection Type *', lang)}</Label>
   260→   260→              <Select value={form.selectionType} onValueChange={(v: SelectionType) => setForm(f => ({ ...f, selectionType: v, selectionIds: [] }))}>
   261→   261→                <SelectTrigger><SelectValue /></SelectTrigger>
   262→   262→                <SelectContent>
   263→   263→                  <SelectItem value="ALL">{t('جميع الموظفين', 'All Employees', lang)}</SelectItem>
   264→   264→                  <SelectItem value="TEAM">{t('حسب فريق العمل', 'By Work Team', lang)}</SelectItem>
   265→   265→                  <SelectItem value="PROJECT">{t('حسب المشروع', 'By Project', lang)}</SelectItem>
   266→   266→                  <SelectItem value="EMPLOYEES">{t('موظفون محددون', 'Specific Employees', lang)}</SelectItem>
   267→   267→                  <SelectItem value="DEPARTMENT">{t('قسم', 'Department', lang)}</SelectItem>
   268→   268→                </SelectContent>
   269→   269→              </Select>
   270→   270→            </div>
   271→   271→
   272→   272→            {/* Team Selection */}
   273→   273→            {form.selectionType === 'TEAM' && (
   274→   274→              <div className="space-y-2">
   275→   275→                <Label>{t('اختر فرق العمل', 'Select Work Teams', lang)}</Label>
   276→   276→                <div className="border rounded-lg max-h-48 overflow-y-auto p-2 space-y-1">
   277→   277→                  {teams.length === 0 ? (
   278→   278→                    <p className="text-sm text-muted-foreground text-center py-4">{t('لا توجد فرق عمل', 'No work teams found', lang)}</p>
   279→   279→                  ) : teams.map(team => (
   280→   280→                    <label key={team.id} className="flex items-center gap-2 p-2 rounded hover:bg-gray-50 cursor-pointer">
   281→   281→                      <input
   282→   282→                        type="checkbox"
   283→   283→                        checked={form.selectionIds.includes(team.id)}
   284→   284→                        onChange={() => toggleSelection(team.id)}
   285→   285→                        className="rounded border-gray-300"
   286→   286→                      />
   287→   287→                      <span className="text-sm">{team.code} - {team.name}{team.project ? ` (${team.project.name})` : ''}</span>
   288→   288→                    </label>
   289→   289→                  ))}
   290→   290→                </div>
   291→   291→                {form.selectionIds.length > 0 && (
   292→   292→                  <p className="text-xs text-muted-foreground">{t(`${form.selectionIds.length} فرق محددة`, `${form.selectionIds.length} teams selected`, lang)}</p>

   294→   294→              </div>
   295→   295→            )}
   296→   296→

   298→   298→            {form.selectionType === 'PROJECT' && (

   301→   301→                <div className="border rounded-lg max-h-48 overflow-y-auto p-2 space-y-1">

   303→   303→                    <p className="text-sm text-muted-foreground text-center py-4">{t('لا توجد مشاريع', 'No projects found', lang)}</p>
   304→   304→                  ) : projects.map(project => (
   305→   305→                    <label key={project.id} className="flex items-center gap-2 p-2 rounded hover:bg-gray-50 cursor-pointer">
   306→   306→                      <input
   307→   307→                        type="checkbox"
   308→   308→                        checked={form.selectionIds.includes(project.id)}
   309→   309→                        onChange={() => toggleSelection(project.id)}
   310→   310→                        className="rounded border-gray-300"
   311→   311→                      />
   312→   312→                      <span className="text-sm">{project.code} - {project.name}</span>
   313→   313→                    </label>
   314→   314→                  ))}
   315→   315→                </div>
   316→   316→                {form.selectionIds.length > 0 && (
   317→   317→                  <p className="text-xs text-muted-foreground">{t(`${form.selectionIds.length} مشاريع محددة`, `${form.selectionIds.length} projects selected`, lang)}</p>

   325→   325→                <div className="flex items-center justify-between">
   326→   326→                  <Label>{t('اختر الموظفين', 'Select Employees', lang)}</Label>

   329→   329→                    <Button type="button" variant="ghost" size="sm" onClick={() => setForm(f => ({ ...f, selectionIds: [] }))}>{t('إلغاء التحديد', 'Deselect All', lang)}</Button>

   332→   332→                <div className="relative">

   336→   336→                    value={employeeSearch}
   337→   337→                    onChange={e => setEmployeeSearch(e.target.value)}
   338→   338→                    className="pr-9"
   339→   339→                  />
   340→   340→                </div>
   341→   341→                <div className="border rounded-lg max-h-48 overflow-y-auto p-2 space-y-1">

   343→   343→                    <p className="text-sm text-muted-foreground text-center py-4">{t('لا يوجد موظفون', 'No employees found', lang)}</p>

   350→   350→                        className="rounded border-gray-300"
   351→   351→                      />

   353→   353→                    </label>

   355→   355→                </div>

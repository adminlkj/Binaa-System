   639→// ============ PAYROLL RUNS (مسيرات الرواتب) ============
   640→// مسير الرواتب يجمع عدة موظفين لفترة شهرية معينة.
   641→// المصدر الوحيد للحقيقة في قيود الرواتب هو JournalEntry المرتبط عبر journalEntryId.
   642→
   643→model PayrollRun {
   644→  id                   String           @id @default(cuid())
   645→  code                 String           @unique // PAY-YYYY-NNNN
   646→  month                Int              // 1-12
   647→  year                 Int
   648→  status               PayrollRunStatus @default(DRAFT)
   649→  totalAmount          Decimal          @default(0) // إجمالي الاستحقاق
   650→  totalDeductions      Decimal          @default(0) // إجمالي الخصومات
   651→  totalGosi            Decimal          @default(0) // إجمالي التأمينات
   652→  totalNet             Decimal          @default(0) // إجمالي الصافي
   653→  notes                String?
   654→  journalEntryId       String?          // قيد الاستحقاق (مدين 8110/7120/7210 / دائن 3310)

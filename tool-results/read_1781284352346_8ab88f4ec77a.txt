   264→   264→export async function resolveCostCenterId(projectId: string | undefined | null): Promise<string | undefined> {

   322→   322→export async function initializeChartOfAccounts() {
   323→   323→  let created = 0
   324→   324→  let updated = 0
   325→   325→
   326→   326→  // Create parent accounts first, then children (sorted by code length then code)
   327→   327→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   328→   328→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   329→   329→    return a.code.localeCompare(b.code)
   330→   330→  })
   331→   331→
   332→   332→  for (const tmpl of sorted) {
   333→   333→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   334→   334→    if (existing) {
   335→   335→      // Update existing account with new fields
   336→   336→      await db.account.update({
   337→   337→        where: { code: tmpl.code },
   338→   338→        data: {
   339→   339→          name: tmpl.name,
   340→   340→          nameAr: tmpl.nameAr,
   341→   341→          type: tmpl.type,
   342→   342→          activityType: tmpl.activityType || null,
   343→   343→          isSystem: tmpl.isSystem || false,
   344→   344→          allowPosting: tmpl.allowPosting || false,
   345→   345→          level: tmpl.level || 0,
   346→   346→          isActive: true,
   347→   347→        },
   348→   348→      })
   349→   349→      // Update parent reference if needed
   350→   350→      if (tmpl.parentId) {
   351→   351→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   352→   352→        if (parent && existing.parentId !== parent.id) {
   353→   353→          await db.account.update({
   354→   354→            where: { code: tmpl.code },
   355→   355→            data: { parentId: parent.id },
   356→   356→          })
   357→   357→        }
   358→   358→      }
   359→   359→      updated++
   360→   360→    } else {
   361→   361→      await ensureAccountExists(tmpl)
   362→   362→      created++
   363→   363→    }
   364→   364→  }
   365→   365→
   366→   366→  const total = await db.account.count()
   367→   367→  return { created, updated, total }
   368→   368→}

   370→   370→// ============ JOURNAL ENTRY CREATION ============
   371→   371→
   372→   372→export async function createJournalEntry(template: JournalEntryTemplate) {
   373→   373→  // Validate: total debits must equal total credits
   374→   374→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   375→   375→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   376→   376→
   377→   377→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   378→   378→    throw new Error(
   379→   379→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   380→   380→    )
   381→   381→  }
   382→   382→
   383→   383→  // Ensure all referenced accounts exist
   384→   384→  for (const line of template.lines) {
   385→   385→    const account = await getAccountByCode(line.accountCode)
   386→   386→    if (!account) {
   387→   387→      throw new Error(`Account not found: ${line.accountCode}`)
   388→   388→    }
   389→   389→  }
   390→   390→
   391→   391→  // Create the journal entry with lines
   392→   392→  const entry = await db.journalEntry.create({
   393→   393→    data: {
   394→   394→      entryNo: template.entryNo,
   395→   395→      date: template.date,
   396→   396→      description: template.description,
   397→   397→      status: 'POSTED', // Auto-entries are posted immediately
   398→   398→      lines: {
   399→   399→        create: await Promise.all(
   400→   400→          template.lines.map(async (line) => {
   401→   401→            const account = await getAccountByCode(line.accountCode)
   402→   402→            return {
   403→   403→              accountId: account!.id,
   404→   404→              costCenterId: line.costCenterId,
   405→   405→              debit: line.debit,
   406→   406→              credit: line.credit,
   407→   407→              description: line.description,
   408→   408→            }
   409→   409→          })
   410→   410→        ),
   411→   411→      },
   412→   412→    },
   413→   413→    include: { lines: true },
   414→   414→  })
   415→   415→
   416→   416→  return entry
   417→   417→}

   419→   419→// ============ AUTO-ENTRY FUNCTIONS ============
   420→   420→// Each function creates the appropriate journal entries for a business transaction
   421→   421→
   422→   422→/**
   423→   423→ * فاتورة مبيعات - Sales Invoice (from Extract)
   424→   424→ * Dr: Clients Receivable (1210) - totalAmount
   425→   425→ * Cr: Progress Claims Revenue (6110) - subtotal
   426→   426→ * Cr: VAT Payable (3200) - vatAmount
   427→   427→ */
   428→   428→export async function autoEntrySalesInvoice(data: {
   429→   429→  invoiceNo: string
   430→   430→  clientId: string
   431→   431→  subtotal: number
   432→   432→  vatRate: number
   433→   433→  vatAmount: number
   434→   434→  totalAmount: number
   435→   435→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   436→   436→  date: Date
   437→   437→  projectId?: string
   438→   438→  costCenterId?: string
   439→   439→}) {
   440→   440→  // Determine revenue account based on invoice type
   441→   441→  let revenueAccountCode: string
   442→   442→  switch (data.invoiceType) {
   443→   443→    case 'RENTAL':
   444→   444→      revenueAccountCode = '6210' // Equipment Rental Revenue
   445→   445→      break
   446→   446→    case 'PROGRESS_CLAIM':
   447→   447→      revenueAccountCode = '6110' // Progress Claims Revenue
   448→   448→      break
   449→   449→    default:
   450→   450→      revenueAccountCode = '6340' // Service Revenue
   451→   451→  }
   452→   452→
   453→   453→  const lines = [
   454→   454→    { accountCode: '1210', debit: data.totalAmount, credit: 0, costCenterId: data.costCenterId }, // Clients Receivable
   455→   455→    { accountCode: revenueAccountCode, debit: 0, credit: data.subtotal, costCenterId: data.costCenterId }, // Revenue
   456→   456→  ]
   457→   457→
   458→   458→  // Add VAT line only if VAT > 0 (Output VAT)
   459→   459→  if (data.vatAmount > 0) {
   460→   460→    lines.push({ accountCode: '3110', debit: 0, credit: data.vatAmount }) // Output VAT (ضريبة مخرجات)
   461→   461→  }
   462→   462→
   463→   463→  return createJournalEntry({
   464→   464→    entryNo: `JE-SI-${Date.now()}`,
   465→   465→    date: data.date,
   466→   466→    description: `Sales Invoice ${data.invoiceNo}`,
   467→   467→    descriptionAr: `فاتورة مبيعات ${data.invoiceNo}`,
   468→   468→    lines,
   469→   469→    sourceType: 'SALES_INVOICE',
   470→   470→    sourceId: data.invoiceNo,
   471→   471→  })
   472→   472→}
   473→   473→
   474→   474→/**
   475→   475→ * فاتورة مشتريات - Purchase Invoice

  1301→  1301→    totalDebit: number

  1309→  1309→        accountBalances.set(key, {
  1310→  1310→          account: line.account,
  1311→  1311→          totalDebit: 0,
  1312→  1312→          totalCredit: 0,
  1313→  1313→        })
  1314→  1314→      }
  1315→  1315→      const bal = accountBalances.get(key)!
  1316→  1316→      bal.totalDebit += line.debit
  1317→  1317→      bal.totalCredit += line.credit
  1318→  1318→    }
  1319→  1319→  }

  1323→  1323→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'

  1416→  1416→  })
  1417→  1417→}

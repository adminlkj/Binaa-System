   272→export async function ensureAccountExists(template: AccountTemplate, tx?: PrismaTransaction) {
   273→  const client = tx || db
   274→  const existing = await client.account.findUnique({ where: { code: template.code } })
   275→  if (existing) {
   276→    // Update existing account with new fields if they are missing
   277→    if (
   278→      existing.activityType !== (template.activityType || null) ||
   279→      existing.isSystem !== (template.isSystem || false) ||
   280→      existing.allowPosting !== (template.allowPosting || false) ||
   281→      existing.level !== (template.level || 0)
   282→    ) {
   283→      await client.account.update({
   284→        where: { code: template.code },
   285→        data: {
   286→          name: template.name,
   287→          nameAr: template.nameAr,
   288→          type: template.type,
   289→          activityType: template.activityType || null,
   290→          isSystem: template.isSystem || false,
   291→          allowPosting: template.allowPosting || false,
   292→          level: template.level || 0,
   293→        },
   294→      })
   295→    }
   296→    return client.account.findUnique({ where: { code: template.code } })
   297→  }
   298→
   299→  let parentId: string | undefined
   300→  if (template.parentId) {
   301→    const parent = await client.account.findUnique({ where: { code: template.parentId } })
   302→    if (parent) parentId = parent.id
   303→  }
   304→
   305→  return client.account.create({
   306→    data: {
   307→      code: template.code,
   308→      name: template.name,
   309→      nameAr: template.nameAr,
   310→      type: template.type,
   311→      parentId,
   312→      isActive: true,
   313→      activityType: template.activityType || null,
   314→      isSystem: template.isSystem || false,
   315→      allowPosting: template.allowPosting || false,
   316→      level: template.level || 0,
   317→    },
   318→  })
   319→}

   321→// ============ INITIALIZE CHART OF ACCOUNTS ============

   323→export async function initializeChartOfAccounts() {
   324→  let created = 0
   325→  let updated = 0
   326→
   327→  // Create parent accounts first, then children (sorted by code length then code)
   328→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   329→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   330→    return a.code.localeCompare(b.code)
   331→  })
   332→
   333→  for (const tmpl of sorted) {
   334→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   335→    if (existing) {
   336→      // Update existing account with new fields
   337→      await db.account.update({
   338→        where: { code: tmpl.code },
   339→        data: {
   340→          name: tmpl.name,
   341→          nameAr: tmpl.nameAr,
   342→          type: tmpl.type,
   343→          activityType: tmpl.activityType || null,
   344→          isSystem: tmpl.isSystem || false,
   345→          allowPosting: tmpl.allowPosting || false,
   346→          level: tmpl.level || 0,
   347→          isActive: true,
   348→        },
   349→      })
   350→      // Update parent reference if needed
   351→      if (tmpl.parentId) {
   352→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   353→        if (parent && existing.parentId !== parent.id) {
   354→          await db.account.update({
   355→            where: { code: tmpl.code },
   356→            data: { parentId: parent.id },
   357→          })
   358→        }
   359→      }
   360→      updated++
   361→    } else {
   362→      await ensureAccountExists(tmpl)
   363→      created++
   364→    }
   365→  }
   366→
   367→  const total = await db.account.count()
   368→  return { created, updated, total }
   369→}

   373→export async function createJournalEntry(template: JournalEntryTemplate, tx?: PrismaTransaction) {
   374→  const client = tx || db
   375→  // Validate: total debits must equal total credits
   376→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   377→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   378→
   379→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   380→    throw new Error(
   381→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   382→    )
   383→  }
   384→
   385→  // Ensure all referenced accounts exist
   386→  const accountMap = new Map<string, { id: string; code: string; nameAr: string }>()
   387→  for (const line of template.lines) {
   388→    const account = await getAccountByCode(line.accountCode, tx)
   389→    if (!account) {
   390→      throw new Error(`Account not found: ${line.accountCode}`)
   391→    }
   392→    accountMap.set(line.accountCode, { id: account.id, code: account.code, nameAr: account.nameAr || '' })
   393→  }
   394→
   395→  const now = new Date()
   396→
   397→  // Create the journal entry with lines and audit log
   398→  const entry = await client.journalEntry.create({
   399→    data: {
   400→      entryNo: template.entryNo,
   401→      date: template.date,
   402→      description: template.description,
   403→      descriptionAr: template.descriptionAr,
   404→      status: 'POSTED', // Auto-entries are posted immediately
   405→      sourceType: template.sourceType,
   406→      sourceId: template.sourceId,
   407→      sourceDocumentNo: template.sourceDocumentNo,
   408→      activityType: template.activityType,
   409→      projectId: template.projectId,
   410→      clientId: template.clientId,
   411→      supplierId: template.supplierId,
   412→      isReversal: template.isReversal || false,
   413→      reversedEntryId: template.reversedEntryId,
   414→      reversalReason: template.reversalReason,
   415→      postedAt: now,
   416→      lines: {
   417→        create: template.lines.map((line) => {
   418→          const acct = accountMap.get(line.accountCode)!
   419→          return {
   420→            accountId: acct.id,
   421→            accountCode: acct.code,
   422→            accountNameAr: acct.nameAr,
   423→            costCenterId: line.costCenterId,
   424→            debit: line.debit,
   425→            credit: line.credit,
   426→            description: line.description,
   427→          };
   428→        }),
   429→      },
   430→      auditLogs: {
   431→        create: [
   432→          {
   433→            action: 'CREATED',
   434→            actionAr: 'إنشاء القيد',
   435→            performedBy: 'System',
   436→            details: `Auto-created for ${template.sourceType}`,
   437→          },
   438→          {
   439→            action: 'POSTED',
   440→            actionAr: 'ترحيل القيد',
   441→            performedBy: 'System',
   442→            details: `Auto-posted at ${now.toISOString()}`,
   443→          },
   444→        ],
   445→      },
   446→    },
   447→    include: { lines: true, auditLogs: true },
   448→  })
   449→
   450→  return entry
   451→}

   456→/**
   457→ * فاتورة مبيعات - Sales Invoice (from Extract)
   458→ * Dr: Clients Receivable (1210) - totalAmount
   459→ * Cr: Progress Claims Revenue (6110) - subtotal
   460→ * Cr: VAT Payable (3200) - vatAmount
   461→ */
   462→export async function autoEntrySalesInvoice(data: {
   463→  invoiceNo: string
   464→  clientId: string
   465→  subtotal: number
   466→  vatRate: number
   467→  vatAmount: number
   468→  totalAmount: number
   469→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   470→  date: Date
   471→  projectId?: string
   472→  costCenterId?: string
   473→}, tx?: PrismaTransaction) {
   474→  // Determine revenue account based on invoice type
   475→  let revenueAccountCode: string
   476→  switch (data.invoiceType) {
   477→    case 'RENTAL':
   478→      revenueAccountCode = '6210' // Equipment Rental Revenue
   479→      break
   480→    case 'PROGRESS_CLAIM':
   481→      revenueAccountCode = '6110' // Progress Claims Revenue
   482→      break
   483→    default:
   484→      revenueAccountCode = '6340' // Service Revenue
   485→  }
   486→
   487→  const lines = [
   488→    { accountCode: '1210', debit: data.totalAmount, credit: 0, costCenterId: data.costCenterId }, // Clients Receivable
   489→    { accountCode: revenueAccountCode, debit: 0, credit: data.subtotal, costCenterId: data.costCenterId }, // Revenue
   490→  ]
   491→
   492→  // Add VAT line only if VAT > 0 (Output VAT)
   493→  if (data.vatAmount > 0) {
   494→    lines.push({ accountCode: '3110', debit: 0, credit: data.vatAmount, costCenterId: data.costCenterId }) // Output VAT (ضريبة مخرجات)
   495→  }
   496→
   497→  return createJournalEntry({
   498→    entryNo: `JE-SI-${Date.now()}`,
   499→    date: data.date,
   500→    description: `Sales Invoice ${data.invoiceNo}`,
   501→    descriptionAr: `فاتورة مبيعات ${data.invoiceNo}`,
   502→    lines,
   503→    sourceType: 'SALES_INVOICE',
   504→    sourceId: data.invoiceNo,
   505→  }, tx)
   506→}

   931→/**
   932→ * رواتب - Salary Entry (Accrual + Payment)
   933→ *
   934→ * Supports two modes via the `accrualOnly` parameter:
   935→ *
   936→ * ACCRUAL MODE (accrualOnly = true):
   937→ *   Used during payroll run to record the expense and liability.
   938→ *   Dr: Salaries & Wages (8110) - grossSalary
   939→ *   Dr: GOSI Expense (8210) - gosiEmployerContribution
   940→ *   Cr: Salaries Payable (3310) - netSalary (gross - gosiEmployeeDeduction)
   941→ *   Cr: GOSI Payable (3830) - total GOSI (employee + employer)
   942→ *
   943→ * PAYMENT MODE (accrualOnly = false, default):
   944→ *   Used when salary is actually paid to clear the payable.
   945→ *   Dr: Salaries Payable (3310) - netSalary
   946→ *   Cr: Cash/Bank (1110/1120) - netCashPaid (netSalary)
   947→ *
   948→ * FULL COMBINED MODE (for backward compatibility, use accrualOnly = undefined):
   949→ *   Creates a single entry that records both expense and payment.
   950→ *   Dr: Salaries & Wages (8110) - grossSalary
   951→ *   Dr: GOSI Expense (8210) - gosiEmployerContribution
   952→ *   Cr: Cash/Bank (1110/1120) - netCashPaid
   953→ *   Cr: GOSI Payable (3830) - total GOSI
   954→ *   Cr: Salaries & Wages (8110) - gosiEmployeeDeduction (contra-entry)
   955→ */
   956→export async function autoEntrySalary(data: {
   957→  employeeName: string
   958→  grossSalary: number
   959→  gosiEmployeeDeduction: number
   960→  gosiEmployerContribution: number
   961→  date: Date
   962→  payFrom: 'TREASURY' | 'BANK'
   963→  costCenterId?: string
   964→  accrualOnly?: boolean // true = accrual entry only, false/undefined = payment entry
   965→}, tx?: PrismaTransaction) {
   966→  const cashAccountCode = data.payFrom === 'BANK' ? '1120' : '1110'
   967→  const netSalary = data.grossSalary - data.gosiEmployeeDeduction
   968→
   969→  // ACCRUAL MODE: Record expense and liability without cash payment
   970→  if (data.accrualOnly === true) {
   971→    const lines = [
   972→      { accountCode: '8110', debit: data.grossSalary, credit: 0, costCenterId: data.costCenterId }, // Salaries & Wages (gross)
   973→      { accountCode: '8210', debit: data.gosiEmployerContribution, credit: 0, costCenterId: data.costCenterId }, // GOSI Expense (employer)
   974→    ]
   975→
   976→    // Credit Salaries Payable for the net amount
   977→    lines.push({ accountCode: '3310', debit: 0, credit: netSalary, costCenterId: data.costCenterId }) // Salaries Payable (رواتب مستحقة)
   978→
   979→    // Credit GOSI Payable for total GOSI (employee deduction + employer contribution)
   980→    const totalGOSI = data.gosiEmployeeDeduction + data.gosiEmployerContribution
   981→    if (totalGOSI > 0) {
   982→      lines.push({ accountCode: '3830', debit: 0, credit: totalGOSI, costCenterId: data.costCenterId }) // GOSI Payable
   983→    }
   984→
   985→    return createJournalEntry({
   986→      entryNo: `JE-SAL-ACC-${Date.now()}`,
   987→      date: data.date,
   988→      description: `Salary accrual - ${data.employeeName}`,
   989→      descriptionAr: `استحقاق راتب - ${data.employeeName}`,
   990→      lines,
   991→      sourceType: 'SALARY_ACCRUAL',
   992→      sourceId: `SAL-ACC-${Date.now()}`,
   993→    }, tx)
   994→  }
   995→
   996→  // PAYMENT MODE: Clear the payable with cash
   997→  // When accrualOnly is explicitly false, this is the payment step
   998→  if (data.accrualOnly === false) {
   999→    return createJournalEntry({
  1000→      entryNo: `JE-SAL-PAY-${Date.now()}`,
  1001→      date: data.date,
  1002→      description: `Salary payment - ${data.employeeName}`,
  1003→      descriptionAr: `صرف راتب - ${data.employeeName}`,
  1004→      lines: [
  1005→        { accountCode: '3310', debit: netSalary, credit: 0 }, // Dr: Salaries Payable
  1006→        { accountCode: cashAccountCode, debit: 0, credit: netSalary }, // Cr: Cash/Bank
  1007→      ],
  1008→      sourceType: 'SALARY_PAYMENT',
  1009→      sourceId: `SAL-PAY-${Date.now()}`,
  1010→    }, tx)
  1011→  }
  1012→
  1013→  // FULL COMBINED MODE (backward compatible): Record expense + payment in one entry
  1014→  const lines = [
  1015→    { accountCode: '8110', debit: data.grossSalary, credit: 0, costCenterId: data.costCenterId }, // Salaries & Wages (gross)
  1016→    { accountCode: '8210', debit: data.gosiEmployerContribution, credit: 0, costCenterId: data.costCenterId }, // GOSI Expense (employer)
  1017→  ]
  1018→
  1019→  if (data.gosiEmployeeDeduction > 0) {
  1020→    lines.push({ accountCode: '8110', debit: 0, credit: data.gosiEmployeeDeduction, costCenterId: data.costCenterId }) // Reduce salary by GOSI employee share
  1021→  }
  1022→
  1023→  lines.push({ accountCode: cashAccountCode, debit: 0, credit: netSalary, costCenterId: data.costCenterId }) // Cash/Bank paid
  1024→  lines.push({ accountCode: '3830', debit: 0, credit: data.gosiEmployeeDeduction + data.gosiEmployerContribution, costCenterId: data.costCenterId }) // GOSI Payable (total)
  1025→
  1026→  return createJournalEntry({
  1027→    entryNo: `JE-SAL-${Date.now()}`,
  1028→    date: data.date,
  1029→    description: `Salary payment - ${data.employeeName}`,
  1030→    descriptionAr: `صرف راتب - ${data.employeeName}`,
  1031→    lines,
  1032→    sourceType: 'SALARY',
  1033→    sourceId: `SAL-${Date.now()}`,
  1034→  }, tx)
  1035→}
  1036→
  1037→/**
  1038→ * تأمينات اجتماعية - GOSI Contribution
  1039→ * Dr: GOSI Expense (8210)

  1373→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1374→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1375→  if (dateFrom || dateTo) {
  1376→    dateFilter.date = {
  1377→      ...(dateFrom && { gte: dateFrom }),
  1378→      ...(dateTo && { lte: dateTo }),
  1379→    }
  1380→  }
  1381→  const entries = await db.journalEntry.findMany({
  1382→    where: {
  1383→      status: 'POSTED',
  1384→      ...dateFilter,
  1385→    },
  1386→    include: {
  1387→      lines: {
  1388→        include: {
  1389→          account: true,
  1390→        },
  1391→      },
  1392→    },
  1393→  })
  1394→
  1395→  // Aggregate by account
  1396→  const accountBalances = new Map<string, {
  1397→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1398→    totalDebit: number
  1399→    totalCredit: number
  1400→  }>()
  1401→
  1402→  for (const entry of entries) {
  1403→    for (const line of entry.lines) {
  1404→      const key = line.accountId
  1405→      if (!accountBalances.has(key)) {
  1406→        accountBalances.set(key, {
  1407→          account: line.account,
  1408→          totalDebit: 0,
  1409→          totalCredit: 0,
  1410→        })
  1411→      }
  1412→      const bal = accountBalances.get(key)!
  1413→      bal.totalDebit += toNumber(line.debit)
  1414→      bal.totalCredit += toNumber(line.credit)
  1415→    }
  1416→  }
  1417→
  1418→  // Calculate net balance for each account
  1419→  const results = Array.from(accountBalances.values()).map(bal => {
  1420→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1421→    let netDebit = 0
  1422→    let netCredit = 0
  1423→
  1424→    if (normalBalance === 'DEBIT') {
  1425→      const net = bal.totalDebit - bal.totalCredit
  1426→      if (net >= 0) netDebit = net
  1427→      else netCredit = Math.abs(net)
  1428→    } else {
  1429→      const net = bal.totalCredit - bal.totalDebit
  1430→      if (net >= 0) netCredit = net
  1431→      else netDebit = Math.abs(net)
  1432→    }
  1433→
  1434→    return {
  1435→      ...bal,
  1436→      netDebit,
  1437→      netCredit,
  1438→    }
  1439→  })
  1440→
  1441→  // Sort by account code
  1442→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1443→
  1444→  return results
  1445→}

  1491→export async function getGeneralLedger(accountCode: string, dateFrom?: Date, dateTo?: Date) {
  1492→  const account = await getAccountByCode(accountCode)
  1493→  if (!account) return []
  1494→
  1495→  const glDateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1496→  if (dateFrom || dateTo) {
  1497→    glDateFilter.date = {
  1498→      ...(dateFrom && { gte: dateFrom }),
  1499→      ...(dateTo && { lte: dateTo }),
  1500→    }
  1501→  }
  1502→  const lines = await db.journalLine.findMany({
  1503→    where: {
  1504→      accountId: account.id,
  1505→      journalEntry: {
  1506→        status: 'POSTED',
  1507→        ...glDateFilter,
  1508→      },
  1509→    },
  1510→    include: { journalEntry: true },
  1511→    orderBy: { journalEntry: { date: 'asc' } },
  1512→  })
  1513→
  1514→  let runningBalance = 0
  1515→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1516→
  1517→  return lines.map(line => {
  1518→    const debit = toNumber(line.debit)
  1519→    const credit = toNumber(line.credit)
  1520→    if (normalBalance === 'DEBIT') {
  1521→      runningBalance += debit - credit
  1522→    } else {
  1523→      runningBalance += credit - debit
  1524→    }
  1525→
  1526→    return {
  1527→      date: line.journalEntry.date,
  1528→      entryNo: line.journalEntry.entryNo,
  1529→      description: line.journalEntry.description,
  1530→      debit,
  1531→      credit,
  1532→      balance: runningBalance,
  1533→    }
  1534→  })
  1535→}

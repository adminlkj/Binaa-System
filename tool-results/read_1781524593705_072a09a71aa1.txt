     5→     5→// Helper: safely convert Prisma Decimal (or null) to a JavaScript number.
     6→     6→// Prisma Decimal.valueOf() returns a string, so using `+` between two
     7→     7→// Decimal values causes string concatenation instead of numeric addition.
     8→     8→// ALWAYS use this helper when performing arithmetic on Decimal fields.
     9→     9→function toNum(value: { valueOf(): string | number } | null | undefined): number {
    10→    10→  return Number(value ?? 0)
    11→    11→}

    13→    13→// Inline account balance calculation to avoid importing the heavy accounting engine
    14→    14→async function getAccountBalance(accountCode: string): Promise<number> {
    15→    15→  const account = await db.account.findUnique({ where: { code: accountCode } })
    16→    16→  if (!account) return 0
    17→    17→
    18→    18→  const result = await db.journalLine.aggregate({
    19→    19→    _sum: { debit: true, credit: true },
    20→    20→    where: {
    21→    21→      accountId: account.id,
    22→    22→      journalEntry: { status: 'POSTED' },
    23→    23→    },
    24→    24→  })
    25→    25→
    26→    26→  const totalDebit = toNum(result._sum.debit)
    27→    27→  const totalCredit = toNum(result._sum.credit)
    28→    28→  const normalBalanceMap: Record<string, 'DEBIT' | 'CREDIT'> = {
    29→    29→    ASSET: 'DEBIT', LIABILITY: 'CREDIT', EQUITY: 'CREDIT', REVENUE: 'CREDIT', EXPENSE: 'DEBIT',
    30→    30→  }
    31→    31→  const normalBalance = normalBalanceMap[account.type] || 'DEBIT'
    32→    32→
    33→    33→  const debit = totalDebit
    34→    34→  const credit = totalCredit
    35→    35→  return normalBalance === 'DEBIT' ? debit - credit : credit - debit
    36→    36→}

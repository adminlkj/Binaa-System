    34→function formatMonthYear(month: number, year: number, lang: 'ar' | 'en'): string {
    35→  if (lang === 'ar') return `${arabicMonths[month - 1]} ${year}`
    36→  return `${englishMonths[month - 1]} ${year}`
    37→}

   142→// ============ Create Rental Invoice Page ============
   143→function CreateRentalInvoicePage({
   144→  approvedTimesheets, onBack,
   145→}: {
   146→  approvedTimesheets: TimesheetOption[]; onBack: () => void
   147→}) {
   148→  const queryClient = useQueryClient()
   149→  const { lang } = useAppStore()
   150→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   151→
   152→  const [timesheetId, setTimesheetId] = useState('')
   153→  const [date, setDate] = useState('')
   154→  const [dueDate, setDueDate] = useState('')
   155→  const [notes, setNotes] = useState('')
   156→
   157→  // Selected timesheet - RULE: User selects ONLY a Timesheet
   158→  const selectedTimesheet = approvedTimesheets.find(ts => ts.id === timesheetId)
   159→
   160→  // RULE: System auto-fills from contract and timesheet
   161→  // Auto-filled from contract: client, project, equipment, contract no, sales order no, hourly rate, delivery fees, payment terms
   162→  // Auto-filled from timesheet: month, year, operating hours
   163→  // RULE: Rate is read-only (from contract)
   164→  const contract = selectedTimesheet?.contract
   165→  const rental = selectedTimesheet?.rental
   166→  const hourlyRate = rental?.hourlyRate || contract?.hourlyRate || 0
   167→  const operatingHours = selectedTimesheet?.operatingHours || 0
   168→  const deliveryFees = rental?.deliveryFees || contract?.deliveryFees || 0
   169→  const deliveryFeesTaxable = rental?.deliveryFeesTaxable ?? contract?.deliveryFeesTaxable ?? true
   170→  const clientId = rental?.clientId || contract?.clientId || ''
   171→  const clientName = selectedTimesheet?.clientName || selectedTimesheet?.clientNameAr || ''
   172→  const projectId = selectedTimesheet?.projectId || ''
   173→  const projectName = selectedTimesheet?.project?.name || ''
   174→  const equipmentName = selectedTimesheet?.equipment?.nameAr || selectedTimesheet?.equipment?.name || ''
   175→  const contractNo = contract?.contractNo || ''
   176→  const salesOrderNo = rental?.salesOrderNo || contract?.salesOrderNo || ''
   177→  const paymentTerms = rental?.paymentDuration || contract?.paymentTerms || '30 days'
   178→
   179→  // RULE: System auto-calculates: hours × rate = subtotal
   180→  const subtotal = operatingHours * hourlyRate
   181→  const vatRate = 0.15
   182→  const rentalVat = Math.round(subtotal * vatRate * 100) / 100
   183→  // Delivery fees (with/without VAT based on contract setting)
   184→  const deliveryVat = deliveryFeesTaxable && deliveryFees > 0 ? Math.round(deliveryFees * vatRate * 100) / 100 : 0
   185→  const totalVat = rentalVat + deliveryVat
   186→  const totalAmount = subtotal + deliveryFees + totalVat
   187→
   188→  const createMutation = useMutation({
   189→    mutationFn: (data: Record<string, unknown>) =>
   190→      fetch('/api/sales-invoices', {
   191→        method: 'POST',
   192→        headers: { 'Content-Type': 'application/json' },
   193→        body: JSON.stringify(data),
   194→      }).then(async r => {
   195→        if (!r.ok) {
   196→          const err = await r.json().catch(() => ({}))
   197→          throw new Error((err as { error?: string }).error || 'فشل في إنشاء الفاتورة')
   198→        }
   199→        return r.json()
   200→      }),
   201→    onSuccess: () => {
   202→      queryClient.invalidateQueries({ queryKey: ['rental-invoices'] })
   203→      queryClient.invalidateQueries({ queryKey: ['equipment-timesheets'] })
   204→      onBack()
   205→    },
   206→  })
   207→
   208→  const handleSubmit = (e: React.FormEvent) => {
   209→    e.preventDefault()
   210→    if (!timesheetId || !date || !dueDate || !clientId) return
   211→
   212→    // Accounting Integration: When rental invoice is created,
   213→    // the accounting engine should create journal entries:
   214→    // Debit: Accounts Receivable (totalAmount)
   215→    // Credit: Rental Revenue (subtotal)
   216→    // Credit: Delivery Revenue (deliveryFees) if applicable
   217→    // Credit: Output VAT (totalVat)
   218→    // Also, timesheet status should be updated to INVOICED
   219→    // This is handled via POST /api/journal-entries with sourceType: SALES_INVOICE
   220→    createMutation.mutate({
   221→      sourceType: 'TIMESHEET',
   222→      timesheetId,
   223→      date,
   224→      dueDate,
   225→      notes,
   226→    })
   227→  }
   228→
   229→  // RULE: Cannot create invoice without an approved timesheet
   230→  if (approvedTimesheets.length === 0) {
   231→    return (
   232→      <div className="space-y-6">
   233→        <div className="flex items-center gap-3">
   234→          <Button variant="outline" size="icon" onClick={onBack}>
   235→            <ArrowRight className="size-4" />
   236→          </Button>
   237→          <div className="flex-1">
   238→            <h1 className="text-2xl font-bold text-gray-900">{t('فاتورة إيجار جديدة', 'New Rental Invoice')}</h1>
   239→          </div>
   240→        </div>
   241→        <Card className="bg-amber-50 border-amber-200">
   242→          <CardContent className="p-6 text-center">
   243→            <p className="text-amber-700 font-medium">{t(labels.noApprovedTimesheets.ar, labels.noApprovedTimesheets.en)}</p>
   244→            <p className="text-sm text-amber-600 mt-1">{t('يجب اعتماد سجل ساعات عمل أولاً من وحدة ساعات العمل', 'An approved timesheet must be created first from the Timesheets module')}</p>
   245→          </CardContent>
   246→        </Card>
   247→      </div>
   248→    )
   249→  }
   250→
   251→  return (
   252→    <div className="space-y-6">
   253→      <div className="flex items-center gap-3">
   254→        <Button variant="outline" size="icon" onClick={onBack}>
   255→          <ArrowRight className="size-4" />
   256→        </Button>
   257→        <div className="flex-1">
   258→          <h1 className="text-2xl font-bold text-gray-900">{t('فاتورة إيجار جديدة', 'New Rental Invoice')}</h1>
   259→          <p className="text-sm text-muted-foreground">{t('إنشاء فاتورة إيجار من سجل ساعات عمل معتمد', 'Create rental invoice from an approved timesheet')}</p>
   260→        </div>
   261→      </div>
   262→
   263→      <form onSubmit={handleSubmit} className="space-y-6">
   264→        {/* RULE: User selects ONLY a Timesheet */}
   265→        <Card>
   266→          <CardHeader className="pb-4">
   267→            <CardTitle className="text-lg">{t('اختيار سجل الساعات', 'Select Timesheet')}</CardTitle>
   268→          </CardHeader>
   269→          <CardContent className="space-y-4">
   270→            <div className="space-y-2">
   271→              <Label>{t(labels.selectTimesheet.ar, labels.selectTimesheet.en)} *</Label>
   272→              <Select value={timesheetId} onValueChange={setTimesheetId}>
   273→                <SelectTrigger className="w-full"><SelectValue placeholder={t(labels.selectTimesheet.ar, labels.selectTimesheet.en)} /></SelectTrigger>
   274→                <SelectContent>
   275→                  {approvedTimesheets.map(ts => (
   276→                    <SelectItem key={ts.id} value={ts.id}>
   277→                      {ts.equipment?.name || '—'} - {formatMonthYear(ts.month, ts.year, lang)} ({formatNumber(ts.operatingHours)} {t('ساعة', 'hrs')}) [{ts.contract?.contractNo}]
   278→                    </SelectItem>
   279→                  ))}
   280→                </SelectContent>
   281→              </Select>
   282→            </div>
   283→
   284→            {/* Auto-filled info from timesheet & contract */}
   285→            {selectedTimesheet && (
   286→              <div className="p-4 rounded-lg border bg-blue-50 space-y-3">
   287→                <div className="flex items-center gap-2 text-sm font-medium text-blue-700 mb-2">
   288→                  <FileText className="size-4" />
   289→                  {t(labels.autoFilledFromContract.ar, labels.autoFilledFromContract.en)} / {t(labels.autoFilledFromTimesheet.ar, labels.autoFilledFromTimesheet.en)}
   290→                </div>
   291→                <div className="grid grid-cols-2 sm:grid-cols-4 gap-3 text-sm">
   292→                  <div><span className="text-muted-foreground">{t(labels.client.ar, labels.client.en)}:</span><p className="font-medium">{clientName}</p></div>
   293→                  <div><span className="text-muted-foreground">{t(labels.project.ar, labels.project.en)}:</span><p className="font-medium">{projectName}</p></div>
   294→                  <div><span className="text-muted-foreground">{t(labels.equipment.ar, labels.equipment.en)}:</span><p className="font-medium">{equipmentName}</p></div>
   295→                  <div><span className="text-muted-foreground">{t(labels.deliveryMonth.ar, labels.deliveryMonth.en)}:</span><p className="font-medium">{formatMonthYear(selectedTimesheet.month, selectedTimesheet.year, lang)}</p></div>
   296→                  <div><span className="text-muted-foreground">{t(labels.contractNo.ar, labels.contractNo.en)}:</span><p className="font-medium font-mono">{contractNo}</p></div>
   297→                  <div><span className="text-muted-foreground">{t(labels.salesOrderNo.ar, labels.salesOrderNo.en)}:</span><p className="font-medium font-mono">{salesOrderNo || '—'}</p></div>
   298→                  <div><span className="text-muted-foreground">{t(labels.paymentTerms.ar, labels.paymentTerms.en)}:</span><p className="font-medium">{paymentTerms}</p></div>
   299→                </div>
   300→              </div>
   301→            )}
   302→          </CardContent>
   303→        </Card>
   304→
   305→        {/* Operating Hours & Rate - RULE: No manual editing of rate or hours */}
   306→        <Card>
   307→          <CardHeader className="pb-4">
   308→            <CardTitle className="text-lg">{t('ساعات التشغيل والسعر', 'Operating Hours & Rate')}</CardTitle>
   309→          </CardHeader>
   310→          <CardContent>
   311→            <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   312→              <div className="space-y-2">
   313→                <Label>{t(labels.operatingHours.ar, labels.operatingHours.en)}</Label>
   314→                {/* Auto-filled from timesheet - read only */}
   315→                <Input type="number" value={operatingHours || ''} readOnly className="bg-gray-100" dir="ltr" />
   316→                <p className="text-xs text-muted-foreground">{t('من سجل الساعات - للقراءة فقط', 'From timesheet - read only')}</p>
   317→              </div>
   318→              <div className="space-y-2">
   319→                <Label>{t(labels.hourlyRate.ar, labels.hourlyRate.en)}</Label>

   322→                <p className="text-xs text-muted-foreground">{t(labels.readOnlyFromContract.ar, labels.readOnlyFromContract.en)}</p>
   323→              </div>
   324→              <div className="space-y-2">
   325→                <Label>{t(labels.subtotal.ar, labels.subtotal.en)} ({t('ساعات × سعر', 'hrs × rate')})</Label>
   326→                <div className="h-9 flex items-center">
   327→                  <MoneyDisplay value={subtotal} lang={lang} size="md" bold />
   328→                </div>
   329→              </div>
   330→            </div>
   331→          </CardContent>
   332→        </Card>
   333→
   334→        {/* Delivery Fees (from contract) */}
   335→        {deliveryFees > 0 && (
   336→          <Card>
   337→            <CardHeader className="pb-4">
   338→              <CardTitle className="text-lg">{t(labels.deliveryFees.ar, labels.deliveryFees.en)}</CardTitle>
   339→            </CardHeader>
   340→            <CardContent>
   341→              <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   342→                <div className="space-y-2">
   343→                  <Label>{t(labels.deliveryFees.ar, labels.deliveryFees.en)}</Label>
   344→                  <Input type="number" value={deliveryFees} readOnly className="bg-gray-100" dir="ltr" />
   345→                </div>
   346→                <div className="space-y-2 flex items-center gap-2 pt-6">
   347→                  <Badge variant="outline" className={deliveryFeesTaxable ? 'bg-emerald-100 text-emerald-700' : 'bg-gray-100 text-gray-700'}>
   348→                    {deliveryFeesTaxable ? t('خاضعة للضريبة', 'Taxable') : t('غير خاضعة للضريبة', 'Not Taxable')}
   349→                  </Badge>
   350→                </div>
   351→              </div>
   352→            </CardContent>
   353→          </Card>
   354→        )}
   355→
   356→        {/* Dates */}
   357→        <Card>
   358→          <CardHeader className="pb-4">
   359→            <CardTitle className="text-lg">{t('التواريخ', 'Dates')}</CardTitle>
   360→          </CardHeader>
   361→          <CardContent>
   362→            <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   363→              <div className="space-y-2">
   364→                <Label>{t(labels.date.ar, labels.date.en)} *</Label>

   366→              </div>
   367→              <div className="space-y-2">

   369→                <Input type="date" value={dueDate} onChange={e => setDueDate(e.target.value)} required />
   370→              </div>
   371→              <div className="space-y-2">
   372→                <Label>{t(labels.notes.ar, labels.notes.en)}</Label>
   373→                <Input value={notes} onChange={e => setNotes(e.target.value)} placeholder={t('ملاحظات اختيارية', 'Optional notes')} />

   378→
   379→        {/* Invoice Summary */}
   380→        {selectedTimesheet && subtotal > 0 && (
   381→          <Card className="bg-gray-50 border-dashed">
   382→            <CardContent className="p-6 space-y-3">
   383→              <h3 className="text-lg font-semibold mb-2">{t(labels.invoiceSummary.ar, labels.invoiceSummary.en)}</h3>

   387→              </div>

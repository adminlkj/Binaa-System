    63→// ============ ImageUploadField Component ============
    64→function ImageUploadField({ 
    65→  value, 
    66→  onChange, 
    67→  label, 
    68→  labelAr,
    69→  lang,
    70→  hint,
    71→  hintAr,
    72→  previewHeight = 'h-24',
    73→  accept = 'image/svg+xml,image/png,image/jpeg,image/webp',
    74→}: {
    75→  value: string | null
    76→  onChange: (url: string | null) => void
    77→  label: string
    78→  labelAr: string
    79→  lang: 'ar' | 'en'
    80→  hint?: string
    81→  hintAr?: string
    82→  previewHeight?: string
    83→  accept?: string
    84→}) {
    85→  const [uploading, setUploading] = useState(false)
    86→  const [dragOver, setDragOver] = useState(false)
    87→  const fileInputRef = useRef<HTMLInputElement>(null)
    88→
    89→  const [error, setError] = useState<string | null>(null)
    90→
    91→  const uploadFile = useCallback(async (file: File) => {
    92→    if (file.size > 5 * 1024 * 1024) {
    93→      setError(lang === 'ar' ? 'حجم الملف يتجاوز 5 ميجابايت' : 'File size exceeds 5MB')
    94→      return
    95→    }
    96→    setUploading(true)
    97→    setError(null)
    98→    try {
    99→      const formData = new FormData()
   100→      formData.append('file', file)
   101→      const res = await fetch('/api/upload', { method: 'POST', body: formData })
   102→      if (!res.ok) {
   103→        const errData = await res.json().catch(() => ({}))
   104→        throw new Error(errData.error || 'Upload failed')
   105→      }
   106→      const data = await res.json()
   107→      onChange(data.url)
   108→    } catch (err) {
   109→      const msg = err instanceof Error ? err.message : 'Unknown error'
   110→      setError(lang === 'ar' ? `فشل في رفع الملف: ${msg}` : `Failed to upload file: ${msg}`)
   111→    } finally {
   112→      setUploading(false)
   113→    }
   114→  }, [lang, onChange])
   115→
   116→  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
   117→    const file = e.target.files?.[0]
   118→    if (file) {
   119→      setError(null)
   120→      uploadFile(file)
   121→    }
   122→    // Reset input so same file can be re-selected
   123→    setTimeout(() => { if (fileInputRef.current) fileInputRef.current.value = '' }, 100)
   124→  }
   125→
   126→  const handleDrop = (e: React.DragEvent) => {
   127→    e.preventDefault()
   128→    setDragOver(false)
   129→    const file = e.dataTransfer.files?.[0]
   130→    if (file) uploadFile(file)
   131→  }
   132→
   133→  const handleDragOver = (e: React.DragEvent) => {
   134→    e.preventDefault()
   135→    setDragOver(true)
   136→  }
   137→
   138→  const handleDragLeave = () => setDragOver(false)
   139→
   140→  const displayLabel = lang === 'ar' ? labelAr : label
   141→  const displayHint = lang === 'ar' ? hintAr : hint
   142→
   143→  return (
   144→    <div className="space-y-2">
   145→      <Label>{displayLabel}</Label>
   146→      {value ? (
   147→        <div className={`relative rounded-lg border bg-white p-2 ${previewHeight} flex items-center justify-center group overflow-hidden`}>
   148→          <img
   149→            src={value}
   150→            alt={displayLabel}
   151→            className="max-h-20 max-w-full object-contain"
   152→          />
   153→          <Button
   154→            variant="destructive"
   155→            size="icon"
   156→            className="absolute top-1 left-1 size-6 opacity-0 group-hover:opacity-100 transition-opacity z-10"
   157→            onClick={() => { setError(null); onChange(null) }}
   158→            type="button"
   159→          >
   160→            <X className="size-3" />
   161→          </Button>
   162→        </div>
   163→      ) : (
   164→        <div
   165→          className={`rounded-lg border-2 border-dashed transition-colors cursor-pointer ${
   166→            dragOver ? 'border-emerald-400 bg-emerald-50' : 'border-gray-300 bg-gray-50 hover:border-emerald-300 hover:bg-emerald-50/50'
   167→          } ${previewHeight} flex flex-col items-center justify-center gap-1`}
   168→          onClick={() => fileInputRef.current?.click()}
   169→          onDrop={handleDrop}
   170→          onDragOver={handleDragOver}
   171→          onDragLeave={handleDragLeave}
   172→        >
   173→          {uploading ? (
   174→            <>
   175→              <Loader2 className="size-5 text-emerald-500 animate-spin" />
   176→              <span className="text-xs text-muted-foreground">
   177→                {lang === 'ar' ? 'جاري الرفع...' : 'Uploading...'}
   178→              </span>
   179→            </>
   180→          ) : (
   181→            <>
   182→              <Upload className="size-5 text-gray-400" />
   183→              <span className="text-xs text-muted-foreground">
   184→                {lang === 'ar' ? 'اسحب الملف هنا أو انقر للاختيار' : 'Drag & drop or click to upload'}
   185→              </span>
   186→              <span className="text-[10px] text-muted-foreground">
   187→                {lang === 'ar' ? 'SVG, PNG, JPG - حد أقصى 5MB' : 'SVG, PNG, JPG - Max 5MB'}
   188→              </span>
   189→            </>
   190→          )}
   191→        </div>
   192→      )}
   193→      <input
   194→        ref={fileInputRef}
   195→        type="file"
   196→        accept={accept}
   197→        onChange={handleFileChange}
   198→        className="hidden"
   199→      />
   200→      {error && (
   201→        <p className="text-xs text-red-600">{error}</p>
   202→      )}
   203→      {displayHint && !error && (
   204→        <p className="text-xs text-muted-foreground">{displayHint}</p>
   205→      )}
   206→    </div>
   207→  )
   208→}

   264→   264→export async function ensureAccountExists(template: AccountTemplate, tx?: PrismaTransaction) {

   266→   266→  const existing = await client.account.findUnique({ where: { code: template.code } })

   268→   268→    // Update existing account with new fields if they are missing

   271→   271→      existing.isSystem !== (template.isSystem || false) ||
   272→   272→      existing.allowPosting !== (template.allowPosting || false) ||
   273→   273→      existing.level !== (template.level || 0) ||
   274→   274→      (template.accountRole && existing.accountRole !== template.accountRole) ||
   275→   275→      (template.parentId && existing.parentCode !== template.parentId)
   276→   276→    ) {
   277→   277→      await client.account.update({
   278→   278→        where: { code: template.code },
   279→   279→        data: {
   280→   280→          name: template.name,

   282→   282→          type: template.type,

   285→   285→          allowPosting: template.allowPosting || false,
   286→   286→          level: template.level || 0,
   287→   287→          accountRole: template.accountRole || existing.accountRole,
   288→   288→          parentCode: template.parentId || existing.parentCode,
   289→   289→        },
   290→   290→      })
   291→   291→    }
   292→   292→    return client.account.findUnique({ where: { code: template.code } })
   293→   293→  }
   294→   294→
   295→   295→  let parentId: string | undefined
   296→   296→  if (template.parentId) {

   298→   298→    if (parent) parentId = parent.id
   299→   299→  }
   300→   300→
   301→   301→  return client.account.create({
   302→   302→    data: {
   303→   303→      code: template.code,

   316→   316→  })

   321→   321→export async function initializeChartOfAccounts() {
   322→   322→  let created = 0
   323→   323→  let updated = 0
   324→   324→
   325→   325→  // Create parent accounts first, then children (sorted by code length then code)
   326→   326→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   327→   327→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   328→   328→    return a.code.localeCompare(b.code)
   329→   329→  })
   330→   330→
   331→   331→  for (const tmpl of sorted) {
   332→   332→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   333→   333→    if (existing) {
   334→   334→      // Update existing account with new fields
   335→   335→      await db.account.update({
   336→   336→        where: { code: tmpl.code },
   337→   337→        data: {
   338→   338→          name: tmpl.name,
   339→   339→          nameAr: tmpl.nameAr,
   340→   340→          type: tmpl.type,
   341→   341→          activityType: tmpl.activityType || null,
   342→   342→          accountRole: tmpl.accountRole || null,
   343→   343→          parentCode: tmpl.parentId || null,
   344→   344→          isSystem: tmpl.isSystem || false,
   345→   345→          allowPosting: tmpl.allowPosting || false,
   346→   346→          level: tmpl.level || 0,
   347→   347→          isActive: true,
   348→   348→        },
   349→   349→      })
   350→   350→      // Update parent reference if needed
   351→   351→      if (tmpl.parentId) {
   352→   352→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   353→   353→        if (parent && existing.parentId !== parent.id) {
   354→   354→          await db.account.update({
   355→   355→            where: { code: tmpl.code },
   356→   356→            data: { parentId: parent.id },
   357→   357→          })
   358→   358→        }
   359→   359→      }
   360→   360→      updated++
   361→   361→    } else {
   362→   362→      await ensureAccountExists(tmpl)
   363→   363→      created++
   364→   364→    }
   365→   365→  }
   366→   366→
   367→   367→  const total = await db.account.count()
   368→   368→  return { created, updated, total }
   369→   369→}

   371→   371→// ============ JOURNAL ENTRY REVERSAL ============
   372→   372→
   373→   373→/**
   374→   374→ * عكس قيد محاسبي - Reverse a posted journal entry
   375→   375→ *
   376→   376→ * Creates a reversal entry with flipped debit/credit on all lines,
   377→   377→ * marks the original entry as CANCELLED, and links them together.
   378→   378→ *
   379→   379→ * IMPORTANT: This function MUST be called inside a $transaction callback.
   380→   380→ * The `tx` parameter is required — no standalone calls allowed.
   381→   381→ *
   382→   382→ * @param journalEntryId - ID of the journal entry to reverse
   383→   383→ * @param tx - Prisma transaction client (required)
   384→   384→ * @returns The reversal journal entry with its lines
   385→   385→ * @throws Error if entry not found, not POSTED, or already reversed
   386→   386→ */
   387→   387→export async function reverseEntry(journalEntryId: string, tx: PrismaTransaction) {
   388→   388→  // 1. Find the original entry with its lines
   389→   389→  const originalEntry = await tx.journalEntry.findUnique({
   390→   390→    where: { id: journalEntryId },
   391→   391→    include: { lines: true },
   392→   392→  })
   393→   393→
   394→   394→  if (!originalEntry) {
   395→   395→    throw new Error(`القيد المحاسبي غير موجود: ${journalEntryId}`)
   396→   396→  }
   397→   397→
   398→   398→  // 2. Check that the entry is POSTED (not DRAFT or CANCELLED)
   399→   399→  if (originalEntry.status !== 'POSTED') {
   400→   400→    throw new Error(
   401→   401→      `لا يمكن عكس قيد بحالة ${originalEntry.status} - يجب أن يكون القيد مرحّلاً (POSTED)`
   402→   402→    )
   403→   403→  }
   404→   404→
   405→   405→  // 3. Check that it hasn't already been reversed
   406→   406→  if (originalEntry.isReversal) {
   407→   407→    throw new Error('لا يمكن عكس قيد عكسي - القيود العكسية لا يمكن عكسها')
   408→   408→  }
   409→   409→
   410→   410→  const existingReversal = await tx.journalEntry.findFirst({
   411→   411→    where: { reversedEntryId: journalEntryId },
   412→   412→  })
   413→   413→  if (existingReversal) {
   414→   414→    throw new Error('القيد محسوب عكسه مسبقاً - يوجد قيد عكسي مرتبط بهذا القيد')
   415→   415→  }
   416→   416→
   417→   417→  // 4. Generate next entry number
   418→   418→  const last = await tx.journalEntry.findFirst({ orderBy: { entryNo: 'desc' } })
   419→   419→  const next = last ? parseInt(last.entryNo.replace('JE-', '')) + 1 : 1
   420→   420→  const entryNo = `JE-${next.toString().padStart(6, '0')}`
   421→   421→
   422→   422→  // 5. Build reversal description with "عكس" prefix
   423→   423→  const reversalDescription = originalEntry.description
   424→   424→    ? `عكس - ${originalEntry.description}`
   425→   425→    : 'عكس قيد محاسبي'
   426→   426→
   427→   427→  // 6. Create the reversal entry with flipped debit/credit lines
   428→   428→  const reversalEntry = await tx.journalEntry.create({
   429→   429→    data: {
   430→   430→      entryNo,
   431→   431→      date: originalEntry.date,
   432→   432→      description: reversalDescription,
   433→   433→      status: 'POSTED',
   434→   434→      sourceType: originalEntry.sourceType,
   435→   435→      sourceId: originalEntry.sourceId,
   436→   436→      isReversal: true,
   437→   437→      reversedEntryId: originalEntry.id,
   438→   438→      isSystem: true,
   439→   439→      lines: {
   440→   440→        create: originalEntry.lines.map((line) => ({
   441→   441→          accountId: line.accountId,
   442→   442→          costCenterId: line.costCenterId,
   443→   443→          // Flip debit and credit
   444→   444→          debit: line.credit,
   445→   445→          credit: line.debit,
   446→   446→          description: line.description
   447→   447→            ? `عكس - ${line.description}`
   448→   448→            : 'عكس بند قيد',
   449→   449→        })),
   450→   450→      },
   451→   451→    },
   452→   452→    include: { lines: true },
   453→   453→  })
   454→   454→
   455→   455→  // 7. Update the original entry status to CANCELLED
   456→   456→  await tx.journalEntry.update({
   457→   457→    where: { id: originalEntry.id },
   458→   458→    data: { status: 'CANCELLED' },
   459→   459→  })
   460→   460→
   461→   461→  return reversalEntry
   462→   462→}

   464→   464→// ============ JOURNAL ENTRY CREATION ============
   465→   465→
   466→   466→export async function createJournalEntry(template: JournalEntryTemplate, tx?: PrismaTransaction) {
   467→   467→  const client = tx || db
   468→   468→  // Validate: total debits must equal total credits
   469→   469→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   470→   470→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   471→   471→
   472→   472→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   473→   473→    throw new Error(
   474→   474→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   475→   475→    )
   476→   476→  }
   477→   477→
   478→   478→  // Ensure all referenced accounts exist
   479→   479→  for (const line of template.lines) {
   480→   480→    const account = await getAccountByCode(line.accountCode, tx)
   481→   481→    if (!account) {
   482→   482→      throw new Error(`Account not found: ${line.accountCode}`)
   483→   483→    }
   484→   484→  }
   485→   485→
   486→   486→  // Create the journal entry with lines
   487→   487→  const entry = await client.journalEntry.create({
   488→   488→    data: {
   489→   489→      entryNo: template.entryNo,
   490→   490→      date: template.date,
   491→   491→      description: template.description,
   492→   492→      status: 'POSTED', // Auto-entries are posted immediately
   493→   493→      lines: {
   494→   494→        create: await Promise.all(
   495→   495→          template.lines.map(async (line) => {
   496→   496→            const account = await getAccountByCode(line.accountCode, tx)
   497→   497→            return {
   498→   498→              accountId: account!.id,
   499→   499→              costCenterId: line.costCenterId,
   500→   500→              debit: line.debit,
   501→   501→              credit: line.credit,
   502→   502→              description: line.description,
   503→   503→            };
   504→   504→          })
   505→   505→        ),
   506→   506→      },
   507→   507→    },
   508→   508→    include: { lines: true },
   509→   509→  })
   510→   510→
   511→   511→  return entry
   512→   512→}

  1251→  1251→export async function autoEntryAssetDisposal(data: {

  1275→  1275→  }

  1292→  1292→ * Cr: VAT Due (3130) - net VAT payable (if output > input)
  1293→  1293→ * OR
  1294→  1294→ * Dr: Input VAT (3120) - if input > output (refund scenario)
  1295→  1295→ * Cr: Output VAT (3110)
  1296→  1296→ * Dr: VAT Refund Receivable (1410) - if refund due
  1297→  1297→ */
  1298→  1298→export async function autoEntryVATDeclaration(data: {
  1299→  1299→  period: string
  1300→  1300→  outputVat: number
  1301→  1301→  inputVat: number
  1302→  1302→  netVat: number // positive = payable, negative = refundable
  1303→  1303→  date: Date
  1304→  1304→}, tx?: PrismaTransaction) {
  1305→  1305→  const lines: { accountCode: string; debit: number; credit: number }[] = []
  1306→  1306→
  1307→  1307→  // Close Output VAT (3110) - debit to zero it out
  1308→  1308→  lines.push({ accountCode: '3110', debit: data.outputVat, credit: 0 })
  1309→  1309→
  1310→  1310→  // Close Input VAT (3120) - credit to zero it out
  1311→  1311→  lines.push({ accountCode: '3120', debit: 0, credit: data.inputVat })
  1312→  1312→
  1313→  1313→  // Net VAT position
  1314→  1314→  if (data.netVat > 0) {
  1315→  1315→    // More output than input → owe VAT
  1316→  1316→    lines.push({ accountCode: '3130', debit: 0, credit: data.netVat }) // VAT Due
  1317→  1317→  } else if (data.netVat < 0) {
  1318→  1318→    // More input than output → refund due
  1319→  1319→    lines.push({ accountCode: '1410', debit: Math.abs(data.netVat), credit: 0 }) // VAT Refund Receivable
  1320→  1320→  }
  1321→  1321→
  1322→  1322→  return createJournalEntry({
  1323→  1323→    entryNo: `JE-VAT-${Date.now()}`,
  1324→  1324→    date: data.date,
  1325→  1325→    description: `VAT Declaration - ${data.period}`,
  1326→  1326→    descriptionAr: `إقرار ضريبي - ${data.period}`,
  1327→  1327→    lines,
  1328→  1328→    sourceType: 'VAT_DECLARATION',
  1329→  1329→    sourceId: `VAT-${data.period}`,
  1330→  1330→  }, tx)
  1331→  1331→}

  1360→  1360→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1361→  1361→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1362→  1362→  if (dateFrom || dateTo) {
  1363→  1363→    dateFilter.date = {
  1364→  1364→      ...(dateFrom && { gte: dateFrom }),
  1365→  1365→      ...(dateTo && { lte: dateTo }),
  1366→  1366→    }
  1367→  1367→  }
  1368→  1368→  const entries = await db.journalEntry.findMany({
  1369→  1369→    where: {
  1370→  1370→      status: 'POSTED',
  1371→  1371→      ...dateFilter,
  1372→  1372→    },
  1373→  1373→    include: {
  1374→  1374→      lines: {
  1375→  1375→        include: {
  1376→  1376→          account: true,
  1377→  1377→        },
  1378→  1378→      },
  1379→  1379→    },
  1380→  1380→  })
  1381→  1381→
  1382→  1382→  // Aggregate by account
  1383→  1383→  const accountBalances = new Map<string, {
  1384→  1384→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1385→  1385→    totalDebit: number
  1386→  1386→    totalCredit: number
  1387→  1387→  }>()
  1388→  1388→
  1389→  1389→  for (const entry of entries) {
  1390→  1390→    for (const line of entry.lines) {
  1391→  1391→      const key = line.accountId
  1392→  1392→      if (!accountBalances.has(key)) {
  1393→  1393→        accountBalances.set(key, {
  1394→  1394→          account: line.account,
  1395→  1395→          totalDebit: 0,
  1396→  1396→          totalCredit: 0,
  1397→  1397→        })
  1398→  1398→      }
  1399→  1399→      const bal = accountBalances.get(key)!
  1400→  1400→      bal.totalDebit += toNumber(line.debit)
  1401→  1401→      bal.totalCredit += toNumber(line.credit)
  1402→  1402→    }
  1403→  1403→  }
  1404→  1404→
  1405→  1405→  // Calculate net balance for each account
  1406→  1406→  const results = Array.from(accountBalances.values()).map(bal => {
  1407→  1407→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1408→  1408→    let netDebit = 0
  1409→  1409→    let netCredit = 0
  1410→  1410→
  1411→  1411→    if (normalBalance === 'DEBIT') {
  1412→  1412→      const net = bal.totalDebit - bal.totalCredit
  1413→  1413→      if (net >= 0) netDebit = net
  1414→  1414→      else netCredit = Math.abs(net)
  1415→  1415→    } else {
  1416→  1416→      const net = bal.totalCredit - bal.totalDebit
  1417→  1417→      if (net >= 0) netCredit = net
  1418→  1418→      else netDebit = Math.abs(net)
  1419→  1419→    }
  1420→  1420→
  1421→  1421→    return {
  1422→  1422→      ...bal,
  1423→  1423→      netDebit,
  1424→  1424→      netCredit,
  1425→  1425→    }
  1426→  1426→  })
  1427→  1427→
  1428→  1428→  // Sort by account code
  1429→  1429→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1430→  1430→
  1431→  1431→  return results
  1432→  1432→}

  1508→  1508→      runningBalance += debit - credit
  1509→  1509→    } else {
  1510→  1510→      runningBalance += credit - debit
  1511→  1511→    }
  1512→  1512→
  1513→  1513→    return {
  1514→  1514→      date: line.journalEntry.date,
  1515→  1515→      entryNo: line.journalEntry.entryNo,
  1516→  1516→      description: line.journalEntry.description,
  1517→  1517→      debit,

  1520→  1520→    }
  1521→  1521→  })
  1522→  1522→}

   595→   595→// ============ Main Function: generateDocument ============
   596→   596→
   597→   597→/**
   598→   598→ * Generate a complete print-ready HTML document.
   599→   599→ *
   600→   600→ * This is the single entry point for all document printing in the system.
   601→   601→ * It routes to the appropriate template based on document type, applies
   602→   602→ * company settings, and includes ZATCA QR codes for invoices.
   603→   603→ *
   604→   604→ * @param documentType - The type of document to generate
   605→   605→ * @param data - The document data (from API/database)
   606→   606→ * @param companySettings - Company settings from CompanySetting model
   607→   607→ * @param lang - Language for the document ('ar' or 'en')
   608→   608→ * @returns Complete HTML document string ready for printing
   609→   609→ */
   610→   610→export function generateDocument(
   611→   611→  documentType: UnifiedDocumentType,
   612→   612→  data: Record<string, unknown>,
   613→   613→  companySettings: PrintSettings,
   614→   614→  lang: 'ar' | 'en' = 'ar',
   615→   615→): string {
   616→   616→  // Resolve type aliases
   617→   617→  const resolvedType = (typeAliases[documentType] || documentType) as PrintDocumentType
   618→   618→
   619→   619→  // Check if this is a custom template (not in modular system)
   620→   620→  const customTemplate = customTemplates[documentType]
   621→   621→
   622→   622→  if (customTemplate) {
   623→   623→    // Generate using custom template
   624→   624→    return generateCustomDocument(documentType, customTemplate, data, companySettings, lang)
   625→   625→  }
   626→   626→
   627→   627→  // For invoice types, automatically add ZATCA QR data if not already present
   628→   628→  if (INVOICE_TYPES_REQUIRING_QR.includes(documentType) && companySettings.taxNumber && !data.qrDataUrl) {
   629→   629→    const sellerName = lang === 'ar' ? companySettings.nameAr : companySettings.nameEn
   630→   630→    const vatNumber = companySettings.taxNumber
   631→   631→    const invoiceDate = data.date
   632→   632→      ? new Date(data.date as string).toISOString().split('T')[0]
   633→   633→      : ''
   634→   634→    const totalAmount = Number(data.totalAmount) || 0
   635→   635→    const vatAmount = Number(data.vatAmount) || 0
   636→   636→
   637→   637→    try {
   638→   638→      const tlvBase64 = generateZatcaTLV({
   639→   639→        sellerName,
   640→   640→        vatNumber,
   641→   641→        invoiceDate,
   642→   642→        totalAmount: totalAmount.toFixed(2),
   643→   643→        vatAmount: vatAmount.toFixed(2),
   644→   644→      })
   645→   645→      // Store the TLV base64 so the template can generate QR from it
   646→   646→      data._zatcaTlvBase64 = tlvBase64
   647→   647→    } catch {
   648→   648→      // ZATCA QR generation failed, continue without it
   649→   649→    }
   650→   650→  }
   651→   651→
   652→   652→  // Use the modular printing system
   653→   653→  return generatePrintHTML({
   654→   654→    type: resolvedType,
   655→   655→    data,
   656→   656→    settings: companySettings,
   657→   657→    lang,
   658→   658→  })
   659→   659→}

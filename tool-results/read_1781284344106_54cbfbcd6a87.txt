   264→export async function resolveCostCenterId(projectId: string | undefined | null): Promise<string | undefined> {
   265→  if (!projectId) return undefined

   268→  const costCenter = await db.costCenter.findFirst({ where: { code: project.code } })
   269→  return costCenter?.id

   272→export async function ensureAccountExists(template: AccountTemplate) {
   273→  const existing = await db.account.findUnique({ where: { code: template.code } })
   274→  if (existing) {
   275→    // Update existing account with new fields if they are missing
   276→    if (
   277→      existing.activityType !== (template.activityType || null) ||
   278→      existing.isSystem !== (template.isSystem || false) ||
   279→      existing.allowPosting !== (template.allowPosting || false) ||
   280→      existing.level !== (template.level || 0)
   281→    ) {
   282→      await db.account.update({
   283→        where: { code: template.code },
   284→        data: {
   285→          name: template.name,
   286→          nameAr: template.nameAr,
   287→          type: template.type,
   288→          activityType: template.activityType || null,
   289→          isSystem: template.isSystem || false,
   290→          allowPosting: template.allowPosting || false,
   291→          level: template.level || 0,
   292→        },
   293→      })
   294→    }
   295→    return db.account.findUnique({ where: { code: template.code } })
   296→  }
   297→
   298→  let parentId: string | undefined
   299→  if (template.parentId) {
   300→    const parent = await db.account.findUnique({ where: { code: template.parentId } })
   301→    if (parent) parentId = parent.id
   302→  }
   303→
   304→  return db.account.create({
   305→    data: {
   306→      code: template.code,
   307→      name: template.name,
   308→      nameAr: template.nameAr,
   309→      type: template.type,
   310→      parentId,
   311→      isActive: true,
   312→      activityType: template.activityType || null,
   313→      isSystem: template.isSystem || false,
   314→      allowPosting: template.allowPosting || false,
   315→      level: template.level || 0,
   316→    },
   317→  })
   318→}

   322→export async function initializeChartOfAccounts() {
   323→  let created = 0
   324→  let updated = 0
   325→
   326→  // Create parent accounts first, then children (sorted by code length then code)
   327→  const sorted = [...CHART_OF_ACCOUNTS_TEMPLATE].sort((a, b) => {
   328→    if (a.code.length !== b.code.length) return a.code.length - b.code.length
   329→    return a.code.localeCompare(b.code)
   330→  })
   331→
   332→  for (const tmpl of sorted) {
   333→    const existing = await db.account.findUnique({ where: { code: tmpl.code } })
   334→    if (existing) {
   335→      // Update existing account with new fields
   336→      await db.account.update({
   337→        where: { code: tmpl.code },
   338→        data: {
   339→          name: tmpl.name,
   340→          nameAr: tmpl.nameAr,
   341→          type: tmpl.type,
   342→          activityType: tmpl.activityType || null,
   343→          isSystem: tmpl.isSystem || false,
   344→          allowPosting: tmpl.allowPosting || false,
   345→          level: tmpl.level || 0,
   346→          isActive: true,
   347→        },
   348→      })
   349→      // Update parent reference if needed
   350→      if (tmpl.parentId) {
   351→        const parent = await db.account.findUnique({ where: { code: tmpl.parentId } })
   352→        if (parent && existing.parentId !== parent.id) {
   353→          await db.account.update({
   354→            where: { code: tmpl.code },
   355→            data: { parentId: parent.id },
   356→          })
   357→        }
   358→      }
   359→      updated++
   360→    } else {
   361→      await ensureAccountExists(tmpl)
   362→      created++
   363→    }
   364→  }
   365→
   366→  const total = await db.account.count()
   367→  return { created, updated, total }
   368→}

   372→export async function createJournalEntry(template: JournalEntryTemplate) {
   373→  // Validate: total debits must equal total credits
   374→  const totalDebit = template.lines.reduce((sum, l) => sum + l.debit, 0)
   375→  const totalCredit = template.lines.reduce((sum, l) => sum + l.credit, 0)
   376→
   377→  if (Math.abs(totalDebit - totalCredit) > 0.01) {
   378→    throw new Error(
   379→      `Journal entry not balanced: Debits=${totalDebit}, Credits=${totalCredit}, Diff=${Math.abs(totalDebit - totalCredit)}`
   380→    )
   381→  }
   382→
   383→  // Ensure all referenced accounts exist
   384→  for (const line of template.lines) {
   385→    const account = await getAccountByCode(line.accountCode)
   386→    if (!account) {
   387→      throw new Error(`Account not found: ${line.accountCode}`)
   388→    }
   389→  }
   390→
   391→  // Create the journal entry with lines
   392→  const entry = await db.journalEntry.create({
   393→    data: {
   394→      entryNo: template.entryNo,
   395→      date: template.date,
   396→      description: template.description,
   397→      status: 'POSTED', // Auto-entries are posted immediately
   398→      lines: {
   399→        create: await Promise.all(
   400→          template.lines.map(async (line) => {
   401→            const account = await getAccountByCode(line.accountCode)
   402→            return {
   403→              accountId: account!.id,
   404→              costCenterId: line.costCenterId,
   405→              debit: line.debit,
   406→              credit: line.credit,
   407→              description: line.description,
   408→            }
   409→          })
   410→        ),
   411→      },
   412→    },
   413→    include: { lines: true },
   414→  })
   415→
   416→  return entry
   417→}

   422→/**
   423→ * فاتورة مبيعات - Sales Invoice (from Extract)
   424→ * Dr: Clients Receivable (1210) - totalAmount
   425→ * Cr: Progress Claims Revenue (6110) - subtotal
   426→ * Cr: VAT Payable (3200) - vatAmount
   427→ */
   428→export async function autoEntrySalesInvoice(data: {
   429→  invoiceNo: string
   430→  clientId: string
   431→  subtotal: number
   432→  vatRate: number
   433→  vatAmount: number
   434→  totalAmount: number
   435→  invoiceType: string // TAX_INVOICE, PROGRESS_CLAIM, RENTAL
   436→  date: Date
   437→  projectId?: string
   438→  costCenterId?: string
   439→}) {
   440→  // Determine revenue account based on invoice type
   441→  let revenueAccountCode: string
   442→  switch (data.invoiceType) {
   443→    case 'RENTAL':
   444→      revenueAccountCode = '6210' // Equipment Rental Revenue
   445→      break
   446→    case 'PROGRESS_CLAIM':
   447→      revenueAccountCode = '6110' // Progress Claims Revenue
   448→      break
   449→    default:
   450→      revenueAccountCode = '6340' // Service Revenue
   451→  }
   452→
   453→  const lines = [
   454→    { accountCode: '1210', debit: data.totalAmount, credit: 0, costCenterId: data.costCenterId }, // Clients Receivable
   455→    { accountCode: revenueAccountCode, debit: 0, credit: data.subtotal, costCenterId: data.costCenterId }, // Revenue
   456→  ]
   457→
   458→  // Add VAT line only if VAT > 0 (Output VAT)
   459→  if (data.vatAmount > 0) {
   460→    lines.push({ accountCode: '3110', debit: 0, credit: data.vatAmount }) // Output VAT (ضريبة مخرجات)
   461→  }
   462→
   463→  return createJournalEntry({
   464→    entryNo: `JE-SI-${Date.now()}`,
   465→    date: data.date,
   466→    description: `Sales Invoice ${data.invoiceNo}`,
   467→    descriptionAr: `فاتورة مبيعات ${data.invoiceNo}`,
   468→    lines,
   469→    sourceType: 'SALES_INVOICE',
   470→    sourceId: data.invoiceNo,
   471→  })
   472→}

  1161→ * Dr: Cash/Bank (1110/1120) - sale price
  1162→ * Cr: Asset account (2110/2120/2130/2140) - original cost (or net book value)
  1163→ * Dr/Cr: Gain/Loss on disposal (6310 gain / 8610 loss)
  1164→ */
  1165→export async function autoEntryAssetDisposal(data: {
  1166→  assetAccountCode: string // e.g., '2110', '2120', '2130', '2140'
  1167→  accumulatedDepAccountCode: string // e.g., '2210', '2220', '2230', '2240'
  1168→  originalCost: number
  1169→  accumulatedDepreciation: number
  1170→  salePrice: number
  1171→  date: Date
  1172→  receivedIn: 'TREASURY' | 'BANK'
  1173→}) {
  1174→  const cashAccountCode = data.receivedIn === 'BANK' ? '1120' : '1110'
  1175→  const netBookValue = data.originalCost - data.accumulatedDepreciation
  1176→  const gainLoss = data.salePrice - netBookValue
  1177→
  1178→  const lines = [
  1179→    { accountCode: cashAccountCode, debit: data.salePrice, credit: 0 }, // Cash received
  1180→    { accountCode: data.accumulatedDepAccountCode, debit: data.accumulatedDepreciation, credit: 0 }, // Remove accumulated depreciation

  1185→  if (gainLoss > 0) {

  1187→  } else if (gainLoss < 0) {
  1188→    lines.push({ accountCode: '8610', debit: Math.abs(gainLoss), credit: 0 }) // Loss on Asset Disposal
  1189→  }

  1191→  return createJournalEntry({
  1192→    entryNo: `JE-DSP-${Date.now()}`,
  1193→    date: data.date,

  1195→    descriptionAr: `التخلص من أصل - ${data.assetAccountCode}`,
  1196→    lines,
  1197→    sourceType: 'ASSET_DISPOSAL',
  1198→    sourceId: `DSP-${Date.now()}`,
  1199→  })
  1200→}

  1204→/**
  1205→ * إقرار ضريبي - VAT Declaration
  1206→ * Dr: Output VAT (3110) - total output VAT
  1207→ * Cr: Input VAT (3120) - total input VAT
  1208→ * Cr: VAT Due (3130) - net VAT payable (if output > input)
  1209→ * OR
  1210→ * Dr: Input VAT (3120) - if input > output (refund scenario)
  1211→ * Cr: Output VAT (3110)
  1212→ * Dr: VAT Refund Receivable (1410) - if refund due
  1213→ */
  1214→export async function autoEntryVATDeclaration(data: {
  1215→  period: string
  1216→  outputVat: number
  1217→  inputVat: number
  1218→  netVat: number // positive = payable, negative = refundable
  1219→  date: Date
  1220→}) {
  1221→  const lines: { accountCode: string; debit: number; credit: number }[] = []
  1222→
  1223→  // Close Output VAT (3110) - debit to zero it out
  1224→  lines.push({ accountCode: '3110', debit: data.outputVat, credit: 0 })
  1225→
  1226→  // Close Input VAT (3120) - credit to zero it out
  1227→  lines.push({ accountCode: '3120', debit: 0, credit: data.inputVat })
  1228→
  1229→  // Net VAT position
  1230→  if (data.netVat > 0) {
  1231→    // More output than input → owe VAT
  1232→    lines.push({ accountCode: '3130', debit: 0, credit: data.netVat }) // VAT Due
  1233→  } else if (data.netVat < 0) {
  1234→    // More input than output → refund due
  1235→    lines.push({ accountCode: '1410', debit: Math.abs(data.netVat), credit: 0 }) // VAT Refund Receivable
  1236→  }
  1237→
  1238→  return createJournalEntry({
  1239→    entryNo: `JE-VAT-${Date.now()}`,
  1240→    date: data.date,
  1241→    description: `VAT Declaration - ${data.period}`,
  1242→    descriptionAr: `إقرار ضريبي - ${data.period}`,
  1243→    lines,
  1244→    sourceType: 'VAT_DECLARATION',
  1245→    sourceId: `VAT-${data.period}`,
  1246→  })
  1247→}

  1276→export async function getTrialBalance(dateFrom?: Date, dateTo?: Date) {
  1277→  const dateFilter: { date?: { gte?: Date; lte?: Date } } = {}
  1278→  if (dateFrom || dateTo) {
  1279→    dateFilter.date = {
  1280→      ...(dateFrom && { gte: dateFrom }),
  1281→      ...(dateTo && { lte: dateTo }),
  1282→    }
  1283→  }
  1284→  const entries = await db.journalEntry.findMany({
  1285→    where: {
  1286→      status: 'POSTED',
  1287→      ...dateFilter,
  1288→    },
  1289→    include: {
  1290→      lines: {
  1291→        include: {
  1292→          account: true,
  1293→        },
  1294→      },
  1295→    },
  1296→  })
  1297→
  1298→  // Aggregate by account
  1299→  const accountBalances = new Map<string, {
  1300→    account: { id: string; code: string; name: string; nameAr: string | null; type: string }
  1301→    totalDebit: number
  1302→    totalCredit: number
  1303→  }>()
  1304→
  1305→  for (const entry of entries) {
  1306→    for (const line of entry.lines) {
  1307→      const key = line.accountId
  1308→      if (!accountBalances.has(key)) {
  1309→        accountBalances.set(key, {
  1310→          account: line.account,
  1311→          totalDebit: 0,
  1312→          totalCredit: 0,
  1313→        })
  1314→      }
  1315→      const bal = accountBalances.get(key)!
  1316→      bal.totalDebit += line.debit
  1317→      bal.totalCredit += line.credit
  1318→    }
  1319→  }
  1320→
  1321→  // Calculate net balance for each account
  1322→  const results = Array.from(accountBalances.values()).map(bal => {
  1323→    const normalBalance = NORMAL_BALANCE[bal.account.type as AccountTypeValue] || 'DEBIT'
  1324→    let netDebit = 0
  1325→    let netCredit = 0
  1326→
  1327→    if (normalBalance === 'DEBIT') {
  1328→      const net = bal.totalDebit - bal.totalCredit
  1329→      if (net >= 0) netDebit = net
  1330→      else netCredit = Math.abs(net)
  1331→    } else {
  1332→      const net = bal.totalCredit - bal.totalDebit
  1333→      if (net >= 0) netCredit = net
  1334→      else netDebit = Math.abs(net)
  1335→    }
  1336→
  1337→    return {
  1338→      ...bal,
  1339→      netDebit,
  1340→      netCredit,
  1341→    }
  1342→  })
  1343→
  1344→  // Sort by account code
  1345→  results.sort((a, b) => a.account.code.localeCompare(b.account.code))
  1346→
  1347→  return results
  1348→}

  1380→  if (dateFrom || dateTo) {

  1383→      ...(dateTo && { lte: dateTo }),
  1384→    }
  1385→  }
  1386→  const lines = await db.journalLine.findMany({
  1387→    where: {
  1388→      accountId: account.id,
  1389→      journalEntry: {
  1390→        status: 'POSTED',
  1391→        ...glDateFilter,
  1392→      },
  1393→    },
  1394→    include: { journalEntry: true },
  1395→    orderBy: { journalEntry: { date: 'asc' } },
  1396→  })
  1397→
  1398→  let runningBalance = 0
  1399→  const normalBalance = NORMAL_BALANCE[account.type as AccountTypeValue] || 'DEBIT'
  1400→
  1401→  return lines.map(line => {
  1402→    if (normalBalance === 'DEBIT') {
  1403→      runningBalance += line.debit - line.credit
  1404→    } else {
  1405→      runningBalance += line.credit - line.debit
  1406→    }
  1407→
  1408→    return {
  1409→      date: line.journalEntry.date,
  1410→      entryNo: line.journalEntry.entryNo,
  1411→      description: line.journalEntry.description,
  1412→      debit: line.debit,
  1413→      credit: line.credit,
  1414→      balance: runningBalance,
  1415→    }
  1416→  })
  1417→}

    63→    63→// ============ ImageUploadField Component ============
    64→    64→function ImageUploadField({ 
    65→    65→  value, 
    66→    66→  onChange, 
    67→    67→  label, 
    68→    68→  labelAr,
    69→    69→  lang,
    70→    70→  hint,
    71→    71→  hintAr,
    72→    72→  previewHeight = 'h-24',
    73→    73→  accept = 'image/svg+xml,image/png,image/jpeg,image/webp',
    74→    74→}: {
    75→    75→  value: string | null
    76→    76→  onChange: (url: string | null) => void
    77→    77→  label: string
    78→    78→  labelAr: string
    79→    79→  lang: 'ar' | 'en'
    80→    80→  hint?: string
    81→    81→  hintAr?: string
    82→    82→  previewHeight?: string
    83→    83→  accept?: string
    84→    84→}) {
    85→    85→  const [uploading, setUploading] = useState(false)
    86→    86→  const [dragOver, setDragOver] = useState(false)
    87→    87→  const fileInputRef = useRef<HTMLInputElement>(null)
    88→    88→
    89→    89→  const [error, setError] = useState<string | null>(null)
    90→    90→
    91→    91→  const uploadFile = useCallback(async (file: File) => {
    92→    92→    if (file.size > 5 * 1024 * 1024) {
    93→    93→      setError(lang === 'ar' ? 'حجم الملف يتجاوز 5 ميجابايت' : 'File size exceeds 5MB')
    94→    94→      return
    95→    95→    }
    96→    96→    setUploading(true)
    97→    97→    setError(null)
    98→    98→    try {
    99→    99→      const formData = new FormData()
   100→   100→      formData.append('file', file)
   101→   101→      const res = await fetch('/api/upload', { method: 'POST', body: formData })
   102→   102→      if (!res.ok) {
   103→   103→        const errData = await res.json().catch(() => ({}))
   104→   104→        throw new Error(errData.error || 'Upload failed')
   105→   105→      }
   106→   106→      const data = await res.json()
   107→   107→      onChange(data.url)
   108→   108→    } catch (err) {
   109→   109→      const msg = err instanceof Error ? err.message : 'Unknown error'
   110→   110→      setError(lang === 'ar' ? `فشل في رفع الملف: ${msg}` : `Failed to upload file: ${msg}`)
   111→   111→    } finally {
   112→   112→      setUploading(false)
   113→   113→    }
   114→   114→  }, [lang, onChange])
   115→   115→
   116→   116→  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
   117→   117→    const file = e.target.files?.[0]
   118→   118→    if (file) {
   119→   119→      setError(null)
   120→   120→      uploadFile(file)
   121→   121→    }
   122→   122→    // Reset input so same file can be re-selected
   123→   123→    setTimeout(() => { if (fileInputRef.current) fileInputRef.current.value = '' }, 100)
   124→   124→  }
   125→   125→
   126→   126→  const handleDrop = (e: React.DragEvent) => {
   127→   127→    e.preventDefault()
   128→   128→    setDragOver(false)
   129→   129→    const file = e.dataTransfer.files?.[0]
   130→   130→    if (file) uploadFile(file)
   131→   131→  }
   132→   132→
   133→   133→  const handleDragOver = (e: React.DragEvent) => {
   134→   134→    e.preventDefault()
   135→   135→    setDragOver(true)
   136→   136→  }
   137→   137→
   138→   138→  const handleDragLeave = () => setDragOver(false)
   139→   139→
   140→   140→  const displayLabel = lang === 'ar' ? labelAr : label
   141→   141→  const displayHint = lang === 'ar' ? hintAr : hint
   142→   142→
   143→   143→  return (
   144→   144→    <div className="space-y-2">
   145→   145→      <Label>{displayLabel}</Label>
   146→   146→      {value ? (
   147→   147→        <div className={`relative rounded-lg border bg-white p-2 ${previewHeight} flex items-center justify-center group overflow-hidden`}>
   148→   148→          <img
   149→   149→            src={value}
   150→   150→            alt={displayLabel}
   151→   151→            className="max-h-20 max-w-full object-contain"
   152→   152→          />
   153→   153→          <Button
   154→   154→            variant="destructive"
   155→   155→            size="icon"
   156→   156→            className="absolute top-1 left-1 size-6 opacity-0 group-hover:opacity-100 transition-opacity z-10"
   157→   157→            onClick={() => { setError(null); onChange(null) }}
   158→   158→            type="button"
   159→   159→          >
   160→   160→            <X className="size-3" />
   161→   161→          </Button>
   162→   162→        </div>
   163→   163→      ) : (
   164→   164→        <div
   165→   165→          className={`rounded-lg border-2 border-dashed transition-colors cursor-pointer ${
   166→   166→            dragOver ? 'border-emerald-400 bg-emerald-50' : 'border-gray-300 bg-gray-50 hover:border-emerald-300 hover:bg-emerald-50/50'
   167→   167→          } ${previewHeight} flex flex-col items-center justify-center gap-1`}
   168→   168→          onClick={() => fileInputRef.current?.click()}
   169→   169→          onDrop={handleDrop}
   170→   170→          onDragOver={handleDragOver}
   171→   171→          onDragLeave={handleDragLeave}
   172→   172→        >
   173→   173→          {uploading ? (
   174→   174→            <>
   175→   175→              <Loader2 className="size-5 text-emerald-500 animate-spin" />
   176→   176→              <span className="text-xs text-muted-foreground">
   177→   177→                {lang === 'ar' ? 'جاري الرفع...' : 'Uploading...'}
   178→   178→              </span>
   179→   179→            </>
   180→   180→          ) : (
   181→   181→            <>
   182→   182→              <Upload className="size-5 text-gray-400" />
   183→   183→              <span className="text-xs text-muted-foreground">
   184→   184→                {lang === 'ar' ? 'اسحب الملف هنا أو انقر للاختيار' : 'Drag & drop or click to upload'}
   185→   185→              </span>
   186→   186→              <span className="text-[10px] text-muted-foreground">
   187→   187→                {lang === 'ar' ? 'SVG, PNG, JPG - حد أقصى 5MB' : 'SVG, PNG, JPG - Max 5MB'}
   188→   188→              </span>
   189→   189→            </>
   190→   190→          )}
   191→   191→        </div>
   192→   192→      )}
   193→   193→      <input
   194→   194→        ref={fileInputRef}
   195→   195→        type="file"
   196→   196→        accept={accept}
   197→   197→        onChange={handleFileChange}
   198→   198→        className="hidden"
   199→   199→      />
   200→   200→      {error && (
   201→   201→        <p className="text-xs text-red-600">{error}</p>
   202→   202→      )}
   203→   203→      {displayHint && !error && (
   204→   204→        <p className="text-xs text-muted-foreground">{displayHint}</p>
   205→   205→      )}
   206→   206→    </div>
   207→   207→  )
   208→   208→}

   811→   811→function BranchDialog({ open, onOpenChange }: { open: boolean; onOpenChange: (v: boolean) => void }) {
   812→   812→  const { lang } = useAppStore()
   813→   813→  const queryClient = useQueryClient()
   814→   814→  const [name, setName] = useState('')
   815→   815→  const [address, setAddress] = useState('')
   816→   816→
   817→   817→  React.useEffect(() => { if (open) { setName(''); setAddress('') } }, [open])
   818→   818→
   819→   819→  const createMutation = useMutation({
   820→   820→    mutationFn: (data: Record<string, unknown>) =>
   821→   821→      fetch('/api/branches', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) })
   822→   822→        .then(r => { if (!r.ok) throw new Error(); return r.json() }),
   823→   823→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['branches'] }); onOpenChange(false) },
   824→   824→  })
   825→   825→
   826→   826→  const handleSubmit = (e: React.FormEvent) => {
   827→   827→    e.preventDefault()
   828→   828→    createMutation.mutate({ name, address: address || null })
   829→   829→  }
   830→   830→
   831→   831→  return (
   832→   832→    <Dialog open={open} onOpenChange={onOpenChange}>
   833→   833→      <DialogContent className="max-w-md">
   834→   834→        <DialogHeader>
   835→   835→          <DialogTitle>{lang === 'ar' ? 'فرع جديد' : 'New Branch'}</DialogTitle>
   836→   836→          <DialogDescription>{lang === 'ar' ? 'إضافة فرع جديد' : 'Add new branch'}</DialogDescription>
   837→   837→        </DialogHeader>
   838→   838→        <form onSubmit={handleSubmit} className="space-y-4">
   839→   839→          <div className="space-y-2">
   840→   840→            <Label>{lang === 'ar' ? 'اسم الفرع *' : 'Branch Name *'}</Label>
   841→   841→            <Input value={name} onChange={e => setName(e.target.value)} required />
   842→   842→          </div>
   843→   843→          <div className="space-y-2">
   844→   844→            <Label>{lang === 'ar' ? 'العنوان' : 'Address'}</Label>
   845→   845→            <Input value={address} onChange={e => setAddress(e.target.value)} />
   846→   846→          </div>
   847→   847→          <DialogFooter>
   848→   848→            <Button type="button" variant="outline" onClick={() => onOpenChange(false)}>{lang === 'ar' ? 'إلغاء' : 'Cancel'}</Button>
   849→   849→            <Button type="submit" disabled={createMutation.isPending || !name} className="bg-emerald-600 hover:bg-emerald-700">
   850→   850→              {createMutation.isPending ? (lang === 'ar' ? 'جاري الإنشاء...' : 'Creating...') : (lang === 'ar' ? 'إضافة' : 'Add')}
   851→   851→            </Button>
   852→   852→          </DialogFooter>
   853→   853→        </form>
   854→   854→      </DialogContent>
   855→   855→    </Dialog>
   856→   856→  )

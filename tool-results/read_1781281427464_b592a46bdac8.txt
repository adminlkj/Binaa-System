     5→export async function GET(request: Request) {
     6→  try {
     7→    const { searchParams } = new URL(request.url)
     8→    const clientId = searchParams.get('clientId')
     9→    const projectId = searchParams.get('projectId')
    10→    const status = searchParams.get('status')
    11→    const invoiceType = searchParams.get('invoiceType')
    12→    const sourceType = searchParams.get('sourceType')
    13→
    14→    const where: Record<string, unknown> = {}
    15→    if (clientId) where.clientId = clientId
    16→    if (projectId) where.projectId = projectId
    17→    if (status) where.status = status
    18→    if (invoiceType) where.invoiceType = invoiceType
    19→    if (sourceType) where.sourceType = sourceType
    20→
    21→    const invoices = await db.salesInvoice.findMany({
    22→      where,
    23→      include: {
    24→        client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
    25→        project: { select: { id: true, name: true, nameAr: true, code: true, projectType: true } },
    26→        contract: { select: { id: true, contractNo: true } },
    27→        timesheet: {
    28→          select: {
    29→            id: true, operatingHours: true, month: true, year: true, status: true,
    30→            project: { select: { id: true, name: true, code: true, projectType: true, client: { select: { id: true, name: true } } } },
    31→            equipment: { select: { id: true, name: true, code: true, nameAr: true } },
    32→            rental: { select: { id: true, hourlyRate: true, deliveryFees: true, deliveryFeesTaxable: true } },
    33→            contract: { select: { id: true, contractNo: true, hourlyRate: true, paymentTerms: true } },
    34→          },
    35→        },
    36→        progressClaim: {
    37→          select: {
    38→            id: true, claimNo: true, date: true, amount: true, vatAmount: true,
    39→            totalAmount: true, status: true, invoiced: true,
    40→            project: { select: { id: true, name: true, code: true, projectType: true, client: { select: { id: true, name: true } } } },
    41→            contract: { select: { id: true, contractNo: true } },
    42→          },
    43→        },
    44→        items: true,
    45→      },
    46→      orderBy: { date: 'desc' },
    47→    })
    48→    return NextResponse.json(invoices)
    49→  } catch (error) {
    50→    console.error('Error fetching sales invoices:', error)
    51→    return NextResponse.json({ error: 'فشل في تحميل فواتير المبيعات' }, { status: 500 })
    52→  }
    53→}

    55→export async function POST(request: Request) {
    56→  try {
    57→    const body = await request.json()
    58→    const { sourceType } = body
    59→
    60→    // ===== MODE A: Create from ProgressClaim (Extract) =====
    61→    if (sourceType === 'EXTRACT') {
    62→      return await createInvoiceFromExtract(body)
    63→    }
    64→
    65→    // ===== MODE B: Create from TimeSheet =====
    66→    if (sourceType === 'TIMESHEET') {
    67→      return await createInvoiceFromTimesheet(body)
    68→    }
    69→
    70→    // ===== LEGACY MODE: Manual creation with items array =====
    71→    return await createInvoiceManual(body)
    72→  } catch (error) {
    73→    console.error('Error creating sales invoice:', error)
    74→    return NextResponse.json({ error: 'فشل في إنشاء فاتورة المبيعات' }, { status: 500 })
    75→  }
    76→}

    79→// MODE A: Create invoice from ProgressClaim (Extract)

    81→async function createInvoiceFromExtract(body: Record<string, unknown>) {
    82→  const { progressClaimId, date, dueDate, notes } = body
    83→
    84→  if (!progressClaimId || typeof progressClaimId !== 'string') {
    85→    return NextResponse.json({ error: 'معرف المستخلص مطلوب' }, { status: 400 })
    86→  }
    87→
    88→  if (!date || !dueDate) {
    89→    return NextResponse.json({ error: 'التاريخ وتاريخ الاستحقاق مطلوبان' }, { status: 400 })
    90→  }
    91→
    92→  // 1. Fetch the ProgressClaim and verify it exists
    93→  const claim = await db.progressClaim.findUnique({
    94→    where: { id: progressClaimId },
    95→    include: {
    96→      project: { select: { id: true, name: true, code: true, clientId: true, client: { select: { id: true, name: true } } } },
    97→      contract: { select: { id: true, contractNo: true } },
    98→    },
    99→  })
   100→
   101→  if (!claim) {
   102→    return NextResponse.json({ error: 'المستخلص غير موجود' }, { status: 404 })
   103→  }
   104→
   105→  // 2. Verify the claim is APPROVED
   106→  if (claim.status !== 'APPROVED') {
   107→    return NextResponse.json({ error: 'يجب اعتماد المستخلص أولاً قبل إنشاء الفاتورة' }, { status: 400 })
   108→  }
   109→
   110→  // 3. DUPLICATE PREVENTION: Check that the claim is NOT already invoiced
   111→  if (claim.invoiced) {
   112→    return NextResponse.json({ error: 'تم إصدار فاتورة لهذا المستخلص بالفعل' }, { status: 400 })
   113→  }
   114→
   115→  // Also check via SalesInvoice table
   116→  const existingInvoice = await db.salesInvoice.findFirst({
   117→    where: { progressClaimId: claim.id },
   118→  })
   119→  if (existingInvoice) {
   120→    return NextResponse.json({ error: 'يوجد فاتورة مرتبطة بهذا المستخلص بالفعل' }, { status: 400 })
   121→  }
   122→
   123→  // 4. Auto-populate from claim
   124→  const clientId = claim.project.clientId
   125→  if (!clientId) {
   126→    return NextResponse.json({ error: 'العميل غير محدد للمشروع' }, { status: 400 })
   127→  }
   128→  const projectId = claim.projectId
   129→  const contractId = claim.contractId
   130→  const subtotal = claim.amount
   131→  const vatAmount = claim.vatAmount
   132→  const totalAmount = claim.totalAmount
   133→
   134→  // 5. Generate invoice number: PCL-YEAR-SEQ
   135→  const prefix = 'PCL'
   136→  const year = new Date().getFullYear()
   137→  const yearStr = String(year)
   138→  const likePattern = `${prefix}-${yearStr}-`
   139→
   140→  const lastInvoice = await db.salesInvoice.findFirst({
   141→    where: { invoiceNo: { startsWith: likePattern } },
   142→    orderBy: { invoiceNo: 'desc' },
   143→    select: { invoiceNo: true },
   144→  })
   145→
   146→  let seq = 1
   147→  if (lastInvoice) {
   148→    const parts = lastInvoice.invoiceNo.split('-')
   149→    const parsedSeq = parseInt(parts[2])
   150→    if (!isNaN(parsedSeq)) {
   151→      seq = parsedSeq + 1
   152→    }
   153→  }
   154→  const invoiceNo = `${prefix}-${yearStr}-${String(seq).padStart(4, '0')}`
   155→
   156→  const invoiceDate = new Date(date as string)
   157→  const invoiceDueDate = new Date(dueDate as string)
   158→
   159→  // 6. Create the invoice
   160→  const invoice = await db.salesInvoice.create({
   161→    data: {
   162→      invoiceNo,
   163→      clientId,
   164→      projectId,
   165→      contractId,
   166→      date: invoiceDate,
   167→      dueDate: invoiceDueDate,
   168→      subtotal,
   169→      discountRate: 0,
   170→      discountAmount: 0,
   171→      netAmount: subtotal,
   172→      vatRate: claim.vatRate,
   173→      vatAmount,
   174→      totalAmount,
   175→      paidAmount: 0,
   176→      status: 'DRAFT',
   177→      invoiceType: 'PROGRESS_CLAIM',
   178→      sourceType: 'EXTRACT',
   179→      progressClaimId: claim.id,
   180→      contractNo: claim.contract.contractNo,
   181→      notes: (notes as string) || `فاتورة مستخلص رقم ${claim.claimNo}`,
   182→      items: {
   183→        create: [{
   184→          description: `مستخلص رقم ${claim.claimNo} - ${claim.project.name}`,
   185→          descriptionEn: `Progress Claim No. ${claim.claimNo} - ${claim.project.name}`,
   186→          quantity: 1,
   187→          unit: 'مستخلص',
   188→          unitPrice: subtotal,
   189→          totalPrice: subtotal,
   190→          itemType: 'PROGRESS_CLAIM',
   191→        }],
   192→      },
   193→    },
   194→    include: {
   195→      client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   196→      project: { select: { id: true, name: true, nameAr: true, code: true } },
   197→      contract: { select: { id: true, contractNo: true } },
   198→      timesheet: { select: { id: true, month: true, year: true, operatingHours: true, status: true } },
   199→      progressClaim: { select: { id: true, claimNo: true, status: true, invoiced: true } },
   200→      items: true,
   201→    },
   202→  })
   203→
   204→  // 7. Mark the ProgressClaim as invoiced
   205→  await db.progressClaim.update({
   206→    where: { id: claim.id },
   207→    data: { invoiced: true },
   208→  })
   209→
   210→  // 8. Create auto accounting entry
   211→  try {
   212→    await initializeChartOfAccounts()
   213→    const journalEntry = await autoEntrySalesInvoice({
   214→      invoiceNo: invoice.invoiceNo,
   215→      clientId: invoice.clientId,
   216→      subtotal: invoice.subtotal,
   217→      vatRate: invoice.vatRate,
   218→      vatAmount: invoice.vatAmount,
   219→      totalAmount: invoice.totalAmount,
   220→      invoiceType: 'PROGRESS_CLAIM',
   221→      date: invoice.date,
   222→      projectId: invoice.projectId || undefined,
   223→    })
   224→
   225→    await db.salesInvoice.update({
   226→      where: { id: invoice.id },
   227→      data: { journalEntryId: journalEntry.id },
   228→    })
   229→  } catch (accountingError) {
   230→    console.error('Accounting entry failed for sales invoice from extract:', accountingError)
   231→  }
   232→
   233→  // Re-fetch to include journalEntryId
   234→  const updatedInvoice = await db.salesInvoice.findUnique({
   235→    where: { id: invoice.id },
   236→    include: {
   237→      client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   238→      project: { select: { id: true, name: true, nameAr: true, code: true } },
   239→      contract: { select: { id: true, contractNo: true } },
   240→      timesheet: { select: { id: true, month: true, year: true, operatingHours: true, status: true } },
   241→      progressClaim: { select: { id: true, claimNo: true, status: true, invoiced: true } },
   242→      items: true,
   243→    },
   244→  })
   245→
   246→  return NextResponse.json(updatedInvoice, { status: 201 })
   247→}

   250→// MODE B: Create invoice from TimeSheet
   251→// ============================================================================
   252→async function createInvoiceFromTimesheet(body: Record<string, unknown>) {
   253→  const { timesheetId, date, dueDate, notes } = body
   254→
   255→  if (!timesheetId || typeof timesheetId !== 'string') {
   256→    return NextResponse.json({ error: 'معرف التايم شيت مطلوب' }, { status: 400 })
   257→  }
   258→
   259→  if (!date || !dueDate) {
   260→    return NextResponse.json({ error: 'التاريخ وتاريخ الاستحقاق مطلوبان' }, { status: 400 })
   261→  }
   262→
   263→  // 1. Fetch the Timesheet with its rental/contract/equipment relations
   264→  const timesheet = await db.timesheet.findUnique({
   265→    where: { id: timesheetId },
   266→    include: {
   267→      contract: {
   268→        include: {
   269→          rental: {
   270→            select: { id: true, hourlyRate: true, pricingType: true, deliveryFees: true, deliveryFeesTaxable: true, clientId: true, status: true, salesOrderNo: true, paymentDuration: true },
   271→          },
   272→        },
   273→      },
   274→      equipment: { select: { id: true, code: true, name: true, nameAr: true } },
   275→      project: { select: { id: true, name: true, code: true, clientId: true } },
   276→      rental: {
   277→        select: { id: true, hourlyRate: true, pricingType: true, deliveryFees: true, deliveryFeesTaxable: true, clientId: true, status: true, salesOrderNo: true, paymentDuration: true },
   278→      },
   279→    },
   280→  })
   281→
   282→  if (!timesheet) {
   283→    return NextResponse.json({ error: 'التايم شيت غير موجود' }, { status: 404 })
   284→  }
   285→
   286→  // 2. Verify the Timesheet status is APPROVED (not DRAFT or INVOICED)
   287→  if (timesheet.status !== 'APPROVED') {
   288→    if (timesheet.status === 'INVOICED') {
   289→      return NextResponse.json({ error: 'تم إصدار فاتورة لهذا التايم شيت بالفعل' }, { status: 400 })
   290→    }
   291→    return NextResponse.json({ error: 'يجب اعتماد التايم شيت أولاً قبل إنشاء الفاتورة' }, { status: 400 })
   292→  }
   293→
   294→  // Also check if there's already an invoice linked
   295→  const existingInvoice = await db.salesInvoice.findUnique({
   296→    where: { timesheetId: timesheet.id },
   297→  })
   298→  if (existingInvoice) {
   299→    return NextResponse.json({ error: 'تم إصدار فاتورة لهذا التايم شيت بالفعل' }, { status: 400 })
   300→  }
   301→
   302→  // 2b. Verify the rental contract is ACTIVE
   303→  const rentalContract = timesheet.rental || timesheet.contract.rental
   304→  if (rentalContract && rentalContract.status !== 'ACTIVE') {
   305→    return NextResponse.json({ 
   306→      error: `لا يمكن إنشاء فاتورة لعقد إيجار ${rentalContract.status === 'EXPIRED' ? 'منتهي' : rentalContract.status === 'TERMINATED' ? 'ملغي' : 'غير نشط'}` 
   307→    }, { status: 400 })
   308→  }
   309→
   310→  // 2c. Verify a delivered delivery order exists for this rental
   311→  if (rentalContract) {
   312→    const deliveryOrder = await db.equipmentDeliveryOrder.findFirst({
   313→      where: { 
   314→        rentalId: rentalContract.id, 
   315→        status: 'DELIVERED' 
   316→      },
   317→    })
   318→    if (!deliveryOrder) {
   319→      return NextResponse.json({ 
   320→        error: 'يجب وجود أمر توصيل تم تنفيذه قبل إنشاء فاتورة التأجير' 
   321→      }, { status: 400 })
   322→    }
   323→  }
   324→
   325→  // 4. Calculate: subtotal = operatingHours × hourlyRate (from contract)
   326→  const rental = timesheet.rental || timesheet.contract.rental
   327→  const hourlyRate = rental?.hourlyRate || timesheet.contract.hourlyRate || 0
   328→  const operatingHours = timesheet.operatingHours
   329→  const subtotal = Math.round(operatingHours * hourlyRate * 100) / 100
   330→
   331→  // 5. Add delivery fees if applicable (from rental contract)
   332→  const deliveryFees = rental?.deliveryFees || timesheet.contract.deliveryFees || 0
   333→  const deliveryFeesTaxable = rental?.deliveryFeesTaxable ?? timesheet.contract.deliveryFeesTaxable ?? true
   334→  const includeDelivery = deliveryFees > 0
   335→
   336→  const vatRate = 0.15
   337→  let deliveryVat = 0
   338→  if (includeDelivery && deliveryFeesTaxable) {
   339→    deliveryVat = Math.round(deliveryFees * vatRate * 100) / 100
   340→  }
   341→  const rentalVat = Math.round(subtotal * vatRate * 100) / 100
   342→  const vatAmount = Math.round((rentalVat + deliveryVat) * 100) / 100
   343→  const totalAmount = Math.round((subtotal + deliveryFees + vatAmount) * 100) / 100
   344→
   345→  // Get clientId from rental or project
   346→  const clientId = rental?.clientId || timesheet.project.clientId
   347→
   348→  // Month names for description
   349→  const monthNamesAr = [
   350→    'يناير', 'فبراير', 'مارس', 'أبريل', 'مايو', 'يونيو',
   351→    'يوليو', 'أغسطس', 'سبتمبر', 'أكتوبر', 'نوفمبر', 'ديسمبر',
   352→  ]
   353→  const monthLabel = `${monthNamesAr[timesheet.month - 1]} ${timesheet.year}`
   354→  const equipmentName = timesheet.equipment.nameAr || timesheet.equipment.name
   355→
   356→  // 6. Generate invoice number: RNT-YEAR-SEQ
   357→  const prefix = 'RNT'
   358→  const year = new Date().getFullYear()
   359→  const yearStr = String(year)
   360→  const likePattern = `${prefix}-${yearStr}-`
   361→
   362→  const lastInvoice = await db.salesInvoice.findFirst({
   363→    where: { invoiceNo: { startsWith: likePattern } },
   364→    orderBy: { invoiceNo: 'desc' },
   365→    select: { invoiceNo: true },
   366→  })
   367→
   368→  let seq = 1
   369→  if (lastInvoice) {
   370→    const parts = lastInvoice.invoiceNo.split('-')
   371→    const parsedSeq = parseInt(parts[2])
   372→    if (!isNaN(parsedSeq)) {
   373→      seq = parsedSeq + 1
   374→    }
   375→  }
   376→  const invoiceNo = `${prefix}-${yearStr}-${String(seq).padStart(4, '0')}`
   377→
   378→  const invoiceDate = new Date(date as string)
   379→  const invoiceDueDate = new Date(dueDate as string)
   380→
   381→  // 7. Create the invoice
   382→  const invoiceItems = [
   383→    {
   384→      description: `تأجير ${equipmentName} - ${monthLabel} - ${operatingHours} ساعة`,
   385→      descriptionEn: `Equipment Rental - ${timesheet.equipment.name} - ${monthLabel} - ${operatingHours} hours`,
   386→      quantity: operatingHours,
   387→      unit: 'ساعة',
   388→      unitPrice: hourlyRate,
   389→      totalPrice: subtotal,
   390→      itemType: 'RENTAL',
   391→    },
   392→  ]
   393→
   394→  // Add delivery fee as separate item if applicable
   395→  if (includeDelivery && deliveryFees > 0) {
   396→    invoiceItems.push({
   397→      description: `رسوم نقل وتنزيل - ${equipmentName}`,
   398→      descriptionEn: `Delivery Fees - ${timesheet.equipment.name}`,
   399→      quantity: 1,
   400→      unit: 'خدمة',
   401→      unitPrice: deliveryFees,
   402→      totalPrice: deliveryFees,
   403→      itemType: 'DELIVERY',
   404→    })
   405→  }
   406→
   407→  const invoice = await db.salesInvoice.create({
   408→    data: {
   409→      invoiceNo,
   410→      clientId,
   411→      projectId: timesheet.projectId,
   412→      contractId: timesheet.contractId,
   413→      date: invoiceDate,
   414→      dueDate: invoiceDueDate,
   415→      subtotal,
   416→      discountRate: 0,
   417→      discountAmount: 0,
   418→      netAmount: subtotal,
   419→      vatRate,
   420→      vatAmount,
   421→      totalAmount,
   422→      paidAmount: 0,
   423→      status: 'DRAFT',
   424→      invoiceType: 'RENTAL',
   425→      sourceType: 'TIMESHEET',
   426→      timesheetId: timesheet.id,
   427→      contractNo: timesheet.contract.contractNo,
   428→      contractType: timesheet.contract.contractType,
   429→      contractPeriodStart: timesheet.contract.startDate,
   430→      contractPeriodEnd: timesheet.contract.endDate,
   431→      salesOrderNo: rental?.salesOrderNo || null,
   432→      equipmentName: timesheet.equipment.name,
   433→      operatingHours,
   434→      hourlyRate,
   435→      includeDelivery,
   436→      deliveryAmount: deliveryFees,
   437→      deliveryFeesTaxable,
   438→      includeVat: true,
   439→      deliveryMonth: `${timesheet.year}-${String(timesheet.month).padStart(2, '0')}`,
   440→      paymentTerms: rental?.paymentDuration || timesheet.contract.paymentTerms || null,
   441→      notes: (notes as string) || `فاتورة تأجير معدات - عقد ${timesheet.contract.contractNo} - ${monthLabel}`,
   442→      items: {
   443→        create: invoiceItems,
   444→      },
   445→    },
   446→    include: {
   447→      client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   448→      project: { select: { id: true, name: true, nameAr: true, code: true } },
   449→      contract: { select: { id: true, contractNo: true } },
   450→      timesheet: { select: { id: true, month: true, year: true, operatingHours: true, status: true } },
   451→      progressClaim: { select: { id: true, claimNo: true, status: true, invoiced: true } },
   452→      items: true,
   453→    },
   454→  })
   455→
   456→  // 8. Mark the Timesheet as INVOICED
   457→  await db.timesheet.update({
   458→    where: { id: timesheet.id },
   459→    data: { status: 'INVOICED', approvedDate: timesheet.approvedDate || new Date() },
   460→  })
   461→
   462→  // 9. Create auto accounting entry
   463→  try {
   464→    await initializeChartOfAccounts()
   465→    const journalEntry = await autoEntryRentalInvoice({
   466→      invoiceNo: invoice.invoiceNo,
   467→      subtotal: invoice.subtotal,
   468→      vatAmount: invoice.vatAmount,
   469→      totalAmount: invoice.totalAmount,
   470→      date: invoice.date,
   471→      costCenterId: invoice.projectId || undefined,
   472→    })
   473→
   474→    await db.salesInvoice.update({
   475→      where: { id: invoice.id },
   476→      data: { journalEntryId: journalEntry.id },
   477→    })
   478→  } catch (accountingError) {
   479→    console.error('Accounting entry failed for sales invoice from timesheet:', accountingError)
   480→  }
   481→
   482→  // Re-fetch to include journalEntryId
   483→  const updatedInvoice = await db.salesInvoice.findUnique({
   484→    where: { id: invoice.id },
   485→    include: {
   486→      client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   487→      project: { select: { id: true, name: true, nameAr: true, code: true } },
   488→      contract: { select: { id: true, contractNo: true } },
   489→      timesheet: { select: { id: true, month: true, year: true, operatingHours: true, status: true } },
   490→      progressClaim: { select: { id: true, claimNo: true, status: true, invoiced: true } },
   491→      items: true,
   492→    },
   493→  })
   494→
   495→  return NextResponse.json(updatedInvoice, { status: 201 })
   496→}

   501→async function createInvoiceManual(body: Record<string, unknown>) {
   502→  const {
   503→    clientId, projectId, contractId, date, dueDate, notes, items,
   504→    vatRate = 0.15, discountRate = 0, discountAmount = 0,
   505→    invoiceType = 'TAX_INVOICE', paymentTerms,
   506→    referenceNo, contractNo, contractType, contractPeriodStart, contractPeriodEnd,
   507→    deliveryMonth, includeDelivery = false, deliveryAmount = 0, includeVat = true,
   508→  } = body as Record<string, unknown>
   509→
   510→  const typedItems = items as Array<{ description: string; descriptionEn?: string; quantity: number; unit?: string; unitPrice: number; itemType?: string }> | undefined
   511→
   512→  if (!clientId || !date || !dueDate || !typedItems?.length) {
   513→    return NextResponse.json({ error: 'البيانات المطلوبة غير مكتملة' }, { status: 400 })
   514→  }
   515→
   516→  // Calculate totals from items
   517→  const subtotal = typedItems.reduce((sum: number, item: { quantity: number; unitPrice: number }) => {
   518→    return sum + (item.quantity * item.unitPrice)
   519→  }, 0)
   520→
   521→  const finalDiscountAmount = (discountAmount as number) || (subtotal * (discountRate as number))
   522→  const netAmount = subtotal - finalDiscountAmount
   523→  const deliveryTotal = includeDelivery ? (deliveryAmount as number) : 0
   524→  const vatAmount = includeVat ? (netAmount + deliveryTotal) * (vatRate as number) : 0
   525→  const totalAmount = netAmount + deliveryTotal + vatAmount
   526→
   527→  // Auto-generate invoice number: TYPE-YEAR-SEQ format
   528→  const typePrefixMap: Record<string, string> = {
   529→    TAX_INVOICE: 'SRV',
   530→    PROGRESS_CLAIM: 'PCL',
   531→    RENTAL: 'RNT',
   532→    SERVICE: 'SVC',
   533→  }
   534→  const prefix = typePrefixMap[invoiceType as string] || 'INV'
   535→  const year = new Date().getFullYear()
   536→  const yearStr = String(year)
   537→  const likePattern = `${prefix}-${yearStr}-`
   538→
   539→  const lastInvoice = await db.salesInvoice.findFirst({
   540→    where: { invoiceNo: { startsWith: likePattern } },
   541→    orderBy: { invoiceNo: 'desc' },
   542→    select: { invoiceNo: true },
   543→  })
   544→
   545→  let seq = 1
   546→  if (lastInvoice) {
   547→    const parts = lastInvoice.invoiceNo.split('-')
   548→    const parsedSeq = parseInt(parts[2])
   549→    if (!isNaN(parsedSeq)) {
   550→      seq = parsedSeq + 1
   551→    }
   552→  }
   553→  const invoiceNo = `${prefix}-${yearStr}-${String(seq).padStart(4, '0')}`
   554→
   555→  const invoice = await db.salesInvoice.create({
   556→    data: {
   557→      invoiceNo,
   558→      clientId: clientId as string,
   559→      projectId: (projectId as string) || null,
   560→      contractId: (contractId as string) || null,
   561→      date: new Date(date as string),
   562→      dueDate: new Date(dueDate as string),
   563→      subtotal,
   564→      discountRate: discountRate as number,
   565→      discountAmount: finalDiscountAmount,
   566→      netAmount,
   567→      vatRate: vatRate as number,
   568→      vatAmount,
   569→      totalAmount,
   570→      paidAmount: 0,
   571→      status: 'DRAFT',
   572→      invoiceType: invoiceType as string,
   573→      notes: (notes as string) || null,
   574→      paymentTerms: (paymentTerms as string) || null,
   575→      referenceNo: (referenceNo as string) || null,
   576→      contractNo: (contractNo as string) || null,
   577→      contractType: (contractType as string) || null,
   578→      contractPeriodStart: contractPeriodStart ? new Date(contractPeriodStart as string) : null,
   579→      contractPeriodEnd: contractPeriodEnd ? new Date(contractPeriodEnd as string) : null,
   580→      deliveryMonth: (deliveryMonth as string) || null,
   581→      includeDelivery: includeDelivery as boolean,
   582→      deliveryAmount: includeDelivery ? (deliveryAmount as number) : 0,
   583→      includeVat: includeVat as boolean,
   584→      items: {
   585→        create: typedItems.map((item: { description: string; descriptionEn?: string; quantity: number; unit?: string; unitPrice: number; itemType?: string }) => ({
   586→          description: item.description,
   587→          descriptionEn: item.descriptionEn || null,
   588→          quantity: item.quantity,
   589→          unit: item.unit || null,
   590→          unitPrice: item.unitPrice,
   591→          totalPrice: item.quantity * item.unitPrice,
   592→          itemType: item.itemType || 'PRODUCT',
   593→        })),
   594→      },
   595→    },
   596→    include: {
   597→      client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   598→      project: { select: { id: true, name: true, nameAr: true, code: true } },
   599→      contract: { select: { id: true, contractNo: true } },
   600→      items: true,
   601→    },
   602→  })
   603→
   604→  // Auto-create accounting journal entry and store journalEntryId
   605→  try {
   606→    await initializeChartOfAccounts()
   607→
   608→    let journalEntry
   609→    if (invoiceType === 'RENTAL') {
   610→      journalEntry = await autoEntryRentalInvoice({
   611→        invoiceNo: invoice.invoiceNo,
   612→        subtotal: invoice.subtotal,
   613→        vatAmount: invoice.vatAmount,
   614→        totalAmount: invoice.totalAmount,
   615→        date: invoice.date,
   616→        costCenterId: invoice.projectId || undefined,
   617→      })
   618→    } else {
   619→      journalEntry = await autoEntrySalesInvoice({
   620→        invoiceNo: invoice.invoiceNo,
   621→        clientId: invoice.clientId,
   622→        subtotal: invoice.subtotal,
   623→        vatRate: invoice.vatRate,
   624→        vatAmount: invoice.vatAmount,
   625→        totalAmount: invoice.totalAmount,
   626→        invoiceType: invoice.invoiceType,
   627→        date: invoice.date,
   628→        projectId: invoice.projectId || undefined,
   629→      })
   630→    }
   631→
   632→    // Store the journalEntryId on the invoice
   633→    await db.salesInvoice.update({
   634→      where: { id: invoice.id },
   635→      data: { journalEntryId: journalEntry.id },
   636→    })
   637→  } catch (accountingError) {
   638→    console.error('Accounting entry failed for sales invoice:', accountingError)
   639→    // Don't fail the invoice creation, just log the error
   640→  }
   641→
   642→  // Re-fetch to include journalEntryId
   643→  const updatedInvoice = await db.salesInvoice.findUnique({
   644→    where: { id: invoice.id },
   645→    include: {
   646→      client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   647→      project: { select: { id: true, name: true, nameAr: true, code: true } },
   648→      contract: { select: { id: true, contractNo: true } },
   649→      items: true,
   650→    },
   651→  })
   652→
   653→  return NextResponse.json(updatedInvoice, { status: 201 })
   654→}

   656→// PUT: Update a sales invoice (with reversal for approved/posted invoices)
   657→export async function PUT(request: Request) {
   658→  try {
   659→    const body = await request.json()
   660→    const { id, ...updateData } = body
   661→
   662→    if (!id) {
   663→      return NextResponse.json({ error: 'معرف الفاتورة مطلوب' }, { status: 400 })
   664→    }
   665→
   666→    const existing = await db.salesInvoice.findUnique({ where: { id } })
   667→    if (!existing) {
   668→      return NextResponse.json({ error: 'الفاتورة غير موجودة' }, { status: 404 })
   669→    }
   670→
   671→    // Cannot modify CANCELLED invoices
   672→    if (existing.status === 'CANCELLED') {
   673→      return NextResponse.json({ error: 'لا يمكن تعديل فاتورة ملغاة' }, { status: 400 })
   674→    }
   675→
   676→    // If the invoice has been approved/posted (has a journal entry), create a reversal + new entry
   677→    if (existing.journalEntryId && (updateData.subtotal !== undefined || updateData.totalAmount !== undefined || updateData.vatAmount !== undefined)) {
   678→      // Create reversal entry for the original
   679→      const originalEntry = await db.journalEntry.findUnique({
   680→        where: { id: existing.journalEntryId },
   681→        include: { lines: true },
   682→      })
   683→
   684→      if (originalEntry) {
   685→        // Create reversal entry (swap debit/credit)
   686→        const accountIds = originalEntry.lines.map(l => l.accountId)
   687→        const accounts = await db.account.findMany({ where: { id: { in: accountIds } } })
   688→        const accountMap = new Map(accounts.map(a => [a.id, a.code]))
   689→
   690→        const resolvedReversalLines = originalEntry.lines.map(line => ({
   691→          accountCode: accountMap.get(line.accountId) || '',
   692→          debit: line.credit,
   693→          credit: line.debit,
   694→          costCenterId: line.costCenterId || undefined,
   695→          description: `Reversal: ${line.description || ''}`,
   696→        }))
   697→
   698→        await createJournalEntry({
   699→          entryNo: `JE-REV-SI-${Date.now()}`,
   700→          date: new Date(),
   701→          description: `Reversal for Sales Invoice ${existing.invoiceNo}`,
   702→          descriptionAr: `قيد عكسي لفاتورة مبيعات ${existing.invoiceNo}`,
   703→          lines: resolvedReversalLines,
   704→          sourceType: 'SALES_INVOICE_REVERSAL',
   705→          sourceId: existing.invoiceNo,
   706→        })
   707→
   708→        // Cancel the original entry
   709→        await db.journalEntry.update({
   710→          where: { id: existing.journalEntryId },
   711→          data: { status: 'CANCELLED' },
   712→        })
   713→      }
   714→
   715→      // Create new entry with updated values
   716→      const newSubtotal = updateData.subtotal ?? existing.subtotal
   717→      const newVatAmount = updateData.vatAmount ?? existing.vatAmount
   718→      const newTotalAmount = updateData.totalAmount ?? existing.totalAmount
   719→      const newInvoiceType = updateData.invoiceType ?? existing.invoiceType
   720→      const newDate = updateData.date ? new Date(updateData.date) : existing.date
   721→
   722→      await initializeChartOfAccounts()
   723→
   724→      let newJournalEntry
   725→      if (newInvoiceType === 'RENTAL') {
   726→        newJournalEntry = await autoEntryRentalInvoice({
   727→          invoiceNo: existing.invoiceNo,
   728→          subtotal: newSubtotal,
   729→          vatAmount: newVatAmount,
   730→          totalAmount: newTotalAmount,
   731→          date: newDate,
   732→          costCenterId: existing.projectId || undefined,
   733→        })
   734→      } else {
   735→        newJournalEntry = await autoEntrySalesInvoice({
   736→          invoiceNo: existing.invoiceNo,
   737→          clientId: existing.clientId,
   738→          subtotal: newSubtotal,
   739→          vatRate: existing.vatRate,
   740→          vatAmount: newVatAmount,
   741→          totalAmount: newTotalAmount,
   742→          invoiceType: newInvoiceType,
   743→          date: newDate,
   744→          projectId: existing.projectId || undefined,
   745→        })
   746→      }
   747→
   748→      updateData.journalEntryId = newJournalEntry.id
   749→    }
   750→
   751→    // Update the invoice
   752→    const updated = await db.salesInvoice.update({
   753→      where: { id },
   754→      data: {
   755→        ...updateData,
   756→        ...(updateData.date && { date: new Date(updateData.date) }),
   757→        ...(updateData.dueDate && { dueDate: new Date(updateData.dueDate) }),
   758→      },
   759→      include: {
   760→        client: { select: { id: true, name: true, nameAr: true, code: true, taxNumber: true, phone: true, email: true, address: true } },
   761→        project: { select: { id: true, name: true, nameAr: true, code: true } },
   762→        contract: { select: { id: true, contractNo: true } },
   763→        timesheet: {
   764→          select: {
   765→            id: true, operatingHours: true, month: true, year: true, status: true,
   766→            project: { select: { id: true, name: true, code: true, client: { select: { id: true, name: true } } } },
   767→            equipment: { select: { id: true, name: true, code: true, nameAr: true } },
   768→            rental: { select: { id: true, hourlyRate: true, deliveryFees: true, deliveryFeesTaxable: true } },
   769→            contract: { select: { id: true, contractNo: true, hourlyRate: true, paymentTerms: true } },
   770→          },
   771→        },
   772→        progressClaim: {
   773→          select: {
   774→            id: true, claimNo: true, date: true, amount: true, vatAmount: true,
   775→            totalAmount: true, status: true, invoiced: true,
   776→            project: { select: { id: true, name: true, code: true, client: { select: { id: true, name: true } } } },
   777→            contract: { select: { id: true, contractNo: true } },
   778→          },
   779→        },
   780→        items: true,
   781→      },
   782→    })
   783→
   784→    return NextResponse.json(updated)
   785→  } catch (error) {
   786→    console.error('Error updating sales invoice:', error)
   787→    return NextResponse.json({ error: 'فشل في تحديث فاتورة المبيعات' }, { status: 500 })
   788→  }
   789→}

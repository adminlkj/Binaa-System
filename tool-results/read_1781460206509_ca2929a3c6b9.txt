   386→    queryKey: ['equipment-timesheets'],
   387→    queryFn: async () => {
   388→      const res = await fetch('/api/equipment/timesheets')
   389→      if (!res.ok) throw new Error('Failed to fetch')
   390→      return res.json()
   391→    },
   392→  })
   393→
   394→  // Fetch active rental contracts for create form
   395→  const { data: rentalContracts = [] } = useQuery<RentalContractOption[]>({
   396→    queryKey: ['rental-contracts-for-timesheets'],
   397→    queryFn: async () => {
   398→      const res = await fetch('/api/equipment/rental-contracts')
   399→      if (!res.ok) return []
   400→      const data = await res.json()
   401→      // Only active contracts
   402→      return (Array.isArray(data) ? data : []).filter((c: { status: string }) => c.status === 'ACTIVE')
   403→    },
   404→  })
   405→
   406→  // Delete mutation
   407→  const deleteMutation = useMutation({
   408→    mutationFn: (id: string) =>
   409→      fetch(`/api/equipment/timesheets/${id}`, { method: 'DELETE' }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   410→    onSuccess: () => {
   411→      queryClient.invalidateQueries({ queryKey: ['equipment-timesheets'] })
   412→      setDeleteId(null)
   413→    },
   414→  })
   415→
   416→  // Status workflow mutation
   417→  const statusMutation = useMutation({
   418→    mutationFn: ({ id, status }: { id: string; status: string }) =>
   419→      fetch(`/api/equipment/timesheets/${id}`, {
   420→        method: 'PUT',
   421→        headers: { 'Content-Type': 'application/json' },
   422→        body: JSON.stringify({ status }),
   423→      }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   424→    onSuccess: () => {
   425→      queryClient.invalidateQueries({ queryKey: ['equipment-timesheets'] })

   427→      queryClient.invalidateQueries({ queryKey: ['approved-timesheets'] })
   428→    },
   429→  })
   430→
   431→  // Filters
   432→  const filtered = timesheets.filter(ts => {
   433→    const monthYear = formatMonthYear(ts.month, ts.year, 'ar')
   434→    const monthYearEn = formatMonthYear(ts.month, ts.year, 'en')
   435→    const projectName = ts.project?.name || ''
   436→    const contractNo = ts.contract?.contractNo || ''
   437→    const matchSearch = !search ||
   438→      monthYear.includes(search) ||
   439→      monthYearEn.toLowerCase().includes(search.toLowerCase()) ||
   440→      projectName.includes(search) ||
   441→      contractNo.toLowerCase().includes(search.toLowerCase())
   442→    const matchStatus = statusFilter === 'all' || ts.status === statusFilter
   443→    return matchSearch && matchStatus
   444→  })
   445→
   446→  // Summary
   447→  const totalTimesheets = timesheets.length
   448→  const draftCount = timesheets.filter(ts => ts.status === 'DRAFT').length
   449→  const submittedCount = timesheets.filter(ts => ts.status === 'SUBMITTED').length
   450→  const approvedCount = timesheets.filter(ts => ts.status === 'APPROVED').length
   451→  const invoicedCount = timesheets.filter(ts => ts.status === 'INVOICED').length
   452→
   453→  // ============ CREATE VIEW ============
   454→  if (viewState.type === 'create') {
   455→    return (
   456→      <CreateTimesheetPage
   457→        contracts={rentalContracts}
   458→        onBack={() => setViewState({ type: 'list' })}
   459→      />
   460→    )
   461→  }
   462→
   463→  // ============ DETAIL VIEW ============
   464→  if (viewState.type === 'detail') {

   466→    if (!timesheet) {
   467→      return (

   470→          <Button variant="outline" onClick={() => setViewState({ type: 'list' })}>{t('العودة', 'Back')}</Button>
   471→        </div>
   472→      )
   473→    }
   474→
   475→    const hourlyRate = timesheet.rental?.hourlyRate || timesheet.contract?.hourlyRate || 0
   476→    const subtotal = timesheet.operatingHours * hourlyRate
   477→    const vatAmount = Math.round(subtotal * 0.15 * 100) / 100
   478→    const totalAmount = subtotal + vatAmount
   479→    const isInvoiced = timesheet.status === 'INVOICED'
   480→
   481→    return (
   482→      <div className="space-y-6">
   483→        <div className="flex items-center gap-3">
   484→          <Button variant="outline" size="icon" onClick={() => setViewState({ type: 'list' })}>
   485→            <ArrowRight className="size-4" />
   486→          </Button>
   487→          <div className="flex-1">
   488→            <div className="flex items-center gap-3">
   489→              <h2 className="text-xl font-bold">
   490→                {t('سجل ساعات عمل', 'Timesheet')} - {formatMonthYear(timesheet.month, timesheet.year, lang)}
   491→              </h2>
   492→              <StatusBadge status={timesheet.status} lang={lang} />
   493→            </div>
   494→            <p className="text-sm text-muted-foreground">
   495→              {timesheet.project?.name || '—'} - {timesheet.contract?.contractNo || '—'}
   496→            </p>
   497→          </div>
   498→        </div>
   499→
   500→        {/* Info Cards */}
   501→        <div className="grid grid-cols-2 sm:grid-cols-5 gap-3">
   502→          <Card><CardContent className="p-3"><p className="text-xs text-muted-foreground">{t(labels.client.ar, labels.client.en)}</p><p className="text-sm font-medium truncate">{timesheet.clientName || timesheet.rental?.client?.name || '—'}</p></CardContent></Card>
   503→          <Card><CardContent className="p-3"><p className="text-xs text-muted-foreground">{t(labels.project.ar, labels.project.en)}</p><p className="text-sm font-medium truncate">{timesheet.project?.name || '—'}</p></CardContent></Card>
   504→          <Card><CardContent className="p-3"><p className="text-xs text-muted-foreground">{t(labels.equipment.ar, labels.equipment.en)}</p><p className="text-sm font-medium truncate">{timesheet.equipment?.name || '—'}</p></CardContent></Card>
   505→          <Card><CardContent className="p-3"><p className="text-xs text-muted-foreground">{t(labels.operatingHours.ar, labels.operatingHours.en)}</p><p className="text-sm font-medium">{formatNumber(timesheet.operatingHours)} {t('ساعة', 'hrs')}</p></CardContent></Card>
   506→          <Card><CardContent className="p-3"><p className="text-xs text-muted-foreground">{t(labels.deliveryMonth.ar, labels.deliveryMonth.en)}</p><p className="text-sm font-medium">{formatMonthYear(timesheet.month, timesheet.year, lang)}</p></CardContent></Card>
   507→        </div>
   508→
   509→        {/* Billing Details */}
   510→        <Card>
   511→          <CardHeader><CardTitle className="text-lg">{t('تفاصيل الفوترة', 'Billing Details')}</CardTitle></CardHeader>
   512→          <CardContent className="space-y-2">
   513→            <div className="flex justify-between text-sm">
   514→              <span>{t(labels.operatingHours.ar, labels.operatingHours.en)}</span>
   515→              <span className="font-medium">{formatNumber(timesheet.operatingHours)} {t('ساعة', 'hours')}</span>
   516→            </div>
   517→            <div className="flex justify-between text-sm">
   518→              <span>{t(labels.hourlyRate.ar, labels.hourlyRate.en)} ({t('من العقد', 'From Contract')})</span>
   519→              <span className="font-medium"><MoneyDisplay value={hourlyRate} lang={lang} size="sm" inline /></span>
   520→            </div>
   521→            <Separator />
   522→            <div className="flex justify-between text-sm">
   523→              <span>{t('المجموع الفرعي', 'Subtotal')}</span>
   524→              <span className="font-medium"><MoneyDisplay value={subtotal} lang={lang} size="sm" inline /></span>
   525→            </div>
   526→            <div className="flex justify-between text-sm">
   527→              <span>{t(labels.vat.ar, labels.vat.en)}</span>
   528→              <span className="font-medium"><MoneyDisplay value={vatAmount} lang={lang} size="sm" inline /></span>
   529→            </div>
   530→            <Separator />
   531→            <div className="flex justify-between text-lg font-bold">
   532→              <span>{t('الإجمالي', 'Total')}</span>
   533→              <span className="text-emerald-700"><MoneyDisplay value={totalAmount} lang={lang} size="lg" inline bold /></span>
   534→            </div>
   535→          </CardContent>
   536→        </Card>
   537→
   538→        {timesheet.notes && (
   539→          <Card>
   540→            <CardContent className="p-4">
   541→              <p className="text-sm text-muted-foreground mb-1">{t(labels.notes.ar, labels.notes.en)}</p>
   542→              <p className="text-sm">{timesheet.notes}</p>
   543→            </CardContent>
   544→          </Card>

   546→

   548→        {/* RULE: Once INVOICED, timesheet cannot be modified */}
   549→        <Card className="border-emerald-200 bg-emerald-50/50">
   550→          <CardContent className="p-4">
   551→            <div className="flex flex-wrap items-center gap-3">
   552→              <span className="text-sm font-semibold">{t('إجراءات سير العمل:', 'Workflow Actions:')}</span>
   553→              {timesheet.status === 'DRAFT' && (
   554→                <Button className="gap-2 bg-blue-600 hover:bg-blue-700" onClick={() => statusMutation.mutate({ id: timesheet.id, status: 'SUBMITTED' })} disabled={statusMutation.isPending}>
   555→                  <Send className="size-4" /> {t(labels.submit.ar, labels.submit.en)}
   556→                </Button>
   557→              )}
   558→              {timesheet.status === 'SUBMITTED' && (
   559→                <Button className="gap-2 bg-emerald-600 hover:bg-emerald-700" onClick={() => statusMutation.mutate({ id: timesheet.id, status: 'APPROVED' })} disabled={statusMutation.isPending}>
   560→                  <CheckCircle className="size-4" /> {t(labels.approve.ar, labels.approve.en)}
   561→                </Button>
   562→              )}
   563→              {timesheet.status === 'APPROVED' && (
   564→                <p className="text-sm text-emerald-700 font-medium">{t('جاهز للفاتورة - يمكن إنشاء فاتورة إيجار من وحدة فواتير الإيجار', 'Ready for invoicing - Create rental invoice from Rental Invoices module')}</p>
   565→              )}
   566→              {isInvoiced && (
   567→                <Badge className="bg-purple-100 text-purple-700 border-purple-200 text-sm px-3 py-1">
   568→                  <FileText className="size-4 ml-1" />
   569→                  {timesheet.invoice ? `${t(labels.invoiceGenerated.ar, labels.invoiceGenerated.en)} - ${timesheet.invoice.invoiceNo}` : t(labels.invoiceGenerated.ar, labels.invoiceGenerated.en)}
   570→                </Badge>
   571→              )}
   572→              {isInvoiced && (
   573→                <p className="text-xs text-muted-foreground">{t(labels.cannotModifyInvoiced.ar, labels.cannotModifyInvoiced.en)}</p>
   574→              )}
   575→            </div>
   576→          </CardContent>
   577→        </Card>
   578→      </div>
   579→    )
   580→  }
   581→
   582→  // ============ LIST VIEW ============
   583→  return (
   584→    <ModuleLayout
   585→      title={labels.title}
   586→      subtitle={labels.subtitle}
   587→      actions={
   588→        <>
   589→          <Button variant="outline" size="icon" onClick={() => refetch()} title={t('تحديث', 'Refresh')}>
   590→            <RefreshCw className="size-4" />
   591→          </Button>
   592→          <Button className="gap-2 bg-emerald-600 hover:bg-emerald-700" onClick={() => setViewState({ type: 'create' })}>
   593→            <Plus className="size-4" /> {t(labels.newTimesheet.ar, labels.newTimesheet.en)}
   594→          </Button>
   595→        </>
   596→      }
   597→    >
   598→      {/* Summary Cards */}
   599→      <div className="grid grid-cols-2 sm:grid-cols-5 gap-3">
   600→        <Card className="bg-gray-50 border-gray-200">
   601→          <CardContent className="p-4">
   602→            <p className="text-sm text-gray-500">{t('إجمالي السجلات', 'Total')}</p>
   603→            <p className="text-xl font-bold text-gray-700">{formatNumber(totalTimesheets)}</p>
   604→          </CardContent>
   605→        </Card>
   606→        <Card className="bg-gray-50 border-gray-200">
   607→          <CardContent className="p-4">
   608→            <p className="text-sm text-gray-500">{t('مسودة', 'Draft')}</p>
   609→            <p className="text-xl font-bold text-gray-700">{formatNumber(draftCount)}</p>
   610→          </CardContent>
   611→        </Card>
   612→        <Card className="bg-blue-50 border-blue-200">
   613→          <CardContent className="p-4">
   614→            <p className="text-sm text-blue-500">{t('مقدمة', 'Submitted')}</p>
   615→            <p className="text-xl font-bold text-blue-700">{formatNumber(submittedCount)}</p>
   616→          </CardContent>
   617→        </Card>
   618→        <Card className="bg-emerald-50 border-emerald-200">
   619→          <CardContent className="p-4">
   620→            <p className="text-sm text-emerald-500">{t('معتمدة', 'Approved')}</p>
   621→            <p className="text-xl font-bold text-emerald-700">{formatNumber(approvedCount)}</p>
   622→          </CardContent>
   623→        </Card>
   624→        <Card className="bg-purple-50 border-purple-200">
   625→          <CardContent className="p-4">
   626→            <p className="text-sm text-purple-500">{t('مفوترة', 'Invoiced')}</p>
   627→            <p className="text-xl font-bold text-purple-700">{formatNumber(invoicedCount)}</p>
   628→          </CardContent>
   629→        </Card>
   630→      </div>
   631→
   632→      {/* Filters */}
   633→      <Card>
   634→        <CardContent className="p-4">
   635→          <div className="flex flex-col sm:flex-row gap-3">
   636→            <div className="relative flex-1">
   637→              <Search className="absolute right-3 top-1/2 -translate-y-1/2 size-4 text-muted-foreground" />
   638→              <Input placeholder={t(labels.search.ar, labels.search.en)} value={search} onChange={e => setSearch(e.target.value)} className="pr-9" />
   639→            </div>
   640→            <Select value={statusFilter} onValueChange={setStatusFilter}>
   641→              <SelectTrigger className="w-full sm:w-40"><SelectValue placeholder={t(labels.allStatus.ar, labels.allStatus.en)} /></SelectTrigger>
   642→              <SelectContent>
   643→                <SelectItem value="all">{t(labels.allStatus.ar, labels.allStatus.en)}</SelectItem>
   644→                <SelectItem value="DRAFT">{t('مسودة', 'Draft')}</SelectItem>
   645→                <SelectItem value="SUBMITTED">{t('مقدمة', 'Submitted')}</SelectItem>
   646→                <SelectItem value="APPROVED">{t('معتمدة', 'Approved')}</SelectItem>
   647→                <SelectItem value="INVOICED">{t('مفوترة', 'Invoiced')}</SelectItem>
   648→              </SelectContent>
   649→            </Select>
   650→          </div>
   651→        </CardContent>
   652→      </Card>
   653→
   654→      {/* Table */}
   655→      <Card>
   656→        <CardContent className="p-0">
   657→          {isLoading ? (
   658→            <TableSkeleton />
   659→          ) : isError ? (
   660→            <div className="flex flex-col items-center gap-3 py-10">
   661→              <p className="text-rose-600">{commonText.error[lang]}</p>
   662→              <Button variant="outline" onClick={() => refetch()}>{commonText.retry[lang]}</Button>
   663→            </div>
   664→          ) : filtered.length === 0 ? (
   665→            <div className="flex flex-col items-center gap-3 py-10">
   666→              <Clock className="size-12 text-gray-300" />
   667→              <p className="text-muted-foreground">{t(labels.noTimesheets.ar, labels.noTimesheets.en)}</p>
   668→              <Button className="bg-emerald-600 hover:bg-emerald-700" onClick={() => setViewState({ type: 'create' })}>
   669→                <Plus className="size-4 mr-1" /> {t(labels.newTimesheet.ar, labels.newTimesheet.en)}
   670→              </Button>
   671→            </div>
   672→          ) : (
   673→            <div className="overflow-x-auto">
   674→              <Table>
   675→                <TableHeader>
   676→                  <TableRow>
   677→                    <TableHead className="text-right">{t(labels.deliveryMonth.ar, labels.deliveryMonth.en)}</TableHead>
   678→                    <TableHead className="text-right">{t(labels.project.ar, labels.project.en)}</TableHead>
   679→                    <TableHead className="text-right">{t(labels.equipment.ar, labels.equipment.en)}</TableHead>
   680→                    <TableHead className="text-right">{t(labels.operatingHours.ar, labels.operatingHours.en)}</TableHead>
   681→                    <TableHead className="text-right">{t(labels.hourlyRate.ar, labels.hourlyRate.en)}</TableHead>
   682→                    <TableHead className="text-right">{t(labels.billingAmount.ar, labels.billingAmount.en)}</TableHead>
   683→                    <TableHead className="text-right">{t(labels.status.ar, labels.status.en)}</TableHead>
   684→                    <TableHead className="text-right">{t(labels.actions.ar, labels.actions.en)}</TableHead>
   685→                  </TableRow>
   686→                </TableHeader>
   687→                <TableBody>
   688→                  {filtered.map(ts => {
   689→                    const rate = ts.rental?.hourlyRate || ts.contract?.hourlyRate || 0
   690→                    const amount = ts.operatingHours * rate
   691→                    return (
   692→                      <TableRow key={ts.id} className="cursor-pointer hover:bg-emerald-50/50" onClick={() => setViewState({ type: 'detail', timesheetId: ts.id })}>
   693→                        <TableCell className="font-medium">{formatMonthYear(ts.month, ts.year, lang)}</TableCell>
   694→                        <TableCell>{ts.project?.name || '—'}</TableCell>
   695→                        <TableCell>{ts.equipment?.name || '—'}</TableCell>
   696→                        <TableCell>{formatNumber(ts.operatingHours)}</TableCell>
   697→                        <TableCell><MoneyDisplay value={rate} lang={lang} size="sm" inline /></TableCell>
   698→                        <TableCell className="font-semibold"><MoneyDisplay value={amount} lang={lang} size="sm" inline bold /></TableCell>
   699→                        <TableCell><StatusBadge status={ts.status} lang={lang} /></TableCell>
   700→                        <TableCell>
   701→                          <div className="flex items-center gap-1" onClick={e => e.stopPropagation()}>
   702→                            <PrintButton type="timesheet-report" documentId={ts.id} size="icon" />
   703→                            <Button variant="ghost" size="icon" className="size-8" onClick={() => setViewState({ type: 'detail', timesheetId: ts.id })} title={t('عرض', 'View')}>
   704→                              <Eye className="size-4" />
   705→                            </Button>
   706→                            {/* RULE: Only DRAFT can be deleted; Once INVOICED, cannot be modified */}
   707→                            {ts.status === 'DRAFT' && (
   708→                              <Button variant="ghost" size="icon" className="size-8 text-rose-500" onClick={() => setDeleteId(ts.id)} title={t('حذف', 'Delete')}>
   709→                                <Trash2 className="size-4" />
   710→                              </Button>
   711→                            )}
   712→                          </div>
   713→                        </TableCell>
   714→                      </TableRow>
   715→                    )
   716→                  })}
   717→                </TableBody>
   718→              </Table>
   719→            </div>
   720→          )}
   721→        </CardContent>
   722→      </Card>
   723→
   724→      {/* Delete Confirmation */}
   725→      <AlertDialog open={!!deleteId} onOpenChange={() => setDeleteId(null)}>
   726→        <AlertDialogContent>
   727→          <AlertDialogHeader>
   728→            <AlertDialogTitle>{t(labels.deleteTitle.ar, labels.deleteTitle.en)}</AlertDialogTitle>
   729→            <AlertDialogDescription>{t(labels.deleteConfirm.ar, labels.deleteConfirm.en)}</AlertDialogDescription>
   730→          </AlertDialogHeader>
   731→          <AlertDialogFooter>
   732→            <AlertDialogCancel>{commonText.cancel[lang]}</AlertDialogCancel>
   733→            <AlertDialogAction className="bg-rose-600 hover:bg-rose-700" onClick={() => deleteId && deleteMutation.mutate(deleteId)}>
   734→              {commonText.delete[lang]}
   735→            </AlertDialogAction>
   736→          </AlertDialogFooter>
   737→        </AlertDialogContent>
   738→      </AlertDialog>
   739→    </ModuleLayout>
   740→  )

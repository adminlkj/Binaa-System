   103→  const { lang } = useAppStore()
   104→  const queryClient = useQueryClient()

   106→  const [purchaseRequestId, setPurchaseRequestId] = useState('')
   107→  const [supplierId, setSupplierId] = useState('')
   108→  const [projectId, setProjectId] = useState('')
   109→  const [date, setDate] = useState(new Date().toISOString().split('T')[0])
   110→  const [deliveryDate, setDeliveryDate] = useState('')
   111→  const [notes, setNotes] = useState('')
   112→  const [lineItems, setLineItems] = useState<LineItemForm[]>([{ ...defaultLineItem }])
   113→
   114→  const { data: suppliers = [] } = useQuery<SupplierOption[]>({
   115→    queryKey: ['suppliers-simple'],

   124→  const { data: projects = [] } = useQuery<ProjectOption[]>({

   326→// ============ Detail View ============
   327→function PODetailView({ id, onBack }: { id: string; onBack: () => void }) {
   328→  const { lang } = useAppStore()
   329→  const queryClient = useQueryClient()
   330→
   331→  const { data: order, isLoading, isError } = useQuery<PurchaseOrder>({
   332→    queryKey: ['purchase-orders', id],
   333→    queryFn: async () => {
   334→      const res = await fetch(`/api/purchase-orders/${id}`)
   335→      if (!res.ok) throw new Error()
   336→      return res.json()
   337→    },
   338→    enabled: !!id,
   339→  })
   340→
   341→  // Approve mutation: DRAFT → PENDING_APPROVAL → APPROVED
   342→  const approveMutation = useMutation({
   343→    mutationFn: (targetStatus: string) => fetch(`/api/purchase-orders/${id}`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ status: targetStatus }) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   344→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['purchase-orders', id] }); queryClient.invalidateQueries({ queryKey: ['purchase-orders'] }); toast.success(t('تم الاعتماد بنجاح', 'Approved successfully', lang)) },
   345→    onError: () => toast.error(t('فشل في الاعتماد', 'Failed to approve', lang)),
   346→  })
   347→
   348→  const deleteMutation = useMutation({
   349→    mutationFn: () => fetch(`/api/purchase-orders/${id}`, { method: 'DELETE' }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   350→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['purchase-orders'] }); toast.success(t('تم الحذف', 'Deleted', lang)); onBack() },
   351→    onError: () => toast.error(t('فشل في الحذف', 'Failed to delete', lang)),
   352→  })
   353→
   354→  if (isLoading) return <div className="p-6"><TableSkeleton /></div>
   355→  if (isError || !order) return (
   356→    <div className="flex flex-col items-center gap-3 py-10">
   357→      <AlertCircle className="size-12 text-rose-300" />
   358→      <p className="text-rose-600">{t('حدث خطأ أثناء تحميل البيانات', 'Error loading data', lang)}</p>
   359→      <Button variant="outline" onClick={onBack}>{t('رجوع', 'Go Back', lang)}</Button>
   360→    </div>
   361→  )
   362→
   363→  const statusCfg = poStatusConfig[order.status] || poStatusConfig.DRAFT
   364→
   365→  return (
   366→    <div className="space-y-6">
   367→      <div className="flex items-center gap-3">
   368→        <Button variant="outline" size="icon" onClick={onBack}><ArrowRight className="size-4" /></Button>
   369→        <div className="flex-1">
   370→          <div className="flex items-center gap-3">
   371→            <h2 className="text-xl font-bold">{t('أمر شراء', 'Purchase Order', lang)} {order.orderNo}</h2>
   372→            <Badge className={`${statusCfg.bg} ${statusCfg.color} border-0`}>{statusCfg.label[lang]}</Badge>
   373→          </div>
   374→          <p className="text-sm text-muted-foreground">{order.supplier.name}</p>
   375→        </div>
   376→        {/* Approval buttons */}
   377→        {order.status === 'DRAFT' && (
   378→          <Button className="gap-2 bg-orange-600 hover:bg-orange-700" onClick={() => approveMutation.mutate('PENDING_APPROVAL')} disabled={approveMutation.isPending}>
   379→            <Clock className="size-4" /> {t('إرسال للاعتماد', 'Submit for Approval', lang)}
   380→          </Button>
   381→        )}
   382→        {order.status === 'PENDING_APPROVAL' && (
   383→          <Button className="gap-2 bg-emerald-600 hover:bg-emerald-700" onClick={() => approveMutation.mutate('APPROVED')} disabled={approveMutation.isPending}>
   384→            <CheckCircle className="size-4" /> {t('اعتماد', 'Approve', lang)}
   385→          </Button>
   386→        )}
   387→        {order.status === 'DRAFT' && (
   388→          <Button variant="ghost" size="icon" className="size-8 text-rose-600" onClick={() => { if (confirm(t('هل أنت متأكد من الحذف؟', 'Are you sure?', lang))) deleteMutation.mutate() }}><Trash2 className="size-4" /></Button>
   389→        )}
   390→        <PrintButton type="purchase-order" documentId={order.id} />
   391→      </div>
   392→
   393→      {/* Info Cards */}
   394→      <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   395→        <Card><CardContent className="p-3">
   396→          <p className="text-xs text-muted-foreground">{t('المورد', 'Supplier', lang)}</p>
   397→          <p className="text-sm font-medium truncate">{order.supplier.name}</p>
   398→        </CardContent></Card>
   399→        <Card><CardContent className="p-3">
   400→          <p className="text-xs text-muted-foreground">{t('المشروع', 'Project', lang)}</p>
   401→          <p className="text-sm font-medium truncate">{order.project?.name || '—'}</p>
   402→        </CardContent></Card>

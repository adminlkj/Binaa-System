   380→// ============ Main Expenses Module ============
   381→export function ExpensesModule() {
   382→  const { lang } = useAppStore()
   383→
   384→  const [activeTab, setActiveTab] = useState<'project' | 'admin'>('project')
   385→  const [search, setSearch] = useState('')
   386→  const [projectFilter, setProjectFilter] = useState<string>('all')
   387→  const [categoryFilter, setCategoryFilter] = useState<string>('all')
   388→  const [dialogOpen, setDialogOpen] = useState(false)
   389→
   390→  const { data: expenses = [], isLoading, isError, refetch } = useQuery<Expense[]>({
   391→    queryKey: ['expenses'],
   392→    queryFn: async () => {
   393→      const res = await fetch('/api/expenses')
   394→      if (!res.ok) throw new Error('Failed to fetch')
   395→      return res.json()
   396→    },
   397→  })
   398→
   399→  const { data: projects = [] } = useQuery<ProjectOption[]>({
   400→    queryKey: ['projects-list'],
   401→    queryFn: async () => {
   402→      const res = await fetch('/api/projects/list')
   403→      if (!res.ok) return []
   404→      return res.json()
   405→    },
   406→  })
   407→
   408→  const { data: suppliers = [] } = useQuery<{ id: string; code: string; name: string; nameAr?: string | null }[]>({
   409→    queryKey: ['suppliers-for-expenses'],
   410→    queryFn: async () => {
   411→      const res = await fetch('/api/suppliers')
   412→      if (!res.ok) return []
   413→      return res.json()
   414→    },
   415→  })
   416→
   417→  // Split expenses by type
   418→  const projectExpenses = useMemo(() => expenses.filter(isProjectExpense), [expenses])
   419→  const adminExpenses = useMemo(() => expenses.filter(isAdminExpense), [expenses])
   420→
   421→  // Filter based on active tab
   422→  const tabExpenses = activeTab === 'project' ? projectExpenses : adminExpenses
   423→
   424→  const filtered = tabExpenses.filter(exp => {
   425→    const matchProject = projectFilter === 'all' || exp.projectId === projectFilter
   426→    const matchCategory = categoryFilter === 'all' || exp.category === categoryFilter
   427→    const matchSearch = !search ||
   428→      exp.description.toLowerCase().includes(search.toLowerCase()) ||
   429→      (exp.project?.name || '').toLowerCase().includes(search.toLowerCase()) ||
   430→      (allCategoryLabels[exp.category]?.[lang] || '').toLowerCase().includes(search.toLowerCase())
   431→    return matchProject && matchCategory && matchSearch
   432→  })
   433→
   434→  // Summary
   435→  const totalProjectExpenses = projectExpenses.reduce((s, e) => s + Number(e.amount || 0), 0)
   436→  const totalAdminExpenses = adminExpenses.reduce((s, e) => s + Number(e.amount || 0), 0)

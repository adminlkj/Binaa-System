   366→// ============ Main Timesheets Module ============
   367→export function TimesheetsModule() {
   368→  const { lang } = useAppStore()
   369→  const queryClient = useQueryClient()
   370→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   371→
   372→  const [viewState, setViewState] = useState<ViewState>({ type: 'list' })
   373→  const [search, setSearch] = useState('')
   374→  const [statusFilter, setStatusFilter] = useState<string>('all')
   375→  const [deleteId, setDeleteId] = useState<string | null>(null)
   376→
   377→  // Fetch timesheets
   378→  const { data: timesheets = [], isLoading, isError, refetch } = useQuery<TimesheetItem[]>({
   379→    queryKey: ['equipment-timesheets'],
   380→    queryFn: async () => {
   381→      const res = await fetch('/api/equipment/timesheets')
   382→      if (!res.ok) throw new Error('Failed to fetch')
   383→      return res.json()
   384→    },
   385→  })
   386→
   387→  // Fetch active rental contracts for create form
   388→  const { data: rentalContracts = [] } = useQuery<RentalContractOption[]>({
   389→    queryKey: ['rental-contracts-for-timesheets'],
   390→    queryFn: async () => {
   391→      const res = await fetch('/api/equipment/rental-contracts')
   392→      if (!res.ok) return []
   393→      const data = await res.json()
   394→      // Only active contracts
   395→      return (Array.isArray(data) ? data : []).filter((c: { status: string }) => c.status === 'ACTIVE')
   396→    },
   397→  })
   398→
   399→  // Delete mutation
   400→  const deleteMutation = useMutation({
   401→    mutationFn: (id: string) =>
   402→      fetch(`/api/equipment/timesheets/${id}`, { method: 'DELETE' }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   403→    onSuccess: () => {
   404→      queryClient.invalidateQueries({ queryKey: ['equipment-timesheets'] })
   405→      setDeleteId(null)
   406→    },
   407→  })
   408→
   409→  // Status workflow mutation
   410→  const statusMutation = useMutation({
   411→    mutationFn: ({ id, status }: { id: string; status: string }) =>
   412→      fetch(`/api/equipment/timesheets/${id}`, {
   413→        method: 'PUT',
   414→        headers: { 'Content-Type': 'application/json' },
   415→        body: JSON.stringify({ status }),
   416→      }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   417→    onSuccess: () => {
   418→      queryClient.invalidateQueries({ queryKey: ['equipment-timesheets'] })
   419→    },
   420→  })
   421→
   422→  // Filters
   423→  const filtered = timesheets.filter(ts => {

   436→
   437→  // Summary
   438→  const totalTimesheets = timesheets.length
   439→  const draftCount = timesheets.filter(ts => ts.status === 'DRAFT').length
   440→  const submittedCount = timesheets.filter(ts => ts.status === 'SUBMITTED').length
   441→  const approvedCount = timesheets.filter(ts => ts.status === 'APPROVED').length

   443→
   444→  // ============ CREATE VIEW ============
   445→  if (viewState.type === 'create') {

   474→        <div className="flex items-center gap-3">

   476→            <ArrowRight className="size-4" />
   477→          </Button>
   478→          <div className="flex-1">
   479→            <div className="flex items-center gap-3">
   480→              <h2 className="text-xl font-bold">
   481→                {t('سجل ساعات عمل', 'Timesheet')} - {formatMonthYear(timesheet.month, timesheet.year, lang)}
   482→              </h2>
   483→              <StatusBadge status={timesheet.status} lang={lang} />
   484→            </div>
   485→            <p className="text-sm text-muted-foreground">
   486→              {timesheet.project?.name || '—'} - {timesheet.contract?.contractNo || '—'}
   487→            </p>
   488→          </div>
   489→        </div>
   490→
   491→        {/* Info Cards */}
   492→        <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   493→          <Card><CardContent className="p-3"><p className="text-xs text-muted-foreground">{t(labels.project.ar, labels.project.en)}</p><p className="text-sm font-medium truncate">{timesheet.project?.name || '—'}</p></CardContent></Card>
   494→          <Card><CardContent className="p-3"><p className="text-xs text-muted-foreground">{t(labels.equipment.ar, labels.equipment.en)}</p><p className="text-sm font-medium truncate">{timesheet.equipment?.name || '—'}</p></CardContent></Card>
   495→          <Card><CardContent className="p-3"><p className="text-xs text-muted-foreground">{t(labels.operatingHours.ar, labels.operatingHours.en)}</p><p className="text-sm font-medium">{formatNumber(timesheet.operatingHours)} {t('ساعة', 'hrs')}</p></CardContent></Card>
   496→          <Card><CardContent className="p-3"><p className="text-xs text-muted-foreground">{t(labels.deliveryMonth.ar, labels.deliveryMonth.en)}</p><p className="text-sm font-medium">{formatMonthYear(timesheet.month, timesheet.year, lang)}</p></CardContent></Card>
   497→        </div>
   498→
   499→        {/* Billing Details */}
   500→        <Card>
   501→          <CardHeader><CardTitle className="text-lg">{t('تفاصيل الفوترة', 'Billing Details')}</CardTitle></CardHeader>
   502→          <CardContent className="space-y-2">
   503→            <div className="flex justify-between text-sm">
   504→              <span>{t(labels.operatingHours.ar, labels.operatingHours.en)}</span>
   505→              <span className="font-medium">{formatNumber(timesheet.operatingHours)} {t('ساعة', 'hours')}</span>
   506→            </div>
   507→            <div className="flex justify-between text-sm">
   508→              <span>{t(labels.hourlyRate.ar, labels.hourlyRate.en)} ({t('من العقد', 'From Contract')})</span>

   520→            <Separator />
   521→            <div className="flex justify-between text-lg font-bold">

   525→          </CardContent>
   526→        </Card>
   527→
   528→        {timesheet.notes && (
   529→          <Card>
   530→            <CardContent className="p-4">
   531→              <p className="text-sm text-muted-foreground mb-1">{t(labels.notes.ar, labels.notes.en)}</p>

   553→              {timesheet.status === 'APPROVED' && (
   554→                <p className="text-sm text-emerald-700 font-medium">{t('جاهز للفاتورة - يمكن إنشاء فاتورة إيجار من وحدة فواتير الإيجار', 'Ready for invoicing - Create rental invoice from Rental Invoices module')}</p>

   557→                <Badge className="bg-purple-100 text-purple-700 border-purple-200 text-sm px-3 py-1">
   558→                  <FileText className="size-4 ml-1" />
   559→                  {timesheet.invoice ? `${t(labels.invoiceGenerated.ar, labels.invoiceGenerated.en)} - ${timesheet.invoice.invoiceNo}` : t(labels.invoiceGenerated.ar, labels.invoiceGenerated.en)}

   562→              {isInvoiced && (
   563→                <p className="text-xs text-muted-foreground">{t(labels.cannotModifyInvoiced.ar, labels.cannotModifyInvoiced.en)}</p>
   564→              )}
   565→            </div>
   566→          </CardContent>
   567→        </Card>
   568→      </div>
   569→    )
   570→  }
   571→
   572→  // ============ LIST VIEW ============
   573→  return (
   574→    <ModuleLayout
   575→      title={labels.title}
   576→      subtitle={labels.subtitle}
   577→      actions={
   578→        <>
   579→          <PrintButton type="timesheet-report" size="icon" />
   580→          <Button variant="outline" size="icon" onClick={() => refetch()} title={t('تحديث', 'Refresh')}>
   581→            <RefreshCw className="size-4" />
   582→          </Button>
   583→          <Button className="gap-2 bg-emerald-600 hover:bg-emerald-700" onClick={() => setViewState({ type: 'create' })}>
   584→            <Plus className="size-4" /> {t(labels.newTimesheet.ar, labels.newTimesheet.en)}
   585→          </Button>
   586→        </>
   587→      }
   588→    >
   589→      {/* Summary Cards */}
   590→      <div className="grid grid-cols-2 sm:grid-cols-5 gap-3">
   591→        <Card className="bg-gray-50 border-gray-200">
   592→          <CardContent className="p-4">
   593→            <p className="text-sm text-gray-500">{t('إجمالي السجلات', 'Total')}</p>
   594→            <p className="text-xl font-bold text-gray-700">{formatNumber(totalTimesheets)}</p>
   595→          </CardContent>
   596→        </Card>
   597→        <Card className="bg-gray-50 border-gray-200">
   598→          <CardContent className="p-4">
   599→            <p className="text-sm text-gray-500">{t('مسودة', 'Draft')}</p>
   600→            <p className="text-xl font-bold text-gray-700">{formatNumber(draftCount)}</p>
   601→          </CardContent>
   602→        </Card>
   603→        <Card className="bg-blue-50 border-blue-200">
   604→          <CardContent className="p-4">
   605→            <p className="text-sm text-blue-500">{t('مقدمة', 'Submitted')}</p>
   606→            <p className="text-xl font-bold text-blue-700">{formatNumber(submittedCount)}</p>
   607→          </CardContent>
   608→        </Card>
   609→        <Card className="bg-emerald-50 border-emerald-200">
   610→          <CardContent className="p-4">
   611→            <p className="text-sm text-emerald-500">{t('معتمدة', 'Approved')}</p>
   612→            <p className="text-xl font-bold text-emerald-700">{formatNumber(approvedCount)}</p>
   613→          </CardContent>
   614→        </Card>
   615→        <Card className="bg-purple-50 border-purple-200">
   616→          <CardContent className="p-4">
   617→            <p className="text-sm text-purple-500">{t('مفوترة', 'Invoiced')}</p>
   618→            <p className="text-xl font-bold text-purple-700">{formatNumber(invoicedCount)}</p>
   619→          </CardContent>
   620→        </Card>
   621→      </div>
   622→
   623→      {/* Filters */}
   624→      <Card>
   625→        <CardContent className="p-4">
   626→          <div className="flex flex-col sm:flex-row gap-3">
   627→            <div className="relative flex-1">
   628→              <Search className="absolute right-3 top-1/2 -translate-y-1/2 size-4 text-muted-foreground" />
   629→              <Input placeholder={t(labels.search.ar, labels.search.en)} value={search} onChange={e => setSearch(e.target.value)} className="pr-9" />
   630→            </div>
   631→            <Select value={statusFilter} onValueChange={setStatusFilter}>
   632→              <SelectTrigger className="w-full sm:w-40"><SelectValue placeholder={t(labels.allStatus.ar, labels.allStatus.en)} /></SelectTrigger>
   633→              <SelectContent>
   634→                <SelectItem value="all">{t(labels.allStatus.ar, labels.allStatus.en)}</SelectItem>
   635→                <SelectItem value="DRAFT">{t('مسودة', 'Draft')}</SelectItem>
   636→                <SelectItem value="SUBMITTED">{t('مقدمة', 'Submitted')}</SelectItem>
   637→                <SelectItem value="APPROVED">{t('معتمدة', 'Approved')}</SelectItem>
   638→                <SelectItem value="INVOICED">{t('مفوترة', 'Invoiced')}</SelectItem>
   639→              </SelectContent>
   640→            </Select>
   641→          </div>
   642→        </CardContent>
   643→      </Card>
   644→
   645→      {/* Table */}
   646→      <Card>
   647→        <CardContent className="p-0">
   648→          {isLoading ? (
   649→            <TableSkeleton />
   650→          ) : isError ? (
   651→            <div className="flex flex-col items-center gap-3 py-10">
   652→              <p className="text-rose-600">{commonText.error[lang]}</p>
   653→              <Button variant="outline" onClick={() => refetch()}>{commonText.retry[lang]}</Button>
   654→            </div>
   655→          ) : filtered.length === 0 ? (
   656→            <div className="flex flex-col items-center gap-3 py-10">
   657→              <Clock className="size-12 text-gray-300" />
   658→              <p className="text-muted-foreground">{t(labels.noTimesheets.ar, labels.noTimesheets.en)}</p>
   659→              <Button className="bg-emerald-600 hover:bg-emerald-700" onClick={() => setViewState({ type: 'create' })}>
   660→                <Plus className="size-4 mr-1" /> {t(labels.newTimesheet.ar, labels.newTimesheet.en)}
   661→              </Button>
   662→            </div>
   663→          ) : (
   664→            <div className="overflow-x-auto">
   665→              <Table>
   666→                <TableHeader>
   667→                  <TableRow>
   668→                    <TableHead className="text-right">{t(labels.deliveryMonth.ar, labels.deliveryMonth.en)}</TableHead>
   669→                    <TableHead className="text-right">{t(labels.project.ar, labels.project.en)}</TableHead>
   670→                    <TableHead className="text-right">{t(labels.equipment.ar, labels.equipment.en)}</TableHead>
   671→                    <TableHead className="text-right">{t(labels.operatingHours.ar, labels.operatingHours.en)}</TableHead>
   672→                    <TableHead className="text-right">{t(labels.hourlyRate.ar, labels.hourlyRate.en)}</TableHead>
   673→                    <TableHead className="text-right">{t(labels.billingAmount.ar, labels.billingAmount.en)}</TableHead>
   674→                    <TableHead className="text-right">{t(labels.status.ar, labels.status.en)}</TableHead>
   675→                    <TableHead className="text-right">{t(labels.actions.ar, labels.actions.en)}</TableHead>
   676→                  </TableRow>
   677→                </TableHeader>
   678→                <TableBody>
   679→                  {filtered.map(ts => {
   680→                    const rate = ts.rental?.rate || ts.contract?.hourlyRate || 0
   681→                    const amount = ts.operatingHours * rate
   682→                    return (
   683→                      <TableRow key={ts.id} className="cursor-pointer hover:bg-emerald-50/50" onClick={() => setViewState({ type: 'detail', timesheetId: ts.id })}>
   684→                        <TableCell className="font-medium">{formatMonthYear(ts.month, ts.year, lang)}</TableCell>
   685→                        <TableCell>{ts.project?.name || '—'}</TableCell>
   686→                        <TableCell>{ts.equipment?.name || '—'}</TableCell>
   687→                        <TableCell>{formatNumber(ts.operatingHours)}</TableCell>
   688→                        <TableCell><MoneyDisplay value={rate} lang={lang} size="sm" inline /></TableCell>
   689→                        <TableCell className="font-semibold"><MoneyDisplay value={amount} lang={lang} size="sm" inline bold /></TableCell>
   690→                        <TableCell><StatusBadge status={ts.status} lang={lang} /></TableCell>
   691→                        <TableCell>
   692→                          <div className="flex items-center gap-1" onClick={e => e.stopPropagation()}>
   693→                            <PrintButton type="timesheet-report" documentId={ts.id} size="icon" />
   694→                            <Button variant="ghost" size="icon" className="size-8" onClick={() => setViewState({ type: 'detail', timesheetId: ts.id })} title={t('عرض', 'View')}>
   695→                              <Eye className="size-4" />
   696→                            </Button>
   697→                            {/* RULE: Only DRAFT can be deleted; Once INVOICED, cannot be modified */}
   698→                            {ts.status === 'DRAFT' && (
   699→                              <Button variant="ghost" size="icon" className="size-8 text-rose-500" onClick={() => setDeleteId(ts.id)} title={t('حذف', 'Delete')}>
   700→                                <Trash2 className="size-4" />
   701→                              </Button>
   702→                            )}
   703→                          </div>
   704→                        </TableCell>
   705→                      </TableRow>
   706→                    )
   707→                  })}
   708→                </TableBody>
   709→              </Table>
   710→            </div>
   711→          )}
   712→        </CardContent>
   713→      </Card>
   714→
   715→      {/* Delete Confirmation */}
   716→      <AlertDialog open={!!deleteId} onOpenChange={() => setDeleteId(null)}>
   717→        <AlertDialogContent>
   718→          <AlertDialogHeader>
   719→            <AlertDialogTitle>{t(labels.deleteTitle.ar, labels.deleteTitle.en)}</AlertDialogTitle>
   720→            <AlertDialogDescription>{t(labels.deleteConfirm.ar, labels.deleteConfirm.en)}</AlertDialogDescription>
   721→          </AlertDialogHeader>
   722→          <AlertDialogFooter>
   723→            <AlertDialogCancel>{commonText.cancel[lang]}</AlertDialogCancel>
   724→            <AlertDialogAction className="bg-rose-600 hover:bg-rose-700" onClick={() => deleteId && deleteMutation.mutate(deleteId)}>
   725→              {commonText.delete[lang]}
   726→            </AlertDialogAction>
   727→          </AlertDialogFooter>
   728→        </AlertDialogContent>

   112→// ============ Service Invoice Creation Form ============
   113→function ServiceInvoiceFormPage({
   114→  clients, projects, onBack,
   115→}: {
   116→  clients: ClientOption[]; projects: ProjectOption[]; onBack: () => void
   117→}) {
   118→  const queryClient = useQueryClient()
   119→  const { lang } = useAppStore()
   120→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   121→
   122→  const [clientId, setClientId] = useState('')
   123→  const [projectId, setProjectId] = useState('')
   124→  const [date, setDate] = useState('')
   125→  const [dueDate, setDueDate] = useState('')
   126→  const [notes, setNotes] = useState('')
   127→  const [paymentTerms, setPaymentTerms] = useState('30 days')
   128→  const [referenceNo, setReferenceNo] = useState('')
   129→  const [contractNo, setContractNo] = useState('')
   130→  const [contractType, setContractType] = useState('')
   131→  const [contractPeriodStart, setContractPeriodStart] = useState('')
   132→  const [contractPeriodEnd, setContractPeriodEnd] = useState('')
   133→  const [includeVat, setIncludeVat] = useState(true)
   134→  const [discountType, setDiscountType] = useState<'none' | 'rate' | 'amount'>('none')
   135→  const [discountRate, setDiscountRate] = useState(0)
   136→  const [discountAmount, setDiscountAmount] = useState(0)
   137→  const [lineItems, setLineItems] = useState<LineItemForm[]>([{ ...defaultLineItem }])
   138→
   139→  // Auto-suggest reference number
   140→  React.useEffect(() => {
   141→    const year = new Date().getFullYear()
   142→    const rand = Math.floor(1000 + Math.random() * 9000)
   143→    setReferenceNo(`SVC-REF-${year}-${rand}`)
   144→  }, [])
   145→
   146→  // Auto-calculate
   147→  const subtotal = useMemo(() => lineItems.reduce((s, i) => s + (i.quantity * i.unitPrice), 0), [lineItems])
   148→  const vatRate = 0.15
   149→  const finalDiscountAmount = useMemo(() => {
   150→    if (discountType === 'rate') return subtotal * (discountRate / 100)
   151→    if (discountType === 'amount') return discountAmount
   152→    return 0
   153→  }, [subtotal, discountType, discountRate, discountAmount])
   154→  const netAmount = subtotal - finalDiscountAmount
   155→  const vatAmount = includeVat ? netAmount * vatRate : 0
   156→  const totalAmount = netAmount + vatAmount
   157→
   158→  const addLine = () => setLineItems([...lineItems, { ...defaultLineItem }])
   159→  const removeLine = (idx: number) => { if (lineItems.length > 1) setLineItems(lineItems.filter((_, i) => i !== idx)) }
   160→  const updateLine = (idx: number, field: keyof LineItemForm, value: string | number) => {
   161→    setLineItems(lineItems.map((item, i) => i === idx ? { ...item, [field]: value } : item))
   162→  }
   163→
   164→  const createMutation = useMutation({
   165→    mutationFn: (data: Record<string, unknown>) =>
   166→      fetch('/api/sales-invoices', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }).then(r => { if (!r.ok) throw new Error(); return r.json() }),
   167→    onSuccess: () => { queryClient.invalidateQueries({ queryKey: ['service-invoices'] }); onBack() },
   168→  })
   169→
   170→  const handleSubmit = (e: React.FormEvent) => {
   171→    e.preventDefault()
   172→    createMutation.mutate({
   173→      clientId, projectId: projectId || null, date, dueDate, notes, paymentTerms,
   174→      invoiceType: 'SERVICE',
   175→      referenceNo: referenceNo || null,
   176→      contractNo: contractNo || null,
   177→      contractType: contractType || null,
   178→      contractPeriodStart: contractPeriodStart || null,
   179→      contractPeriodEnd: contractPeriodEnd || null,
   180→      includeDelivery: false,
   181→      deliveryAmount: 0,
   182→      includeVat,
   183→      discountRate: discountType === 'rate' ? discountRate / 100 : 0,
   184→      discountAmount: finalDiscountAmount,
   185→      items: lineItems.map(i => ({
   186→        description: i.description,
   187→        quantity: i.quantity,
   188→        unitPrice: i.unitPrice,
   189→        unit: i.unit || undefined,
   190→        itemType: 'SERVICE',
   191→      })),
   192→    })
   193→  }
   194→
   195→  return (
   196→    <div className="space-y-6">
   197→      <div className="flex items-center gap-3 no-print">
   198→        <Button variant="outline" size="icon" onClick={onBack}>
   199→          <ArrowRight className="size-4" />
   200→        </Button>
   201→        <div className="flex-1">
   202→          <h1 className="text-2xl font-bold text-gray-900">{t('فاتورة خدمة جديدة', 'New Service Invoice')}</h1>
   203→          <p className="text-sm text-muted-foreground">{t('إنشاء فاتورة خدمات استشارية أو فنية', 'Create consulting or technical service invoice')}</p>
   204→        </div>
   205→      </div>
   206→
   207→      <form onSubmit={handleSubmit} className="space-y-6">
   208→        {/* Section 1: Basic Info */}
   209→        <Card>
   210→          <CardHeader className="pb-4">
   211→            <CardTitle className="text-lg">{t('المعلومات الأساسية', 'Basic Information')}</CardTitle>
   212→          </CardHeader>
   213→          <CardContent className="space-y-4">
   214→            <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   215→              <div className="space-y-2">
   216→                <Label>{t('العميل *', 'Client *')}</Label>
   217→                <Select value={clientId} onValueChange={setClientId}>
   218→                  <SelectTrigger className="w-full"><SelectValue placeholder={t('اختر العميل', 'Select client')} /></SelectTrigger>
   219→                  <SelectContent>
   220→                    {clients.map(c => <SelectItem key={c.id} value={c.id}>{c.name}</SelectItem>)}
   221→                  </SelectContent>
   222→                </Select>
   223→              </div>
   224→              <div className="space-y-2">
   225→                <Label>{t('المشروع', 'Project')}</Label>
   226→                <Select value={projectId} onValueChange={setProjectId}>
   227→                  <SelectTrigger className="w-full"><SelectValue placeholder={t('اختر المشروع', 'Select project')} /></SelectTrigger>
   228→                  <SelectContent>
   229→                    {projects.map(p => <SelectItem key={p.id} value={p.id}>{p.name}</SelectItem>)}
   230→                  </SelectContent>
   231→                </Select>
   232→              </div>
   233→            </div>
   234→            <div className="grid grid-cols-1 sm:grid-cols-4 gap-4">
   235→              <div className="space-y-2">
   236→                <Label>{t('تاريخ الفاتورة *', 'Invoice Date *')}</Label>
   237→                <Input type="date" value={date} onChange={e => setDate(e.target.value)} required />
   238→              </div>
   239→              <div className="space-y-2">
   240→                <Label>{t('تاريخ الاستحقاق *', 'Due Date *')}</Label>
   241→                <Input type="date" value={dueDate} onChange={e => setDueDate(e.target.value)} required />
   242→              </div>
   243→              <div className="space-y-2">
   244→                <Label>{t('شروط السداد', 'Payment Terms')}</Label>
   245→                <Input value={paymentTerms} onChange={e => setPaymentTerms(e.target.value)} />
   246→              </div>
   247→              <div className="space-y-2">
   248→                <Label>{t('رقم المرجع', 'Reference No.')}</Label>
   249→                <Input value={referenceNo} onChange={e => setReferenceNo(e.target.value)} dir="ltr" />
   250→              </div>
   251→            </div>
   252→          </CardContent>
   253→        </Card>
   254→
   255→        {/* Section 2: Contract Data */}
   256→        <Card>
   257→          <CardHeader className="pb-4">
   258→            <CardTitle className="text-lg">{t('بيانات العقد', 'Contract Data')} ({t('اختياري', 'Optional')})</CardTitle>
   259→          </CardHeader>
   260→          <CardContent>
   261→            <div className="grid grid-cols-1 sm:grid-cols-4 gap-4">
   262→              <div className="space-y-2">
   263→                <Label>{t('رقم العقد', 'Contract No.')}</Label>
   264→                <Input value={contractNo} onChange={e => setContractNo(e.target.value)} placeholder="CTR-001" dir="ltr" />
   265→              </div>
   266→              <div className="space-y-2">
   267→                <Label>{t('نوع العقد', 'Contract Type')}</Label>
   268→                <Select value={contractType} onValueChange={setContractType}>
   269→                  <SelectTrigger className="w-full"><SelectValue placeholder={t('اختر النوع', 'Select type')} /></SelectTrigger>
   270→                  <SelectContent>
   271→                    <SelectItem value="Lump Sum">{t('مقطوعية / Lump Sum', 'Lump Sum')}</SelectItem>
   272→                    <SelectItem value="Unit Rate">{t('أسعار وحدوية / Unit Rate', 'Unit Rate')}</SelectItem>
   273→                    <SelectItem value="Time & Materials">{t('وقت ومواد / Time & Materials', 'Time & Materials')}</SelectItem>
   274→                    <SelectItem value="Cost Plus">{t('تكلفة + هامش / Cost Plus', 'Cost Plus')}</SelectItem>
   275→                  </SelectContent>
   276→                </Select>
   277→              </div>
   278→              <div className="space-y-2">
   279→                <Label>{t('بداية فترة العمل', 'Period Start')}</Label>
   280→                <Input type="date" value={contractPeriodStart} onChange={e => setContractPeriodStart(e.target.value)} />
   281→              </div>
   282→              <div className="space-y-2">
   283→                <Label>{t('نهاية فترة العمل', 'Period End')}</Label>
   284→                <Input type="date" value={contractPeriodEnd} onChange={e => setContractPeriodEnd(e.target.value)} />
   285→              </div>
   286→            </div>
   287→          </CardContent>
   288→        </Card>
   289→
   290→        {/* Section 3: VAT */}
   291→        <Card>
   292→          <CardHeader className="pb-4">
   293→            <CardTitle className="text-lg">{t('الضريبة', 'Tax')}</CardTitle>
   294→          </CardHeader>
   295→          <CardContent>
   296→            <div className="p-4 rounded-lg border bg-gray-50">
   297→              <div className="flex items-center justify-between">
   298→                <Label className="text-sm font-semibold">{t('ضريبة القيمة المضافة (15%)', 'VAT (15%)')}</Label>
   299→                <Switch checked={includeVat} onCheckedChange={setIncludeVat} />
   300→              </div>
   301→              <p className="text-xs text-muted-foreground mt-1">
   302→                {includeVat ? t('سيتم احتساب ضريبة القيمة المضافة', 'VAT will be calculated') : t('لن يتم احتساب ضريبة القيمة المضافة', 'VAT will not be calculated')}
   303→              </p>
   304→            </div>
   305→          </CardContent>
   306→        </Card>
   307→
   308→        {/* Section 4: Line Items */}
   309→        <Card>
   310→          <CardHeader className="pb-4">
   311→            <div className="flex items-center justify-between">
   312→              <CardTitle className="text-lg">{t('بنود الخدمة', 'Service Items')}</CardTitle>
   313→              <Button type="button" variant="outline" size="sm" className="gap-1" onClick={addLine}>
   314→                <Plus className="size-3" /> {t('إضافة بند', 'Add Item')}
   315→              </Button>
   316→            </div>
   317→          </CardHeader>
   318→          <CardContent>
   319→            <div className="space-y-3">
   320→              {lineItems.map((item, idx) => (
   321→                <div key={idx} className="flex items-end gap-2 p-3 rounded-lg border bg-gray-50">
   322→                  <div className="flex-1 min-w-0">
   323→                    <Label className="text-xs">{t('وصف الخدمة', 'Service Description')}</Label>
   324→                    <Input value={item.description} onChange={e => updateLine(idx, 'description', e.target.value)} placeholder={t('وصف الخدمة المقدمة', 'Service description')} className="h-9" />
   325→                  </div>
   326→                  <div className="w-20">
   327→                    <Label className="text-xs">{t('الكمية', 'Qty')}</Label>
   328→                    <Input type="number" min="0" step="0.0001" value={item.quantity || ''} onChange={e => updateLine(idx, 'quantity', parseFloat(e.target.value) || 0)} className="h-9" dir="ltr" />
   329→                  </div>
   330→                  <div className="w-24">
   331→                    <Label className="text-xs">{t('الوحدة', 'Unit')}</Label>
   332→                    <Select value={item.unit} onValueChange={v => updateLine(idx, 'unit', v)}>
   333→                      <SelectTrigger className="h-9"><SelectValue /></SelectTrigger>
   334→                      <SelectContent>
   335→                        {serviceUnitOptions.map(o => <SelectItem key={o.value} value={o.value}>{o.label}</SelectItem>)}
   336→                      </SelectContent>
   337→                    </Select>
   338→                  </div>
   339→                  <div className="w-28">
   340→                    <Label className="text-xs">{t('سعر الوحدة', 'Unit Price')}</Label>
   341→                    <Input type="number" min="0" step="0.0001" value={item.unitPrice || ''} onChange={e => updateLine(idx, 'unitPrice', parseFloat(e.target.value) || 0)} className="h-9" dir="ltr" />
   342→                  </div>
   343→                  <div className="w-28 text-left">
   344→                    <Label className="text-xs">{t('الإجمالي', 'Total')}</Label>
   345→                    <p className="text-sm font-medium mt-1.5">{formatSAR(item.quantity * item.unitPrice, lang)}</p>
   346→                  </div>
   347→                  <Button type="button" variant="ghost" size="icon" className="size-9 shrink-0 text-rose-500" onClick={() => removeLine(idx)} disabled={lineItems.length <= 1}>
   348→                    <X className="size-4" />
   349→                  </Button>
   350→                </div>
   351→              ))}
   352→            </div>
   353→          </CardContent>
   354→        </Card>
   355→
   356→        {/* Section 5: Discount */}
   357→        <Card>
   358→          <CardHeader className="pb-4">
   359→            <CardTitle className="text-lg">{t('الخصم', 'Discount')}</CardTitle>
   360→          </CardHeader>
   361→          <CardContent>
   362→            <div className="flex items-center gap-3 flex-wrap">
   363→              <Select value={discountType} onValueChange={(v) => setDiscountType(v as 'none' | 'rate' | 'amount')}>
   364→                <SelectTrigger className="w-40"><SelectValue /></SelectTrigger>
   365→                <SelectContent>
   366→                  <SelectItem value="none">{t('بدون خصم', 'No Discount')}</SelectItem>
   367→                  <SelectItem value="rate">{t('نسبة مئوية (%)', 'Percentage (%)')}</SelectItem>
   368→                  <SelectItem value="amount">{t('مبلغ ثابت', 'Fixed Amount')}</SelectItem>
   369→                </SelectContent>
   370→              </Select>
   371→              {discountType === 'rate' && (
   372→                <Input type="number" min="0" max="100" step="0.1" value={discountRate || ''} onChange={e => setDiscountRate(parseFloat(e.target.value) || 0)} className="w-24" dir="ltr" placeholder="%" />
   373→              )}
   374→              {discountType === 'amount' && (
   375→                <Input type="number" min="0" step="0.0001" value={discountAmount || ''} onChange={e => setDiscountAmount(parseFloat(e.target.value) || 0)} className="w-32" dir="ltr" placeholder="0.00" />
   376→              )}
   377→              {finalDiscountAmount > 0 && (
   378→                <span className="text-sm text-rose-600 font-medium">-{formatSAR(finalDiscountAmount, lang)}</span>
   379→              )}
   380→            </div>
   381→          </CardContent>
   382→        </Card>
   383→
   384→        {/* Section 6: Summary */}
   385→        <Card className="bg-gray-50 border-dashed">
   386→          <CardContent className="p-6 space-y-3">
   387→            <h3 className="text-lg font-semibold mb-2">{t('ملخص الفاتورة', 'Invoice Summary')}</h3>
   388→            <div className="flex justify-between text-sm">
   389→              <span>{t('المجموع قبل الضريبة', 'Subtotal')}</span>
   390→              <span className="font-medium">{formatSAR(subtotal, lang)}</span>
   391→            </div>
   392→            {finalDiscountAmount > 0 && (
   393→              <div className="flex justify-between text-sm text-rose-600">

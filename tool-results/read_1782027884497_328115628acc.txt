    69→    69→function ImageUploadField({ 
    70→    70→  value, 
    71→    71→  onChange, 
    72→    72→  label, 
    73→    73→  labelAr,
    74→    74→  lang,
    75→    75→  hint,
    76→    76→  hintAr,
    77→    77→  previewHeight = 'h-24',
    78→    78→  accept = 'image/svg+xml,image/png,image/jpeg,image/webp',
    79→    79→}: {
    80→    80→  value: string | null
    81→    81→  onChange: (url: string | null) => void
    82→    82→  label: string
    83→    83→  labelAr: string
    84→    84→  lang: 'ar' | 'en'
    85→    85→  hint?: string
    86→    86→  hintAr?: string
    87→    87→  previewHeight?: string
    88→    88→  accept?: string
    89→    89→}) {
    90→    90→  const [uploading, setUploading] = useState(false)
    91→    91→  const [dragOver, setDragOver] = useState(false)
    92→    92→  const fileInputRef = useRef<HTMLInputElement>(null)
    93→    93→
    94→    94→  const [error, setError] = useState<string | null>(null)
    95→    95→
    96→    96→  const uploadFile = useCallback(async (file: File) => {
    97→    97→    if (file.size > 5 * 1024 * 1024) {
    98→    98→      setError(lang === 'ar' ? 'حجم الملف يتجاوز 5 ميجابايت' : 'File size exceeds 5MB')
    99→    99→      return
   100→   100→    }
   101→   101→    setUploading(true)
   102→   102→    setError(null)
   103→   103→    try {
   104→   104→      const formData = new FormData()
   105→   105→      formData.append('file', file)
   106→   106→      const res = await fetch('/api/upload', { method: 'POST', body: formData })
   107→   107→      if (!res.ok) {
   108→   108→        const errData = await res.json().catch(() => ({}))
   109→   109→        throw new Error(errData.error || 'Upload failed')
   110→   110→      }
   111→   111→      const data = await res.json()
   112→   112→      onChange(data.url)
   113→   113→    } catch (err) {
   114→   114→      const msg = err instanceof Error ? err.message : 'Unknown error'
   115→   115→      setError(lang === 'ar' ? `فشل في رفع الملف: ${msg}` : `Failed to upload file: ${msg}`)
   116→   116→    } finally {
   117→   117→      setUploading(false)
   118→   118→    }
   119→   119→  }, [lang, onChange])
   120→   120→
   121→   121→  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
   122→   122→    const file = e.target.files?.[0]
   123→   123→    if (file) {
   124→   124→      setError(null)
   125→   125→      uploadFile(file)
   126→   126→    }
   127→   127→    // Reset input so same file can be re-selected
   128→   128→    setTimeout(() => { if (fileInputRef.current) fileInputRef.current.value = '' }, 100)
   129→   129→  }
   130→   130→
   131→   131→  const handleDrop = (e: React.DragEvent) => {
   132→   132→    e.preventDefault()
   133→   133→    setDragOver(false)
   134→   134→    const file = e.dataTransfer.files?.[0]
   135→   135→    if (file) uploadFile(file)
   136→   136→  }
   137→   137→
   138→   138→  const handleDragOver = (e: React.DragEvent) => {
   139→   139→    e.preventDefault()
   140→   140→    setDragOver(true)
   141→   141→  }
   142→   142→
   143→   143→  const handleDragLeave = () => setDragOver(false)
   144→   144→
   145→   145→  const displayLabel = lang === 'ar' ? labelAr : label
   146→   146→  const displayHint = lang === 'ar' ? hintAr : hint
   147→   147→
   148→   148→  return (
   149→   149→    <div className="space-y-2">
   150→   150→      <Label>{displayLabel}</Label>
   151→   151→      {value ? (
   152→   152→        <div className={`relative rounded-lg border bg-white p-2 ${previewHeight} flex items-center justify-center group overflow-hidden`}>
   153→   153→          <img
   154→   154→            src={value}
   155→   155→            alt={displayLabel}
   156→   156→            className="max-h-20 max-w-full object-contain"
   157→   157→          />
   158→   158→          <Button
   159→   159→            variant="destructive"
   160→   160→            size="icon"
   161→   161→            className="absolute top-1 left-1 size-6 opacity-0 group-hover:opacity-100 transition-opacity z-10"
   162→   162→            onClick={() => { setError(null); onChange(null) }}
   163→   163→            type="button"
   164→   164→          >
   165→   165→            <X className="size-3" />
   166→   166→          </Button>
   167→   167→        </div>
   168→   168→      ) : (
   169→   169→        <div
   170→   170→          className={`rounded-lg border-2 border-dashed transition-colors cursor-pointer ${
   171→   171→            dragOver ? 'border-emerald-400 bg-emerald-50' : 'border-gray-300 bg-gray-50 hover:border-emerald-300 hover:bg-emerald-50/50'
   172→   172→          } ${previewHeight} flex flex-col items-center justify-center gap-1`}
   173→   173→          onClick={() => fileInputRef.current?.click()}
   174→   174→          onDrop={handleDrop}
   175→   175→          onDragOver={handleDragOver}
   176→   176→          onDragLeave={handleDragLeave}
   177→   177→        >
   178→   178→          {uploading ? (
   179→   179→            <>
   180→   180→              <Loader2 className="size-5 text-emerald-500 animate-spin" />
   181→   181→              <span className="text-xs text-muted-foreground">
   182→   182→                {lang === 'ar' ? 'جاري الرفع...' : 'Uploading...'}
   183→   183→              </span>
   184→   184→            </>
   185→   185→          ) : (
   186→   186→            <>
   187→   187→              <Upload className="size-5 text-gray-400" />
   188→   188→              <span className="text-xs text-muted-foreground">
   189→   189→                {lang === 'ar' ? 'اسحب الملف هنا أو انقر للاختيار' : 'Drag & drop or click to upload'}
   190→   190→              </span>
   191→   191→              <span className="text-[10px] text-muted-foreground">
   192→   192→                {lang === 'ar' ? 'SVG, PNG, JPG - حد أقصى 5MB' : 'SVG, PNG, JPG - Max 5MB'}
   193→   193→              </span>
   194→   194→            </>
   195→   195→          )}
   196→   196→        </div>
   197→   197→      )}
   198→   198→      <input
   199→   199→        ref={fileInputRef}
   200→   200→        type="file"
   201→   201→        accept={accept}
   202→   202→        onChange={handleFileChange}
   203→   203→        className="hidden"
   204→   204→      />
   205→   205→      {error && (
   206→   206→        <p className="text-xs text-red-600">{error}</p>
   207→   207→      )}
   208→   208→      {displayHint && !error && (
   209→   209→        <p className="text-xs text-muted-foreground">{displayHint}</p>
   210→   210→      )}
   211→   211→    </div>
   212→   212→  )
   213→   213→}

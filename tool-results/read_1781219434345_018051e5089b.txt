   595→// ============ Rental Workflow Chain Component ============
   596→function RentalWorkflowChain({ activeRentals, lang, onNavigate }: { activeRentals: EquipmentRental[]; lang: 'ar' | 'en'; onNavigate: (navItem: NavItem) => void }) {
   597→  const t = (ar: string, en: string) => lang === 'ar' ? ar : en
   598→  const hasActive = activeRentals.length > 0
   599→
   600→  const workflowSteps = RENTAL_WORKFLOW.map((step, idx) => {
   601→    const isActive = hasActive
   602→    const isLast = idx === RENTAL_WORKFLOW.length - 1
   603→    let stepCount = 0
   604→
   605→    if (step.step === 'rental-contract') stepCount = activeRentals.length
   606→    if (step.step === 'delivery' && hasActive) stepCount = activeRentals.reduce((s, r) => s + (r.deliveryOrders?.length || 0), 0)
   607→    if (step.step === 'timesheet' && hasActive) stepCount = activeRentals.reduce((s, r) => s + (r.timesheets?.length || 0), 0)
   608→    if (step.step === 'invoice' && hasActive) stepCount = activeRentals.reduce((s, r) => s + (r.timesheets?.filter(ts => ts.invoice)?.length || 0), 0)
   609→
   610→    return { ...step, isActive, stepCount, isLast }
   611→  })
   612→
   613→  return (
   614→    <div className="space-y-4">
   615→      <div className="flex items-center gap-2 mb-4">
   616→        <Link2 className="size-5 text-cyan-600" />
   617→        <h3 className="font-semibold text-cyan-700">{t('سلسلة التأجير', 'Rental Workflow Chain')}</h3>
   618→        {hasActive && <Badge className="bg-emerald-100 text-emerald-700 border-0">{t('نشط', 'Active')} ({activeRentals.length})</Badge>}
   619→      </div>
   620→
   621→      {/* Visual Chain */}
   622→      <div className="overflow-x-auto pb-2">
   623→        <div className="flex items-center gap-1 min-w-max">
   624→          {workflowSteps.map((step, idx) => (
   625→            <React.Fragment key={step.step}>
   626→              <button
   627→                onClick={() => onNavigate(step.navItem)}
   628→                className={`flex flex-col items-center gap-1 px-3 py-2 rounded-lg border-2 transition-all min-w-[80px] hover:shadow-md cursor-pointer ${
   629→                  step.isActive
   630→                    ? 'border-cyan-400 bg-cyan-50 shadow-sm'
   631→                    : 'border-gray-200 bg-gray-50 opacity-60'
   632→                }`}
   633→              >
   634→                {step.isActive ? (
   635→                  <CheckCircle2 className="size-5 text-cyan-600" />
   636→                ) : (
   637→                  <Circle className="size-5 text-gray-400" />
   638→                )}
   639→                <span className={`text-xs font-medium text-center leading-tight ${step.isActive ? 'text-cyan-700' : 'text-gray-500'}`}>
   640→                  {step.label[lang]}
   641→                </span>
   642→                {step.stepCount > 0 && (
   643→                  <Badge className="bg-cyan-600 text-white text-[10px] px-1.5 py-0 border-0">{step.stepCount}</Badge>
   644→                )}
   645→              </button>
   646→              {!step.isLast && (
   647→                <ChevronLeft className="size-4 text-gray-400 shrink-0 rotate-180" />
   648→              )}
   649→            </React.Fragment>
   650→          ))}
   651→        </div>
   652→      </div>
   653→
   654→      {/* Active Rentals Detail */}
   655→      {activeRentals.length > 0 && (
   656→        <Card className="border-cyan-200 bg-cyan-50/30">
   657→          <CardHeader className="pb-2">
   658→            <CardTitle className="text-sm text-cyan-700">{t('عقود التأجير النشطة', 'Active Rental Contracts')}</CardTitle>
   659→          </CardHeader>
   660→          <CardContent>
   661→            <div className="space-y-3">
   662→              {activeRentals.map(rental => (
   663→                <div key={rental.id} className="p-3 rounded-lg bg-white border border-cyan-200 space-y-2">
   664→                  <div className="flex items-center justify-between">
   665→                    <div className="flex items-center gap-2">
   666→                      <Badge className="bg-cyan-100 text-cyan-700 border-0">{rental.contract?.contractNo || '—'}</Badge>
   667→                      <span className="text-sm font-medium">{rental.client?.name || '—'}</span>
   668→                    </div>
   669→                    <MoneyDisplay value={rental.totalAmount} lang={lang} size="sm" bold className="text-cyan-700" />
   670→                  </div>
   671→                  <div className="grid grid-cols-2 sm:grid-cols-4 gap-2 text-xs text-muted-foreground">
   672→                    <div>{t('البداية', 'Start')}: {formatDate(rental.startDate, lang)}</div>
   673→                    <div>{t('النهاية', 'End')}: {rental.endDate ? formatDate(rental.endDate, lang) : '—'}</div>
   674→                    <div>{t('أوامر التوصيل', 'Delivery Orders')}: {rental.deliveryOrders?.length || 0}</div>
   675→                    <div>{t('سجلات الساعات', 'Timesheets')}: {rental.timesheets?.length || 0}</div>

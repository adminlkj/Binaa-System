    20→/**
    21→ * يحوّل لون hex إلى نسخة بشفافية معينة (rgba).
    22→ * يستخدم لِتوليد ألوان خلفية فاتحة من اللون الأساسي للقالب.
    23→ */
    24→function hexToRgba(hex: string, alpha: number): string {
    25→  const clean = hex.replace('#', '')
    26→  const r = parseInt(clean.substring(0, 2), 16) || 0
    27→  const g = parseInt(clean.substring(2, 4), 16) || 0
    28→  const b = parseInt(clean.substring(4, 6), 16) || 0
    29→  return `rgba(${r}, ${g}, ${b}, ${alpha})`
    30→}
    31→
    32→/**
    33→ * يولّد لوناً أغمق من اللون الأساسي (للحالات hover/active أو الحدود).
    34→ */
    35→function darkenHex(hex: string, factor = 0.15): string {
    36→  const clean = hex.replace('#', '')
    37→  const r = Math.max(0, Math.floor((parseInt(clean.substring(0, 2), 16) || 0) * (1 - factor)))
    38→  const g = Math.max(0, Math.floor((parseInt(clean.substring(2, 4), 16) || 0) * (1 - factor)))
    39→  const b = Math.max(0, Math.floor((parseInt(clean.substring(4, 6), 16) || 0) * (1 - factor)))
    40→  return `#${r.toString(16).padStart(2, '0')}${g.toString(16).padStart(2, '0')}${b.toString(16).padStart(2, '0')}`
    41→}

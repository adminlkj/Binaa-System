   155→// ============ Expense Form Dialog ============
   156→interface ExpenseFormDialogProps {
   157→  open: boolean; onOpenChange: (open: boolean) => void

   161→
   162→function ExpenseFormDialog({
   163→  open, onOpenChange, projects, activeTab: initialTab, suppliers,
   164→}: ExpenseFormDialogProps) {
   165→  const { lang } = useAppStore()
   166→  const queryClient = useQueryClient()
   167→  const { toast } = useToast()
   168→
   169→  const [tab, setTab] = useState<'project' | 'admin'>(initialTab)
   170→  const [projectId, setProjectId] = useState('')
   171→  const [supplierId, setSupplierId] = useState('')
   172→  const [category, setCategory] = useState('')
   173→  const [description, setDescription] = useState('')
   174→  const [amount, setAmount] = useState('')
   175→  const [vatAmount, setVatAmount] = useState('')
   176→  const [date, setDate] = useState('')
   177→  const [reference, setReference] = useState('')
   178→  const [payFrom, setPayFrom] = useState('TREASURY')
   179→  const [attachmentPath, setAttachmentPath] = useState('')
   180→  const [vatRate, setVatRate] = useState('0.15')
   181→
   182→  React.useEffect(() => {
   183→    if (open) {
   184→      setTab(initialTab)
   185→      setProjectId('')
   186→      setSupplierId('')
   187→      setCategory(''); setDescription('')
   188→      setAmount(''); setVatAmount(''); setDate('')
   189→      setReference(''); setPayFrom('TREASURY')
   190→      setAttachmentPath(''); setVatRate('0.15')
   191→    }
   192→  }, [open, initialTab])
   193→
   194→  const categoryOptions = tab === 'project' ? projectCategoryOptions : adminCategoryOptions
   195→
   196→  // Auto-calc VAT if applicable
   197→  const parsedAmount = parseFloat(amount) || 0
   198→  const parsedVatRate = parseFloat(vatRate) || 0.15
   199→  const autoVat = useMemo(() => {
   200→    const manualVat = parseFloat(vatAmount)
   201→    if (!isNaN(manualVat)) return manualVat
   202→    return parsedAmount * parsedVatRate
   203→  }, [amount, vatAmount, parsedAmount, parsedVatRate])
   204→
   205→  const totalAmount = parsedAmount + autoVat
   206→
   207→  // ── Accounting Account Mapping ──
   208→  const categoryAccount = category ? EXPENSE_CATEGORY_ACCOUNT_MAP[category] : null
   209→  const paymentAccount = payFrom === 'PAY_LATER'
   210→    ? SUPPLIER_ACCOUNT
   211→    : PAYMENT_METHOD_ACCOUNT_MAP[payFrom]
   212→
   213→  // ── Accounting Impact Preview Lines ──
   214→  const journalPreview = useMemo(() => {
   215→    if (!category || parsedAmount <= 0) return null
   216→
   217→    const debitAccount = categoryAccount || { code: '8630', nameAr: 'مصروفات أخرى', nameEn: 'Other Expenses' }
   218→    const creditAccount = paymentAccount || { code: '1110', nameAr: 'الصندوق', nameEn: 'Cash - Treasury' }
   219→
   220→    const debitLines: Array<{ code: string; name: string; amount: number }> = []
   221→    const creditLines: Array<{ code: string; name: string; amount: number }> = []
   222→
   223→    // Debit: Expense account
   224→    debitLines.push({
   225→      code: debitAccount.code,
   226→      name: lang === 'ar' ? debitAccount.nameAr : debitAccount.nameEn,
   227→      amount: parsedAmount,
   228→    })
   229→
   230→    // Debit: Input VAT (if any)
   231→    if (autoVat > 0) {
   232→      debitLines.push({
   233→        code: '3120',
   234→        name: lang === 'ar' ? 'ضريبة المدخلات' : 'Input VAT',
   235→        amount: autoVat,
   236→      })
   237→    }
   238→
   239→    // Credit: Payment account
   240→    creditLines.push({
   241→      code: creditAccount.code,
   242→      name: lang === 'ar' ? creditAccount.nameAr : creditAccount.nameEn,
   243→      amount: totalAmount,
   244→    })
   245→
   246→    const totalDebit = debitLines.reduce((s, l) => s + l.amount, 0)
   247→    const totalCredit = creditLines.reduce((s, l) => s + l.amount, 0)
   248→    const isBalanced = Math.abs(totalDebit - totalCredit) < 0.01
   249→
   250→    return { debitLines, creditLines, totalDebit, totalCredit, isBalanced }
   251→  }, [category, parsedAmount, autoVat, totalAmount, categoryAccount, paymentAccount, lang])
   252→
   253→  const createMutation = useMutation({
   254→    mutationFn: (data: Record<string, unknown>) =>
   255→      fetch('/api/expenses', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) })
   256→        .then(r => { if (!r.ok) throw new Error(); return r.json() }),
   257→    onSuccess: (result: { journalEntryId?: string | null; id?: string }) => {
   258→      queryClient.invalidateQueries({ queryKey: ['expenses'] })
   259→      onOpenChange(false)
   260→      // Show toast about journal entry creation
   261→      if (result?.journalEntryId) {
   262→        toast({
   263→          title: t(lang, 'تم إنشاء القيد المحاسبي', 'Accounting Entry Created'),
   264→          description: t(lang, 'تم إنشاء القيد المحاسبي بنجاح', 'The accounting journal entry has been created successfully'),
   265→        })
   266→      } else {
   267→        toast({
   268→          title: t(lang, 'تم إضافة المصروف', 'Expense Added'),
   269→          description: t(lang, 'تم إضافة المصروف بنجاح', 'The expense has been added successfully'),
   270→        })
   271→      }
   272→    },
   273→    onError: () => {
   274→      toast({
   275→        title: t(lang, 'خطأ', 'Error'),
   276→        description: t(lang, 'فشل في إضافة المصروف', 'Failed to add expense'),
   277→        variant: 'destructive',
   278→      })
   279→    },
   280→  })
   281→
   282→  const handleSubmit = (e: React.FormEvent) => {
   283→    e.preventDefault()
   284→    const isProjectTab = tab === 'project'
   285→    createMutation.mutate({
   286→      projectId: isProjectTab ? projectId : null,
   287→      expenseType: isProjectTab ? 'PROJECT' : 'INTERNAL',
   288→      supplierId: (isProjectTab || payFrom === 'PAY_LATER') ? (supplierId || null) : null,
   289→      category, description, amount,
   290→      vatRate: parsedVatRate,
   291→      vatAmount: autoVat || null,
   292→      totalAmount,
   293→      date, reference: reference || null,
   294→      payFrom,
   295→      attachmentPath: attachmentPath || null,
   296→    })
   297→  }
   298→
   299→  // PAY_LATER requires a supplier
   300→  const isPayLater = payFrom === 'PAY_LATER'
   301→  const needsSupplier = isPayLater || (tab === 'project' && ['SERVICES', 'RENT', 'MAINTENANCE', 'INSURANCE', 'CONSUMABLES'].includes(category))
   302→
   303→  return (
   304→    <Dialog open={open} onOpenChange={onOpenChange}>
   305→      <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
   306→        <DialogHeader>
   307→          <DialogTitle>{t(lang, 'مصروف جديد', 'New Expense')}</DialogTitle>
   308→          <DialogDescription>{t(lang, 'إضافة مصروف جديد', 'Add a new expense')}</DialogDescription>
   309→        </DialogHeader>
   310→        <form onSubmit={handleSubmit} className="space-y-4">
   311→          {/* Type toggle */}
   312→          <div className="flex gap-2 p-1 bg-gray-100 rounded-lg">
   313→            <button
   314→              type="button"
   315→              onClick={() => { setTab('project'); setCategory(''); setProjectId('') }}
   316→              className={`flex-1 flex items-center justify-center gap-2 px-3 py-2 rounded-md text-sm font-medium transition-all ${
   317→                tab === 'project' ? 'bg-emerald-600 text-white shadow-sm' : 'text-gray-600 hover:text-gray-800'
   318→              }`}
   319→            >
   320→              <Building2 className="size-4" />
   321→              {t(lang, 'مصروف مشروع', 'Project Expense')}
   322→            </button>
   323→            <button
   324→              type="button"
   325→              onClick={() => { setTab('admin'); setCategory(''); setProjectId('') }}
   326→              className={`flex-1 flex items-center justify-center gap-2 px-3 py-2 rounded-md text-sm font-medium transition-all ${
   327→                tab === 'admin' ? 'bg-emerald-600 text-white shadow-sm' : 'text-gray-600 hover:text-gray-800'
   328→              }`}
   329→            >
   330→              <Briefcase className="size-4" />
   331→              {t(lang, 'مصروف إداري', 'Admin Expense')}
   332→            </button>
   333→          </div>
   334→
   335→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
   336→            {/* Project selector - only for project tab */}
   337→            {tab === 'project' && (
   338→              <div className="space-y-2">
   339→                <Label>{t(lang, 'المشروع *', 'Project *')}</Label>
   340→                <Select value={projectId} onValueChange={setProjectId}>
   341→                  <SelectTrigger className="w-full"><SelectValue placeholder={t(lang, 'اختر المشروع', 'Select project')} /></SelectTrigger>
   342→                  <SelectContent>
   343→                    {projects.map(p => <SelectItem key={p.id} value={p.id}>{p.name}</SelectItem>)}
   344→                  </SelectContent>
   345→                </Select>
   346→              </div>
   347→            )}
   348→
   349→            {/* Supplier - for project tab or PAY_LATER */}
   350→            {(tab === 'project' || isPayLater) && (
   351→              <div className="space-y-2">
   352→                <Label>
   353→                  {t(lang, 'المورد', 'Supplier')}
   354→                  {needsSupplier ? ' *' : ''}
   355→                </Label>
   356→                <Select value={supplierId} onValueChange={setSupplierId}>
   357→                  <SelectTrigger className="w-full"><SelectValue placeholder={t(lang, 'اختر المورد', 'Select supplier')} /></SelectTrigger>
   358→                  <SelectContent>
   359→                    {suppliers.map(s => <SelectItem key={s.id} value={s.id}>{s.nameAr || s.name}</SelectItem>)}
   360→                  </SelectContent>
   361→                </Select>
   362→                {isPayLater && !supplierId && (
   363→                  <p className="text-xs text-amber-600">{t(lang, 'المورد إلزامي للسداد الآجل', 'Supplier is required for pay later')}</p>
   364→                )}
   365→                {tab === 'project' && ['SERVICES', 'RENT', 'MAINTENANCE', 'INSURANCE', 'CONSUMABLES'].includes(category) && !supplierId && !isPayLater && (
   366→                  <p className="text-xs text-amber-600">{t(lang, 'المورد إلزامي لهذه الفئة', 'Supplier is required for this category')}</p>
   367→                )}
   368→              </div>
   369→            )}
   370→
   371→            {/* Admin info */}
   372→            {tab === 'admin' && !isPayLater && (
   373→              <div className="space-y-2 sm:col-span-2">
   374→                <div className="flex items-center gap-2 rounded-lg bg-amber-50 border border-amber-200 px-3 py-2">
   375→                  <Briefcase className="size-4 text-amber-600 shrink-0" />
   376→                  <span className="text-sm text-amber-700">
   377→                    {t(lang, 'المصروفات الإدارية لا ترتبط بمشروع معين', 'Administrative expenses are not linked to a specific project')}
   378→                  </span>
   379→                </div>
   380→              </div>
   381→            )}
   382→
   383→            {/* Category + Account Badge */}
   384→            <div className="space-y-2">
   385→              <Label>{t(lang, 'الفئة *', 'Category *')}</Label>
   386→              <Select value={category} onValueChange={setCategory}>
   387→                <SelectTrigger className="w-full"><SelectValue placeholder={t(lang, 'اختر الفئة', 'Select category')} /></SelectTrigger>
   388→                <SelectContent>
   389→                  {categoryOptions.map(c => <SelectItem key={c.value} value={c.value}>{c[lang]}</SelectItem>)}
   390→                </SelectContent>
   391→              </Select>
   392→            </div>
   393→            {/* Account mapping badge */}
   394→            {category && categoryAccount && (
   395→              <div className="space-y-2">
   396→                <Label>{t(lang, 'الحساب', 'Account')}</Label>
   397→                <div className="flex items-center gap-2 h-10">
   398→                  <ArrowLeftRight className="size-4 text-gray-400 shrink-0" />
   399→                  <Badge className="bg-teal-50 text-teal-800 border-teal-200 border font-mono gap-1.5 px-3 py-1.5 text-sm">
   400→                    <span className="font-bold">{categoryAccount.code}</span>
   401→                    <span>{lang === 'ar' ? categoryAccount.nameAr : categoryAccount.nameEn}</span>
   402→                  </Badge>
   403→                </div>
   404→              </div>
   405→            )}
   406→
   407→            <div className="space-y-2 sm:col-span-2">
   408→              <Label>{t(lang, 'الوصف *', 'Description *')}</Label>
   409→              <Input value={description} onChange={e => setDescription(e.target.value)} placeholder={t(lang, 'وصف المصروف', 'Expense description')} required />
   410→            </div>
   411→            <div className="space-y-2">
   412→              <Label>{t(lang, 'المبلغ *', 'Amount *')}</Label>
   413→              <Input type="number" min="0" step="0.01" value={amount} onChange={e => setAmount(e.target.value)} dir="ltr" required />
   414→            </div>
   415→            <div className="space-y-2">
   416→              <Label>{t(lang, 'ضريبة القيمة المضافة (15%)', 'VAT Amount (15%)')}</Label>
   417→              <Input type="number" min="0" step="0.01" value={vatAmount} onChange={e => setVatAmount(e.target.value)} dir="ltr" placeholder={t(lang, 'تلقائي', 'Auto')} />
   418→            </div>
   419→            <div className="space-y-2">
   420→              <Label>{t(lang, 'التاريخ *', 'Date *')}</Label>
   421→              <Input type="date" value={date} onChange={e => setDate(e.target.value)} required />
   422→            </div>
   423→            <div className="space-y-2">
   424→              <Label>{t(lang, 'طريقة الدفع', 'Payment Method')}</Label>
   425→              <Select value={payFrom} onValueChange={setPayFrom}>
   426→                <SelectTrigger className="w-full"><SelectValue /></SelectTrigger>
   427→                <SelectContent>
   428→                  {payFromOptions.map(p => (
   429→                    <SelectItem key={p.value} value={p.value}>
   430→                      <div className="flex items-center gap-2">
   431→                        <span>{lang === 'ar' ? p.labelAr : p.labelEn}</span>
   432→                        <span className="text-xs text-gray-400 font-mono">({p.accountCode})</span>
   433→                      </div>
   434→                    </SelectItem>
   435→                  ))}
   436→                </SelectContent>
   437→              </Select>
